935 std::vector<long> PatchKernel::collapseCoincidentVertices(int nBins)
936 {
937  std::vector<long> collapsedVertices;
938  if (!isExpert()) {
939  return collapsedVertices;
940  }
941 
942  std::vector<std::vector<std::array<long, 2> > > bins;
943 
944  // ====================================================================== //
945  // INITIALIZE LOCAL VARIABLES //
946  // ====================================================================== //
947 
948  // Random number generator
949  srand(1223145611);
950 
951  // Resize variables
952  bins.resize(nBins * nBins * nBins);
953 
954  // ====================================================================== //
955  // SORT VERTICES ON BINS //
956  // ====================================================================== //
957 
958  // Sort vertices
959  std::unordered_map<long, long> bin_index = binSortVertex(nBins);
960 
961  // Sort cells
962  std::array<long, 2> binEntry;
963  for (const Cell &cell : m_cells) {
964  int nCellVertices = cell.getVertexCount();
965  for (int j = 0; j < nCellVertices; ++j) {
966  binEntry[0] = cell.getId();
967  binEntry[1] = j;
968 
969  long vertexId = cell.getVertex(j);
970  long binId = bin_index[vertexId];
971  bins[binId].push_back(binEntry);
972  }
973  }
974 
975  // Free memory
976  std::unordered_map<long, long>().swap(bin_index);
977 
978  // ====================================================================== //
979  // COLLAPSE DOUBLE VERTICES //
980  // ====================================================================== //
981  std::unordered_set<long> uniqueVertices;
982  uniqueVertices.reserve(getVertexCount());
983 
984  for (auto &bin : bins) {
985  int nBinCells = bin.size();
986  if (nBinCells > 0) {
987  // Randomize vertex insertion
988  std::vector<int> list;
989  utils::extractWithoutReplacement(nBinCells, nBinCells - 1, list);
990 
991  // Vertex insertion
992  KdTree<3, Vertex, long> kd(nBinCells);
993  for (int j = 0; j < nBinCells; ++j) {
994  long cellId = bin[list[j]][0];
995  Cell &cell = m_cells[cellId];
996 
997  long k = bin[list[j]][1];
998  long vertexId = cell.getVertex(k);
999  if (uniqueVertices.count(vertexId) != 0) {
1000  continue;
1001  }
1002 
1003  Vertex &vertex = m_vertices[vertexId];
1004 
1005  long collapsedVertexId;
1006  if (kd.exist(&vertex, collapsedVertexId) >= 0) {
1007  cell.setVertex(k, collapsedVertexId);
1008  } else {
1009  kd.insert(&vertex, vertexId);
1010  uniqueVertices.insert(vertexId);
1011  }
1012  }
1013  }
1014  }
1015 
1016  // Create the list of collapsed vertices
1017  size_t nCollapsedVertices = getVertexCount() - uniqueVertices.size();
1018  collapsedVertices.resize(nCollapsedVertices);
1019 
1020  size_t k = 0;
1021  for (auto vertexItr = m_vertices.begin(); vertexItr != m_vertices.end(); ++vertexItr) {
1022  long vertexId = vertexItr.getId();
1023  if (uniqueVertices.count(vertexId) != 0) {
1024  continue;
1025  }
1026 
1027  collapsedVertices[k++] = vertexId;
1028  }
1029 
1030  return collapsedVertices;
1031 }
