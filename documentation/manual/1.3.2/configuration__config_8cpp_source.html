210 Config::MultiSection Config::getSections(const std::string &key)
211 {
212  MultiSection sections;

215  auto range = m_sections->equal_range(key);

356 void Config::dump(std::ostream &out, int indentLevel) const
357 {
358  const int INDENT_SIZE = 2;
359 
360  std::string padding(INDENT_SIZE, ' ');
361  std::string indent(INDENT_SIZE * indentLevel, ' ');
362 
363  out << std::endl;
364  out << indent << "Options..." << std::endl;
365  if (getOptionCount() > 0) {
366  for (auto &entry : getOptions()) {
367  out << indent << padding << entry.first << " = " << entry.second << std::endl;
368  }
369  } else {
370  out << indent << padding << "No options." << std::endl;
371  }
372 
373  ++indentLevel;
374  for (auto &entry : getSections()) {
375  out << std::endl;
376  out << indent << padding << "::: Section " << entry.first << " :::" << std::endl;
377  entry.second->dump(out, indentLevel);
378  }
379 }
