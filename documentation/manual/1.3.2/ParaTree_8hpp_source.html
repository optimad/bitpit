485  template<class Impl>
486  void
487  communicate(DataCommInterface<Impl> & userData){
488  //BUILD SEND BUFFERS
489  std::map<int,CommBuffer> sendBuffers;
490  size_t fixedDataSize = userData.fixedSize();
491  std::map<int,u32vector >::iterator bitend = m_bordersPerProc.end();
492  std::map<int,u32vector >::iterator bitbegin = m_bordersPerProc.begin();
493  for(std::map<int,u32vector >::iterator bit = bitbegin; bit != bitend; ++bit){
494  const int & key = bit->first;
495  const u32vector & pborders = bit->second;
496  size_t buffSize = 0;
497  size_t nofPbordersPerProc = pborders.size();
498  if(fixedDataSize != 0){
499  buffSize = fixedDataSize*nofPbordersPerProc;
500  }
501  else{
502  for(size_t i = 0; i < nofPbordersPerProc; ++i){
503  buffSize += userData.size(pborders[i]);
504  }
505  }
506  //enlarge buffer to store number of pborders from this proc
507  buffSize += sizeof(int);
508  //build buffer for this proc
509  sendBuffers[key] = CommBuffer(buffSize,'a',m_comm);
510  //store number of pborders from this proc at the begining
511  MPI_Pack(&nofPbordersPerProc,1,MPI_INT,sendBuffers[key].m_commBuffer,sendBuffers[key].m_commBufferSize,&sendBuffers[key].m_pos,m_comm);
512 
513  //WRITE SEND BUFFERS
514  for(size_t j = 0; j < nofPbordersPerProc; ++j){
515  userData.gather(sendBuffers[key],pborders[j]);
516  }
517  }
518 
519  //Communicate Buffers Size
520  MPI_Request* req = new MPI_Request[sendBuffers.size()*2];
521  MPI_Status* stats = new MPI_Status[sendBuffers.size()*2];
522  int nReq = 0;
523  std::map<int,int> recvBufferSizePerProc;
524  std::map<int,CommBuffer>::iterator sitend = sendBuffers.end();
525  for(std::map<int,CommBuffer>::iterator sit = sendBuffers.begin(); sit != sitend; ++sit){
526  recvBufferSizePerProc[sit->first] = 0;
527  m_errorFlag = MPI_Irecv(&recvBufferSizePerProc[sit->first],1,MPI_UINT32_T,sit->first,m_rank,m_comm,&req[nReq]);
528  ++nReq;
529  }
530  std::map<int,CommBuffer>::reverse_iterator rsitend = sendBuffers.rend();
531  for(std::map<int,CommBuffer>::reverse_iterator rsit = sendBuffers.rbegin(); rsit != rsitend; ++rsit){
532  m_errorFlag = MPI_Isend(&rsit->second.m_commBufferSize,1,MPI_UINT32_T,rsit->first,rsit->first,m_comm,&req[nReq]);
533  ++nReq;
534  }
535  MPI_Waitall(nReq,req,stats);
536 
537  //Communicate Buffers
538  std::map<int,CommBuffer> recvBuffers;
539  std::map<int,int>::iterator ritend = recvBufferSizePerProc.end();
540  for(std::map<int,int>::iterator rit = recvBufferSizePerProc.begin(); rit != ritend; ++rit){
541  recvBuffers[rit->first] = CommBuffer(rit->second,'a',m_comm);
542  }
543  nReq = 0;
544  for(std::map<int,CommBuffer>::iterator sit = sendBuffers.begin(); sit != sitend; ++sit){
545  m_errorFlag = MPI_Irecv(recvBuffers[sit->first].m_commBuffer,recvBuffers[sit->first].m_commBufferSize,MPI_PACKED,sit->first,m_rank,m_comm,&req[nReq]);
546  ++nReq;
547  }
548  for(std::map<int,CommBuffer>::reverse_iterator rsit = sendBuffers.rbegin(); rsit != rsitend; ++rsit){
549  m_errorFlag = MPI_Isend(rsit->second.m_commBuffer,rsit->second.m_commBufferSize,MPI_PACKED,rsit->first,rsit->first,m_comm,&req[nReq]);
550  ++nReq;
551  }
552  MPI_Waitall(nReq,req,stats);
553 
554  //READ RECEIVE BUFFERS
555  int ghostOffset = 0;
556  std::map<int,CommBuffer>::iterator rbitend = recvBuffers.end();
557  std::map<int,CommBuffer>::iterator rbitbegin = recvBuffers.begin();
558  for(std::map<int,CommBuffer>::iterator rbit = rbitbegin; rbit != rbitend; ++rbit){
559  int nofGhostFromThisProc = 0;
560  MPI_Unpack(rbit->second.m_commBuffer,rbit->second.m_commBufferSize,&rbit->second.m_pos,&nofGhostFromThisProc,1,MPI_INT,m_comm);
561  for(int k = 0; k < nofGhostFromThisProc; ++k){
562  userData.scatter(rbit->second, k+ghostOffset);
563  }
564  ghostOffset += nofGhostFromThisProc;
565  }
566  delete [] req; req = NULL;
567  delete [] stats; stats = NULL;
568 
569  };
