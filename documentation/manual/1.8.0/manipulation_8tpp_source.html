40template <class T>
41void transpose(
42 std::vector< std::vector< T > > &A,
43 std::vector< std::vector< T > > &B
44) {
45
46// ========================================================================== //
47// VARIABLES DECLARATION //
48// ========================================================================== //
49
50// Local variables
51int m, n;
52
53// Counters
54int i, j;
55
56// ========================================================================== //
57// CHECK INPUT //
58// ========================================================================== //
59m = A.size();
60if (m == 0) { return; }
61n = A[0].size();
62if (n == 0) { return; }
63
64// ========================================================================== //
65// MATRIX TRANSPOSITION //
66// ========================================================================== //
67
68// Resize output variables
69B.resize(n);
70for (i = 0; i < n; ++i) {
71 B[i].resize(m, (T) 0.0);
72} //next i
73
74// Transposition
75for (i = 0; i < m; i++) {
76 for (j = 0; j < n; j++) {
77 B[j][i] = A[i][j];
78 } //next j
79} //next i
80
81return; };
