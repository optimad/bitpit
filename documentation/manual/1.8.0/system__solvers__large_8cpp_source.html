328bool PetscManager::initialize(bool debug)
329{
330 // Early return if PETSc is already initialized
331 PetscBool isPetscInitialized;
332 PetscInitialized(&isPetscInitialized);
333 if (isPetscInitialized) {
334 return false;
335 }
336
337#if BITPIT_ENABLE_MPI==1
338 // Early return if MPI is already finalized
339 int isMPIFinalized;
340 MPI_Finalized(&isMPIFinalized);
341 if (isMPIFinalized) {
342 throw std::runtime_error("PETSc finalization cannot be called after MPI finaliation.");
343 }
344#endif
345
346 // Generate command line arguments
347 //
348 // The first argument is the executable name and it is set to a
349 // dummy value.
350 std::string help = "None";
351 std::string programName = "bitpit_petsc_manager";
352
353 int argc = 1 + m_options.size();
354 char **argv = new char*[argc + 1];
355 argv[0] = strdup(programName.data());
356 for (std::size_t i = 0; i < m_options.size(); i++) {
357 argv[1 + i] = strdup(m_options[i].data());
358 }
359 argv[argc] = nullptr;
360
361#if BITPIT_ENABLE_MPI==1
362 // Initialize MPI
363 int isMPIInitialized;
364 MPI_Initialized(&isMPIInitialized);
365 if (!isMPIInitialized) {
366 MPI_Init(&argc, &argv);
367 m_externalMPIInitialization = false;
368 }
369#endif
370
371 // Initialize PETSc
372 PetscInitialize(&argc, &argv, 0, help.data());
373 m_externalPETScInitialization = false;
374
375 // Enable log view
376 if (debug) {
377 enableLogView();
378 }
379
380 // Clean-up command line arguments
381 for (int i = 0; i < argc; ++i) {
382 free(argv[i]);
383 }
384
385 delete[] argv;
386
387 // Initialization completed
388 return true;
389}

416bool PetscManager::finalize(bool permanent)
417{
418#if BITPIT_ENABLE_MPI==1
419 // Early return if MPI is already finalized
420 int isMPIFinalized;
421 MPI_Finalized(&isMPIFinalized);
422 if (isMPIFinalized) {
423 return false;
424 }
425#endif
426
427 // Early return if PETSc was initialized externally
428 if (m_externalPETScInitialization) {
429 return false;
430 }
431
432 // Finalize PETSc
433 PetscBool isPetscFinalized;
434 PetscFinalized(&isPetscFinalized);
435 if (!isPetscFinalized) {
436 PetscFinalize();
437 }
438
439#if BITPIT_ENABLE_MPI==1
440 // Finalize MPI
441 if (permanent) {
442 if (!m_externalMPIInitialization) {
443 MPI_Finalize();
444 }
445 }
446#endif
447
448 return (!isPetscFinalized);
449}

590void SystemSolver::setPermutations(long nRows, const long *rowRanks, long nCols, const long *colRanks)
591{
592 // Permutation has to be set before assembling the system
593 if (isAssembled()) {
594 throw std::runtime_error("Unable to set the permutations. The system is already assembled.");
595 }
596
597 // Reset existing permutations
598 resetPermutations();
599
600 // Create new permutations
601 PetscInt *rowPermutationsStorage;
602 PetscMalloc(nRows * sizeof(PetscInt), &rowPermutationsStorage);
603 for (long i = 0; i < nRows; ++i) {
604 rowPermutationsStorage[i] = rowRanks[i];
605 }
606
607#if BITPIT_ENABLE_MPI == 1
608 ISCreateGeneral(m_communicator, nRows, rowPermutationsStorage, PETSC_OWN_POINTER, &m_rowPermutation);
609#else
610 ISCreateGeneral(PETSC_COMM_SELF, nRows, rowPermutationsStorage, PETSC_OWN_POINTER, &m_rowPermutation);
611#endif
612 ISSetPermutation(m_rowPermutation);
613
614 PetscInt *colPermutationsStorage;
615 PetscMalloc(nCols * sizeof(PetscInt), &colPermutationsStorage);
616 for (long j = 0; j < nCols; ++j) {
617 colPermutationsStorage[j] = colRanks[j];
618 }
619
620#if BITPIT_ENABLE_MPI == 1
621 ISCreateGeneral(m_communicator, nCols, colPermutationsStorage, PETSC_OWN_POINTER, &m_colPermutation);
622#else
623 ISCreateGeneral(PETSC_COMM_SELF, nCols, colPermutationsStorage, PETSC_OWN_POINTER, &m_colPermutation);
624#endif
625 ISSetPermutation(m_colPermutation);
626}

886void SystemSolver::solve()
887{
888 // Check if the system is assembled
889 if (!isAssembled()) {
890 throw std::runtime_error("Unable to solve the system. The system is not yet assembled.");
891 }
892
893 // Check if the system is set up
894 if (!isSetUp()) {
895 setUp();
896 }
897
898 // Perfrom actions before KSP solution
899 preKSPSolveActions();
900
901 // Force consistency
902 if (m_forceConsistency) {
903 removeNullSpaceFromRHS();
904 }
905
906 // Solve the system
907 if (!m_transpose) {
908 m_KSPStatus.error = KSPSolve(m_KSP, m_rhs, m_solution);
909 } else {
910 m_KSPStatus.error = KSPSolveTranspose(m_KSP, m_rhs, m_solution);
911 }
912
913 // Set solver info
914 if (m_KSPStatus.error == 0) {
915 KSPGetIterationNumber(m_KSP, &m_KSPStatus.its);
916 KSPGetConvergedReason(m_KSP, &m_KSPStatus.convergence);
917 } else {
918 m_KSPStatus.its = -1;
919 m_KSPStatus.convergence = KSP_DIVERGED_BREAKDOWN;
920 }
921
922 // Perfrom actions after KSP solution
923 postKSPSolveActions();
924}

968void SystemSolver::matrixCreate(const SystemMatrixAssembler &assembler)
969{
970 const PetscInt *rowRanks = nullptr;
971 if (m_rowPermutation) {
972 ISGetIndices(m_rowPermutation, &rowRanks);
973 }
974
975 // Set sizes
976 long nRows = assembler.getRowCount();
977 long nCols = assembler.getColCount();
978
979#if BITPIT_ENABLE_MPI == 1
980 long nGlobalRows = assembler.getRowGlobalCount();
981 long nGlobalCols = assembler.getColGlobalCount();
982#endif
983
984 // Preallocation information
985 std::vector<int> d_nnz(nRows, 0);
986#if BITPIT_ENABLE_MPI == 1
987 std::vector<int> o_nnz(nRows, 0);
988
989 long firstColGlobalId = assembler.getColGlobalOffset();
990 long lastColGlobalId = firstColGlobalId + nCols - 1;
991
992 ConstProxyVector<long> rowPattern;
993#endif
994
995 for (long row = 0; row < nRows; ++row) {
996 long matrixRow = row;
997 if (m_rowPermutation) {
998 matrixRow = rowRanks[matrixRow];
999 }
1000
1001 d_nnz[row] = assembler.getRowNZCount(matrixRow);
1002#if BITPIT_ENABLE_MPI == 1
1003 if (m_partitioned) {
1004 assembler.getRowPattern(matrixRow, &rowPattern);
1005
1006 int nRowNZ = rowPattern.size();
1007 for (int k = 0; k < nRowNZ; ++k) {
1008 long columnGlobalId = rowPattern[k];
1009 if (columnGlobalId < firstColGlobalId || columnGlobalId > lastColGlobalId) {
1010 ++o_nnz[row];
1011 }
1012 }
1013
1014 d_nnz[row] -= o_nnz[row];
1015 }
1016#endif
1017 }
1018
1019 // Create the matrix
1020#if BITPIT_ENABLE_MPI == 1
1021 MatCreateAIJ(m_communicator, nRows, nCols, nGlobalRows, nGlobalCols, 0, d_nnz.data(), 0, o_nnz.data(), &m_A);
1022#else
1023 MatCreateSeqAIJ(PETSC_COMM_SELF, nRows, nCols, 0, d_nnz.data(), &m_A);
1024#endif
1025
1026 // Cleanup
1027 if (m_rowPermutation) {
1028 ISRestoreIndices(m_rowPermutation, &rowRanks);
1029 }
1030}

1037void SystemSolver::matrixFill(const SystemMatrixAssembler &assembler)
1038{
1039 const long nRows = assembler.getRowCount();
1040 const long nCols = assembler.getColCount();
1041 const long maxRowNZ = assembler.getMaxRowNZCount();
1042
1043 const PetscInt *rowRanks = nullptr;
1044 if (m_rowPermutation) {
1045 ISGetIndices(m_rowPermutation, &rowRanks);
1046 }
1047
1048 IS invColPermutation;
1049 const PetscInt *colInvRanks = nullptr;
1050 if (m_colPermutation) {
1051 ISInvertPermutation(m_colPermutation, nCols, &invColPermutation);
1052 ISGetIndices(invColPermutation, &colInvRanks);
1053 }
1054
1055 // Create the matrix
1056 if (maxRowNZ > 0) {
1057 std::vector<PetscInt> rowNZGlobalIds(maxRowNZ);
1058 std::vector<PetscScalar> rowNZValues(maxRowNZ);
1059
1060 PetscInt rowGlobalOffset;
1061 MatGetOwnershipRangeColumn(m_A, &rowGlobalOffset, nullptr);
1062
1063 PetscInt colGlobalBegin;
1064 PetscInt colGlobalEnd;
1065 MatGetOwnershipRangeColumn(m_A, &colGlobalBegin, &colGlobalEnd);
1066
1067 ConstProxyVector<long> rowPattern;
1068 ConstProxyVector<double> rowValues;
1069 for (long row = 0; row < nRows; ++row) {
1070 long matrixRow = row;
1071 if (m_rowPermutation) {
1072 matrixRow = rowRanks[matrixRow];
1073 }
1074
1075 assembler.getRowData(matrixRow, &rowPattern, &rowValues);
1076
1077 const int nRowNZ = rowPattern.size();
1078 const PetscInt globalRow = rowGlobalOffset + row;
1079 for (int k = 0; k < nRowNZ; ++k) {
1080 long matrixGlobalCol = rowPattern[k];
1081
1082 long globalCol = matrixGlobalCol;
1083 if (m_colPermutation) {
1084 if (globalCol >= colGlobalBegin && globalCol < colGlobalEnd) {
1085 long col = globalCol - colGlobalBegin;
1086 col = colInvRanks[col];
1087 globalCol = colGlobalBegin + col;
1088 }
1089 }
1090
1091 rowNZGlobalIds[k] = globalCol;
1092 rowNZValues[k] = rowValues[k];
1093 }
1094
1095 MatSetValues(m_A, 1, &globalRow, nRowNZ, rowNZGlobalIds.data(), rowNZValues.data(), INSERT_VALUES);
1096 }
1097 }
1098
1099 // Let petsc build the matrix
1100 MatAssemblyBegin(m_A, MAT_FINAL_ASSEMBLY);
1101 MatAssemblyEnd(m_A, MAT_FINAL_ASSEMBLY);
1102
1103 // No new allocations are now allowed
1104 //
1105 // When updating the matrix it will not be possible to alter the pattern,
1106 // it will be possible to change only the values.
1107 MatSetOption(m_A, MAT_NEW_NONZERO_LOCATIONS, PETSC_FALSE);
1108
1109 // Cleanup
1110 if (m_rowPermutation) {
1111 ISRestoreIndices(m_rowPermutation, &rowRanks);
1112 }
1113
1114 if (m_colPermutation) {
1115 ISDestroy(&invColPermutation);
1116 }
1117}

1131void SystemSolver::matrixUpdate(long nRows, const long *rows, const SystemMatrixAssembler &assembler)
1132{
1133 // Update element values
1134 PetscInt rowGlobalOffset;
1135 MatGetOwnershipRange(m_A, &rowGlobalOffset, nullptr);
1136
1137 const long maxRowNZ = std::max(assembler.getMaxRowNZCount(), 0L);
1138
1139 std::vector<PetscInt> rawRowPattern(maxRowNZ);
1140 std::vector<PetscScalar> rawRowValues(maxRowNZ);
1141
1142 ConstProxyVector<long> rowPattern;
1143 ConstProxyVector<double> rowValues;
1144 for (long n = 0; n < nRows; ++n) {
1145 assembler.getRowData(n, &rowPattern, &rowValues);
1146 const int nRowElements = rowPattern.size();
1147 if (nRowElements == 0) {
1148 continue;
1149 }
1150
1151 // Get global row
1152 long row;
1153 if (rows) {
1154 row = rows[n];
1155 } else {
1156 row = n;
1157 }
1158
1159 const PetscInt globalRow = rowGlobalOffset + row;

1579void SystemSolver::preKSPSetupActions()
1580{
1581 // Preconditioner configuration
1582 PCType preconditionerType;
1583#if BITPIT_ENABLE_MPI == 1
1584 if (isPartitioned()) {
1585 preconditionerType = PCASM;
1586 } else {
1587 preconditionerType = PCILU;
1588 }
1589#else

1592
1593 PC preconditioner;
1594 KSPGetPC(m_KSP, &preconditioner);
1595 PCSetType(preconditioner, preconditionerType);
1596 if (strcmp(preconditionerType, PCASM) == 0) {
1597 if (m_KSPOptions.overlap != PETSC_DEFAULT) {
1598 PCASMSetOverlap(preconditioner, m_KSPOptions.overlap);
1599 }
1600 } else if (strcmp(preconditionerType, PCILU) == 0) {
1601 if (m_KSPOptions.levels != PETSC_DEFAULT) {
1602 PCFactorSetLevels(preconditioner, m_KSPOptions.levels);
1603 }
1604 }
1605
1606 // Solver configuration
1607 KSPSetType(m_KSP, KSPFGMRES);
1608 if (m_KSPOptions.restart != PETSC_DEFAULT) {
1609 KSPGMRESSetRestart(m_KSP, m_KSPOptions.restart);
1610 }
1611 if (m_KSPOptions.rtol != PETSC_DEFAULT || m_KSPOptions.atol != PETSC_DEFAULT || m_KSPOptions.maxits != PETSC_DEFAULT) {
1612 KSPSetTolerances(m_KSP, m_KSPOptions.rtol, m_KSPOptions.atol, PETSC_DEFAULT, m_KSPOptions.maxits);
1613 }
1614 KSPSetInitialGuessNonzero(m_KSP, PETSC_TRUE);
