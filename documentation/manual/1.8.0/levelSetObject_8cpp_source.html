300LevelSetIntersectionStatus LevelSetObject::intersectSurface(long id, LevelSetIntersectionMode mode) const{
301
302 double incircle, circumcircle;
303
304 double absoluteDistance = std::abs(getValue(id));
305 double distanceTolerance = m_kernel->getMesh()->getTol();
306
307 switch(mode){
308 case LevelSetIntersectionMode::FAST_GUARANTEE_TRUE:
309 {
310 incircle = m_kernel->computeCellIncircle(id) ;
311 if(utils::DoubleFloatingLessEqual()(absoluteDistance, incircle, distanceTolerance, distanceTolerance)){
312 return LevelSetIntersectionStatus::TRUE;
313 } else {
314 return LevelSetIntersectionStatus::FALSE;
315 }
316
317 break;
318 }
319
320 case LevelSetIntersectionMode::FAST_GUARANTEE_FALSE:
321 {
322 circumcircle = m_kernel->computeCellCircumcircle(id) ;
323 if(utils::DoubleFloatingGreater()(absoluteDistance, circumcircle, distanceTolerance, distanceTolerance)){
324 return LevelSetIntersectionStatus::FALSE;
325 } else {
326 return LevelSetIntersectionStatus::TRUE;
327 }
328
329 break;
330 }
331
332 case LevelSetIntersectionMode::FAST_FUZZY:
333 {
334 circumcircle = m_kernel->computeCellCircumcircle(id) ;
335 if(utils::DoubleFloatingGreater()(absoluteDistance, circumcircle, distanceTolerance, distanceTolerance)){
336 return LevelSetIntersectionStatus::FALSE;
337 }
338
339 incircle = m_kernel->computeCellIncircle(id) ;
340 if(utils::DoubleFloatingLessEqual()(absoluteDistance, incircle, distanceTolerance, distanceTolerance)){
341 return LevelSetIntersectionStatus::TRUE;
342 }
343
344 return LevelSetIntersectionStatus::CLOSE;
345
346 break;
347 }
348
349 case LevelSetIntersectionMode::ACCURATE:
350 {
351 circumcircle = m_kernel->computeCellCircumcircle(id) ;
352 if(utils::DoubleFloatingGreater()(absoluteDistance, circumcircle, distanceTolerance, distanceTolerance)){
353 return LevelSetIntersectionStatus::FALSE;
354 }
355
356 incircle = m_kernel->computeCellIncircle(id) ;
357 if(utils::DoubleFloatingLessEqual()(absoluteDistance, incircle, distanceTolerance, distanceTolerance)){
358 return LevelSetIntersectionStatus::TRUE;
359 }
360
361 std::array<double,3> root = computeProjectionPoint(id);
362 std::array<double,3> normal = getGradient(id);
363 if( m_kernel->intersectCellPlane(id,root,normal, distanceTolerance) ){
364 return LevelSetIntersectionStatus::TRUE;
365 } else {
366 return LevelSetIntersectionStatus::FALSE;
367 }
368
369 break;
370 }
371 }
372
373 BITPIT_UNREACHABLE("cannot reach");
374
375}

524void LevelSetObject::enableVTKOutput( LevelSetWriteField fieldset, const std::string &objectName, bool enable) {
525
526 std::vector<LevelSetWriteField> fields;
527
528 if( fieldset==LevelSetWriteField::ALL){
529 fields.push_back(LevelSetWriteField::VALUE);
530 fields.push_back(LevelSetWriteField::GRADIENT);
531 fields.push_back(LevelSetWriteField::NORMAL);
532 fields.push_back(LevelSetWriteField::PART);
533
534 } else if ( fieldset==LevelSetWriteField::DEFAULT){
535 fields.push_back(LevelSetWriteField::VALUE);
536 fields.push_back(LevelSetWriteField::GRADIENT);
537
538 } else {
539 fields.push_back(fieldset);
540 }
541
542 VTK &vtkWriter = m_kernel->getMesh()->getVTK() ;
543 for( const LevelSetWriteField &field : fields){
544 // Update the list of enabled VTK outputs
545 if (enable) {
546 m_enabledVTKOutputs.insert(field) ;
547 } else {
548 m_enabledVTKOutputs.erase(field) ;
549 }
550
551
552 // Get name of the field
553 std::stringstream nameStream;
554 nameStream << "levelset";
555 switch(field){
556
557 case LevelSetWriteField::VALUE:
558 nameStream << "Value_" << objectName;
559 break;
560
561 case LevelSetWriteField::GRADIENT:
562 nameStream << "Gradient_" << objectName;
563 break;
564
565 case LevelSetWriteField::NORMAL:
566 nameStream << "Normal_" << objectName;
567 break;
568
569 case LevelSetWriteField::PART:
570 nameStream << "PartId_" << objectName;
571 break;
572
573 default:
574 std::cout << " field " << (int) field << std::endl;
575 throw std::runtime_error ("Unsupported value of field in LevelSetObject::addDataToVTK() ");
576 break;
577 }
578
579 std::string name = nameStream.str();
580
581 // Check if the state of the filed is already the requested one
582 if (enable == vtkWriter.hasData(name)) {
583 continue;
584 }
585
586 // Process the field
587 if (!enable) {
588 vtkWriter.removeData( name);
589 } else {
590 switch(field){
591
592 case LevelSetWriteField::VALUE:
593 vtkWriter.addData<double>( name, VTKFieldType::SCALAR, VTKLocation::CELL, this);
594 break;
595
596 case LevelSetWriteField::GRADIENT:
597 vtkWriter.addData<double>( name, VTKFieldType::VECTOR, VTKLocation::CELL, this);
598 break;
599
600 case LevelSetWriteField::NORMAL:
601 vtkWriter.addData<double>( name, VTKFieldType::VECTOR, VTKLocation::CELL, this);
602 break;
603
604 case LevelSetWriteField::PART:
605 vtkWriter.addData<int>( name, VTKFieldType::SCALAR, VTKLocation::CELL, this);
606 break;
607
608 default:
609 throw std::runtime_error ("Unsupported value of field in LevelSetObject::addDataToVTK() ");
610 break;
611
612 }
613 }
614 }

627void LevelSetObject::flushData( std::fstream &stream, const std::string &name, VTKFormat format){
628
629
630 if(utils::string::keywordInString(name,"levelsetValue")){
631
632 void (*writeFunctionPtr)(std::fstream &, const double &) = nullptr;
633
634 if(format==VTKFormat::APPENDED){
635 writeFunctionPtr = genericIO::flushBINARY<double>;
636 } else if(format==VTKFormat::ASCII){
637 writeFunctionPtr = genericIO::flushASCII<double>;
638 } else {
639 BITPIT_UNREACHABLE("Non-existent VTK format.");
640 }
641
642 for( const Cell &cell : m_kernel->getMesh()->getVTKCellWriteRange() ){
643 long cellId = cell.getId();
644 double value = getValue(cellId);
645 (*writeFunctionPtr)(stream,value);
646 }
647
648 } else if( utils::string::keywordInString(name,"levelsetGradient")){
649
650 void (*writeFunctionPtr)(std::fstream &, const std::array<double,3> &) = nullptr;
651
652 if(format==VTKFormat::APPENDED){
653 writeFunctionPtr = genericIO::flushBINARY<std::array<double,3>>;
654 } else if(format==VTKFormat::ASCII){
655 writeFunctionPtr = genericIO::flushASCII<std::array<double,3>>;
656 } else {
657 BITPIT_UNREACHABLE("Non-existent VTK format.");
658 }
659
660 for( const Cell &cell : m_kernel->getMesh()->getVTKCellWriteRange() ){
661 long cellId = cell.getId();
662 const std::array<double,3> &value = getGradient(cellId);
663 (*writeFunctionPtr)(stream,value);
664 }
665
666 } else if( utils::string::keywordInString(name,"levelsetNormal")){
667
668 void (*writeFunctionPtr)(std::fstream &, const std::array<double,3> &) = nullptr;
669
670 if(format==VTKFormat::APPENDED){
671 writeFunctionPtr = genericIO::flushBINARY<std::array<double,3>>;

693 BITPIT_UNREACHABLE("Non-existent VTK format.");

697 long cellId = cell.getId();
698 int value = getPart(cellId);

734void LevelSetObject::startExchange( const std::unordered_map<int,std::vector<long>> &sendList,
735 DataCommunicator *dataCommunicator){
736
737 // Fill the send buffer with the content from the LevelSetObject base
738 // class and the specific derived class.
739 for (const auto &entry : sendList) {
740 // Create an empty send
741 int rank = entry.first;
742 dataCommunicator->setSend(rank, 0);
743
744 // Write data in the buffer
745 SendBuffer &buffer = dataCommunicator->getSendBuffer(rank);
746 writeCommunicationBuffer(entry.second, buffer);
747 }
748
749 // Discover the receives
750 dataCommunicator->discoverRecvs();
751
752 // Start the sends
753 dataCommunicator->startAllRecvs();
754
755 // Start the sends
756 dataCommunicator->startAllSends();
757
758}
759
766void LevelSetObject::completeExchange( const std::unordered_map<int,std::vector<long>> &recvList,
767 DataCommunicator *dataCommunicator){
768
769 // Check which data communications have arrived. For those which are
770 // available start reading the databuffer into the data structure of
771 // LevelSetObject and its derived classes.
772 int nCompletedRecvs = 0;
773 while (nCompletedRecvs < dataCommunicator->getRecvCount()) {
774 int rank = dataCommunicator->waitAnyRecv();
775
776 RecvBuffer &dataBuffer = dataCommunicator->getRecvBuffer(rank);
777 readCommunicationBuffer(recvList.at(rank), dataBuffer);
778
779 ++nCompletedRecvs;
780 }
781
782 dataCommunicator->waitAllSends();
783 dataCommunicator->finalize();
784}
