152void ReferenceElementInfo::initializeFaceEdges(const std::vector<const ReferenceElementInfo *> &facesInfo,
153 const std::vector<const ReferenceElementInfo *> &edgesInfo)
154{
155 for (int k = 0; k < nFaces; ++k) {
156 const ReferenceElementInfo &faceInfo = *(facesInfo[k]);
157
158 int nFaceEdges = faceInfo.nFaces;
159 int faceEdgeCounter = 0;
160 for (int i = 0; i < nFaceEdges; ++i) {
161 const ReferenceElementInfo &faceEdgeInfo = *(edgesInfo[i]);
162
163 // Connectivity of the edge associated to the face
164 const int *localFaceEdgeConnect = faceInfo.faceConnectStorage[i].data();
165
166 std::set<int> faceEdgeConnect;
167 for (int n = 0; n < faceEdgeInfo.nVertices; ++n) {
168 int localVertexId = localFaceEdgeConnect[n];
169 int vertexId = faceConnectStorage[k][localVertexId];
170
171 faceEdgeConnect.insert(vertexId);
172 }
173
174 // Search the edge that has the same connectivity of the face edge
175 for (int j = 0; j < nEdges; ++j) {
176 const ReferenceElementInfo &guessEdgeInfo = *(edgesInfo[j]);
177
178 // If face edge and the guess edge have a different type, the
179 // two edge cannot be the same.
180 if (guessEdgeInfo.type != faceEdgeInfo.type) {
181 continue;
182 }
183
184 // If the connecitivity of the face edge and the one of the
185 // guess edge are the same, the two edges coincides.
186 const std::set<int> guessEdgeConnect = std::set<int>(edgeConnectStorage[j].begin(), edgeConnectStorage[j].begin() + guessEdgeInfo.nVertices);
187 if (faceEdgeConnect == guessEdgeConnect) {
188 faceEdgeStorage[k][faceEdgeCounter] = j;
189 ++faceEdgeCounter;
190 }
191 }
192 }
193
194 assert(faceEdgeCounter == nFaceEdges);
195 }
196}

230double Reference3DElementInfo::evalSurfaceArea(const std::array<double, 3> *vertexCoords) const
231{
232 std::array<std::array<double, 3>, MAX_ELEM_VERTICES> faceVertexCoords;
233
234 double area = 0;
235 for (int i = 0; i < nFaces; ++i) {
236 ElementType faceType = faceTypeStorage[i];
237 const Reference2DElementInfo &faceInfo = static_cast<const Reference2DElementInfo &>(getInfo(faceType));
238 for (int n = 0; n < faceInfo.nVertices; ++n) {
239 faceVertexCoords[n] = vertexCoords[faceConnectStorage[i][n]];
240 }
241
242 area += faceInfo.evalArea(faceVertexCoords.data());
243 }
244
245 return area;

281void Reference3DElementInfo::evalPointProjection(const std::array<double, 3> &point, const std::array<double, 3> *vertexCoords,
282 std::array<double, 3> *projection, double *distance) const
283{
284 std::array<std::array<double, 3>, MAX_ELEM_VERTICES> faceVertexCoords;
285
286 *distance = std::numeric_limits<double>::max();
287 for (int i = 0; i < nFaces; ++i) {
288 ElementType faceType = faceTypeStorage[i];
289 const Reference2DElementInfo &faceInfo = static_cast<const Reference2DElementInfo &>(getInfo(faceType));
290 for (int n = 0; n < faceInfo.nVertices; ++n) {
291 faceVertexCoords[n] = vertexCoords[faceConnectStorage[i][n]];
292 }
293
294 double faceDistance;
295 std::array<double, 3> faceProjection;
296 faceInfo.evalPointProjection(point, faceVertexCoords.data(), &faceProjection, &faceDistance);
297
298 if (faceDistance < *distance) {
299 *distance = faceDistance;
300 *projection = faceProjection;
301 }
302 }
303}

312double Reference3DElementInfo::evalPointDistance(const std::array<double, 3> &point, const std::array<double, 3> *vertexCoords) const
313{
314 std::array<std::array<double, 3>, MAX_ELEM_VERTICES> faceVertexCoords;
315
316 double distance = std::numeric_limits<double>::max();
317 for (int i = 0; i < nFaces; ++i) {
318 ElementType faceType = faceTypeStorage[i];
319 const Reference2DElementInfo &faceInfo = static_cast<const Reference2DElementInfo &>(getInfo(faceType));
320 for (int n = 0; n < faceInfo.nVertices; ++n) {
321 faceVertexCoords[n] = vertexCoords[faceConnectStorage[i][n]];

323

325 }

708double ReferenceHexahedronInfo::evalVolume(const std::array<double, 3> *vertexCoords) const
709{
710 std::array<std::array<double, 3>, MAX_ELEM_VERTICES> pyramidVertexCoords;
711
712 const ReferenceQuadInfo &quadInfo = static_cast<const ReferenceQuadInfo &>(getInfo(ElementType::QUAD));
713 const ReferencePyramidInfo &pyramidInfo = static_cast<const ReferencePyramidInfo &>(getInfo(ElementType::PYRAMID));
714
715 // The centroid is the apex of all the pyramids
716 pyramidVertexCoords[4] = vertexCoords[0];
717 for (int i = 1; i < nVertices; ++i) {
718 pyramidVertexCoords[4] += vertexCoords[i];
719 }
720 pyramidVertexCoords[4] = pyramidVertexCoords[4] / double(nVertices);
721
722 // Sum the volume of the pyramids having a face as the base and the
723 // centroid as the apex.
724 double volume = 0.;
725 for (int i = 0; i < nFaces; ++i) {
726 for (int n = 0; n < quadInfo.nVertices; ++n) {
727 pyramidVertexCoords[quadInfo.nVertices - n - 1] = vertexCoords[faceConnectStorage[i][n]];
728 }
729
730 volume += pyramidInfo.evalVolume(pyramidVertexCoords.data());
731 }
732
733 return volume;
734}

882double ReferencePyramidInfo::evalVolume(const std::array<double, 3> *vertexCoords) const
883{

885 const std::array<double, 3> DB = vertexCoords[3] - vertexCoords[1];
886 const std::array<double, 3> AC = vertexCoords[2] - vertexCoords[0];

1037 pyramidVertexCoords[4] = vertexCoords[0];

1050 for (int n = 0; n < triaInfo.nVertices; ++n) {
1051 tetraVertexCoords[triaInfo.nVertices - n - 1] = vertexCoords[faceConnectStorage[i][n]];
1052 }

1055 }
1056

1059 pyramidVertexCoords[quadInfo.nVertices - n - 1] = vertexCoords[faceConnectStorage[i][n]];
