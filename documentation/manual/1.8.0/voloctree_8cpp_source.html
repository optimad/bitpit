516void VolOctree::__setDimension(int dimension)
517{
518 if (m_tree->getDim() > 0 && dimension != m_tree->getDim()) {
519 throw std::runtime_error ("The dimension does not match the dimension of the octree.");
520 }
521
522 // Initialize local edges/vertex/faces association
523 std::vector<std::vector<int>>().swap(m_octantLocalFacesOnVertex);
524 std::vector<std::vector<int>>().swap(m_octantLocalEdgesOnVertex);
525 std::vector<std::vector<int>>().swap(m_octantLocalFacesOnEdge);
526
527 if (dimension == 3) {
528 m_octantLocalFacesOnVertex.reserve(8);
529 m_octantLocalFacesOnVertex.push_back({{0, 2, 4}});
530 m_octantLocalFacesOnVertex.push_back({{1, 2, 4}});
531 m_octantLocalFacesOnVertex.push_back({{0, 3, 4}});
532 m_octantLocalFacesOnVertex.push_back({{1, 3, 4}});
533 m_octantLocalFacesOnVertex.push_back({{0, 2, 5}});
534 m_octantLocalFacesOnVertex.push_back({{1, 2, 5}});
535 m_octantLocalFacesOnVertex.push_back({{0, 3, 5}});
536 m_octantLocalFacesOnVertex.push_back({{1, 3, 5}});
537
538 m_octantLocalEdgesOnVertex.reserve(8);
539 m_octantLocalEdgesOnVertex.push_back({{0, 2, 4}});
540 m_octantLocalEdgesOnVertex.push_back({{1, 2, 5}});
541 m_octantLocalEdgesOnVertex.push_back({{0, 3, 6}});
542 m_octantLocalEdgesOnVertex.push_back({{1, 3, 7}});
543 m_octantLocalEdgesOnVertex.push_back({{4, 8, 10}});
544 m_octantLocalEdgesOnVertex.push_back({{5, 9, 10}});
545 m_octantLocalEdgesOnVertex.push_back({{6, 8, 11}});
546 m_octantLocalEdgesOnVertex.push_back({{7, 9, 11}});
547
548 m_octantLocalFacesOnEdge.reserve(12);
549 m_octantLocalFacesOnEdge.push_back({{0, 4}});
550 m_octantLocalFacesOnEdge.push_back({{1, 4}});
551 m_octantLocalFacesOnEdge.push_back({{2, 4}});
552 m_octantLocalFacesOnEdge.push_back({{3, 4}});
553 m_octantLocalFacesOnEdge.push_back({{0, 2}});
554 m_octantLocalFacesOnEdge.push_back({{1, 2}});
555 m_octantLocalFacesOnEdge.push_back({{0, 3}});
556 m_octantLocalFacesOnEdge.push_back({{1, 3}});
557 m_octantLocalFacesOnEdge.push_back({{0, 5}});
558 m_octantLocalFacesOnEdge.push_back({{1, 5}});
559 m_octantLocalFacesOnEdge.push_back({{2, 5}});
560 m_octantLocalFacesOnEdge.push_back({{3, 5}});
561 } else {
562 m_octantLocalFacesOnVertex.reserve(4);
563 m_octantLocalFacesOnVertex.push_back({{0, 2}});
564 m_octantLocalFacesOnVertex.push_back({{1, 2}});
565 m_octantLocalFacesOnVertex.push_back({{0, 3}});
566 m_octantLocalFacesOnVertex.push_back({{1, 3}});
567 }
568
569 // Info of the cell type
570 ElementType cellType;
571 if (dimension == 3) {
572 cellType = ElementType::VOXEL;
573 } else {
574 cellType = ElementType::PIXEL;
575 }
576
577 m_cellTypeInfo = &ReferenceElementInfo::getInfo(cellType);
578
579 // Info on the interface type
580 ElementType interfaceType;
581 if (dimension == 3) {
582 interfaceType = ElementType::PIXEL;
583 } else {
584 interfaceType = ElementType::LINE;
585 }
586
587 m_interfaceTypeInfo = &ReferenceElementInfo::getInfo(interfaceType);
588}

593void VolOctree::setBoundingBox()
594{
595 std::array<double, 3> minPoint;
596 std::array<double, 3> maxPoint;
597
598 // Get the bounding box from the tree
599 m_tree->getBoundingBox(minPoint, maxPoint);
600
601#if BITPIT_ENABLE_MPI==1
602 // The tree is only evaluating the bounding box of the internal octants,
603 // we need to consider also ghosts cells.
604 CellConstIterator endItr = ghostCellConstEnd();
605 for (CellConstIterator ghostCellItr = ghostCellConstBegin(); ghostCellItr != endItr; ++ghostCellItr) {
606 ConstProxyVector<long> ghostVertexIds = ghostCellItr->getVertexIds();
607 int nGhostCellVertices = ghostVertexIds.size();
608 for (int i = 0; i < nGhostCellVertices; ++i) {
609 const std::array<double, 3> coords = m_vertices[ghostVertexIds[i]].getCoords();
610 for (int d = 0; d < 3; ++d) {
611 minPoint[d] = std::min(coords[d], minPoint[d]);
612 maxPoint[d] = std::max(coords[d], maxPoint[d]);
613 }
614 }
615 }
616#endif
617
618 // Set the bounding box
619 setBoundingBox(minPoint, maxPoint);
620}

632void VolOctree::simulateCellUpdate(long id, adaption::Marker marker, std::vector<Cell> *virtualCells, PiercedVector<Vertex, long> *virtualVertices) const
633{
634 // Get virtual post-update octants
635 OctantInfo octantInfo = getCellOctant(id);
636 const Octant *octant = getOctantPointer(octantInfo);
637
638 int8_t markerValue;
639 switch (marker) {
640
641 case adaption::MARKER_COARSEN:
642 markerValue = -1;
643 break;
644
645 case adaption::MARKER_REFINE:
646 markerValue = 1;
647 break;
648
649 default:
650 markerValue = 0;
651
652 }
653
654 int nMaxVirtualOctants;
655 if (getDimension() == 3) {
656 nMaxVirtualOctants = 8;
657 } else {
658 nMaxVirtualOctants = 4;
659 }
660
661 std::vector<Octant> virtualOctants;
662 virtualOctants.reserve(nMaxVirtualOctants);
663 m_tree->expectedOctantAdapt(octant, markerValue, &virtualOctants);
664 std::size_t nVirtualOctants = virtualOctants.size();
665
666 // Create virtual post-update cells
667 int nCellVertices = m_cellTypeInfo->nVertices;
668
669 virtualVertices->clear();
670 virtualVertices->reserve(nCellVertices * nVirtualOctants);
671
672 virtualCells->clear();
673 virtualCells->resize(nVirtualOctants);
674
675 for (std::size_t k = 0; k < nVirtualOctants; ++k) {
676 const Octant *virtualOctant = &(virtualOctants[k]);
677
678 std::unique_ptr<long[]> connectStorage = std::unique_ptr<long[]>(new long[nCellVertices]);
679 for (int i = 0; i < nCellVertices; ++i) {
680 long vertexId = k * nCellVertices + i;
681 connectStorage[i] = vertexId;
682 virtualVertices->emplaceBack(vertexId, vertexId, m_tree->getNode(virtualOctant, i));
683 }
684
685 virtualCells->emplace_back(id, m_cellTypeInfo->type, std::move(connectStorage), true, false, false);
686 }
687}

969std::vector<adaption::Info> VolOctree::_adaptionPrepare(bool trackAdaption)
970{
971 BITPIT_UNUSED(trackAdaption);
972
973 if (getSpawnStatus() == SPAWN_NEEDED) {
974 throw std::runtime_error ("The initial import has not been performed.");
975 }
976
977 // Call pre-adapt routine
978 m_tree->preadapt();
979
980 // Track adaption changes
981 adaption::InfoCollection adaptionData;
982 if (trackAdaption) {
983 // Current rank
984 int currentRank = -1;
985#if BITPIT_ENABLE_MPI==1
986 currentRank = getRank();
987#endif
988
989 // Track internal and ghosts octants that will be coarsend/refined
990 std::vector<uint32_t> treeIds;
991 std::vector<int8_t> treeMarkers;
992 std::vector<bool> treeGhosts;
993 m_tree->getPreMapping(treeIds, treeMarkers, treeGhosts);
994 std::size_t nUpdatedOctants = treeIds.size();
995
996 adaption::Info *adaptionInfo = nullptr;
997 uint64_t previousFatherMorton = PABLO::INVALID_MORTON;
998 for (std::size_t n = 0; n < nUpdatedOctants; ++n) {
999 OctantInfo octantInfo(treeIds[n], !treeGhosts[n]);
1000 Octant *octant = getOctantPointer(octantInfo);
1001
1002 adaption::Type adaptionType;
1003 if (treeMarkers[n] > 0) {
1004 adaptionType = adaption::TYPE_REFINEMENT;
1005 } else {
1006 adaptionType = adaption::TYPE_COARSENING;
1007 }
1008
1009 bool createAdaption = true;
1010 if (adaptionType == adaption::TYPE_COARSENING) {
1011 uint64_t fatherMorton = octant->computeFatherMorton();
1012 if (fatherMorton == previousFatherMorton) {
1013 createAdaption = false;
1014 } else {
1015 previousFatherMorton = fatherMorton;
1016 }
1017 }
1018
1019 if (createAdaption) {
1020 std::size_t adaptionInfoId = adaptionData.create(adaptionType, adaption::ENTITY_CELL, currentRank);
1021 adaptionInfo = &(adaptionData[adaptionInfoId]);
1022 }
1023
1024 assert(adaptionInfo);
1025 adaptionInfo->previous.emplace_back();
1026 long &cellId = adaptionInfo->previous.back();
1027 cellId = getOctantId(octantInfo);
1028 }

1030#if BITPIT_ENABLE_MPI==1
1031 // Ghost cells will be removed

1039 for (CellConstIterator itr = beginItr; itr != endItr; ++itr) {
1040 adaptionInfo.previous.emplace_back();
1041 long &deletedGhostCellId = adaptionInfo.previous.back();
1042 deletedGhostCellId = itr.getId();
1043 }
1044 }
1045#endif
1046 }
1047
1048 return adaptionData.dump();
1049}

1099
1107std::vector<adaption::Info> VolOctree::sync(bool trackChanges)
1108{
1109 log::cout() << ">> Syncing patch..." << std::endl;
1110
1111 // Detect if we are in import-from-scratch mode
1112 //
1113 // In import-from-scratch mode we start form an empty patch and we need
1114 // to import all the octants of the tree. If we are importing the tree
1115 // from scratch there are no cells to delete/renumber.
1116 bool importFromScratch = empty();
1117
1118 // Last operation on the tree
1119 ParaTree::Operation lastTreeOperation = m_tree->getLastOperation();
1120 if (lastTreeOperation == ParaTree::OP_ADAPT_UNMAPPED && !importFromScratch) {
1121 throw std::runtime_error ("Unable to sync the patch after an unmapped adaption");
1122 }
1123
1124 // Info on the tree
1125 long nOctants = m_tree->getNumOctants();
1126 long nPreviousOctants = m_octantToCell.size();
1127
1128 log::cout() << ">> Number of octants : " << nOctants << std::endl;
1129
1130 // Info on the tree
1131 long nGhostsOctants = m_tree->getNumGhosts();
1132 long nPreviousGhosts = m_ghostToCell.size();
1133
1134 // Initialize tracking data
1135 adaption::InfoCollection adaptionData;
1136
1137 // Current rank
1138 int currentRank = -1;
1139#if BITPIT_ENABLE_MPI==1
1140 currentRank = getRank();
1141#endif
1142
1143 // Extract information for transforming the patch
1144 //
1145 // If there are no cells in the mesh we need to import all
1146 // octants.
1147 log::cout() << ">> Extract information for transforming the patch...";
1148
1149 std::vector<bool> unmappedOctants(nPreviousOctants, true);
1150 std::vector<OctantInfo> addedOctants;
1151 std::vector<RenumberInfo> renumberedOctants;
1152 std::vector<DeleteInfo> deletedOctants;
1153
1154 addedOctants.reserve(nOctants + nGhostsOctants);
1155 if (!importFromScratch) {
1156 renumberedOctants.reserve(nPreviousOctants + nPreviousGhosts);
1157 deletedOctants.reserve(nPreviousOctants + nPreviousGhosts);
1158 }
1159
1160 uint32_t treeId = 0;
1161 std::vector<uint32_t> mapper_octantMap;
1162 std::vector<bool> mapper_ghostFlag;
1163 std::vector<int> mapper_octantRank;
1164 while (treeId < (uint32_t) nOctants) {
1165 // Octant mapping
1166 mapper_octantMap.clear();
1167 mapper_ghostFlag.clear();
1168 mapper_octantRank.clear();
1169 if (!importFromScratch) {
1170 m_tree->getMapping(treeId, mapper_octantMap, mapper_ghostFlag, mapper_octantRank);
1171 }
1172
1173 // Adaption type
1174 adaption::Type adaptionType = adaption::TYPE_NONE;
1175 if (importFromScratch) {
1176 adaptionType = adaption::TYPE_CREATION;
1177 } else if (lastTreeOperation == ParaTree::OP_ADAPT_MAPPED) {
1178 bool isNewR = m_tree->getIsNewR(treeId);
1179 if (isNewR) {
1180 adaptionType = adaption::TYPE_REFINEMENT;
1181 } else {
1182 bool isNewC = m_tree->getIsNewC(treeId);
1183 if (isNewC) {
1184 adaptionType = adaption::TYPE_COARSENING;
1185 } else if (treeId != mapper_octantMap.front()) {
1186 adaptionType = adaption::TYPE_RENUMBERING;
1187 }
1188 }
1189#if BITPIT_ENABLE_MPI==1
1190 } else if (lastTreeOperation == ParaTree::OP_LOADBALANCE || lastTreeOperation == ParaTree::OP_LOADBALANCE_FIRST) {
1191 if (currentRank != mapper_octantRank.front()) {
1192 adaptionType = adaption::TYPE_PARTITION_RECV;
1193 } else if (treeId != mapper_octantMap.front()) {
1194 adaptionType = adaption::TYPE_RENUMBERING;
1195 }
1196#endif
1197 }
1198
1199 // If the octant cell has not been modified we can skip to the next
1200 // octant.
1201 if (adaptionType == adaption::TYPE_NONE) {
1202 unmappedOctants[treeId] = false;
1203 ++treeId;
1204 continue;
1205 }
1206
1207 // Re-numbered cells just need to be added to the proper list.
1208 //
1209 // Renumbered cells are not tracked, because the re-numbering
1210 // only happens inside VolOctree.
1211 if (adaptionType == adaption::TYPE_RENUMBERING) {
1212 uint32_t previousTreeId = mapper_octantMap.front();
1213 OctantInfo previousOctantInfo(previousTreeId, !mapper_ghostFlag.front());
1214 long cellId = getOctantId(previousOctantInfo);
1215 renumberedOctants.emplace_back(cellId, treeId);
1216 unmappedOctants[previousTreeId] = false;
1217
1218 // No more work needed, skip to the next octant
1219 ++treeId;
1220 continue;
1221 }
1222
1223 // Handle other kind of adaption
1224 //
1225 // New octants need to be imported into the patch,
1226 // whereas cells associated to previous octants
1227 // need to be removed.
1228 //
1229 // If the user want to track adaption, adaption
1230 // data needs to be filled.
1231
1232 // Current tree ids that will be imported
1233 long nCurrentTreeIds;
1234 if (importFromScratch) {
1235 nCurrentTreeIds = nOctants - treeId;
1236 } else if (adaptionType == adaption::TYPE_REFINEMENT) {
1237 nCurrentTreeIds = pow(2, getDimension());
1238 } else {
1239 nCurrentTreeIds = 1;
1240 }
1241
1242 const long lastCurrentTreeId = treeId + nCurrentTreeIds;
1243 for (int currentTreeId = treeId; currentTreeId < lastCurrentTreeId; ++currentTreeId) {
1244 addedOctants.emplace_back(currentTreeId, true);
1245 }
1246
1247 // Cells that will be removed
1248 //
1249 // Mark the cells associated to previous local octants for deletion.
1250 if (!importFromScratch) {
1251 int nPreviousTreeIds = mapper_octantMap.size();
1252 for (int k = 0; k < nPreviousTreeIds; ++k) {
1253#if BITPIT_ENABLE_MPI==1
1254 // Only local cells can be deleted
1255 if (mapper_octantRank[k] != currentRank) {
1256 continue;
1257 }
1258#endif
1259
1260 // Ghost octants will be processed later
1261 if (mapper_ghostFlag[k]) {
1262 continue;
1263 }
1264
1265 // Mark previous octant for deletion
1266 uint32_t previousTreeId = mapper_octantMap[k];
1267 OctantInfo previousOctantInfo(previousTreeId, !mapper_ghostFlag[k]);
1268 long cellId = getOctantId(previousOctantInfo);
1269 deletedOctants.emplace_back(cellId, adaptionType);
1270
1271 unmappedOctants[previousTreeId] = false;
1272 }
1273 }
1274
1275 // Adaption tracking
1276 //
1277 // The adaption info associated to the octants that has been received
1278 // from external partitions will contain the current octants sorted by
1279 // their tree id (we are looping over the octants in that order), this
1280 // is the same order that will be used on the process that has sent
1281 // the octants. Since the order is the same, the two processes are able
1282 // to exchange cell data without any additional extra communication
1283 // (they already know the list of cells for which data is needed and
1284 // the order in which these data will be sent).
1285 if (trackChanges) {
1286 // Rank assocated to the adaption info
1287 int rank = currentRank;
1288#if BITPIT_ENABLE_MPI==1
1289 if (adaptionType == adaption::TYPE_PARTITION_RECV) {
1290 rank = mapper_octantRank[0];
1291 }
1292#endif
1293
1294 // Get the adaption info
1295 std::size_t infoId = adaptionData.create(adaptionType, adaption::ENTITY_CELL, rank);
1296 adaption::Info &adaptionInfo = adaptionData[infoId];
1297
1298 // Current status
1299 //
1300 // We don't know the id of the current status, because those
1301 // cells are not yet in the mesh. Store the trre id, and
1302 // make the translation later.
1303 //
1304 // WARNING: tree id are uint32_t wherase adaptionInfo stores
1305 // id as long.
1306 adaptionInfo.current.reserve(nCurrentTreeIds);
1307 auto addedOctantsIter = addedOctants.cend() - nCurrentTreeIds;
1308 while (addedOctantsIter != addedOctants.cend()) {
1309 adaptionInfo.current.emplace_back();
1310 long &adaptionId = adaptionInfo.current.back();
1311 adaptionId = (*addedOctantsIter).id;
1312
1313 addedOctantsIter++;
1314 }
1315
1316 // Previous cells
1317 //
1318 // A coarsening can merge togheter cells of different processes.
1319 // However, since the coarsening is limited to one level, the
1320 // previous cells will always be internal or among the ghost of
1321 // the current process.
1322 int nPreviousCellIds = mapper_octantMap.size();
1323 adaptionInfo.previous.reserve(nPreviousCellIds);
1324 for (int k = 0; k < nPreviousCellIds; ++k) {
1325 long previousCellId;
1326#if BITPIT_ENABLE_MPI==1
1327 if (mapper_octantRank[k] != currentRank) {
1328 previousCellId = Cell::NULL_ID;
1329 } else
1330#endif
1331 {
1332 OctantInfo previousOctantInfo(mapper_octantMap[k], !mapper_ghostFlag[k]);
1333 previousCellId = getOctantId(previousOctantInfo);
1334 }
1335
1336 adaptionInfo.previous.emplace_back();
1337 long &adaptionId = adaptionInfo.previous.back();
1338 adaptionId = previousCellId;
1339 }
1340 }
1341
1342 // Incremente tree id
1343 treeId += nCurrentTreeIds;
1344 }
1345
1346 log::cout() << " Done" << std::endl;
1347
1348#if BITPIT_ENABLE_MPI==1
1349 // New ghost octants need to be added
1350 for (uint32_t treeId = 0; treeId < (uint32_t) nGhostsOctants; ++treeId) {
1351 addedOctants.emplace_back(treeId, false);
1352 }
1353#endif
1354
1355 // Remove octants that are no more in the tree
1356 if (!importFromScratch) {
1357#if BITPIT_ENABLE_MPI==1
1358 // Cells that have been send to other processes need to be removed
1359 PabloUniform::LoadBalanceRanges loadBalanceRanges = m_tree->getLoadBalanceRanges();
1360 for (const auto &rankEntry : loadBalanceRanges.sendRanges) {
1361 int rank = rankEntry.first;
1362 if (rank == currentRank) {
1363 continue;
1364 }
1365
1366 adaption::Type deletionType;
1367 if (loadBalanceRanges.sendAction == PabloUniform::LoadBalanceRanges::ACTION_DELETE) {
1368 deletionType = adaption::TYPE_DELETION;
1369 } else {
1370 deletionType = adaption::TYPE_PARTITION_SEND;
1371 }
1372
1373 uint32_t beginTreeId = rankEntry.second[0];
1374 uint32_t endTreeId = rankEntry.second[1];
1375 for (uint32_t treeId = beginTreeId; treeId < endTreeId; ++treeId) {
1376 OctantInfo octantInfo(treeId, true);
1377 long cellId = getOctantId(octantInfo);
1378 deletedOctants.emplace_back(cellId, deletionType, rank);
1379 unmappedOctants[treeId] = false;
1380 }
1381 }
1382
1383 // Previous ghosts cells need to be removed
1384 if (nPreviousGhosts > 0) {
1385 for (uint32_t ghostTreeId = 0; ghostTreeId < nPreviousGhosts; ++ghostTreeId) {
1386 OctantInfo ghostOctantInfo(ghostTreeId, false);
1387 long ghostCellId = getOctantId(ghostOctantInfo);
1388 deletedOctants.emplace_back(ghostCellId, adaption::TYPE_DELETION);
1389 }
1390 }
1391#endif
1392
1393 // Remove unmapped octants
1394 //
1395 // A coarsening that merges cells from different processes, can leave, on
1396 // the processes which own the ghost octants involved in the coarsening,
1397 // some octants that are not mapped.
1398 for (uint32_t previousTreeId = 0; previousTreeId < nPreviousOctants; ++previousTreeId) {
1399 if (unmappedOctants[previousTreeId]) {
1400 OctantInfo octantInfo = OctantInfo(previousTreeId, true);
1401 long cellId = getOctantId(octantInfo);
1402 deletedOctants.emplace_back(cellId, adaption::TYPE_DELETION);
1403 }
1404 }
1405 }
1406
1407 // Enable advanced editing
1408 setExpert(true);
1409
1410 // Renumber cells
1411 renumberCells(renumberedOctants);
1412
1413 // Remove deleted cells
1414 StitchInfo stitchInfo;
1415 if (deletedOctants.size() > 0) {
1416 log::cout() << ">> Removing non-existing cells...";
1417
1418 // Track changes
1419 //
1420 // The adaption info associated to the octants that has been sent
1421 // to external partitions will contain the current octants sorted by
1422 // their tree id (they were added to the deleted octants list in that
1423 // order), this is the same order that will be used on the process
1424 // that has received the octants. Since the order is the same, the two
1425 // processes are able to exchange cell data without any additional
1426 // extra communication (they already know the list of cells for which
1427 // data is needed and the order in which these data will be sent).
1428 if (trackChanges) {
1429 // Track deleted cells
1430 std::unordered_set<long> sendAdaptionInfo;
1431 std::unordered_set<long> removedInterfaces;
1432 for (const DeleteInfo &deleteInfo : deletedOctants) {
1433 // Cell id
1434 long cellId = deleteInfo.cellId;
1435
1436 // Adaption tracking
1437 //
1438 // Only cells deleted from a real deletion or a partition send
1439 // needs to be tracked here, the other cells will be tracked
1440 // where the adaption that deleted the cell is tracked.
1441 adaption::Type adaptionType = deleteInfo.trigger;
1442 bool adaptionIsDeletion = (adaptionType == adaption::TYPE_DELETION);
1443 bool adaptionIsSend = (adaptionType == adaption::TYPE_PARTITION_SEND);
1444
1445 if (adaptionIsDeletion || adaptionIsSend) {
1446 int rank = deleteInfo.rank;
1447 std::size_t adaptionInfoId = adaptionData.create(adaptionType, adaption::ENTITY_CELL, rank);
1448 adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
1449 adaptionInfo.previous.emplace_back();
1450 long &deletedId = adaptionInfo.previous.back();
1451 deletedId = cellId;
1452
1453 // Keep track of adaption info for the send cells
1454 if (adaptionIsSend) {
1455 sendAdaptionInfo.insert(adaptionInfoId);
1456 }
1457 }
1458
1459 // List of deleted interfaces
1460 const Cell &cell = m_cells.at(cellId);
1461 long nCellInterfaces = cell.getInterfaceCount();
1462 const long *interfaces = cell.getInterfaces();
1463 for (int k = 0; k < nCellInterfaces; ++k) {
1464 long interfaceId = interfaces[k];
1465 if (interfaceId >= 0) {
1466 removedInterfaces.insert(interfaceId);
1467 }
1468 }
1469 }
1470
1471#if BITPIT_ENABLE_MPI==1
1472 // Sort sent cells
1473 //
1474 // We cannot use native functions to evaluate the position of the
1475 // cells because the octants associated to the cells no longer
1476 // exist on the octree. The cells are still there, therefore we
1477 // can evaluate the cell positions using generic patch functions.
1478 for (int adaptionInfoId : sendAdaptionInfo) {
1479 adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
1480 std::sort(adaptionInfo.previous.begin(), adaptionInfo.previous.end(), CellPositionLess(*this, false));
1481 }
1482#endif
1483
1484 // Adaption info for the deleted interfaces
1485 std::size_t adaptionInfoId = adaptionData.create(adaption::TYPE_DELETION, adaption::ENTITY_INTERFACE, currentRank);
1486 adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
1487 adaptionInfo.previous.reserve(removedInterfaces.size());
1488 for (long interfaceId : removedInterfaces) {
1489 adaptionInfo.previous.emplace_back();
1490 long &deletedInterfaceId = adaptionInfo.previous.back();
1491 deletedInterfaceId = interfaceId;
1492 }
1493 }
1494
1495 // Delete cells
1496 stitchInfo = deleteCells(deletedOctants);
1497
1498 log::cout() << " Done" << std::endl;
1499 log::cout() << ">> Cells removed: " << deletedOctants.size() << std::endl;
1500 }
1501
1502 std::vector<DeleteInfo>().swap(deletedOctants);
1503
1504 // Import added cells
1505 std::vector<long> createdCells;
1506 if (addedOctants.size() > 0) {
1507 log::cout() << ">> Importing new octants...";
1508
1509 createdCells = importCells(addedOctants, stitchInfo);
1510
1511 log::cout() << " Done" << std::endl;
1512 log::cout() << ">> Octants imported: " << addedOctants.size() << std::endl;
1513 }
1514
1515 StitchInfo().swap(stitchInfo);
1516
1517 // Disable advanced editing
1518 setExpert(false);
1519
1520 // Track mesh adaption
1521 if (trackChanges) {
1522 // Complete mesh adaption info for the cells
1523 for (auto &adaptionInfo : adaptionData.data()) {
1524 if (adaptionInfo.entity != adaption::ENTITY_CELL) {
1525 continue;
1526 }
1527
1528 // Map ids of the added cells
1529 int nCurrentIds = adaptionInfo.current.size();
1530 for (int k = 0; k < nCurrentIds; ++k) {
1531 long cellId = m_octantToCell.at(adaptionInfo.current[k]);
1532 adaptionInfo.current[k] = cellId;
1533 }
1534
1535#if BITPIT_ENABLE_MPI==1
1536 // Sort received cells
1537 //
1538 // To match the sorting done on the procesor that sent the cells,
1539 // we don't use the native functions to evaluate the position of
1540 // the cells.
1541 adaption::Type adaptionType = adaptionInfo.type;
1542 if (adaptionType == adaption::TYPE_PARTITION_RECV) {
1543 std::sort(adaptionInfo.current.begin(), adaptionInfo.current.end(), CellPositionLess(*this, false));
1544 }
1545#endif
1546 }
1547
1548#if BITPIT_ENABLE_MPI==1
1549 // Track created ghosts cells
1550 if (nGhostsOctants > 0) {
1551 std::size_t adaptionInfoId = adaptionData.create(adaption::TYPE_CREATION, adaption::ENTITY_CELL, currentRank);
1552 adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
1553
1554 adaptionInfo.current.reserve(nGhostsOctants);
1555 auto cellIterator = m_cellToGhost.cbegin();
1556 while (cellIterator != m_cellToGhost.cend()) {
1557 adaptionInfo.current.emplace_back();
1558 long &adaptionId = adaptionInfo.current.back();
1559 adaptionId = cellIterator->first;
1560
1561 cellIterator++;
1562 }
1563 }
1564#endif
1565
1566 // Track created interfaces
1567 if (createdCells.size() > 0) {
1568 // List of unique interfaces that have been created
1569 std::unordered_set<long> createdInterfaces;
1570 for (const auto &cellId : createdCells) {
1571 const Cell &cell = m_cells.at(cellId);
1572 long nCellInterfaces = cell.getInterfaceCount();
1573 const long *interfaces = cell.getInterfaces();
1574 for (int k = 0; k < nCellInterfaces; ++k) {
1575 long interfaceId = interfaces[k];
1576 if (interfaceId >= 0) {
1577 createdInterfaces.insert(interfaceId);
1578 }
1579 }
1580 }
1581
1582 // Adaption info
1583 std::size_t infoId = adaptionData.create(adaption::TYPE_CREATION, adaption::ENTITY_INTERFACE, currentRank);
1584 adaption::Info &adaptionInfo = adaptionData[infoId];
1585 adaptionInfo.current.reserve(createdInterfaces.size());
1586 for (long interfaceId : createdInterfaces) {
1587 adaptionInfo.current.emplace_back();
1588 long &createdInterfaceId = adaptionInfo.current.back();
1589 createdInterfaceId = interfaceId;
1590 }
1591 }
1592 }
1593
1594 // Done
1595 return adaptionData.dump();
1596}

1598
1607VolOctree::StitchInfo VolOctree::deleteCells(const std::vector<DeleteInfo> &deletedOctants)
1608{
1609 // Info of the cells
1610 int nCellVertices = m_cellTypeInfo->nVertices;
1611 int nCellFaces = m_cellTypeInfo->nFaces;
1612
1613 // Delete the cells
1614 std::unordered_set<long> deadVertices;
1615
1616 std::vector<long> deadCells;
1617 deadCells.reserve(deletedOctants.size());
1618 for (const DeleteInfo &deleteInfo : deletedOctants) {
1619 long cellId = deleteInfo.cellId;
1620 const Cell &cell = m_cells[cellId];
1621
1622 // List vertices to remove
1623 //
1624 // For now, all cell vertices will be listed. Later, the vertex of
1625 // the dangling faces will be removed from the list.
1626 ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
1627 for (int k = 0; k < nCellVertices; ++k) {
1628 long vertexId = cellVertexIds[k];
1629 deadVertices.insert(vertexId);
1630 }
1631
1632 // Remove patch-tree associations
1633 const OctantInfo &octantInfo = getCellOctant(cellId);
1634 uint32_t treeId = octantInfo.id;
1635
1636 if (cell.isInterior()) {
1637 m_cellToOctant.erase(cellId);
1638
1639 auto octantToCellItr = m_octantToCell.find(treeId);
1640 if (octantToCellItr->second == cellId) {
1641 m_octantToCell.erase(octantToCellItr);
1642 }
1643 } else {
1644 m_cellToGhost.erase(cellId);
1645
1646 auto ghostToCellItr = m_ghostToCell.find(treeId);
1647 if (ghostToCellItr->second == cellId) {
1648 m_ghostToCell.erase(ghostToCellItr);
1649 }
1650 }
1651
1652 // Cell needs to be removed
1653 deadCells.push_back(cellId);
1654 }
1655
1656 PatchKernel::deleteCells(deadCells);
1657
1658 // Prune cell adjacencies and interfaces
1659 //
1660 // At this stage we cannot fully update adjacencies and interfaces, but
1661 // we need to remove stale adjacencies and interfaces.
1662 pruneStaleAdjacencies();
1663
1664 pruneStaleInterfaces();
1665
1666 // All the vertices belonging to the dangling cells has to be kept
1667 //
1668 // We need to keep all the vertices that lie on a dangling face and belong
1669 // to cells that were not deleted. These are the vertices of the dangling
1670 // faces, plus all the vertices that lie on the borders of the dangling
1671 // faces. Latter vertices arise from three-dimensional configurations where
1672 // a small cell has been deleted and, among its non-deleted neighburs, there
1673 // areboth larger cells and cells of the same size. In this case, a vertex
1674 // of the deleted small cell can lie on one of the edges of a bigger cell
1675 // and belong also to a smaller cell. If only the bigger cell is dangling,
1676 // that vertex will lie on a dangling face (it's on the edge of the cell,
1677 // hence on the border of the face), but it's not one of the vertices of
1678 // the dangling cells.
1679 //
1680 // To identify all vertices that need to be kept, we consider the vertices
1681 // of the dangling faces and the vertices of the smaller neighbouring faces
1682 // (i.e., the faces of the smaller neigbours of the dangling cells that lie
1683 // on the faces of the dangling cells).
1684 //
1685 // Morover we need to build a map between the patch numbering and the
1686 // octree numbering of the vertices of the dangling cells. This map will
1687 // be used when imprting the octants to stitch the imported octants to
1688 // the existing cells.
1689 StitchInfo stitchVertices;
1690 for (const auto &entry: m_alteredCells) {
1691 // Consider only dangling faces
1692 if (!testAlterationFlags(entry.second, FLAG_DANGLING)) {
1693 continue;
1694 }
1695
1696 // Vertices of the cell
1697 long cellId = entry.first;
1698 const Cell &cell = m_cells[cellId];
1699 ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
1700
1701 OctantInfo octantInfo = getCellOctant(cellId);
1702 Octant *octant = getOctantPointer(octantInfo);
1703
1704 for (int k = 0; k < nCellVertices; ++k) {
1705 long vertexId = cellVertexIds[k];
1706 uint64_t vertexTreeKey = m_tree->computeNodePersistentKey(octant, k);
1707 stitchVertices.insert({vertexTreeKey, vertexId});
1708 deadVertices.erase(vertexId);
1709 }
1710
1711 // Vertices of neighbours
1712 //
1713 // We need to select the neighbours smaller than the current cell and
1714 // keep the vertices of the faces that are shared with the current cell.
1715 int cellLevel = octant->getLevel();
1716
1717 for (int face = 0; face < nCellFaces; ++face) {
1718 int nFaceAdjacencies = cell.getAdjacencyCount(face);
1719 const long *faceAdjacencies = cell.getAdjacencies(face);
1720 for (int i = 0; i < nFaceAdjacencies; ++i) {
1721 int neighId = faceAdjacencies[i];
1722 OctantInfo neighOctantInfo = getCellOctant(neighId);
1723 Octant *neighOctant = getOctantPointer(neighOctantInfo);
1724 int neighLevel = neighOctant->getLevel();
1725 if (neighLevel <= cellLevel) {
1726 continue;
1727 }
1728
1729 const Cell &neigh = m_cells[neighId];
1730 int neighFace = findAdjoinNeighFace(cell, face, neigh);
1731 const int *localNeighFaceConnect = m_cellTypeInfo->faceConnectStorage[neighFace].data();
1732 ConstProxyVector<long> faceVertexIds = neigh.getFaceVertexIds(neighFace);
1733 std::size_t nFaceVertexIds = faceVertexIds.size();
1734 for (std::size_t k = 0; k < nFaceVertexIds; ++k) {
1735 long vertexId = faceVertexIds[k];
1736 uint64_t vertexTreeKey = m_tree->computeNodePersistentKey(neighOctant, localNeighFaceConnect[k]);
1737 stitchVertices.insert({vertexTreeKey, vertexId});
1738 deadVertices.erase(vertexId);
1739 }
1740 }
1741 }
1742 }
1743
1744 // Delete the vertices
1745 std::vector<long> deadVerticesList(deadVertices.cbegin(), deadVertices.cend());
1746 PatchKernel::deleteVertices(deadVerticesList);
1747
1748 // Done
1749 return stitchVertices;
1750}

1802 for (const OctantInfo &octantInfo : octantInfoList) {
1803 Octant *octant = getOctantPointer(octantInfo);
1804 for (int k = 0; k < nCellVertices; ++k) {
1805 uint64_t vertexTreeKey = m_tree->computeNodePersistentKey(octant, k);
1806 if (stitchInfo.count(vertexTreeKey) == 0) {

1808 std::array<double, 3> nodeCoords = m_tree->getNode(octant, k);
1809

1811 long vertexId;

1813#if BITPIT_ENABLE_MPI==1
1814 VertexIterator vertexIterator = addVertex(std::move(nodeCoords));
1815#else
1816 VertexIterator vertexIterator = addVertex(std::move(nodeCoords));
1817#endif
1818 vertexId = vertexIterator.getId();
1819 } else {
1820 utils::binary::read(*restoreStream, vertexId);
1821
1822#if BITPIT_ENABLE_MPI==1
1823 int rank;
1824 utils::binary::read(*restoreStream, rank);
1825
1826 restoreVertex(std::move(nodeCoords), rank, vertexId);
1827#else
1828 int dummtRank;
1829 utils::binary::read(*restoreStream, dummtRank);
1830
1831 restoreVertex(std::move(nodeCoords), vertexId);
1832#endif
1833 }
1834
1835 // Add the vertex to the stitching info
1836 stitchInfo[vertexTreeKey] = vertexId;
1837 }
1838 }
1839 }
1840
1841 // Reserve space for the maps
1842 long nOctants = m_tree->getNumOctants();
1843 m_cellToOctant.reserve(nOctants);
1844 m_octantToCell.reserve(nOctants);
1845
1846 long nGhostsOctants = m_tree->getNumGhosts();
1847 m_cellToGhost.reserve(nGhostsOctants);
1848 m_ghostToCell.reserve(nGhostsOctants);
1849

1851 size_t octantInfoListSize = octantInfoList.size();
1852 m_cells.reserve(m_cells.size() + octantInfoListSize);
1853
1854 std::vector<long> createdCells(octantInfoListSize);
1855 for (size_t i = 0; i < octantInfoListSize; ++i) {
1856 const OctantInfo &octantInfo = octantInfoList[i];
1857
1858 // Octant connectivity
1859 Octant *octant = getOctantPointer(octantInfo);
1860
1861 // Cell connectivity
1862 std::unique_ptr<long[]> cellConnect = std::unique_ptr<long[]>(new long[nCellVertices]);
1863 for (int k = 0; k < nCellVertices; ++k) {
1864 uint64_t vertexTreeKey = m_tree->computeNodePersistentKey(octant, k);
1865 cellConnect[k] = stitchInfo.at(vertexTreeKey);
1866 }
1867
1868#if BITPIT_ENABLE_MPI==1
1869 // Cell owner
1870 int rank;
1871 if (octantInfo.internal) {
1872 rank = getRank();
1873 } else {
1874 uint64_t globalTreeId = m_tree->getGhostGlobalIdx(octantInfo.id);
1875 rank = m_tree->getOwnerRank(globalTreeId);
1876 }
1877#endif
1878
1879 // Add cell
1880 long cellId;
1881 if (!restoreStream) {
1882#if BITPIT_ENABLE_MPI==1
1883 CellIterator cellIterator = addCell(m_cellTypeInfo->type, std::move(cellConnect), rank);
1884#else
1885 CellIterator cellIterator = addCell(m_cellTypeInfo->type, std::move(cellConnect));
1886#endif
1887 cellId = cellIterator.getId();
1888 } else {
1889 utils::binary::read(*restoreStream, cellId);
1890
1891#if BITPIT_ENABLE_MPI==1
1892 restoreCell(m_cellTypeInfo->type, std::move(cellConnect), rank, cellId);
1893#else
1894 restoreCell(m_cellTypeInfo->type, std::move(cellConnect), cellId);
1895#endif
1896 }
1897
1898 // Create patch-tree associations
1899 if (octantInfo.internal) {
1900 m_cellToOctant.insert({{cellId, octantInfo.id}});
1901 m_octantToCell.insert({{octantInfo.id, cellId}});
1902 } else {
1903 m_cellToGhost.insert({{cellId, octantInfo.id}});
1904 m_ghostToCell.insert({{octantInfo.id, cellId}});
1905 }
1906
1907 // Add the cell to the list of created cells
1908 createdCells[i] = cellId;
1909 }
1910
1911 // Update first ghost and last internal information
1912 if (restoreStream) {
1913 updateLastInternalCellId();
1914#if BITPIT_ENABLE_MPI==1
1915 updateFirstGhostCellId();
1916#endif
1917 }
1918
1919 // Update cell PIDs
1920 if (restoreStream) {
1921 for (Cell &cell : getCells()) {
1922 int PID;
1923 utils::binary::read(*restoreStream, PID);
1924 cell.setPID(PID);
1925 }
1926 }
1927
1928 // Update adjacencies
1929 updateAdjacencies();
1930
1931 // Update interfaces
1932 if (!restoreStream) {
1933 updateInterfaces();
1934 }
1935
1936 // Done
1937 return createdCells;
1938}

1946void VolOctree::_updateAdjacencies()
1947{
1948 // Tree information
1949 int maxLevel = m_tree->getMaxDepth();
1950
1951 // Face information
1952 int nCellFaces = m_cellTypeInfo->nFaces;
1953 const uint8_t *oppositeFace = m_tree->getOppface();
1954
1955 // Count cells with dirty adjacencies
1956 long nDirtyAdjacenciesCells = 0;
1957 std::vector<std::size_t> nDirtyAdjacenciesCellsByLevel(maxLevel + 1, 0);
1958 for (const auto &entry : m_alteredCells) {
1959 AlterationFlags cellAlterationFlags = entry.second;
1960 if (!testAlterationFlags(cellAlterationFlags, FLAG_ADJACENCIES_DIRTY)) {
1961 continue;
1962 }
1963
1964 long cellId = entry.first;
1965 int cellLevel = getCellLevel(cellId);
1966
1967 ++nDirtyAdjacenciesCells;
1968 ++nDirtyAdjacenciesCellsByLevel[cellLevel];
1969 }
1970
1971 // Group altered cells by their tree level
1972 std::vector<std::vector<long>> hierarchicalCellIds(maxLevel + 1);
1973 for (int level = 0; level <= maxLevel; ++level) {
1974 hierarchicalCellIds[level].reserve(nDirtyAdjacenciesCellsByLevel[level]);
1975 }
1976
1977 for (const auto &entry : m_alteredCells) {
1978 AlterationFlags cellAlterationFlags = entry.second;
1979 if (!testAlterationFlags(cellAlterationFlags, FLAG_ADJACENCIES_DIRTY)) {
1980 continue;
1981 }
1982
1983 long cellId = entry.first;
1984 int cellLevel = getCellLevel(cellId);
1985 hierarchicalCellIds[cellLevel].push_back(cellId);
1986 }
1987
1988 // Update the adjacencies
1989 std::vector<uint32_t> neighTreeIds;
1990 std::vector<bool> neighGhostFlags;
1991 for (int level = 0; level <= maxLevel; ++level) {
1992 for (long cellId : hierarchicalCellIds[level]) {
1993 Cell &cell = m_cells[cellId];
1994 OctantInfo octantInfo = getCellOctant(cellId);
1995 Octant *octant = getOctantPointer(octantInfo);
1996 for (int face = 0; face < nCellFaces; ++face) {
1997 // Check if the face needs to be processed
1998 //
1999 // If the face has no adjacencies, we need to process it to
2000 // figure out if it is a border or it has some neighbours.
2001 //
2002 // If the face has only one adjacency and the neighbour's level
2003 // is smaller or equal than the one of the current cell, all
2004 // face adjacencies have already been found.
2005 //
2006 // If the face has only one adjacency and the neighbour's level
2007 // is greater than the one of the current cell, there are still
2008 // adjacencies to be found. The adjacency associated with the
2009 // face is an adjacency that was not created during this update
2010 // (cells are processed in level increasing order, only cells
2011 // with a level smaller or equal to the one of the current cell
2012 // may already have been processed during this update). Since
2013 // the cell is bigger that its neighbour it cannot have only
2014 // one adjacency, there are other adjacencies to be found in
2015 // the cell marked as dirty.
2016 //
2017 // If the face has multiple adjacencies, we need to figure out
2018 // if all the adjacencies have already been found. Current
2019 // adjacencies were not created during this update (since there
2020 // are multiple adjacencies, those adjacencies are cells smaller
2021 // than the current one, however since cells are processed in
2022 // level increasing order, only cells with a level smaller or
2023 // equal to the one of the current cell may already have been
2024 // processed during this update). To check if all adjacencies
2025 // have been found, we check if the area coverd by the current
2026 // adjacencies is equal to the face area.
2027 int nFaceAdjacencies = cell.getAdjacencyCount(face);
2028 if (nFaceAdjacencies > 0) {
2029 if (nFaceAdjacencies == 1) {
2030 long cellLevel = octant->getLevel();
2031 long neighId = cell.getAdjacency(face, 0);
2032 long neighLevel = getCellLevel(neighId);
2033 if (neighLevel <= cellLevel) {
2034 continue;
2035 }
2036 } else {
2037 uint64_t neighsArea = 0;
2038 const long *faceAdjacencies = cell.getAdjacencies(face);
2039 for (int k = 0; k < nFaceAdjacencies; ++k) {
2040 long neighId = faceAdjacencies[k];
2041 OctantInfo neighOctantInfo = getCellOctant(neighId);
2042 Octant *neighOctant = getOctantPointer(neighOctantInfo);
2043 neighsArea += neighOctant->getLogicalArea();
2044 }
2045
2046 uint64_t faceArea = octant->getLogicalArea();
2047 if (faceArea == neighsArea) {
2048 continue;
2049 }
2050 }
2051 }
2052
2053 // Find cell neighbours
2054 neighTreeIds.clear();
2055 neighGhostFlags.clear();
2056 m_tree->findNeighbours(octant, face, 1, neighTreeIds, neighGhostFlags);
2057
2058 // Set the adjacencies
2059 //
2060 // Some of the neighbours may already be among the adjacencies
2061 // of the cell, this is not a problem because the function that
2062 // pushs the adjacency will insert only unique adjacencies.
2063 int nNeighs = neighTreeIds.size();
2064 for (int k = 0; k < nNeighs; ++k) {
2065 OctantInfo neighOctantInfo(neighTreeIds[k], !neighGhostFlags[k]);
2066 long neighId = getOctantId(neighOctantInfo);
2067
2068 // Set cell data
2069 cell.pushAdjacency(face, neighId);
2070
2071 // Set neighbour data
2072 int neighFace = oppositeFace[face];
2073 Cell &neigh = m_cells[neighId];
2074 neigh.pushAdjacency(neighFace, cellId);
2075 }
2076 }
2077 }
2078 }
2079}

2503void VolOctree::scale(const std::array<double, 3> &scaling, const std::array<double, 3> &center)
2504{
2505 bool uniformScaling = true;
2506 uniformScaling &= (std::abs(scaling[0] - scaling[1]) > 1e-14);
2507 uniformScaling &= (std::abs(scaling[0] - scaling[2]) > 1e-14);
2508 assert(uniformScaling);
2509 if (!uniformScaling) {
2510 log::cout() << "VolOctree patch only allows uniform scaling." << std::endl;
2511 return;
2512 }
2513
2514 std::array<double, 3> origin = m_tree->getOrigin();
2515 for (int n = 0; n < 3; ++n) {
2516 origin[n] = center[n] + scaling[n] * (origin[n] - center[n]);
2517 }
2518 m_tree->setOrigin(origin);
2519
2520 m_tree->setL(m_tree->getL() * scaling[0]);
2521
2522 VolumeKernel::scale(scaling, center);
2523
