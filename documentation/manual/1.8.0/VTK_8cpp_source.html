587void VTK::calcAppendedOffsets(){
588
589 uint64_t offset(0) ;
590 uint64_t HeaderByte(0) ;
591
592 if( getHeaderType() == "UInt32"){
593 HeaderByte = sizeof(uint32_t) ;
594 }
595
596 else if( getHeaderType() == "UInt64") {
597 HeaderByte = sizeof(uint64_t) ;
598 }
599
600 for( auto & field : m_data ){
601 if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::POINT ) {
602 field.setOffset( offset) ;
603 offset += HeaderByte + calcFieldSize(field) ;
604 }
605 }
606
607 for( auto & field : m_data ){
608 if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::CELL) {
609 field.setOffset( offset) ;
610 offset += HeaderByte + calcFieldSize(field) ;
611 }
612 }
613
614 for( auto & field : m_geometry ){
615 if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED ) {
616 field.setOffset( offset) ;
617 offset += HeaderByte + calcFieldSize(field) ;
618 }
619 }
620
621}

655void VTK::write( VTKWriteMode writeMode ){
656
657 int counter(0);
658
659 if( writeMode == VTKWriteMode::NO_SERIES ){
660 counter = unsetCounter() ;
661 }
662
663 if( writeMode == VTKWriteMode::NO_INCREMENT ){
664 counter = getCounter() -1 ;
665 setCounter( counter) ;
666 }
667
668 checkAllFields() ;
669 calcAppendedOffsets() ;
670
671 writeMetaInformation() ;
672 writeData() ;
673
674 if( m_procs > 1 && m_rank == 0) writeCollection() ;
675
676 if( writeMode == VTKWriteMode::DEFAULT || writeMode == VTKWriteMode::NO_INCREMENT ){
677 m_fh.incrementCounter() ;
678 }
679
680 if( writeMode == VTKWriteMode::NO_SERIES ){
681 setCounter(counter) ;
682 }
683
684}

706void VTK::writeData( ){
707
708 std::fstream str ;
709 std::fstream::pos_type position_insert ;
710
711 int length;
712 char* buffer ;
713
714
715 //We need to open the file in binary mode in order for the seekg/tellg
716 //methods to work properly. ASCII mode does not guarantee a correct
717 //stream positioning, because of automatic jump of escape characters
718 // (e.g., "\n" on Unix or "\r\n" on Windows).
719 str.open( m_fh.getPath( ), std::ios::in | std::ios::out | std::ios::binary ) ;
720 if (!str.is_open()) {
721 throw std::runtime_error("Cannot create file \"" + m_fh.getName() + "\"" + " inside the directory \"" + m_fh.getDirectory() + "\"");
722 }
723
724 //Position the input stream at the beginning
725 str.seekg(0, std::ios::beg);
726
727 { // Write Ascii
728 //
729 // It is fine to write ASCII on a binary opened file because the method
730 // genericIO::flushBinary is basically a reinterpret_cast of chars into
731 // chars and ensures consistent results.
732
733 position_insert = str.tellg();
734
735 //Writing first point data then cell data
736 for( auto &field : m_data ){
737 if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII && field.getLocation() == VTKLocation::POINT ) {
738 str.seekg( position_insert);
739 readDataArray( str, field ) ;
740
741 str.seekg( field.getPosition() ) ;
742 genericIO::copyUntilEOFInString( str, buffer, length );
743
744 field.write( str ) ;
745 position_insert = str.tellg();
746
747 if(position_insert==field.getPosition()){
748 log::cout() << "Error VTK: No data has been written for field " << field.getName() << std::endl;
749 assert(false);
750 }
751
752 str << std::endl ;
753 genericIO::flushBINARY( str, buffer, length) ;
754
755 delete [] buffer ;
756
757 }
758 }
759
760 for( auto &field : m_data ){
761 if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII && field.getLocation() == VTKLocation::CELL ) {
762 str.seekg( position_insert);
763 readDataArray( str, field ) ;
764
765 str.seekg( field.getPosition() ) ;
766 genericIO::copyUntilEOFInString( str, buffer, length );
767
768 field.write( str ) ;
769 position_insert = str.tellg();
770
771 if(position_insert==field.getPosition()){
772 log::cout() << "Error VTK: No data has been written for field " << field.getName() << std::endl;
773 assert(false);
774 }
775
776 str << std::endl ;
777 genericIO::flushBINARY( str, buffer, length) ;
778
779 delete [] buffer ;
780 }
781 }
782
783 for( auto &field : m_geometry ){
784 if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII ) {
785 str.seekg( position_insert);
786 readDataArray( str, field ) ;
787
788 str.seekg( field.getPosition() ) ;
789 genericIO::copyUntilEOFInString( str, buffer, length );
790
791 field.write( str ) ;
792 position_insert = str.tellg();
793
794 if(position_insert==field.getPosition()){
795 log::cout() << "Error VTK: No data has been written for field " << field.getName() << std::endl;
796 assert(false);
797 }
798
799 str << std::endl ;
800 genericIO::flushBINARY( str, buffer, length) ;
801
802 delete [] buffer ;
803 }
804 }
805
806 // Position the stream at the beginning of the stream
807 str.seekg(0, std::ios::beg);
808
809 }
810
811 { // Write Appended
812
813 char c_;
814 std::string line ;
815 std::fstream::pos_type position_before_write ;
816
817 //Go to the initial position of the appended section
818 while( getline(str, line) && (! bitpit::utils::string::keywordInString( line, "<AppendedData")) ){}
819
820 str >> c_;
821 while( c_ != '_') str >> c_;
822
823 position_insert = str.tellg();
824 genericIO::copyUntilEOFInString( str, buffer, length );
825
826 //Writing first point data then cell data
827 for( auto &field : m_data ){
828 if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::POINT ) {
829 if( getHeaderType() == "UInt32"){
830 uint32_t nbytes = calcFieldSize(field) ;
831 genericIO::flushBINARY(str, nbytes) ;
832 }
833
834 else{
835 uint64_t nbytes = calcFieldSize(field) ;
836 genericIO::flushBINARY(str, nbytes) ;
837 }
838
839 position_before_write = str.tellg();
840 field.write(str) ;
841 if( (uint64_t) str.tellg()-position_before_write != calcFieldSize(field) ){
842 log::cout() << "Error VTK: Data written do not corrispond to size of field " << field.getName() << std::endl;
843 assert(false);
844 }
845 }
846 }
847
848 for( auto &field : m_data ){
849 if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::CELL ) {
850
851 if( getHeaderType() == "UInt32"){
852 uint32_t nbytes = calcFieldSize(field) ;
853 genericIO::flushBINARY(str, nbytes) ;
854 }
855
856 else{
857 uint64_t nbytes = calcFieldSize(field) ;
858 genericIO::flushBINARY(str, nbytes) ;
859 }
860
861 position_before_write = str.tellg();
862 field.write(str) ;
863 if( (uint64_t) str.tellg()-position_before_write != calcFieldSize(field) ){
864 log::cout() << "Error VTK: Data written do not corrispond to size of field " << field.getName() << std::endl;
865 assert(false);
866 }
867
868 }
869 }
870
871 //Writing Geometry Data
872 for( auto &field : m_geometry ){
873 if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED ) {
874 if( getHeaderType() == "UInt32"){
875 uint32_t nbytes = calcFieldSize(field) ;
876 genericIO::flushBINARY(str, nbytes) ;
877 }
878
879 else{
880 uint64_t nbytes = calcFieldSize(field) ;
881 genericIO::flushBINARY(str, nbytes) ;
882 }
883
884 position_before_write = str.tellg();
885 field.write(str) ;
886 if( (uint64_t) str.tellg()-position_before_write != calcFieldSize(field) ){
887 log::cout() << "Error VTK: Data written do not corrispond to size of field " << field.getName() << std::endl;
888 assert(false);
889 }
890 }
891 }
892
893 genericIO::flushBINARY( str, buffer, length) ;
894
895 delete [] buffer ;
896 }
897
898 // Closing Appended Section
899 str.close();
900
901}

908void VTK::writeDataHeader( std::fstream &str, bool parallel ){
909
910 VTKLocation location ;
911 std::stringstream scalars, vectors ;
912
913 for( int j=0; j<2; j++){
914
915 if( j==0 ) location = VTKLocation::POINT ;
916 if( j==1 ) location = VTKLocation::CELL ;
917
918 scalars.str("");
919 vectors.str("");
920
921 //Creating Scalar and Vector Lists
922 scalars << "\"" ;
923 vectors << "\"" ;
924
925 for( auto &field : m_data ){
926 if( field.isEnabled() && field.getLocation() == location){
927 if( field.getFieldType() == VTKFieldType::SCALAR ) scalars << field.getName() << " " ;
928 else if( field.getFieldType() == VTKFieldType::VECTOR ) vectors << field.getName() << " " ;
929 }
930
931 }
932
933 scalars << "\"" ;
934 vectors << "\"" ;
935
936 if( location == VTKLocation::POINT) {
937 str << " <" ;
938 if( parallel ) str << "P" ;
939 str << "PointData " ;
940 }
941
942 else if( location == VTKLocation::CELL ) {
943 str << " <" ;
944 if( parallel ) str << "P" ;
945 str << "CellData " ;
946 }
947
948 str << " Scalars=" << scalars.str()
949 << " Vectors=" << vectors.str()
950 << ">" << std::endl;
951
952 //Writing DataArray
953 for( auto &field : m_data ){
954 if( field.isEnabled() && field.getLocation() == location && !parallel) writeDataArray( str, field ) ;
955 if( field.isEnabled() && field.getLocation() == location && parallel) writePDataArray( str, field );
956 }
957
958 str << " </" ;
959 if( parallel ) str << "P" ;
960
961 if( location == VTKLocation::POINT) str << "PointData> " << std::endl;
962 if( location == VTKLocation::CELL) str << "CellData> " << std::endl;
963
964 }
965
966}

1006void VTK::readData( ){
1007
1008 std::fstream str ;
1009 std::fstream::pos_type position_appended;
1010 std::string line;
1011 char c_ ;
1012 uint32_t nbytes32 ;
1013 uint64_t nbytes64 ;
1014
1015 str.open( m_fh.getPath( ), std::ios::in | std::ios::binary) ;
1016
1017 //Read appended data
1018 //Go to the initial position of the appended section
1019 bool foundAppendedSection = false;
1020 while( !foundAppendedSection && getline(str, line) ){
1021 foundAppendedSection = bitpit::utils::string::keywordInString( line, "<AppendedData");
1022 }
1023
1024 if( foundAppendedSection){
1025 str >> c_;
1026 while( c_ != '_') str >> c_;
1027
1028 position_appended = str.tellg();
1029
1030
1031 //Read appended data
1032 for( auto & field : m_data){
1033 if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED){
1034 str.clear();
1035 str.seekg( position_appended) ;
1036 str.seekg( field.getOffset(), std::ios::cur) ;
1037 if( m_headerType== "UInt32") genericIO::absorbBINARY( str, nbytes32 ) ;
1038 if( m_headerType== "UInt64") genericIO::absorbBINARY( str, nbytes64 ) ;
1039
1040#if BITPIT_ENABLE_DEBUG
1041 std::fstream::pos_type position_before = str.tellg();
1042#endif
1043
1044 field.read( str, calcFieldEntries(field), calcFieldComponents(field) ) ;
1045
1046#if BITPIT_ENABLE_DEBUG
1047 if( uint64_t(str.tellg()-position_before) != calcFieldSize(field) ){
1048 log::cout() << "Warning VTK: Size of data read does not corrispond to size of field " << field.getName() << std::endl;
1049 log::cout() << "Found data chunk dimension of : "<< uint64_t(str.tellg()-position_before)<<" different from estimated field size : " <<calcFieldSize(field)<<std::endl;
1050 }
1051#endif
1052
