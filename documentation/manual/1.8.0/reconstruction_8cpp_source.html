219void ReconstructionPolynomial::evalPointBasisValues(uint8_t degree, uint8_t dimensions, const std::array<double, 3> &origin,
220 const std::array<double, 3> &point, double *csi)
221{
222 // Set 0-th degree coefficients
223 csi[0] = 1.;
224
225 // Set high degree coefficients
226 if (degree >= 1) {
227 int offset = 1;
228 const std::array<double, 3> distance = point - origin;
229
230 // Set 1-st degree coefficients
231 for (int i = 0; i < dimensions; ++i) {
232 csi[offset++] = distance[i];
233 }
234
235 // Set 2-nd degree coefficients
236 if (degree >= 2) {
237 for (int i = 0; i < dimensions; ++i) {
238 csi[offset++] = 0.5 * distance[i] * distance[i];
239 }
240
241 if (dimensions >= 2) {
242 csi[offset++] = distance[0] * distance[1];
243
244 if (dimensions >= 3) {
245 csi[offset++] = distance[0] * distance[2];
246 csi[offset++] = distance[1] * distance[2];
247 }
248 }
249 }
250
251 // Check if all coefficients have been set
252 assert(offset == ReconstructionPolynomial::getCoefficientCount(degree, dimensions));
253 }
254}

268void ReconstructionPolynomial::evalPointBasisDerivatives(uint8_t degree, uint8_t dimensions, const std::array<double, 3> &origin,
269 const std::array<double, 3> &point, const std::array<double, 3> &direction,
270 double *dcsi)
271{
272 // Set 0-th degree coefficients
273 dcsi[0] = 0.;
274
275 // Set high degree coefficients
276 if (degree >= 1) {
277 int offset = 1;
278
279 // Set 1-st degree coefficients
280 for (int i = 0; i < dimensions; ++i) {
281 dcsi[offset++] = direction[i];
282 }
283
284 // Set 2-nd degree coefficients
285 if (degree >= 2) {
286 const std::array<double, 3> distance = point - origin;
287
288 for (int i = 0; i < dimensions; ++i) {
289 dcsi[offset++] = distance[i] * direction[i];
290 }
291
292 if (dimensions >= 2) {
293 dcsi[offset++] = distance[0] * direction[1] + distance[1] * direction[0];
294
295 if (dimensions >= 3) {
296 dcsi[offset++] = distance[0] * direction[2] + distance[2] * direction[0];
297 dcsi[offset++] = distance[1] * direction[2] + distance[2] * direction[1];
298 }
299 }
300 }
301
302 // Check if all coefficients have been set
303 assert(offset == ReconstructionPolynomial::getCoefficientCount(degree, dimensions));
304 }

339void ReconstructionPolynomial::evalCellBasisValues(uint8_t degree, uint8_t dimensions, const std::array<double, 3> &origin,
340 const Cell &cell, const std::array<double, 3> *vertexCoords,
341 double *csi)
342{
343 // Check if cell type is supported
344 //
345 // Polynomials of degree 0 and 1 are supported on all type of cells, for
346 // pixels and voxels also degree 2 is supported.
347 //
348 // Coefficients are evaluated as the volume average of the Taylor series
349 // expansion coefficients. In order to support higher order degrees on
350 // all type of cells, we need to implement numerical integration of the
351 // expansions terms over the volume of the cell. This can be done using
352 // Gauss quadrature rules and evaluating the terms on the integration
353 // points using the function that evaluate the basis function on a
354 // specified point.
355 ElementType cellType = cell.getType();
356
357 bool cellTypeSupported;
358 if (cellType == ElementType::PIXEL) {
359 cellTypeSupported = (degree <= 2);
360 } else if (cellType == ElementType::VOXEL) {
361 cellTypeSupported = (degree <= 2);
362 } else {
363 cellTypeSupported = (degree <= 1);
364 }
365
366 if (!cellTypeSupported) {
367 throw std::runtime_error("Cell type not supported.");
368 }
369
370 // Set 0-th degree coefficients
371 csi[0] = 1.;
372
373 // Set high degree coefficients
374 if (degree >= 1) {
375 int offset = 1;
376 const std::array<double, 3> distance = cell.evalCentroid(vertexCoords) - origin;
377
378 // Set 1-st degree coefficients
379 for (int i = 0; i < dimensions; ++i) {
380 csi[offset++] = distance[i];
381 }
382
383 // Set 2-nd degree coefficients
384 if (degree >= 2) {
385 double cellSize = cell.evalSize(vertexCoords);
386
387 for (int i = 0; i < dimensions; ++i) {
388 csi[offset++] = 0.5 * (distance[i] * distance[i] + cellSize * cellSize / 12.);
389 }
390
391 if (dimensions >= 2) {
392 csi[offset++] = distance[0] * distance[1];
393
394 if (dimensions >= 3) {
395 csi[offset++] = distance[0] * distance[2];
396 csi[offset++] = distance[1] * distance[2];
397 }
398 }
399 }
400
401 // Check if all coefficients have been set
402 assert(offset == ReconstructionPolynomial::getCoefficientCount(degree, dimensions));
403 }
404}

486void ReconstructionPolynomial::initialize(uint8_t degree, uint8_t dimensions,
487 const std::array<double, 3> &origin,
488 int nFields, bool release)
489{
490 assert(degree <= ReconstructionPolynomial::MAX_DEGREE);
491 m_degree = degree;
492
493 assert(dimensions <= ReconstructionPolynomial::MAX_DIMENSIONS);
494 m_dimensions = dimensions;
495
496 int currentStorageSize = m_nCoeffs;
497
498 m_nCoeffs = ReconstructionPolynomial::getCoefficientCount(m_degree, m_dimensions);
499
500 if (nFields > 0) {
501 m_nFields = nFields;
502
503 int storageSize = m_nCoeffs * m_nFields;
504
505 bool reallocate;
506 if (release) {
507 reallocate = (currentStorageSize != storageSize);
508 } else {
509 reallocate = (currentStorageSize < storageSize);
510 }
511
512 if (reallocate) {
513 m_coeffs = std::unique_ptr<double[]>(new double[storageSize]);
514 }
515 } else {
516 clear(release);
517 }
518
519 m_origin = origin;
520}

797void ReconstructionPolynomial::computeValues(int degree, const std::array<double, 3> &point,
798 int nFields, int offset, double *values) const
799{
800 assert(degree <= getDegree());
801
802 // Early return if there are no field to process
803 if (nFields == 0) {
804 return;
805 }
806
807 // Get values
808 const std::size_t fieldValuesStride = 1;
809 double *fieldValue = values;
810 const double *fieldValueEnd = fieldValue + fieldValuesStride * nFields;
811
812 // Get coefficients
813 std::size_t fieldCoeffsStride = getFieldCoefficientsStride();
814 const double *fieldCoeffs = m_coeffs.get() + fieldCoeffsStride * offset;
815
816 // Constant polynomial
817 if (ENABLE_FAST_PATH_OPTIMIZATIONS && degree == 0) {
818 do {
819 *fieldValue = fieldCoeffs[0];
820
821 fieldCoeffs += fieldCoeffsStride;
822 fieldValue += fieldValuesStride;
823 } while (fieldValue != fieldValueEnd);
824
825 return;
826 }
827
828 // Generic polynomial
829 int nCoeffs = ReconstructionPolynomial::getCoefficientCount(degree, m_dimensions);
830
831 BITPIT_CREATE_WORKSPACE(csi, double, nCoeffs, MAX_STACK_WORKSPACE_SIZE);
832 ReconstructionPolynomial::evalPointBasisValues(degree, m_dimensions, m_origin, point, csi);
833
834 do {
835 *fieldValue = fieldCoeffs[0] * csi[0];
836 for (int i = 1; i < nCoeffs; ++i) {
837 *fieldValue += fieldCoeffs[i] * csi[i];
838 }
839
840 fieldCoeffs += fieldCoeffsStride;
841 fieldValue += fieldValuesStride;
842 } while (fieldValue != fieldValueEnd);
843}

1002void ReconstructionPolynomial::computeValuesLimited(int degree, const std::array<double, 3> &point,
1003 const double *limiters, int nFields, int offset,
1004 double *values) const
1005{
1006 assert(degree <= getDegree());
1007
1008 // Early return if there are no field to process
1009 if (nFields == 0) {
1010 return;
1011 }
1012
1013 // Get values
1014 std::size_t fieldValuesStride = 1;
1015 double *fieldValue = values;
1016 double *fieldValueEnd = fieldValue + fieldValuesStride * nFields;
1017
1018 // Get coefficients
1019 std::size_t fieldCoeffsStride = getFieldCoefficientsStride();
1020 const double *fieldCoeffs = m_coeffs.get() + fieldCoeffsStride * offset;
1021
1022 // Constant polynomial
1023 if (ENABLE_FAST_PATH_OPTIMIZATIONS && degree == 0) {
1024 do {
1025 *fieldValue = fieldCoeffs[0];
1026
1027 fieldCoeffs += fieldCoeffsStride;
1028 fieldValue += fieldValuesStride;
1029 } while (fieldValue != fieldValueEnd);
1030
1031 return;
1032 }
1033
1034 // Get limiters
1035 std::size_t fieldLimitersStride = degree;
1036 const double *fieldLimiters = limiters;
1037
1038 // Generic polynomial
1039 int nCoeffs = ReconstructionPolynomial::getCoefficientCount(degree, m_dimensions);
1040 const std::vector<uint16_t> &nDegreeCoeffs = ReconstructionPolynomial::getDegreeCoefficientsCount(m_dimensions);
1041
1042 BITPIT_CREATE_WORKSPACE(csi, double, nCoeffs, MAX_STACK_WORKSPACE_SIZE);
1043 ReconstructionPolynomial::evalPointBasisValues(degree, m_dimensions, m_origin, point, csi);
1044
1045 do {
1046 // Degree 0
1047 *fieldValue = fieldCoeffs[0] * csi[0];
1048
1049 // Degrees greater than 0
1050 int coeffEnd = nDegreeCoeffs[0];
1051 for (int n = 1; n <= degree; ++n) {
1052 int coeffBegin = coeffEnd;
1053 coeffEnd = coeffBegin + nDegreeCoeffs[n];
1054
1055 double fieldDegreeValue = 0;
1056 for (int i = coeffBegin; i < coeffEnd; ++i) {
1057 fieldDegreeValue += fieldCoeffs[i] * csi[i];
1058 }
1059
1060 *fieldValue += fieldDegreeValue * fieldLimiters[n - 1];
1061 }
1062
1063 fieldCoeffs += fieldCoeffsStride;
1064 fieldValue += fieldValuesStride;
1065 fieldLimiters += fieldLimitersStride;
1066 } while (fieldValue != fieldValueEnd);
1067}

1186void ReconstructionPolynomial::computeDerivatives(int degree, const std::array<double, 3> &point,
1187 const std::array<double, 3> &direction,
1188 int nFields, int offset, double *derivatives) const
1189{
1190 assert(degree <= getDegree());
1191
1192 // Early return if there are no field to process
1193 if (nFields == 0) {
1194 return;
1195 }
1196
1197 // Get derivatives
1198 std::size_t fieldDerivativeStride = 1;
1199 double *fieldDerivative = derivatives;
1200 double *fieldDerivativeEnd = fieldDerivative + fieldDerivativeStride * nFields;
1201

3179 // through Lagrange multipliers. The resulting linear system is:
3180 //
3181 // | A^t A w C^t | |x | = |A^t w b|
3182 // | C 0 | |lambda| |d |
3183 //
3184 // with A and C the least-squares and constraints equations respectively,
3185 // and b and d their corresponding RHSs. x are the coefficients of
3186 // the polynomial and lambda the lagrange multipliers. w are the normalized
3187 // east square scale factors.
3188 //
3189 // This system is denoted by S:
3190 //
3191 // | x | |A^t w 0| |b|
3192 // |S| | | = | | | |
3193 // |lambda| |0 I| |d|
3194 //
3195 // The matrices S and S^-1 are symmetric and only the upper portions are

3236 // | x | |A^t w 0| |b| |b|
3237 // | | = S^-1 | | | | = S^-1 Q | |
3238 // |lambda| |0 I| |d| |d|
