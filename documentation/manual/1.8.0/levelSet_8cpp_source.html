319int LevelSet::registerObject( std::unique_ptr<LevelSetObject> &&object ) {
320
321 int id = object->getId();
322 if (id == levelSetDefaults::OBJECT) {
323 id = m_objectIdentifierGenerator.generate();
324 object->setId(id);
325 } else {
326 m_objectIdentifierGenerator.setAssigned(id);
327 }
328
329 if( m_kernel){
330 object->setKernel(m_kernel.get());
331 }
332
333 if (m_narrowBandSize > 0.) {
334 object->setSizeNarrowBand(m_narrowBandSize);
335 }
336
337 m_objects[id] = std::move(object) ;
338
339 setObjectProcessingOrder(id);
340
341 incrementObjectsReferenceCount(id);
342
343 return id;
344}

419void LevelSet::makeObjectImmutable( int id ) {
420
421 LevelSetObject *object = getObjectPtr( id ) ;
422
423 // Get the list of referring objects
424 //
425 // Referring objects are objects for which this object is a source.
426 std::vector<int> referringProxyObjectIds;
427 if( object->getReferenceCount() > 0 ) {
428 for( const auto &entry : m_objects ) {
429 const LevelSetProxyObject *proxyObject = dynamic_cast<const LevelSetProxyObject*>(entry.second.get());
430 if (!proxyObject) {
431 continue;
432 }
433
434 std::vector<const LevelSetObject *> sourceObjects = proxyObject->getSourceObjects() ;
435 std::size_t nSourceObjects = sourceObjects.size() ;
436 for ( std::size_t i = 0; i < nSourceObjects; ++i ){
437 const LevelSetObject *sourceObject = sourceObjects[i];
438 if (sourceObject == object) {
439 referringProxyObjectIds.push_back(proxyObject->getId());
440 }
441 }
442 }
443 }
444
445 // Create the immutable object
446 std::unique_ptr<LevelSetObject> immutableObject = LevelSetObjectFactory::createImmutableObject<LevelSetImmutableNarrowBandCache>( m_kernel.get(), m_storageType, object ) ;
447 assert( immutableObject );
448
449 // Update referring objects
450 for (int referringProxyObjectId : referringProxyObjectIds) {
451 LevelSetProxyObject *referringProxyObject = static_cast<LevelSetProxyObject *>(getObjectPtr( referringProxyObjectId ));
452 referringProxyObject->replaceSourceObject(object, immutableObject.get());
453 }
454
455 // Replace the object with the newly created immutable object
456 removeObject(id, true);
457 addObject(std::move(immutableObject));
458}

472void LevelSet::setObjectProcessingOrder( int id ) {
473
474 // Immutable objects are not processed.
475 if (dynamic_cast<const LevelSetImmutableObjectBase*>(getObjectPtr(id))) {
476 return;
477 }
478
479 // Define the processing order for the object
480 //
481 // The insertion order determines the processing order, however priority
482 // is given to primary objects.
483 std::vector<int>::iterator processingOrderItr;
484 if(getObjectPtr(id)->isPrimary()){
485 std::vector<int>::iterator processingOrderBegin = m_objectsProcessingOrder.begin();
486 std::vector<int>::iterator processingOrderEnd = m_objectsProcessingOrder.end();
487
488 for (processingOrderItr = processingOrderBegin; processingOrderItr != processingOrderEnd; ++processingOrderItr) {
489 int candidateId = *processingOrderItr ;
490 const LevelSetObject *candidateObject = getObjectPtr(candidateId) ;
491 if( !candidateObject->isPrimary() ){
492 break;
493 }
494 }
495 } else {
496 processingOrderItr = m_objectsProcessingOrder.end();
497 }
498
499 m_objectsProcessingOrder.insert(processingOrderItr,id) ;
500
501}

509void LevelSet::unsetObjectProcessingOrder(int id){
510
511 // Immutable objects are not processed.
512 if (dynamic_cast<const LevelSetImmutableObjectBase*>(getObjectPtr(id))) {
513 return;
514 }
515
516 // Remove the object from the list of processed objects
517 std::vector<int>::iterator processingOrderBegin = m_objectsProcessingOrder.begin();
518 std::vector<int>::iterator processingOrderEnd = m_objectsProcessingOrder.end();
519
520 std::vector<int>::iterator processingOrderItr = std::find(processingOrderBegin, processingOrderEnd, id);
521 assert(processingOrderItr != processingOrderEnd);
522 m_objectsProcessingOrder.erase(processingOrderItr);
523

722void LevelSet::compute( const std::unordered_set<LevelSetObject *> &objectProcessList ){
723
724 assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::compute()");
725
726 std::unique_ptr<LevelSetSignPropagator> signPropagator ;
727 if (m_propagateSign) {
728 signPropagator = m_kernel->createSignPropagator() ;
729 }
730
731 for( int id : m_objectsProcessingOrder){
732 LevelSetObject *object = m_objects.at(id).get() ;
733 if (objectProcessList.count(object) == 0) {
734 continue;
735 }
736
737 LevelSetSignedObjectInterface *signPropagationObject;
738 if (m_propagateSign) {
739 signPropagationObject = dynamic_cast<LevelSetSignedObjectInterface *>(object);
740 } else {
741 signPropagationObject = nullptr;
742 }
743
744 // Clear the object
745 object->clear();
746
747 // Compute levelset inside the narrowband
748 object->computeNarrowBand(m_signedDistance) ;
749#if BITPIT_ENABLE_MPI
750 object->exchangeGhosts();
751#endif
752
753 // Propagate sign
754 if(signPropagationObject) {
755 signPropagator->execute(signPropagationObject);
756 }
757 }
758}

821void LevelSet::update( const std::vector<adaption::Info> &adaptionData, const std::unordered_set<LevelSetObject *> &objectProcessList ){
822
823 assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::partition()");
824
825 VolumeKernel *mesh = m_kernel->getMesh();
826#if BITPIT_ENABLE_MPI
827 bool isMeshPartitioned = mesh->isPartitioned();
828#endif
829
830#if BITPIT_ENABLE_MPI
831 // Set the communicator
832 //
833 // The mesh may have been partitioned after being set.
834 if (isMeshPartitioned && !m_kernel->isCommunicatorSet()) {
835 m_kernel->initializeCommunicator();
836 }
837#endif
838
839 // Inspect adaption data to detect what needs to be updated
840 std::unordered_map<int,std::vector<long>> partitioningSendList ;
841 std::unordered_map<int,std::vector<long>> partitioningRecvList ;
842
843 bool updateNarrowBand = false;
844 bool updatePartitioning = false;
845 for( const adaption::Info &adaptionInfo : adaptionData){
846 if( adaptionInfo.entity != adaption::Entity::ENTITY_CELL){
847 continue;
848 }
849
850 if( adaptionInfo.type == adaption::Type::TYPE_PARTITION_SEND){
851 partitioningSendList.insert({{adaptionInfo.rank,adaptionInfo.previous}}) ;
852 updatePartitioning = true;
853 } else if( adaptionInfo.type == adaption::Type::TYPE_PARTITION_RECV){
854 partitioningRecvList.insert({{adaptionInfo.rank,adaptionInfo.current}}) ;
855 updatePartitioning = true;
856 } else if (adaptionInfo.type == adaption::Type::TYPE_DELETION) {
857 updateNarrowBand = true;
858 } else {
859 if (!updateNarrowBand) {
860 for (long cellId : adaptionInfo.current) {
861 const Cell &cell = mesh->getCell(cellId);
862 if (cell.isInterior()) {
863 updateNarrowBand = true;
864 break;
865 }
866 }
867 }
868 }
869 }
870
871#if BITPIT_ENABLE_MPI
872 if (isMeshPartitioned) {
873 MPI_Allreduce(MPI_IN_PLACE, &updateNarrowBand, 1, MPI_C_BOOL, MPI_LOR, m_kernel->getCommunicator());
874 MPI_Allreduce(MPI_IN_PLACE, &updatePartitioning, 1, MPI_C_BOOL, MPI_LOR, m_kernel->getCommunicator());
875 }
876#endif
877
878 // Early return if no update is needed
879 if (!updateNarrowBand && !updatePartitioning) {
880 return;
881 }
882
883#if BITPIT_ENABLE_MPI
884 // Get data communicator for updating partitioning
885 std::unique_ptr<DataCommunicator> dataCommunicator;
886 if (updatePartitioning) {
887 dataCommunicator = m_kernel->createDataCommunicator();
888 }
889#endif
890
891 // Update kernel
892 m_kernel->updateGeometryCache( adaptionData ) ;
893
894 // Create sign propagator
895 std::unique_ptr<LevelSetSignPropagator> signPropagator ;
896 if (m_propagateSign) {
897 signPropagator = m_kernel->createSignPropagator() ;
898 }
899
900 // Update objects
901 for( int id : m_objectsProcessingOrder){
902 LevelSetObject *object = m_objects.at(id).get() ;
903 if (objectProcessList.count(object) == 0) {
904 continue;
905 }
906
907 LevelSetSignedObjectInterface *signPropagationObject;
908 if (m_propagateSign) {
909 signPropagationObject = dynamic_cast<LevelSetSignedObjectInterface *>(object);
910 } else {
911 signPropagationObject = nullptr;
912 }
913
914#if BITPIT_ENABLE_MPI
915 // Start partitioning update
916 if (updatePartitioning) {
917 object->startExchange( partitioningSendList, dataCommunicator.get() );
918 }
919#endif
920
921 // Propagated sign is now dirty
922 //
923 // The propagated sign will be set as non-dirty by the sign propagator.
924 if (signPropagationObject) {
925 signPropagationObject->setSignStorageDirty(true);
926 }
927
928 // Clear data structures after mesh update
929 object->clearAfterMeshAdaption( adaptionData ) ;
930
931#if BITPIT_ENABLE_MPI
932 // Complete partitioning update
933 if (updatePartitioning) {
934 object->completeExchange( partitioningRecvList, dataCommunicator.get() );
935 }
936#endif
937
938 // Update levelset inside narrow band
939 if (updateNarrowBand) {
940 object->updateNarrowBand( adaptionData, m_signedDistance ) ;
941 }
942
943#if BITPIT_ENABLE_MPI
944 // Update data on ghost cells
945 object->exchangeGhosts();
946#endif
947
948 // Propagate sign
949 //
950 // It's not possible to communicate sign information, therefore sign
951 // needs to be propagated also when the mesh is only partitioned.
952 if (signPropagationObject) {
953 signPropagator->execute(adaptionData, signPropagationObject);
954 }
955 }
956
957}

997std::unordered_set<LevelSetObject *> LevelSet::getObjectProcessList(std::size_t nObjects, const int *objectIds) const{
998
999 std::unordered_set<LevelSetObject *> objectProcessList;
1000 for (std::size_t i = 0; i < nObjects; ++i) {
1001 int objectId = objectIds[i] ;
1002 LevelSetObject *object = getObjectPtr(objectId) ;
1003 if (objectProcessList.count(object) != 0) {
1004 continue;
1005 }
1006
1007 objectProcessList.insert(object);
1008 if( const LevelSetProxyObject *proxyObject = dynamic_cast<const LevelSetProxyObject*>(object) ){
1009 std::vector<int> sourceObjectIds = proxyObject->getSourceObjectIds();
1010 std::unordered_set<LevelSetObject *> sourceObjectProcessList = getObjectProcessList(sourceObjectIds.size(), sourceObjectIds.data()) ;
1011 objectProcessList.insert(sourceObjectProcessList.begin(), sourceObjectProcessList.end());
1012 }
1013 }
1014
1015 return objectProcessList;
