268 void
269 ParaTree::_initializeCommunicator(MPI_Comm communicator)
270 {
271 // Communicator can be set just once
272 if (isCommSet()) {
273 throw std::runtime_error ("PABLO communicator can be set just once");
274 }
275
276 // Early return if the communicator is a null communicator
277 if (communicator == MPI_COMM_NULL) {
278 m_comm = MPI_COMM_NULL;
279 return;
280 }
281
282 // Create a copy of the user-specified communicator
283 //
284 // No library routine should use MPI_COMM_WORLD as the communicator;
285 // instead, a duplicate of a user-specified communicator should always
286 // be used.
287 MPI_Comm_dup(communicator, &m_comm);
288 }

295 void
296 ParaTree::_initializePartitions() {
297#if BITPIT_ENABLE_MPI==1
298 // Set MPI information
299 if (isCommSet()) {
300 MPI_Comm_size(m_comm, &m_nproc);
301 MPI_Comm_rank(m_comm, &m_rank);
302 } else {
303 m_rank = 0;
304 m_nproc = 1;
305 }
306#else
307 // Set dummy MPI information
308 m_rank = 0;
309 m_nproc = 1;
310#endif
311
312 // Create the data structures for storing partion information
313 m_partitionFirstDesc.resize(m_nproc);
314 m_partitionLastDesc.resize(m_nproc);
315 m_partitionRangeGlobalIdx.resize(m_nproc);
316 m_partitionRangeGlobalIdx0.resize(m_nproc);
317 uint64_t lastDescMorton = m_octree.getLastDescMorton();
318 uint64_t firstDescMorton = m_octree.getFirstDescMorton();
319 for(int p = 0; p < m_nproc; ++p){
320 m_partitionRangeGlobalIdx0[p] = 0;
321 m_partitionRangeGlobalIdx[p] = m_globalNumOctants - 1;
322 m_partitionLastDesc[p] = lastDescMorton;
323 m_partitionFirstDesc[p] = firstDescMorton;
324 }
325 }

488 void ParaTree::dump(std::ostream &stream, bool full)
489 {
490 // Version
491 utils::binary::write(stream, getDumpVersion());
492
493 // Tree data
494 utils::binary::write(stream, getNproc());
495
496 utils::binary::write(stream, getDim());
497
498 utils::binary::write(stream, getSerial());
499 utils::binary::write(stream, getNofGhostLayers());
500 utils::binary::write(stream, getMaxDepth());
501 utils::binary::write(stream, getStatus());
502 utils::binary::write(stream, getBalanceCodimension());
503
504 for (int i = 0; i < m_treeConstants->nFaces; i++) {
505 utils::binary::write(stream, getPeriodic(i));
506 }
507
508 // Octant data
509 uint32_t nOctants = getNumOctants();
510 utils::binary::write(stream, nOctants);
511
512 uint32_t nGlobalOctants = getGlobalNumOctants();
513 utils::binary::write(stream, nGlobalOctants);
514
515 for (uint32_t i = 0; i < nOctants; i++) {
516 const Octant &octant = m_octree.m_octants[i];
517
518 utils::binary::write(stream, octant.getLevel());
519 utils::binary::write(stream, octant.getLogicalX());
520 utils::binary::write(stream, octant.getLogicalY());
521 utils::binary::write(stream, octant.getLogicalZ());
522 utils::binary::write(stream, octant.getGhostLayer());
523
524 for (size_t k = 0; k < Octant::INFO_ITEM_COUNT; ++k) {
525 utils::binary::write(stream, (bool) octant.m_info[k]);
526 }
527
528 utils::binary::write(stream, octant.getBalance());
529 utils::binary::write(stream, octant.getMarker());
530 }
531
532 // Information about partitioning
533 for (int k = 0; k < m_nproc; ++k) {
534 utils::binary::write(stream, m_partitionFirstDesc[k]);
535 }
536
537 for (int k = 0; k < m_nproc; ++k) {
538 utils::binary::write(stream, m_partitionLastDesc[k]);
539 }
540
541 for (int k = 0; k < m_nproc; ++k) {
542 utils::binary::write(stream, m_partitionRangeGlobalIdx[k]);
543 }
544
545 // Extended information (mapping, ...)
546 utils::binary::write(stream, full);
547 if (full) {
548 utils::binary::write(stream, m_lastOp);
549 if (m_lastOp == OP_ADAPT_MAPPED){
550 for (auto idx : m_mapIdx) {
551 utils::binary::write(stream, idx);
552 }
553
554 utils::binary::write(stream, m_octree.m_lastGhostBros.size());
555 for (auto lastGhostBrother : m_octree.m_lastGhostBros) {
556 utils::binary::write(stream, lastGhostBrother);
557 }
558 }
559 else if (m_lastOp == OP_LOADBALANCE || m_lastOp == OP_LOADBALANCE_FIRST){
560 for (int i = 0; i < m_nproc; ++i) {
561 utils::binary::write(stream, m_partitionRangeGlobalIdx0[i]);
562 }
563 }
564 }
565
566 }

574 void ParaTree::restore(std::istream &stream)
575 {
576 // Version
577 int version;
578 utils::binary::read(stream, version);
579 if (version != getDumpVersion()) {
580 throw std::runtime_error ("The version of the file does not match the required version");
581 }
582
583 // Check if the number of processes matches
584 int nProcs;
585 utils::binary::read(stream, nProcs);
586 if (nProcs != m_nproc) {
587 throw std::runtime_error ("The restart was saved with a different number of processes.");
588 }
589
590 // Initialize the tree
591 uint8_t dimension;
592 utils::binary::read(stream, dimension);
593
594 m_octree.initialize(dimension);
595 m_trans.initialize(dimension);
596 reinitialize(dimension, m_log->getName());
597 reset(false);
598
599 // Set tree properties
600 utils::binary::read(stream, m_serial);
601 utils::binary::read(stream, m_nofGhostLayers);
602 utils::binary::read(stream, m_maxDepth);
603 utils::binary::read(stream, m_status);
604
605 bool balanceCodimension;
606 utils::binary::read(stream, balanceCodimension);
607 setBalanceCodimension(balanceCodimension);
608
609 for (int i = 0; i < m_treeConstants->nFaces; i++) {
610 bool periodicBorder;
611 utils::binary::read(stream, periodicBorder);
612 if (periodicBorder){
613 setPeriodic(i);
614 }
615 }
616
617 // Restore octants
618 uint32_t nOctants;
619 utils::binary::read(stream, nOctants);
620 m_octree.m_sizeOctants = nOctants;
621
622 uint32_t nGlobalOctants;
623 utils::binary::read(stream, nGlobalOctants);
624 m_globalNumOctants = nGlobalOctants;
625
626 m_octree.m_octants.clear();
627 m_octree.m_octants.reserve(nOctants);
628 for (uint32_t i = 0; i < nOctants; i++) {
629 // Create octant
630 uint8_t level;
631 utils::binary::read(stream, level);
632
633 uint32_t x;
634 utils::binary::read(stream, x);
635
636 uint32_t y;
637 utils::binary::read(stream, y);
638
639 uint32_t z;
640 utils::binary::read(stream, z);
641
642 Octant octant(false, m_dim, level, x, y, z);
643
644 int ghost;
645 utils::binary::read(stream, ghost);
646 octant.setGhostLayer(ghost);
647
648 // Set octant info
649 for (size_t k = 0; k < Octant::INFO_ITEM_COUNT; ++k) {
650 bool bit;
651 utils::binary::read(stream, bit);
652 octant.m_info.set(k, bit);
653 }
654
655 // Set octant 2:1 balance
656 bool balance21;
657 utils::binary::read(stream, balance21);
658 octant.setBalance(balance21);
659
660 // Set marker
661 int8_t marker;
662 utils::binary::read(stream, marker);
663 octant.setMarker(marker);
664
665 // Add octant to the list
666 m_octree.m_octants.push_back(std::move(octant));
667 }
668
669 m_octree.updateLocalMaxDepth();
670
671 // Set first last descendant
672 m_partitionFirstDesc.resize(m_nproc);
673 for (int k = 0; k < m_nproc; ++k) {
674 uint64_t descendant;
675 utils::binary::read(stream, descendant);
676 m_partitionFirstDesc[k] = descendant;
677 }
678 m_octree.m_firstDescMorton = m_partitionFirstDesc[m_rank];
679
680 m_partitionLastDesc.resize(m_nproc);
681 for (int k = 0; k < m_nproc; ++k) {
682 uint64_t descendant;
683 utils::binary::read(stream, descendant);
684 m_partitionLastDesc[k] = descendant;
685 }
686 m_octree.m_lastDescMorton = m_partitionLastDesc[m_rank];
687
688 // Set partitions and parallel information
689 m_partitionRangeGlobalIdx.resize(m_nproc);
690 for (int k = 0; k < m_nproc; ++k) {
691 uint64_t rangeGlobalIdx;
692 utils::binary::read(stream, rangeGlobalIdx);
693 m_partitionRangeGlobalIdx[k] = rangeGlobalIdx;
694 }
695
696#if BITPIT_ENABLE_MPI==1
697 if (!m_serial) {
698 computeGhostHalo();
699 }
700#endif
701
702 // Full restore (i.e. restore with mapper of last operation)
703 m_mapIdx.clear();
704
705 for (int i = 0; i < m_nproc; ++i) {
706 m_partitionRangeGlobalIdx0[i] = 0;
707 }
708
709 bool full;
710 utils::binary::read(stream, full);
711 if (full) {
712 utils::binary::read(stream, m_lastOp);
713 if (m_lastOp == OP_ADAPT_MAPPED) {
714 m_mapIdx.resize(m_octree.m_octants.size());
715 for (size_t i = 0; i < m_octree.m_octants.size(); ++i) {
716 utils::binary::read(stream, m_mapIdx[i]);
717 }
718
719 size_t lastGhostBrosSize;
720 utils::binary::read(stream, lastGhostBrosSize);
721 m_octree.m_lastGhostBros.resize(lastGhostBrosSize);
722 for (size_t i = 0; i < lastGhostBrosSize; ++i) {
723 utils::binary::read(stream, m_octree.m_lastGhostBros[i]);
724 }
725 }
726 else if (m_lastOp == OP_LOADBALANCE || m_lastOp == OP_LOADBALANCE_FIRST){
727 for (int i = 0; i < m_nproc; ++i) {
728 utils::binary::read(stream, m_partitionRangeGlobalIdx0[i]);
729 }
730 }
731 }
732 else {
733 m_lastOp = OP_INIT;
734 }
735 }

864 void
865 ParaTree::replaceComm(MPI_Comm communicator, MPI_Comm *previousCommunicator)
866 {
867 // The communicator has to be already set
868 if (!isCommSet()) {
869 throw std::runtime_error ("PABLO communicator is not set");
870 }
871
872 // Check if the communicator is valid
873 //
874 // The communicator should be equaivalent to the one currently set,
875 // i.e., the rank of the processes have to be the same in both
876 // communicators.
877 int updatedRank;
878 MPI_Comm_rank(communicator, &updatedRank);
879 int currentRank = getRank();
880
881 int isValid = (updatedRank == currentRank) ? 1 : 0;
882 MPI_Allreduce(MPI_IN_PLACE, &isValid, 1, MPI_INT, MPI_LAND, m_comm);
883 if (!isValid) {
884 throw std::runtime_error ("New communicator is not valid");
885 }
886
887 // Handle previous communicator
888 if (previousCommunicator) {
889 *previousCommunicator = m_comm;
890 m_comm = MPI_COMM_NULL;
891 } else {
892 freeComm();
893 }
894
895 // Set the communicator
896 setComm(communicator);
897 }

2756 ParaTree::findAllNodeNeighbours(const Octant* oct, uint32_t node, u32vector & neighbours, bvector & isghost) const {
2757
2758 int dim = getDim();
2759 int octantLevel = getLevel(oct);
2760
2761 u32vector codimNeighnours;
2762 bvector codimIsGhost;
2763
2764 // Get vertex neighbours
2765 m_octree.findNodeNeighbours(oct, node, neighbours, isghost, false);
2766
2767 // Get edge neighbours
2768 //
2769 // On non uniform trees the vertex can be inside the edge of the
2770 // neighbour (hanging nodes). To correctly consider these neighbours,
2771 // the following logic can be used:
2772 // - if an edge neighbour has the same level or a lower level than
2773 // the current cell, then it certainly is also a vertex neighbour;
2774 // - if a edge neighbour has a higher level than the current cell,
2775 // it is necessary to check if the neighbour actually contains the
2776 // vertex.
2777 if (dim == 3) {
2778 for (int edge : m_treeConstants->nodeEdge[node]) {
2779 findNeighbours(oct, edge, 2, codimNeighnours, codimIsGhost, false);
2780 for (std::size_t i = 0; i < codimNeighnours.size(); ++i) {
2781 const Octant* neighOctant;
2782 if (codimIsGhost[i]==0) {
2783 neighOctant = &m_octree.m_octants[codimNeighnours[i]];
2784 }
2785 else {
2786 neighOctant = &m_octree.m_ghosts[codimNeighnours[i]];
2787 }
2788 int neighOctantLevel = getLevel(neighOctant);
2789 if (neighOctantLevel <= octantLevel) {
2790 neighbours.push_back(codimNeighnours[i]);
2791 isghost.push_back(codimIsGhost[i]);
2792 } else if (isNodeOnOctant(oct, node, neighOctant)) {
2793 neighbours.push_back(codimNeighnours[i]);
2794 isghost.push_back(codimIsGhost[i]);
2795 }
2796 }
2797 }
2798 }
2799
2800 // Get face neighbours
2801 //
2802 // On non uniform trees the vertex can be inside the face of the
2803 // neighbour (hanging nodes). To correctly consider these neighbours,
2804 // the following logic can be used:
2805 // - if an face neighbour has the same level or a lower level than
2806 // the current cell, then it certainly is also a vertex neighbour;
2807 // - if a face neighbour has a higher level than the current cell,
2808 // it is necessary to check if the neighbour actually contains the
2809 // vertex.
2810 for (int j = 0; j < dim; ++j) {
2811 int face = m_treeConstants->nodeFace[node][j];
2812 findNeighbours(oct, face, 1, codimNeighnours, codimIsGhost, false);
2813 for (std::size_t i = 0; i < codimNeighnours.size(); ++i) {
2814 const Octant* neighOctant;
2815 if (codimIsGhost[i]==0) {
2816 neighOctant = &m_octree.m_octants[codimNeighnours[i]];
2817 }
2818 else {
2819 neighOctant = &m_octree.m_ghosts[codimNeighnours[i]];
2820 }
2821 int neighOctantLevel = getLevel(neighOctant);
2822 if (neighOctantLevel <= octantLevel) {
2823 neighbours.push_back(codimNeighnours[i]);
2824 isghost.push_back(codimIsGhost[i]);
2825 } else if (isNodeOnOctant(oct, node, neighOctant)) {
2826 neighbours.push_back(codimNeighnours[i]);
2827 isghost.push_back(codimIsGhost[i]);

3026 ParaTree::getPointOwnerIdx(const double * point) const {
3027 uint32_t noctants = m_octree.m_octants.size();
3028 if(noctants==0)
3029 return numeric_limits<uint32_t>::max();
3030 uint32_t idxtry = noctants/2;
3031 uint32_t x, y, z;
3032 uint64_t morton, mortontry;
3033 int powner = 0;
3034 //ParaTree works in [0,1] domain
3035 if (point[0] > 1+m_tol || point[1] > 1+m_tol || point[2] > 1+m_tol
3036 || point[0] < -m_tol || point[1] < -m_tol || point[2] < -m_tol){
3037 return numeric_limits<uint32_t>::max();
3038 }
3039
3040 x = m_trans.mapX(std::min(std::max(point[0], 0.0), 1.0));
3041 y = m_trans.mapY(std::min(std::max(point[1], 0.0), 1.0));
3042 z = m_trans.mapZ(std::min(std::max(point[2], 0.0), 1.0));
3043
3044 uint32_t maxLength = getMaxLength();
3045 if (x == maxLength) x = x - 1;
3046 if (y == maxLength) y = y - 1;
3047 if (z == maxLength) z = z - 1;
3048 morton = PABLO::computeMorton(m_dim,x,y,z);
3049
3050
3051 powner = 0;
3052 if(!m_serial) powner = findOwner(morton);
3053
3054 if ((powner!=m_rank) && (!m_serial))
3055 return numeric_limits<uint32_t>::max();
3056
3057 int32_t jump = idxtry;
3058 while(abs(jump) > 0){
3059
3060 mortontry = m_octree.m_octants[idxtry].getMorton();
3061 jump = ((mortontry<morton)-(mortontry>morton))*abs(jump)/2;
3062 idxtry += jump;
3063 if (idxtry > noctants-1){
3064 if (jump > 0){
3065 idxtry = noctants - 1;
3066 jump = 0;
3067 }
3068 else if (jump < 0){
3069 idxtry = 0;
3070 jump = 0;
3071 }
3072 }
3073 }
3074 if(m_octree.m_octants[idxtry].getMorton() == morton){
3075 return idxtry;
3076 }
3077 else{
3078 // Step until the mortontry lower than morton (one idx of distance)
3079 {
3080 while(m_octree.m_octants[idxtry].getMorton() < morton){
3081 idxtry++;
3082 if(idxtry > noctants-1){
3083 idxtry = noctants-1;
3084 break;
3085 }
3086 }
3087 while(m_octree.m_octants[idxtry].getMorton() > morton){
3088 idxtry--;
3089 if(idxtry > noctants-1){
3090 idxtry = 0;

4694 void
4695 ParaTree::computePartition(const dvector *weight, uint32_t *partition){
4696
4697 assert(weight->size() >= m_octree.m_sizeOctants);
4698
4699 // Evaluate global weights
4700 //
4701 // If the tree is serial, all process have all the octants, hence
4702 // global weights and local weights are the same.
4703 std::vector<double> globalWeightsStorage;
4704
4705 const double *globalWeights;
4706 if (m_serial) {
4707 globalWeights = weight->data();
4708 } else {
4709 // Information about current partitioning and displacements should
4710 // be stored using uint32_t, however the maximum number of items
4711 // that can be exchanged using MPI funcitons is limited to INT_MAX.
4712 assert(m_globalNumOctants <= INT_MAX);
4713 std::vector<int> currentPartition(m_nproc);
4714 MPI_Allgather(&(m_octree.m_sizeOctants), 1, MPI_INT, currentPartition.data(), 1, MPI_INT, m_comm);
4715
4716 std::vector<int> displacements(m_nproc);
4717 displacements[0] = 0;
4718 for(int i = 1; i < m_nproc; ++i){
4719 displacements[i] = displacements[i - 1] + currentPartition[i - 1];
4720 }
4721
4722 globalWeightsStorage.resize(m_globalNumOctants);
4723 globalWeights = globalWeightsStorage.data();
4724 MPI_Allgatherv(weight->data(), m_octree.m_sizeOctants, MPI_DOUBLE, globalWeightsStorage.data(),
4725 currentPartition.data(), displacements.data(), MPI_DOUBLE, m_comm);
4726 }
4727
4728 // Initialize partitioning
4729 for (int i = 0; i < m_nproc; ++i) {
4730 partition[i] = 0;
4731 }
4732
4733 // Assign octants to partitions
4734 //
4735 // After evaluationg the target weight of a partition, octants will
4736 // be added to that partition until the weigth of the partition is
4737 // greater or equal the target weigth or until all the octant are
4738 // assigned
4739 uint32_t nAssigendOctants = 0;
4740 for (int i = 0; i < m_nproc - 1; ++i) {
4741 double unassignedWeight = 0.;
4742 for (uint32_t n=nAssigendOctants; n<m_globalNumOctants; n++){
4743 unassignedWeight += globalWeights[n];
4744 }
4745 double targetWeight = unassignedWeight / (m_nproc - i);
4746
4747 double partitionWeight = 0.;
4748 while(partitionWeight < targetWeight){
4749 partitionWeight += globalWeights[nAssigendOctants];
4750 partition[i]++;
4751
4752 nAssigendOctants++;
4753 if (nAssigendOctants == m_globalNumOctants) {
4754 break;
4755 }
4756 }
4757
4758 if (nAssigendOctants == m_globalNumOctants) {
4759 break;
4760 }
4761 }
4762 partition[m_nproc-1] = m_globalNumOctants - nAssigendOctants;
4763 };

4778 ParaTree::computePartition(uint8_t level_, const dvector *weight, uint32_t *partition) {
4779
4780 // Compute partitioning without family constrains
4781 uint32_t* partition_temp = new uint32_t[m_nproc];
4782 if (weight==NULL){
4783 computePartition(partition_temp);
4784 }
4785 else{
4786 computePartition(weight, partition_temp);
4787 }
4788
4789 // Modify partitioning to take into account family constrains
4790 //
4791 // Partitioning is modified to guarantee that families of octants at
4792 // the desired level above the maximum depth reached in the tree
4793 // are retained compact on the same process.
4794 uint8_t level = uint8_t(min(int(max(int(m_maxDepth) - int(level_), int(1))) , int(m_treeConstants->maxLevel)));
4795 uint8_t* new_boundary_owner = new uint8_t[m_nproc-1];
4796 uint8_t new_interfaces_count;
4797 uint8_t first_new_interface_rank_index;

4799 uint8_t* first_new_interface_rank_index_per_rank = new uint8_t[m_nproc];
4800
4801 uint32_t Dh = uint32_t(pow(double(2),double(m_treeConstants->maxLevel-level)));
4802 uint32_t istart, nocts, rest;
4803 uint32_t forw = 0, backw = 0;
4804 uint32_t i = 0, iproc, j;
4805 uint64_t sum;
4806 int32_t* pointercomm;
4807 int32_t* deplace = new int32_t[m_nproc-1];
4808
4809 // Find processes currently owning the new incoming process boundaries
4810 // boundary_proc[i] = j means that the new process interface between i-th and (i+1)-th
4811 // processes is falling currently on j-th process
4812 j = 0;
4813 sum = 0;
4814 for (iproc=0; iproc<(uint32_t)(m_nproc-1); iproc++){
4815 sum += partition_temp[iproc];
4816 while(sum > m_partitionRangeGlobalIdx[j]){
4817 j++;
4818 }
4819 new_boundary_owner[iproc] = j;
4820 }
4821
4822 nocts = getNumOctants();
4823 sum = 0;
4824
4825 // Store how many process interfaces fall in the current rank. For these interfaces
4826 // the current rank has to communicate the info about the correction to the partition
4827 // strcture aimed to maintain the families compact.
4828 new_interfaces_count = 0;
4829
4830 // Store the index of the first process owner of the new process interface
4831 // It will be the starting index for the communication of the corrections on
4832 // the partition structure.
4833 first_new_interface_rank_index = 0;
4834
4835 for (iproc=0; iproc<(uint32_t)(m_nproc-1); iproc++){
4836 deplace[iproc] = 1;
4837 sum += partition_temp[iproc];
4838
4839 // If the current process owns a new process interface, check if
4840 // the family at the interface is compact and store the correction on the
4841 // temporary partition structure.
4842 if (new_boundary_owner[iproc] == m_rank){
4843 if (new_interfaces_count == 0){
4844 first_new_interface_rank_index = iproc;
4845 }
4846 new_interfaces_count++;
4847
4848 // Place istart at index of the last octant at new incoming process interface
4849 if (m_rank!=0)
4850 istart = sum - m_partitionRangeGlobalIdx[m_rank-1] - 1;
4851 else
4852 istart = sum;
4853
4854 // Compute the rest w.r.t. the size of the octant of the same size
4855 // of a compact family of the desired level
4856 i = istart;
4857 rest = m_octree.m_octants[i].getLogicalX()%Dh + m_octree.m_octants[i].getLogicalY()%Dh;
4858 if (m_dim == 3){
4859 rest += m_octree.m_octants[i].getLogicalZ()%Dh;
4860 }
4861
4862 // Deplace forward the index of the octant i defining the process boundary until the previous
4863 // defined rest is zero, i.e. the families are compact up to the target level.
4864 while(rest!=0){
4865 if (i==nocts-1){
4866 i = istart + nocts;
4867 break;
4868 }
4869 i++;
4870 rest = m_octree.m_octants[i].getLogicalX()%Dh + m_octree.m_octants[i].getLogicalY()%Dh;
4871 if (m_dim == 3){
4872 rest += m_octree.m_octants[i].getLogicalZ()%Dh;
4873 }
4874 }
4875
4876 // Store the computed forward correction (= local number of octants if not found)
4877 forw = i - istart;
4878
4879 // Do the same for a backward correction try
4880 i = istart;
4881 rest = m_octree.m_octants[i].getLogicalX()%Dh + m_octree.m_octants[i].getLogicalY()%Dh;
4882 if (m_dim == 3){
4883 rest += m_octree.m_octants[i].getLogicalZ()%Dh;
4884 }
4885 while(rest!=0){
4886 if (i==0){
4887 i = istart - nocts;
4888 break;
4889 }
4890 i--;
4891 rest = m_octree.m_octants[i].getLogicalX()%Dh + m_octree.m_octants[i].getLogicalY()%Dh;
4892 if (m_dim == 3){
4893 rest += m_octree.m_octants[i].getLogicalZ()%Dh;
4894 }
4895 }
4896
4897 // Store the backward correction previously computed
4898 backw = istart - i;
4899
4900 // If no correction are needed forward and backward corrections are zero,
4901 // so the deplace term will be zero and, even if communicated it will no correct
4902 // the partition structure.
4903 // If correction is needed the lower between forward and backward corrections
4904 // is the correct one, beeing the other one overdimensioned with local number of octants value.
4905 if (forw<backw)
4906 deplace[iproc] = forw;
4907 else
4908 deplace[iproc] = -(int32_t)backw;
4909 }
4910 }
4911
4912 // Communicate the right corrections to other processes
4913 m_errorFlag = MPI_Allgather(&new_interfaces_count,1,MPI_UINT8_T,new_interfaces_count_per_rank,1,MPI_UINT8_T,m_comm);
4914 m_errorFlag = MPI_Allgather(&first_new_interface_rank_index,1,MPI_UINT8_T,first_new_interface_rank_index_per_rank,1,MPI_UINT8_T,m_comm);
4915 for (iproc=0; iproc<(uint32_t)(m_nproc); iproc++){
4916 pointercomm = &deplace[first_new_interface_rank_index_per_rank[iproc]];
4917 m_errorFlag = MPI_Bcast(pointercomm, new_interfaces_count_per_rank[iproc], MPI_INT32_T, iproc, m_comm);
4918 }
4919
4920 // Apply the corrections stored in deplace container to the termporary
4921 // computed partition structure.
4922 // The new number of octants for each processors are increased by the
4923 // (signed) deplace value related to itself and decreased by the (signed)
4924 // deplace value of the previous process.
4925 for (iproc=0; iproc<(uint32_t)(m_nproc); iproc++){
4926 if (iproc < (uint32_t)(m_nproc-1))
4927 partition[iproc] = partition_temp[iproc] + deplace[iproc];
4928 else
4929 partition[iproc] = partition_temp[iproc];
4930 if (iproc !=0)
4931 partition[iproc] = partition[iproc] - deplace[iproc-1];
4932 }
4933
4934 delete [] partition_temp; partition_temp = NULL;
4935 delete [] new_boundary_owner; new_boundary_owner = NULL;
4936 delete [] new_interfaces_count_per_rank; new_interfaces_count_per_rank = NULL;
4937 delete [] first_new_interface_rank_index_per_rank; first_new_interface_rank_index_per_rank = NULL;
4938 delete [] deplace; deplace = NULL;
4939 }

5114 void
5115 ParaTree::computeGhostHalo(){
5116 // Build first layer of ghosts
5117 setPboundGhosts();
5118
5119 // Early return if we need to build only one layer
5120 if(m_nofGhostLayers <= 1){
5121 return;
5122 }
5123
5124 //
5125 // Accrete sources
5126 //
5127 // We don't build ghost layers directly, instead we identify the
5128 // internal cells that are ghosts for the neighboring process and
5129 // we use this list to create the ghosts. We use the term "sources"
5130 // to identify internal cells that are ghosts for the neighboring
5131 // process. For each layer of ghosts, a corresponding layer of
5132 // sources exists.
5133 //
5134 // Sources are identified one layer at a time. The first layer is
5135 // already known: the process-border octants. The neighbors of
5136 // process-border octants are the second layer of sources; the
5137 // neighbors of the second layer of sources are the third layer,
5138 // and so on an so forth.
5139 //
5140 // To identify the sources, an auxiliary data structure is used. This
5141 // data structure is called accretion and contains the list of sources
5142 // currently identified (population), a list of octants to be used for
5143 // building the next layer of sources (seeds) and the rank on which
5144 // the sources gathered by the accretion will be ghosts.
5145 //
5146 // The identification of the sources starts creating one accretions for
5147 // each of the neighboring processes. The accretions are initialized
5148 // using the process-border octants already build: those octants are
5149 // the first layer of sources and the seeds for the generation of the
5150 // second layer. Adding the internal neighbors of the internal seeds to
5151 // the population, accretions are grown one layer at a time. When an
5152 // accretion reaches a neighboring processes (i.e., when a first-layer
5153 // ghost enters in the list of foreign seeds), we communicate to the
5154 // owner of the ghost to create a new accretion and continue the search
5155 // for the sources. At the end of the procedure, the population of the
5156 // accretions on each process will contain the desired sources.
5157
5158 // Initialize cache for 1-rings of the internal octants
5159 std::unordered_map<uint32_t, std::vector<uint64_t>> oneRingsCache;
5160 oneRingsCache.reserve(getNumOctants());
5161
5162 // Initialize data communicator
5163 DataCommunicator accretionDataCommunicator(m_comm);
5164
5165 // Initialize accretions
5166 std::vector<AccretionData> accretions;
5167 initializeGhostHaloAccretions(&accretions);
5168
5169 // Grow the accretions
5170 for(std::size_t layer = 1; layer < m_nofGhostLayers; ++layer){
5171 // Exchange accretions
5172 //
5173 // When a ghost is incorporated in the seeds, the accretion
5174 // needs to continue on the process that owns the ghost.
5175 exchangeGhostHaloAccretions(&accretionDataCommunicator, &accretions);
5176
5177 // Grow accretions
5178 growGhostHaloAccretions(&oneRingsCache, &accretions);
5179 }
5180
5181 // To correctly identify the population of the last layer of sources,
5182 // we need to exchange the accretions one more time. This allows to
5183 // communicate the foreign seeds found during the last growth to the
5184 // processes that own them.
5185 exchangeGhostHaloAccretions(&accretionDataCommunicator, &accretions);
5186
5187 //
5188 // Extract list of sources
5189 //
5190 // Sources are internal octants that are ghosts for other processes,
5191 // i.e., internal octants on processes borders (pborder octants).
5192 // The population of the accretions is exaclty made of the sources
5193 // for the target rank of the accretion.
5194 for(const AccretionData &accretion : accretions){
5195 int targetRank = accretion.targetRank;
5196
5197 std::vector<uint32_t> &rankBordersPerProc = m_bordersPerProc[targetRank];
5198 rankBordersPerProc.resize(accretion.population.size());
5199
5200 std::size_t index = 0;
5201 for (const auto &populationEntry : accretion.population) {
5202 uint64_t globalIdx = populationEntry.first;
5203 uint32_t localIdx = getLocalIdx(globalIdx);
5204 rankBordersPerProc[index] = localIdx;
5205 ++index;
5206 }
5207
5208 std::sort(rankBordersPerProc.begin(), rankBordersPerProc.end());
5209 }
5210
5211 //
5212 // Build the ghosts
5213 //
5214 buildGhostOctants(m_bordersPerProc, accretions);
5215 }

5225 void
5226 ParaTree::initializeGhostHaloAccretions(std::vector<AccretionData> *accretions) {
5227

5229

5231 for(const auto &bordersPerProcEntry : m_bordersPerProc){
5232 accretions->emplace_back();
5233 AccretionData &accretion = accretions->back();
5234
5235 // Rank for which the accretion will gather sources
5236 int targetRank = bordersPerProcEntry.first;
5237 accretion.targetRank = targetRank;

5240 //
5241 // The population and the seeds of the first layer are the octants
5242 // on processes borders.
5243 const std::vector<uint32_t> &rankBordersPerProc = bordersPerProcEntry.second;
5244 const std::size_t nRankBordersPerProc = rankBordersPerProc.size();

5270 void
5271 ParaTree::growGhostHaloAccretions(std::unordered_map<uint32_t, std::vector<uint64_t>> *oneRingsCache,
5272 std::vector<AccretionData> *accretions) {
5273
5274 // The neighbour of the internal seeds are the next layer of sources.
5275 u32vector seedNeighLocalIds;
5276 bvector seedNeighGhostFlag;
5277 for(AccretionData &accretion : *accretions){
5278 // If the accretion doesn't have internal seeds we can skip it
5279 std::size_t nSeeds = accretion.internalSeeds.size();
5280 if (nSeeds == 0) {
5281 continue;
5282 }
5283
5284 // Rank for which accretion is gathering data
5285 const int targetRank = accretion.targetRank;
5286
5287 // Seeds
5288 //
5289 // We make a copy of the seeds and then we clear the original
5290 // list in order to generate the seeds for the next layer.
5291 std::unordered_map<uint64_t, int> currentInternalSeeds;
5292 currentInternalSeeds.reserve(accretion.internalSeeds.size());
5293 accretion.internalSeeds.swap(currentInternalSeeds);
5294
5295 // The next layer is obtained adding the 1-ring neighbours of
5296 // the internal octants of the previous layer.
5297 for(const auto &seedEntry : currentInternalSeeds){
5298 // Get seed information
5299 int seedLayer = seedEntry.second;
5300 uint64_t seedGlobalIdx = seedEntry.first;
5301 uint32_t seedLocalIdx = getLocalIdx(seedGlobalIdx);
5302
5303 // Find the 1-ring of the source
5304 auto oneRingsCacheItr = oneRingsCache->find(seedLocalIdx);
5305 if (oneRingsCacheItr == oneRingsCache->end()) {
5306 const Octant *seedOctant = getOctant(seedLocalIdx);
5307 findAllCodimensionNeighbours(seedOctant, seedNeighLocalIds, seedNeighGhostFlag);
5308 size_t nSeedNeighs = seedNeighLocalIds.size();
5309
5310 oneRingsCacheItr = oneRingsCache->insert({seedLocalIdx, std::vector<uint64_t>()}).first;
5311 oneRingsCacheItr->second.resize(nSeedNeighs + 1);
5312 for(size_t n = 0; n < nSeedNeighs; ++n){
5313 if (!seedNeighGhostFlag[n]) {
5314 oneRingsCacheItr->second[n] = getGlobalIdx(seedNeighLocalIds[n]);
5315 } else {
5316 oneRingsCacheItr->second[n] = getGhostGlobalIdx(seedNeighLocalIds[n]);
5317 }
5318 }
5319 oneRingsCacheItr->second[nSeedNeighs] = getGlobalIdx(seedLocalIdx);
5320 }
5321 const std::vector<uint64_t> &seedOneRing = oneRingsCacheItr->second;
5322
5323 // Add the 1-ring of the octant to the sources
5324 for(uint64_t neighGlobalIdx : seedOneRing){
5325 // Discard octants already in the population
5326 if (accretion.population.count(neighGlobalIdx) > 0) {
5327 continue;
5328 }
5329
5330 // Get neighbour information
5331 bool isNeighInternal = isInternal(neighGlobalIdx);
5332
5333 int neighRank;
5334 if (isNeighInternal) {
5335 neighRank = getRank();
5336 } else {
5337 neighRank = getOwnerRank(neighGlobalIdx);
5338 }
5339
5340 // Add the neighbour to the population
5341 //
5342 // Population should only contain internal octants.
5343 if (isNeighInternal) {
5344 accretion.population.insert({neighGlobalIdx, seedLayer + 1});
5345 }
5346
5347 // Add the neighbour to the seeds
5348 //
5349 // Internal seeds contains only internal octants, foreign
5350 // seeds containt ghost octants that are not owned by the
5351 // target rank (if an octant is owned by the target rank,
5352 // by definition it will not be a source for that rank).
5353 if (isNeighInternal) {
5354 accretion.internalSeeds.insert({neighGlobalIdx, seedLayer + 1});
5355 } else if (neighRank != targetRank) {
5356 accretion.foreignSeeds.insert({neighGlobalIdx, seedLayer + 1});
5357 }
5358 }
5359 }

5378 //
5379 // When the accretion reaches a foreign process (i.e., a ghost is
5380 // added to the seeds), the ranks that owns the ghost seed have to
5381 // continue the propagation of those seeds (which will be local
5382 // seeds for that foreign rank) locally. Therefore we need to

5391
5392 auto foreignRankAccretionsItr = foreignRankAccretions.begin();

5394 if (foreignRankAccretionsItr->targetRank!= accretion.targetRank) {
5395 continue;
5396 }
5397
5398 break;
5399 }
5400
5401 if (foreignRankAccretionsItr == foreignRankAccretions.end()) {
5402 foreignRankAccretions.emplace_back();
5403 foreignRankAccretionsItr = foreignRankAccretions.begin() + foreignRankAccretions.size() - 1;
5404 foreignRankAccretionsItr->targetRank = accretion.targetRank;
5405 }
5406

5408

5410 foreignAccretion.internalSeeds.insert(seedEntry);
5411 }
5412 }

5433 std::size_t nSeeds = foreignAccretion.internalSeeds.size();

5452 sendBuffer << seedEntry.first;
5453 sendBuffer << seedEntry.second;
5454 }
5455 }
5456 }
5457
5458 // Start communications
5459 dataCommunicator->discoverRecvs();
5460 dataCommunicator->startAllRecvs();
5461 dataCommunicator->startAllSends();
5462
5463 // Receive the accretiation to grow on behalf of neighbour processes

5468

5470 recvBuffer >> nForeignAccretions;

5484 continue;
5485 }
5486
5487 break;
5488 }
5489
5490 if (accretionsItr == accretions->end()) {
5491 accretions->emplace_back();
5492 accretionsItr = accretions->begin() + accretions->size() - 1;
5493 accretionsItr->targetRank = targetRank;
5494 }
5495
5496 AccretionData &accretion = *accretionsItr;
5497

5499 //
5500 // We are receiving only internal octants, there is no need
5501 // to explicitly check if and octant is internal before add
5502 // it to the population and to the internal seeds.
5503 std::size_t nSeeds;
5504 recvBuffer >> nSeeds;

5531 void
5532 ParaTree::buildGhostOctants(const std::map<int, u32vector> &bordersPerProc,
5533 const std::vector<AccretionData> &accretions){
5534
5535 DataCommunicator ghostDataCommunicator(m_comm);
5536
5537 // Binary size of a ghost entry in the communication buffer
5538 const std::size_t GHOST_ENTRY_BINARY_SIZE = sizeof(uint64_t) + Octant::getBinarySize() + sizeof(int);
5539
5540 // Fill the send buffers with source octants
5541 //
5542 // A source octant is an internal octants that is ghosts on other
5543 // process.
5544 for(const auto &bordersPerProcEntry : bordersPerProc){
5545 int rank = bordersPerProcEntry.first;
5546 const std::vector<uint32_t> &rankBordersPerProc = bordersPerProcEntry.second;
5547 std::size_t nRankBordersPerProc = rankBordersPerProc.size();
5548
5549 // Get the accretion associated with this rank
5550 //
5551 // If there are no accretions it means we are building the first
5552 // layer of ghosts.
5553 std::vector<AccretionData>::const_iterator accretionsItr;
5554 if (accretions.size() > 0) {
5555 bool accretionFound = false;
5556 BITPIT_UNUSED(accretionFound);
5557 accretionsItr = accretions.begin();
5558 for (; accretionsItr != accretions.end(); ++accretionsItr) {
5559 if (accretionsItr->targetRank == rank) {
5560 accretionFound = true;
5561 break;
5562 }
5563 }
5564 assert(accretionFound);
5565 } else {
5566 accretionsItr = accretions.end();
5567 }
5568
5569 // Initialize the send
5570 std::size_t buffSize = GHOST_ENTRY_BINARY_SIZE * nRankBordersPerProc;
5571 ghostDataCommunicator.setSend(rank, buffSize);
5572
5573 // Fill the buffer
5574 SendBuffer &sendBuffer = ghostDataCommunicator.getSendBuffer(rank);
5575 for(uint32_t sourceLocalIdx : rankBordersPerProc){
5576 // Global index
5577 uint64_t sourceGlobalIdx = getGlobalIdx(sourceLocalIdx);
5578 sendBuffer << sourceGlobalIdx;
5579
5580 // Source data
5581 sendBuffer << m_octree.m_octants[sourceLocalIdx];
5582
5583 // Layer information
5584 //
5585 // If there are no accretions it means we are building the
5586 // first layer of ghosts.
5587 int layer;
5588 if (accretionsItr != accretions.end()) {
5589 layer = accretionsItr->population.at(sourceGlobalIdx);
5590 } else {
5591 layer = 0;
5592 }
5593
5594 sendBuffer << layer;
5595 }
5596 }
5597
5598 // Discover the receives
5599 ghostDataCommunicator.discoverRecvs();
5600 ghostDataCommunicator.startAllRecvs();
5601 ghostDataCommunicator.startAllSends();
5602
5603 // Get the ranks from which ghosts will be received
5604 std::vector<int> ghostCommunicatorRecvsRanks = ghostDataCommunicator.getRecvRanks();
5605 std::sort(ghostCommunicatorRecvsRanks.begin(), ghostCommunicatorRecvsRanks.end());
5606
5607 // Prepare ghost data structures
5608 m_octree.m_sizeGhosts = 0;
5609 for (int rank : ghostCommunicatorRecvsRanks) {
5610 RecvBuffer &recvBuffer = ghostDataCommunicator.getRecvBuffer(rank);
5611 std::size_t nRankGhosts = recvBuffer.getSize() / GHOST_ENTRY_BINARY_SIZE;
5612 m_octree.m_sizeGhosts += nRankGhosts;
5613 }
5614
5615 m_octree.m_ghosts.resize(m_octree.m_sizeGhosts);
5616 m_octree.m_globalIdxGhosts.resize(m_octree.m_sizeGhosts);
5617
5618 // Receive the ghosts
5619 //
5620 // Ghosts have to be received following the rank order.
5621 uint32_t ghostLocalIdx = 0;
5622 for (int rank : ghostCommunicatorRecvsRanks) {
5623 ghostDataCommunicator.waitRecv(rank);
5624 RecvBuffer &recvBuffer = ghostDataCommunicator.getRecvBuffer(rank);
5625
5626 std::size_t nRankGhosts = recvBuffer.getSize() / GHOST_ENTRY_BINARY_SIZE;
5627 for(std::size_t n = 0; n < nRankGhosts; ++n){
5628 // Assign the global index
5629 uint64_t ghostGlobalIdx;
5630 recvBuffer >> ghostGlobalIdx;
5631 m_octree.m_globalIdxGhosts[ghostLocalIdx] = ghostGlobalIdx;
5632
5633 // Build the ghosts
5634 //
5635 // The layer of the received octant will be overwritten with
5636 // the actual ghost layer.
5637 Octant &ghostOctant = m_octree.m_ghosts[ghostLocalIdx];
5638 recvBuffer >> ghostOctant;
5639
5640 // Set the layer of the ghost
5641 int ghostLayer;
5642 recvBuffer >> ghostLayer;
5643 ghostOctant.setGhostLayer(ghostLayer);
5644
5645 // Increase the ghost index
5646 ++ghostLocalIdx;
5647 }
5648 }
5649
5650 // Wait for the communications to complete
5651 ghostDataCommunicator.waitAllSends();
