246void SurfUnstructured::extractEdgeNetwork(LineUnstructured &net)
247{
248 // ====================================================================== //
249 // VARIABLES DECLARATION //
250 // ====================================================================== //
251
252 // Local variables
253 bool check;
254 int n_faces, n_adj;
255 long id;
256 const long *adjacencies;
257
258 // Counters
259 int i, j;
260 VertexIterator v_, ve_ = vertexEnd();
261 CellIterator c_, ce_ = cellEnd();
262
263 // ====================================================================== //
264 // INITIALIZE DATA STRUCTURE //
265 // ====================================================================== //
266 net.reserveCells(net.getCellCount() + countFaces());
267 net.reserveVertices(net.getVertexCount() + getVertexCount());
268
269 // ====================================================================== //
270 // ADD VERTEX TO net //
271 // ====================================================================== //
272 for (v_ = vertexBegin(); v_ != ve_; ++v_) {
273 net.addVertex(v_->getCoords(), v_->getId());
274 } //next v_
275
276 // ====================================================================== //
277 // ADD EDGES //
278 // ====================================================================== //
279 for (c_ = cellBegin(); c_ != ce_; ++c_) {
280 id = c_->getId();
281 n_faces = c_->getFaceCount();
282 ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
283 for (i = 0; i < n_faces; ++i) {
284 check = true;
285 n_adj = c_->getAdjacencyCount(i);
286 adjacencies = c_->getAdjacencies(i);
287 for (j = 0; j < n_adj; ++j) {
288 check = check && (id > adjacencies[j]);
289 } //next j
290 if (check) {
291 // Get edge type
292 ElementType edgeType = c_->getFaceType(i);
293
294 // Get edge connect
295 ConstProxyVector<long> faceConnect = c_->getFaceConnect(i);
296 int faceConnectSize = faceConnect.size();
297
298 std::unique_ptr<long[]> edgeConnect = std::unique_ptr<long[]>(new long[faceConnectSize]);
299 for (int k = 0; k < faceConnectSize; ++k) {
300 edgeConnect[k] = faceConnect[k];
301 }
302
303 // Add edge
304 net.addCell(edgeType, std::move(edgeConnect));
305 }
306 } //next i
307 } //next c_
308
309 return;
310}

411int SurfUnstructured::importSTL(const std::string &filename, STLReader::Format format,
412 bool joinFacets, int PIDOffset, bool PIDSquash,
413 std::unordered_map<int, std::string> *PIDNames)
414{
415 int readerError;
416
417 // Initialize reader
418 STLReader reader(filename, format);
419 if (format == STLReader::FormatUnknown) {
420 format = reader.getFormat();
421 }
422
423 // Begin reding the STL file
424 readerError = reader.readBegin();
425 if (readerError != 0) {
426 return readerError;
427 }
428
429 // Initialize cache needed for joinin the facets
430 //
431 // The cache will contain the ids of the vertices that have been added
432 // to the patch. It uses a special comparator that allows to compare
433 // the coordinates of a candidate vertex with the coordinates of the
434 // vertices in the cache.
435 std::array<double, 3> *candidateVertexCoords;
436 Vertex::Less vertexLess(10 * std::numeric_limits<double>::epsilon());
437
438 auto vertexCoordsLess = [this, &vertexLess, &candidateVertexCoords](const long &id_1, const long &id_2)
439 {
440 const std::array<double, 3> &coords_1 = (id_1 >= 0) ? this->getVertex(id_1).getCoords() : *candidateVertexCoords;
441 const std::array<double, 3> &coords_2 = (id_2 >= 0) ? this->getVertex(id_2).getCoords() : *candidateVertexCoords;
442
443 return vertexLess(coords_1, coords_2);
444 };
445
446 std::set<long, decltype(vertexCoordsLess)> vertexCache(vertexCoordsLess);
447
448 // Read all the solids in the STL file
449 int pid = PIDOffset;
450
451 ElementType facetType = ElementType::TRIANGLE;
452 const int nFacetVertices = ReferenceElementInfo::getInfo(ElementType::TRIANGLE).nVertices;
453
454 while (true) {
455 // Read header
456 std::size_t nFacets;
457 std::string name;
458 readerError = reader.readHeader(&name, &nFacets);
459 if (readerError != 0) {
460 if (readerError == -2) {
461 break;
462 } else {
463 return readerError;
464 }
465 }
466
467 // Generate patch cells from STL facets
468 reserveVertices(getVertexCount() + nFacetVertices * nFacets);
469 reserveCells(getCellCount() + nFacets);
470
471 vertexCache.clear();
472
473 for (std::size_t n = 0; n < nFacets; ++n) {
474 // Read facet data
475 std::array<Vertex, 3> faceVertices;
476 std::array<double, 3> facetNormal;
477 readerError = reader.readFacet(&(faceVertices[0].getCoords()), &(faceVertices[1].getCoords()), &(faceVertices[2].getCoords()), &facetNormal);
478 if (readerError != 0) {
479 return readerError;
480 }
481
482 // Add vertices
483 std::unique_ptr<long[]> connectStorage = std::unique_ptr<long[]>(new long[nFacetVertices]);
484 if (joinFacets) {
485 for (int i = 0; i < nFacetVertices; ++i) {
486 // The candidate vertex is set equal to the face vertex
487 // that needs to be added, then the cache is checked: if
488 // a vertex with the same coordinates is already in the
489 // cache, the vertex will not be added and the existing
490 // one will be used.
491 //
492 // The cache contains the ids of the vertices that have
493 // been added. It uses a special comparator that allows
494 // to compare the candidate vertex with the ones in the
495 // cache, in order to do that the null vertex should be
496 // passed to the find function.
497 candidateVertexCoords = &(faceVertices[i].getCoords());
498
499 long vertexId;
500 auto vertexCacheItr = vertexCache.find(Vertex::NULL_ID);
501 if (vertexCacheItr == vertexCache.end()) {
502 VertexIterator vertexItr = addVertex(*candidateVertexCoords);
503 vertexId = vertexItr.getId();
504 vertexCache.insert(vertexId);
505 } else {
506 vertexId = *vertexCacheItr;
507 }
508 connectStorage[i] = vertexId;
509 }
510 } else {
511 for (int i = 0; i < nFacetVertices; ++i) {
512 VertexIterator vertexItr = addVertex(faceVertices[i].getCoords());
513 long vertexId = vertexItr.getId();
514 connectStorage[i] = vertexId;
515 }
516 }
517
518 // Add cell
519 CellIterator cellIterator = addCell(facetType, std::move(connectStorage));
520 cellIterator->setPID(pid);
521 }
522
523 // Read footer
524 readerError = reader.readFooter(name);
525 if (readerError != 0) {
526 return readerError;
527 }
528
529 // Assign PID name
530 if (!PIDSquash) {
531 if (PIDNames && !name.empty()) {
532 PIDNames->insert({{pid, name}});
533 }
534 ++pid;
535 }
536
537 // Multi-body STL files are supported only in ASCII mode
538 if (format == STLReader::FormatBinary) {
539 break;
540 }
541 }
542
543 // End reading
544 readerError = reader.readEnd();
545 if (readerError != 0) {
546 return readerError;
547 }
548
549 return 0;
550}

605int SurfUnstructured::exportSTLSingle(const std::string &filename, bool isBinary)
606{
607 int writerError;
608
609 // Initialize writer
610 STLReader::Format format;
611 if (isBinary) {
612 format = STLReader::FormatBinary;
613 } else {
614 format = STLReader::FormatASCII;
615 }
616
617 STLWriter writer(filename, format);
618
619 // Solid name
620 //
621 // An empty solid name will be used.
622 const std::string name = "";
623
624 // Detect if this is the master writer
625 bool isMasterWriter;
626#if BITPIT_ENABLE_MPI==1
627 if (isPartitioned()) {
628 isMasterWriter = (getRank() == 0);
629 } else {
630#else
631 {
632#endif
633 isMasterWriter = true;
634 }
635
636 // Detect if the file will be written incrementally
637 bool incrementalWrite;
638#if BITPIT_ENABLE_MPI==1
639 incrementalWrite = true;
640#else
641 incrementalWrite = false;
642#endif
643
644 // Count the number of facets
645 long nFacets = getInternalCellCount();
646#if BITPIT_ENABLE_MPI==1
647 if (isPartitioned()) {
648 MPI_Allreduce(MPI_IN_PLACE, &nFacets, 1, MPI_LONG, MPI_SUM, getCommunicator());
649 }
650#endif
651
652 // Write header
653 if (isMasterWriter) {
654 // Begin writing
655 writerError = writer.writeBegin(STLWriter::WriteOverwrite, incrementalWrite);
656 if (writerError != 0) {
657 writer.writeEnd();
658
659 return writerError;
660 }
661
662 // Write header section
663 writerError = writer.writeHeader(name, nFacets);
664 if (writerError != 0) {
665 writer.writeEnd();
666
667 return writerError;
668 }
669
670#if BITPIT_ENABLE_MPI==1
671 // Close the file to let other process write into it
672 writerError = writer.writeEnd();
673 if (writerError != 0) {
674 return writerError;
675 }
676#endif
677 }
678
679 // Write facet data
680#if BITPIT_ENABLE_MPI==1
681 int nRanks = getProcessorCount();
682 for (int i = 0; i < nRanks; ++i) {
683 if (i == getRank()) {
684 // Begin writing
685 writerError = writer.writeBegin(STLWriter::WriteAppend, incrementalWrite);
686 if (writerError != 0) {
687 writer.writeEnd();
688 return writerError;
689 }
690
691#else
692 {
693#endif
694 // Write facet section
695 CellConstIterator cellBegin = internalCellConstBegin();
696 CellConstIterator cellEnd = internalCellConstEnd();
697 for (CellConstIterator cellItr = cellBegin; cellItr != cellEnd; ++cellItr) {
698 // Get cell
699 const Cell &cell = *cellItr;
700
701 // Get vertex coordinates
702 ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
703 assert(cellVertexIds.size() == 3);
704
705 const std::array<double, 3> &coords_0 = getVertex(cellVertexIds[0]).getCoords();
706 const std::array<double, 3> &coords_1 = getVertex(cellVertexIds[1]).getCoords();
707 const std::array<double, 3> &coords_2 = getVertex(cellVertexIds[2]).getCoords();
708
709 // Evaluate normal
710 const std::array<double, 3> normal = evalFacetNormal(cell.getId());
711
712 // Write data
713 writerError = writer.writeFacet(coords_0, coords_1, coords_2, normal);
714 if (writerError != 0) {
715 writer.writeEnd();
716 return writerError;
717 }
718 }
719
720#if BITPIT_ENABLE_MPI==1
721 // Close the file to let other process write into it
722 writerError = writer.writeEnd();
723 if (writerError != 0) {
724 return writerError;
725 }
726 }
727
728 if (isPartitioned()) {
729 MPI_Barrier(getCommunicator());
730 }
731#endif
732 }
733
734 // Write footer
735 if (isMasterWriter) {
736#if BITPIT_ENABLE_MPI==1
737 // Begin writning
738 //
739 // The file was previously closed to let other process write into it.
740 writerError = writer.writeBegin(STLWriter::WriteAppend, incrementalWrite);
741 if (writerError != 0) {
742 writer.writeEnd();
743 return writerError;
744 }
745#endif
746
747 // Write footer section
748 writerError = writer.writeFooter(name);
749 if (writerError != 0) {
750 writer.writeEnd();
751
752 return writerError;
753 }
754
755 // End writing
756 writerError = writer.writeEnd();
757 if (writerError != 0) {
758 return writerError;
759 }
760 }
761
762 return 0;
763}

775int SurfUnstructured::exportSTLMulti(const std::string &filename, std::unordered_map<int, std::string> *PIDNames)
776{
777 int writerError;
778
779 // Initialize writer
780 STLWriter writer(filename, STLReader::FormatASCII);
781
782 // Detect if this is the master writer
783 bool isMasterWriter;
784#if BITPIT_ENABLE_MPI==1
785 if (isPartitioned()) {
786 isMasterWriter = (getRank() == 0);
787 } else {
788#else
789 {
790#endif
791 isMasterWriter = true;
792 }
793
794 // Detect if the file will be written incrementally
795 bool incrementalWrite = true;
796
797#if BITPIT_ENABLE_MPI==1
798 // Count number of ranks
799 int nRanks = getProcessorCount();
800#endif
801
802 // Get the global list of PID
803 std::set<int> cellPIDs;
804#if BITPIT_ENABLE_MPI==1
805 if (isPartitioned()) {
806 std::set<int> internalPIDs = getInternalCellPIDs();
807 std::vector<int> localPIDs(internalPIDs.begin(), internalPIDs.end());
808 int nLocalPids = localPIDs.size();
809
810 std::vector<int> gatherPIDCount(nRanks);
811 MPI_Allgather(&nLocalPids, 1, MPI_INT, gatherPIDCount.data(), 1, MPI_INT, getCommunicator());
812
813 std::vector<int> gatherPIDDispls(nRanks, 0);
814 for (int i = 1; i < nRanks; ++i) {
815 gatherPIDDispls[i] = gatherPIDDispls[i - 1] + gatherPIDCount[i - 1];
816 }
817
818 int gatherPIDsSize = gatherPIDDispls.back() + gatherPIDCount.back();
819
820 std::vector<int> globalPIDs(gatherPIDsSize);
821 MPI_Allgatherv(localPIDs.data(), nLocalPids, MPI_INT, globalPIDs.data(),
822 gatherPIDCount.data(), gatherPIDDispls.data(), MPI_INT, getCommunicator());
823
824 cellPIDs = std::set<int>(globalPIDs.begin(), globalPIDs.end());
825 } else {
826#else
827 {
828#endif
829 cellPIDs = getInternalCellPIDs();
830 }
831
832 // Export cells
833 bool firstPID = true;
834 for (int pid : cellPIDs) {
835 // Cells associated with the PID
836 std::vector<long> cells = getInternalCellsByPID(pid);
837
838 // Count the number of facets
839 long nFacets = cells.size();
840#if BITPIT_ENABLE_MPI==1
841 if (isPartitioned()) {
842 MPI_Allreduce(MPI_IN_PLACE, &nFacets, 1, MPI_LONG, MPI_SUM, getCommunicator());
843 }
844#endif
845
846 // Solid name
847 std::string name;
848 if (PIDNames && PIDNames->count(pid) > 0) {
849 name = PIDNames->at(pid);
850 } else {
851 name = std::to_string(pid);
852 }
853
854 // Write header
855 if (isMasterWriter) {
856 // Begin writing
857 STLWriter::WriteMode writeMode;
858 if (firstPID) {
859 writeMode = STLWriter::WriteOverwrite;
860 } else {
861 writeMode = STLWriter::WriteAppend;
862 }
863
864 writerError = writer.writeBegin(writeMode, incrementalWrite);
865 if (writerError != 0) {
866 writer.writeEnd();
867
868 return writerError;
869 }
870
871 // Write header section
872 writerError = writer.writeHeader(name, nFacets);
873 if (writerError != 0) {
874 writer.writeEnd();
875
876 return writerError;
877 }
878
879#if BITPIT_ENABLE_MPI==1
880 // Close the file to let other process write into it
881 writerError = writer.writeEnd();
882 if (writerError != 0) {
883 return writerError;
884 }
885#endif
886 }
887
888 // Write facet data
889#if BITPIT_ENABLE_MPI==1
890 for (int i = 0; i < nRanks; ++i) {
891 if (i == getRank()) {
892 // Begin writing
893 writerError = writer.writeBegin(STLWriter::WriteAppend, incrementalWrite);
894 if (writerError != 0) {
895 writer.writeEnd();
896 return writerError;
897 }
898
899#else
900 {
901#endif
902 // Write facet section
903 for (long cellId : cells) {
904 // Get cell
905 const Cell &cell = getCell(cellId);
906
907 // Get vertex coordinates
908 ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
909 assert(cellVertexIds.size() == 3);
910
911 const std::array<double, 3> &coords_0 = getVertex(cellVertexIds[0]).getCoords();
912 const std::array<double, 3> &coords_1 = getVertex(cellVertexIds[1]).getCoords();
913 const std::array<double, 3> &coords_2 = getVertex(cellVertexIds[2]).getCoords();
914
915 // Evaluate normal
916 const std::array<double, 3> normal = evalFacetNormal(cell.getId());
917
918 // Write data
919 writerError = writer.writeFacet(coords_0, coords_1, coords_2, normal);
920 if (writerError != 0) {
921 writer.writeEnd();
922 return writerError;
923 }
924 }
925
926 #if BITPIT_ENABLE_MPI==1
927 // Close the file to let other process write into it
928 writerError = writer.writeEnd();
929 if (writerError != 0) {
930 return writerError;
931 }
932 }
933
934 if (isPartitioned()) {
935 MPI_Barrier(getCommunicator());
936 }
937#endif
938 }
939
940 // Write footer
941 if (isMasterWriter) {
942#if BITPIT_ENABLE_MPI==1
943 // Begin writing
944 //
945 // The file was previously closed to let other process write into it.
946 writerError = writer.writeBegin(STLWriter::WriteAppend, incrementalWrite);
947 if (writerError != 0) {
948 writer.writeEnd();
949 return writerError;
950 }
951#endif
952
953 // Write footer section
954 writerError = writer.writeFooter(name);
955 if (writerError != 0) {
956 writer.writeEnd();
957
958 return writerError;
959 }
960
961 // End writing
962 writerError = writer.writeEnd();
963 if (writerError != 0) {
964 return writerError;
965 }
966 }
967
968 // The first PID has been written
969 firstPID = false;
970 }
971
972 return 0;
973}

1022int SurfUnstructured::importDGF(const std::string &filename, bool joinFacets, int PIDOffset, bool PIDSquash)
1023{
1024 // ====================================================================== //
1025 // VARIABLES DECLARATION //
1026 // ====================================================================== //
1027
1028 // Local variables
1029 DGFObj dgf_in(filename);
1030 int nV = 0, nS = 0;
1031 std::vector<std::array<double, 3>> vertex_list;
1032 std::vector<std::vector<int>> simplex_list;
1033 std::vector<int> simplex_PID;
1034 std::vector<long> vertex_map;
1035 std::vector<long> connect;
1036
1037 // Counters
1038 std::vector<std::vector<int>>::iterator c_, ce_;
1039 std::vector<int>::iterator i_, ie_;
1040 std::vector<long>::iterator j_, je_;
1041
1042 // ====================================================================== //
1043 // IMPORT DATA //
1044 // ====================================================================== //
1045
1046 // Read vertices and cells from DGF file
1047 dgf_in.load(nV, nS, vertex_list, simplex_list, simplex_PID);
1048
1049 // Add vertices
1050 vertex_map.resize(nV);
1051 if (joinFacets) {
1052 // Vertices added to the patch are also added into a cache. Before
1053 // inserting a new vertex, the cache is checked: if a vertex with the
1054 // same coordinates is already in the cache, the new vertex will not
1055 // be added and the existing one will be used.
1056 //
1057 // The cache contains the position of the vertices in the list of
1058 // vertices read from the DGF file.
1059 Vertex::Less vertexLess(10 * std::numeric_limits<double>::epsilon());
1060
1061 auto vertexCoordsLess = [&vertexLess, &vertex_list](const int &i, const int &j)
1062 {
1063 return vertexLess(vertex_list[i], vertex_list[j]);
1064 };
1065
1066 std::set<int, decltype(vertexCoordsLess)> vertexCache(vertexCoordsLess);
1067
1068 for (int i = 0; i < nV; ++i) {
1069 long vertexId;
1070 auto vertexCacheItr = vertexCache.find(i);
1071 if (vertexCacheItr == vertexCache.end()) {
1072 VertexIterator vertexItr = addVertex(vertex_list[i]);
