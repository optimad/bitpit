52void Map::initialize(uint8_t dim){
53 m_dim = dim;
54
55 m_origin[0] = 0.;
56 m_origin[1] = 0.;
57 m_origin[2] = 0.;
58
59 m_L = 1.0;
60
61 if (m_dim > 0) {
62 const TreeConstants &treeConstants = TreeConstants::instance(m_dim);
63
64 m_nnodes = treeConstants.nNodes;
65 m_nnodesPerFace = treeConstants.nNodesPerFace;
66
67 m_maxLength = treeConstants.lengths[0];
68 m_maxArea = treeConstants.areas[0];
69 m_maxVolume = treeConstants.volumes[0];
70
71 m_maxLength_1 = 1. / double(m_maxLength);
72 m_maxArea_1 = 1. / double(m_maxArea);
73 m_maxVolume_1 = 1. / double(m_maxVolume);
74 } else {
75 m_nnodes = 0;
76 m_nnodesPerFace = 0;
77
78 m_maxLength = 0;
79 m_maxArea = 0;
80 m_maxVolume = 0;
81
82 m_maxLength_1 = std::numeric_limits<double>::infinity();
83 m_maxArea_1 = std::numeric_limits<double>::infinity();
84 m_maxVolume_1 = std::numeric_limits<double>::infinity();
85 }
86
87};
