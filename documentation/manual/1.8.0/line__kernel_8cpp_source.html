205double LineKernel::evalCellLength(long id) const
206{
207 const Cell &cell = m_cells[id];
208 switch (cell.getType()) {
209
210 case ElementType::LINE:
211 {
212 ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
213 const Vertex &vertex_0 = getVertex(cellVertexIds[0]);
214 const Vertex &vertex_1 = getVertex(cellVertexIds[1]);
215 double length = norm2(vertex_1.getCoords() - vertex_0.getCoords());
216
217 return length;
218 }
219
220 default:
221 {
222 return 0.;
223 }
224
225 }
226}

240std::array<double, 3> LineKernel::evalCellNormal(long id, const std::array<double, 3> &orientation) const
241{
242 const Cell &cell = m_cells[id];
243 switch (cell.getType()) {
244
245 case ElementType::LINE:
246 {
247 ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
248 const Vertex &vertex_0 = getVertex(cellVertexIds[0]);
249 const Vertex &vertex_1 = getVertex(cellVertexIds[1]);
250
251 std::array<double, 3> normal = vertex_1.getCoords() - vertex_0.getCoords();
252 normal = crossProduct(normal, orientation);
253 normal = normal / norm2(normal);
254
255 return normal;
256 }
257
258 default:
259 {
260 return {{0., 0., 0.}};
261 }
262
263 }
264}
