166void VTKRectilinearGrid::readMetaInformation( ){
167
168 std::fstream str;
169 std::string line, temp;
170
171 std::fstream::pos_type position;
172 std::array<int,6> extensions ;
173
174
175 str.open( m_fh.getPath( ), std::ios::in ) ;
176
177 getline( str, line);
178 while( ! bitpit::utils::string::keywordInString( line, "<VTKFile")){
179 getline(str, line);
180 }
181
182 if( bitpit::utils::string::getAfterKeyword( line, "header_type", '\"', temp) ){
183 setHeaderType( temp) ;
184 }
185
186 while( ! bitpit::utils::string::keywordInString( line, "<Piece")){
187 getline(str, line);
188 }
189
190 bitpit::utils::string::getAfterKeyword( line, "Extent", '\"', temp) ;
191 bitpit::utils::string::convertString( temp, extensions );
192
193 m_localIndex[0][0] = extensions[0] ;
194 m_localIndex[0][1] = extensions[1] ;
195 m_localIndex[1][0] = extensions[2] ;
196 m_localIndex[1][1] = extensions[3] ;
197 m_localIndex[2][0] = extensions[4] ;
198 m_localIndex[2][1] = extensions[5] ;
199
200 position = str.tellg() ;
201
202 readDataHeader( str ) ;
203
204 // Read metadata information
205 for( auto &field : m_geometry ){ //int i=0; i<geometry.size(); ++i){
206 str.seekg( position) ;
207 if( ! readDataArray( str, field ) ) {
208#if BITPIT_ENABLE_DEBUG
209 log::cout() <<"Geometry field " << field.getName() << " not found, it will be disabled" << std::endl ;
210#endif
211 field.disable();
212 }
213 }
214
215
216 setDimensions( m_localIndex[0][0], m_localIndex[0][1], m_localIndex[1][0], m_localIndex[1][1], m_localIndex[2][0], m_localIndex[2][1] ) ;
217 str.close() ;
218
219}

224void VTKRectilinearGrid::writeMetaInformation( ){
225
226 std::fstream str;
227
228 str.open( m_fh.getPath( ), std::ios::out ) ;
229 if (!str.is_open()) {
230 throw std::runtime_error("Cannot create file \"" + m_fh.getName() + "\"" + " inside the directory \"" + m_fh.getDirectory() + "\"");
231 }
232
233 //Writing XML header
234 str << "<?xml version=\"1.0\"?>" << std::endl;
235
236 //Writing Piece Information
237 str << "<VTKFile type=\"RectilinearGrid\" version=\"0.1\" byte_order=\"LittleEndian\" header_type=\"" << m_headerType << "\">" << std::endl;
238 str << " <RectilinearGrid WholeExtent= \""
239 << m_globalIndex[0][0] << " " << m_globalIndex[0][1]<< " "
240 << m_globalIndex[1][0] << " " << m_globalIndex[1][1]<< " "
241 << m_globalIndex[2][0] << " " << m_globalIndex[2][1]<< " "
242 << "\" >" << std::endl;
243
244 str << " <Piece Extent= \" "
245 << m_localIndex[0][0] << " " << m_localIndex[0][1]<< " "
246 << m_localIndex[1][0] << " " << m_localIndex[1][1]<< " "
247 << m_localIndex[2][0] << " " << m_localIndex[2][1]<< " "
248 << "\" >" << std::endl;
249
250
251
252 //Header for Data
253 writeDataHeader( str, false ) ;
254
255 //Wring Geometry Information
256 str << " <Coordinates>" << std::endl;
257 writeDataArray( str, m_geometry[0] ) ;
258 writeDataArray( str, m_geometry[1] ) ;
259 writeDataArray( str, m_geometry[2] ) ;
260 str << " </Coordinates>" << std::endl;
261
262 //Closing Piece
263 str << " </Piece>" << std::endl;
264 str << " </RectilinearGrid>" << std::endl;
265
266 //Write Appended Section
267 str << " <AppendedData encoding=\"raw\">" << std::endl;
268 str << "_" ;
269 str << std::endl ;
270
271 //Closing XML
272 str << "</VTKFile>" << std::endl;
273
274 str.close() ;
275
276}

282void VTKRectilinearGrid::writeCollection( ){
283
284 std::fstream str ;
285
286 FileHandler fhp, fho ;
287
288 fhp = m_fh ;
289 fho = m_fh ;

291 fhp.setParallel(false) ;
292 fhp.setAppendix("pvtr") ;
293
294 fho.setDirectory(".") ;
295
296 str.open( fhp.getPath( ), std::ios::out ) ;
297
298 //Writing XML header
299 str << "<?xml version=\"1.0\"?>" << std::endl;
300
301 //Writing Piece Information
302 str << "<VTKFile type=\"PRectilinearGrid\" version=\"0.1\" byte_order=\"LittleEndian\">" << std::endl;
303 str << " <PRectilinearGrid WholeExtent= \""
304 << m_globalIndex[0][0] << " " << m_globalIndex[0][1]<< " "
305 << m_globalIndex[1][0] << " " << m_globalIndex[1][1]<< " "
306 << m_globalIndex[2][0] << " " << m_globalIndex[2][1]<< " "
307 << "GhostLevel=\"0\">" << std::endl;
308
309
310
311 //Header for Data
312 writeDataHeader( str, true );
313
314 //Wring Geometry Information
315 str << " <PCoordinates>" << std::endl;
316 writePDataArray( str, m_geometry[0] ) ;
317 writePDataArray( str, m_geometry[1] ) ;
318 writePDataArray( str, m_geometry[2] ) ;
319 str << " </PCoordinates>" << std::endl;
320
321
322 for( int i=0; i<m_procs; i++){
323 fho.setBlock(i) ;
324 extension3D_t &index = m_procIndex[i] ;
325
326 str << " <Piece Extent= \" "
327 << index[0][0] << " " << index[0][1] << " "
328 << index[1][0] << " " << index[1][1] << " "
329 << index[2][0] << " " << index[2][1] << " "
330 << "\" Source= \"" << fho.getPath() << "\"/>" << std::endl;
331 }
332
333 str << " </PRectilinearGrid>" << std::endl;
334 str << "</VTKFile>" << std::endl;
335
336 str.close() ;

349void VTKRectilinearGrid::setDimensions( int n1, int n2, int m1, int m2, int l1, int l2 ){
350
351 m_dimensions = ( l1 == l2 ) ? 2 :3 ;
352
353 m_localIndex[0][0] = n1 ;
354 m_localIndex[0][1] = n2 ;
355 m_localIndex[1][0] = m1 ;
356 m_localIndex[1][1] = m2 ;
357 m_localIndex[2][0] = l1 ;
358 m_localIndex[2][1] = l2 ;
359
360 if(m_procs==1)
361 m_globalIndex = m_localIndex ;
362
363 m_cells = 1 ;
364 m_points = 1 ;
365
366 for(int d=0; d<m_dimensions; ++d){
367 m_cells = m_cells * ( m_localIndex[d][1] -m_localIndex[d][0] ) ;
368 m_points = m_points * ( m_localIndex[d][1] -m_localIndex[d][0] +1 ) ;
369 }
370
371}

510uint64_t VTKRectilinearGrid::calcFieldEntries( const VTKField &field ){
511
512 uint64_t entries(0) ;
513 std::string name( field.getName() ) ;
514
515 if( name == "x_Coord" ){
516 entries = m_localIndex[0][1] -m_localIndex[0][0] +1 ;
517
518 } else if( name == "y_Coord" ){
519 entries = m_localIndex[1][1] -m_localIndex[1][0] +1 ;
520
521 } else if( name == "z_Coord" ){
522 entries = m_localIndex[2][1] -m_localIndex[2][0] +1 ;
523
524 } else{
525
526 VTKLocation location( field.getLocation() ) ;
527 assert( location != VTKLocation::UNDEFINED) ;
528
529 if( location == VTKLocation::CELL ){
530 entries = m_cells ;
531
532 } else if( location == VTKLocation::POINT ){
533 entries = m_points ;
534
535 }
536
537 VTKFieldType fieldType( field.getFieldType() ) ;
538 assert( fieldType != VTKFieldType::UNDEFINED) ;
539
540 entries *= static_cast<uint64_t>(fieldType) ;
541
542 }
543
544 return entries ;
545
546}
