42template<typename storage_manager_t>
43LevelSetMaskObject<storage_manager_t>::LevelSetMaskObject(int id, const std::unordered_set<long> &mask, const VolumeKernel &mesh ) :LevelSetSegmentationObject<storage_manager_t>(id) {
44
45 std::unordered_map<long,long> meshToEnvelope ;
46
47 std::unique_ptr<SurfUnstructured> segmentation = extractCellEnvelope(mask,mesh,meshToEnvelope) ;
48
49 long intrIndex = meshToEnvelope.begin()->first;
50 long enveIndex = meshToEnvelope.begin()->second;
51
52 bool sameOrientation = sameInterfaceEnvelopeOrientation(mesh, intrIndex, *segmentation, enveIndex);
53
54 auto const &interface = mesh.getInterface(intrIndex);
55 long ownerId = interface.getOwner();
56 bool invert = (mask.count(ownerId)==0) ;
57
58 bool flip = (sameOrientation == invert);
59
60 bool orientable = segmentation->adjustCellOrientation( enveIndex, flip);
61 if( !orientable){
62 throw std::runtime_error ("Error in LevelSetMaskObject");
63 }
64
65 this->setSegmentation(std::move(segmentation));
66}

105template<typename storage_manager_t>
106std::unique_ptr<SurfUnstructured> LevelSetMaskObject<storage_manager_t>::extractCellEnvelope(const std::unordered_set<long> &mask, const VolumeKernel &mesh, std::unordered_map<long,long> &meshToEnvelope){
107
108 std::vector<long> list;
109
110 for( long cellIndex : mask){
111 const auto &cell = mesh.getCell(cellIndex);
112 const long *adjacencies = cell.getAdjacencies();
113 const long *interfaceIndex = cell.getInterfaces();
114
115 int interfaceCount= cell.getInterfaceCount() ;
116
117 for(int i=0; i<interfaceCount; i++){
118 long neigh = adjacencies[i];
119
120 if(mask.count(neigh)==0){
121 list.push_back(interfaceIndex[i]);
122 }
123 }
124 }
125
126 return extractFaceEnvelope(list,mesh,meshToEnvelope);
127}

137template<typename storage_manager_t>
138std::unique_ptr<SurfUnstructured> LevelSetMaskObject<storage_manager_t>::extractFaceEnvelope(const std::vector<long> &list, const VolumeKernel &mesh, std::unordered_map<long,long> &meshToEnvelope){
139
140 std::unique_ptr<SurfUnstructured> envelope;
141#if BITPIT_ENABLE_MPI==1
142 envelope = std::unique_ptr<SurfUnstructured>(new SurfUnstructured(mesh.getDimension()-1,mesh.getCommunicator()));
143#else
144 envelope = std::unique_ptr<SurfUnstructured>(new SurfUnstructured(mesh.getDimension()-1));
145#endif
146
147 // ====================================================================== //
148 // RESIZE DATA STRUCTURES //
149 // ====================================================================== //
150 long nVertices(0);
151 long nCells(list.size());
152
153 for( long faceIndex : list){
154 auto const &interface = mesh.getInterface(faceIndex);
155 nVertices += interface.getVertexCount() ;
156 }
157
158 envelope->reserveVertices(nVertices);
159 envelope->reserveCells(nCells);
160
161 // ====================================================================== //
162 // LOOP OVER CELLS //
163 // ====================================================================== //
164 std::unordered_map<long,long> vertexMap;
165
166 for( long faceIndex : list){
167 auto const &interface = mesh.getInterface(faceIndex);
168 ConstProxyVector<long> faceVertexIds = interface.getVertexIds();
169 int nFaceVertices = faceVertexIds.size();
170
171 // Add face vertices to the envelope and get face
172 // connectivity in the envelope
173 std::unique_ptr<long[]> faceEnvelopeConnect = std::unique_ptr<long[]>(new long[nFaceVertices]);
174 for (int j = 0; j < nFaceVertices; ++j) {
175 long vertexId = faceVertexIds[j];
176
177 // If the vertex is not yet in the envelope
178 // add it.
179 if (vertexMap.count(vertexId) == 0) {
180 const Vertex &vertex = mesh.getVertex(vertexId);
181 auto envelopeVertex = envelope->addVertex(vertex);
182 vertexMap[vertexId] = envelopeVertex->getId();
183 }
184
185 // Update face connectivity in the envelope
186 faceEnvelopeConnect[j] = vertexMap.at(vertexId);
187 }
188
189 // Add face to envelope
190 ElementType faceType = interface.getType();
191 PatchKernel::CellIterator cellItr = envelope->addCell(faceType, std::move(faceEnvelopeConnect));
192 meshToEnvelope.insert({{faceIndex,cellItr.getId()}});
193 }
194
195 envelope->squeeze();
196 envelope->initializeAdjacencies();
197 envelope->initializeInterfaces();
198
199 envelope->getVTK().setName("geometry_002") ;
200 envelope->write() ;
201
202 return envelope;
203
204}

214template<typename storage_manager_t>
215bool LevelSetMaskObject<storage_manager_t>::sameInterfaceEnvelopeOrientation(const VolumeKernel &mesh, long faceIndex, SurfUnstructured &envelope, long enveIndex){
216
217 std::array<double,3> facetNormal = envelope.evalFacetNormal(enveIndex);
218 std::array<double,3> interfaceNormal = mesh.evalInterfaceNormal(faceIndex);
219
220 return (dotProduct(facetNormal,interfaceNormal)>0) ;
221
222}
