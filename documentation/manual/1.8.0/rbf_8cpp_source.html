526bool RBFKernel::removeData(std::vector<int> & list)
527{
528 std::set<int> setList;
529 for(auto && id : list) setList.insert(id);
530
531 int extracted = 0;

533 if(id>=0 && id <m_fields){;
534 m_fields--;

536 assert(index >= 0);
537 if (m_mode == RBFMode::INTERP) {
538 assert((std::size_t) index < m_value.size());
539 auto valueItr = m_value.begin() + index;

573std::vector<double> RBFKernel::evalRBF( const std::array<double,3> &point)
574{
575 std::vector<double> values(m_fields, 0.);
576 int i, j;
577 double dist, basis;
578
579 for( i=0; i<m_nodes; ++i ){
580 if( m_activeNodes[i] ) {
581 dist = calcDist(point, i) / m_supportRadius;
582 basis = evalBasis( dist );
583
584 for( j=0; j<m_fields; ++j) {
585 values[j] += basis * m_weight[j][i];
586 }
587 }
588 }
589
590 return values;
591}

601std::vector<double> RBFKernel::evalRBF(int jnode)
602{
603 std::vector<double> values(m_fields, 0.);
604 int i, j;
605 double dist, basis;
606
607 if(jnode<0 || jnode>= m_nodes ) {
608 return values;
609 }
610
611 for( i=0; i<m_nodes; ++i ) {
612 if( m_activeNodes[i] ) {
613 dist = calcDist(jnode, i) / m_supportRadius;
614 basis = evalBasis( dist );
615
616 for( j=0; j<m_fields; ++j) {
617 values[j] += basis * m_weight[j][i];
618 }
619 }
620 }
621
622 return values;

632int RBFKernel::solve()
633{
634 if(m_mode == RBFMode::PARAM) {
635 return -1;
636 }
637
638 int j, k;
639 double dist;
640
641 int nS = getActiveCount();
642 int nrhs = getDataCount();
643
644 int lda = nS;
645 int ldb = nS;
646 int info;
647
648 std::vector<int> ipiv(nS);
649
650 std::vector<int> activeSet( getActiveSet() );
651
652 std::vector<double> A(lda * nS);
653 std::vector<double> b(ldb * nrhs);
654
655 k=0;
656 for( j=0; j<nrhs; ++j) {
657 for( const auto & i : activeSet ) {
658 b[k] = m_value[j][i];
659 ++k;
660 }
661 }
662
663 k=0;
664 for( const auto &i : activeSet ) {
665 for( const auto &j : activeSet ){
666
667 dist = calcDist(j,i) / m_supportRadius;
668 A[k] = evalBasis( dist );
669 k++;
670 }
671 }
672
673 info = LAPACKE_dgesv( LAPACK_COL_MAJOR, nS, nrhs, A.data(), lda, ipiv.data(), b.data(), ldb );
674
675 if( info > 0 ) {
676 printf( "The diagonal element of the triangular factor of the linear system matrix \n" );
677 printf( "U(%i,%i) is zero, so that matrix is singular;\n", info, info );
678 printf( "the solution could not be computed.\n" );
679 return 1;
680 }
681
682 m_weight.resize(nrhs);
683
684 k=0;
685 for( j=0; j<nrhs; ++j) {
686 m_weight[j].resize(m_nodes,0);
687
688 for( const auto &i : activeSet ) {
689 m_weight[j][i] = b[k];
690 ++k;
691 }
692 }
693
694 return 0;
695}

704int RBFKernel::greedy( double tolerance)
705{
706 if(m_mode == RBFMode::PARAM) return -1;
707
708 int i, j;
709 double error(1.e18);
710 std::vector<double> local(m_fields);
711
712 m_error.resize(m_nodes);
713
714 for( auto && active : m_activeNodes )
715 active = false;
716
717 for( i=0; i<m_nodes; ++i){
718
719 for( j=0; j<m_fields; ++j) {
720 local[j] = m_value[j][i];
721 }
722
723 m_error[i] = norm2(local);
724
725 }
726
727 std::ios::fmtflags streamFlags(log::cout().flags());
728
729 int errorFlag = 0;
730 while( error > tolerance) {
731 i = addGreedyPoint();
732
733 if( i != -1) {
734 m_activeNodes[i] = true;
735
736 //solve();
737 solveLSQ();
738
739 error = evalError();
740
741 log::cout() << std::scientific;
742 log::cout() << " error now " << error << " active nodes" << getActiveCount() << " / " << m_nodes << std::endl;
743 } else {
744 errorFlag = 1;
745 break;
746
747 }
748 }
749
750 log::cout().flags(streamFlags);
751
752 return errorFlag;
753}

804int RBFKernel::addGreedyPoint( )
805{
806 if(m_mode == RBFMode::PARAM) {
807 return -1;
808 }
809
810 int i(0), index(-1);
811 double maxError(0.);
812
813 std::vector<int> active( getActiveSet() );
814
815 for( auto error : m_error ){
816
817 if(!m_activeNodes[i] ){
818
819 if( error > maxError ) {
820 maxError = error;
821 index = i;
822 }
823 }
824
825 ++i;
826
827 }
828
829 return index;

837double RBFKernel::evalError( )
838{
839 if(m_mode == RBFMode::PARAM) {
840 return -1.0;
841 }
842
843 int i(0), j(0);
844 //int index;
845 double maxError(0), relError, realValue;
846 std::vector<double> reconValues;
847
848 for( int iX=0; iX< m_nodes; ++iX ) {
849 reconValues = evalRBF( iX );
850
851 j=0;
852 relError = 0.;
853 for( auto &val : reconValues ) {
854 realValue = m_value[j][i];
855 relError += std::pow( (val - realValue), 2 );
856
857 ++j;
858 }
859
860 relError = sqrt(relError);
861 m_error[i] = relError;
862
863 if( relError > maxError ) {
864 maxError = relError;
865 }
866
867 ++i;
868 }
869
870 return maxError;
871}

879int RBFKernel::solveLSQ()
880{
881 if(m_mode == RBFMode::PARAM) {
882 return -1;
883 }
884
885 int i, j, k;
886 double dist;
887
888 int nR = getActiveCount();
889 int nP = m_nodes;
890 int nrhs = getDataCount();
891
892 std::vector<int> activeSet( getActiveSet() );
893
894 int n ,m, lda, ldb, info, rank;
895 double rcond = -1.0;
896 //double rcond = 1.e-4;
897
898 m = nP;
899 n = nR;
900
901 lda = m;
902 ldb = std::max(n,m);
903
904 std::vector<double> A(lda * n);
905 std::vector<double> b(ldb * nrhs);
906 std::vector<double> s(m);
907
908 for( j=0; j<nrhs; ++j) {
909 for( i=0; i<nP; ++i) {
910 k = j*ldb + i;
911 b[k] = m_value[j][i];
912 }
913 }
914
915 k=0;
916 for( const auto &j : activeSet ) {
917 for( i=0; i<nP; ++i) {
918 dist = calcDist(j,i) / m_supportRadius;
919 A[k] = evalBasis( dist );
920 k++;
921 }
922 }
923
924 info = LAPACKE_dgelsd( LAPACK_COL_MAJOR, nP, nR, nrhs, A.data(), lda, b.data(), ldb, s.data(), rcond, &rank );
925
926 if( info > 0 ) {
927 return 1;
928 }
929
930 m_weight.resize(nrhs);
931
932 for( j=0; j<nrhs; ++j) {
933 m_weight[j].clear();
934 m_weight[j].resize(m_nodes,0);
935
936 k=0;
937 for( const auto &i : activeSet ) {
938 m_weight[j][i] = b[j*ldb+k];
939 ++k;
940 }
941 }
942
943 return 0;
944}
