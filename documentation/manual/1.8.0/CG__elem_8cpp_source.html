64void _projectPointsTriangle( int nPoints, array3D const *points, array3D const &Q0, array3D const &Q1, array3D const &Q2, array3D *proj, double *lambda )
65{
66 assert( validTriangle(Q0,Q1,Q2) );
67
68 array3D v0 = Q1 - Q0;
69 array3D v1 = Q2 - Q0;
70
71 double d00 = dotProduct(v0, v0);
72 double d01 = dotProduct(v0, v1);
73 double d11 = dotProduct(v1, v1);
74
75 double denom = d00 * d11 - d01 * d01;
76
77 for( int i=0; i<nPoints; ++i){
78 array3D v2 = points[i] - Q0;
79
80 double d20 = dotProduct(v2, v0);
81 double d21 = dotProduct(v2, v1);
82
83 double *pointLambda = lambda + 3 * i;
84 pointLambda[1] = (d11 * d20 - d01 * d21) / denom;
85 pointLambda[2] = (d00 * d21 - d01 * d20) / denom;
86 pointLambda[0] = 1. - pointLambda[1] - pointLambda[2];
87
88 array3D *pointProjections = proj + i;
89 *pointProjections = restrictPointTriangle( Q0, Q1, Q2, pointLambda);
90 }
91}

110bool _intersectBoxTriangle(array3D const &A0, array3D const &A1, array3D const &V0, array3D const &V1, array3D const &V2, bool interiorTriangleVertices, bool triangleEdgeBoxHullIntersections, bool triangleBoxEdgeIntersections, std::vector<array3D> *intrPtr, std::vector<int> *flagPtr, int dim, const double distanceTolerance)
111{
112
113 bool intersect(false);
114 bool addFlag( flagPtr!=nullptr);
115 bool computeIntersection(interiorTriangleVertices||triangleBoxEdgeIntersections||triangleEdgeBoxHullIntersections);
116
117 assert( ! (computeIntersection && (intrPtr==nullptr) ) );
118
119 if(computeIntersection){
120 intrPtr->clear();
121 }
122
123 if(addFlag){
124 flagPtr->clear();
125 }
126
127 //check if Triangle Boundig Box and Box overlap -> necessary condition
128 array3D B0, B1;
129 computeAABBTriangle( V0, V1, V2, B0, B1);
130
131 if( !intersectBoxBox( A0, A1, B0, B1, dim, distanceTolerance) ) {
132 return false;
133 }
134
135 //check if triangle vertices are within the box
136 for( int i=0; i<3; ++i){
137 vertexOfTriangle(i, V0, V1, V2, B0);
138 if( intersectPointBox(B0, A0, A1, dim, distanceTolerance) ){
139 intersect = true;
140 if(!interiorTriangleVertices) break;
141
142 intrPtr->push_back(B0);
143 if(addFlag) flagPtr->push_back(0);
144 }
145 }
146
147 //check if triangle edges and box faces intersect
148 if( !intersect || triangleEdgeBoxHullIntersections) {
149
150 for( int edge=0; edge<3; ++edge){
151
152 edgeOfTriangle(edge, V0, V1, V2, B0, B1);
153
154 if(dim==2){
155 array3D p;
156 array3D faceVertex0, faceVertex1;
157
158 for( int face=0; face<4; ++face){
159 edgeOfBox( face, A0, A1, faceVertex0, faceVertex1);
160
161 if( intersectSegmentSegment(B0, B1, faceVertex0, faceVertex1, p, distanceTolerance) ){
162 intersect=true;
163 if(!triangleEdgeBoxHullIntersections) break;
164
165 intrPtr->push_back(p);
166 if(addFlag) flagPtr->push_back(1);
167 }
168
169 }
170
171 } else if(dim==3){
172 array3D p;
173 std::array<array3D, 4> V;
174
175 for( int face=0; face<6; ++face){
176 faceOfBox( face, A0, A1, V[0], V[1], V[2], V[3] );
177
178 if( intersectSegmentPolygon( B0, B1, V.size(), V.data(), p, distanceTolerance ) ) {
179 intersect=true;
180 if(!triangleEdgeBoxHullIntersections) break;
181
182 intrPtr->push_back(p);
183 if(addFlag) flagPtr->push_back(1);
184 }
185 }
186 }
187 }
188 }
189
190 //check if triangle and box edges (dim=3) or box vertices (dim=2) intersect
191 if( !intersect || triangleBoxEdgeIntersections ) {
192
193 array3D p;
194
195 if(dim==2){
196 for( int i=0; i<4; ++i){
197 vertexOfBox( i, A0, A1, B0);
198 if( intersectPointTriangle(B0,V0,V1,V2,distanceTolerance)) {
199 intersect = true;
200 if(!triangleBoxEdgeIntersections) break;
201
202 intrPtr->push_back(B0);
203 if(addFlag) flagPtr->push_back(2);
204 }
205 }
206
207 } else if(dim==3){
208 for( int i=0; i<12; ++i){
209 edgeOfBox( i, A0, A1, B0, B1);
210 if( intersectSegmentTriangle(B0,B1,V0,V1,V2,p,distanceTolerance)) {
211 intersect = true;
212 if(!triangleBoxEdgeIntersections) break;
213
214 intrPtr->push_back(p);
215 if(addFlag) flagPtr->push_back(2);
216 }
217 }
218 }
219 }
220
221 return intersect;
222}

330bool _intersectPlaneBox(array3D const &P, array3D const &N, array3D const &A0, array3D const &A1, std::vector<array3D> *intrPtr, int dim, const double distanceTolerance)
331{
332
333 bool intersect(false);
334 bool computeIntersection(intrPtr);
335
336 if(computeIntersection){
337 intrPtr->clear();
338 } else if( intersectPointBox(P,A0,A1,dim,distanceTolerance)) {
339 return true;
340 }
341
342 // check if plane intersects the edges of the box
343 // and store eventually intersection points
344 int edgeCount = (dim==2) ? 4 : 12;
345 array3D E0, E1, V;
346
347 for(int i=0; i<edgeCount; ++i){
348 edgeOfBox(i, A0, A1, E0, E1);
349
350 if( intersectSegmentPlane( E0, E1, P, N, V, distanceTolerance) ){
351 intersect = true;
352
353 if(intrPtr){
354 intrPtr->push_back(V);
355 } else {
356 break;
357 }
358 }
359 }
360
361
362 // sort intersection points in couterclock-wise sense
363 if( dim==3 && intrPtr && intersect){
364
365 const array3D origin = intrPtr->at(0);
366
367 std::sort(intrPtr->begin(), intrPtr->end(), [&](const array3D &lhs, const array3D &rhs) -> bool {
368 array3D v = crossProduct(lhs-origin,rhs-origin);
369 return dotProduct(v, N) < 0;
370 } );
371 }
372
373 return intersect;
374}

392bool _intersectBoxPolygon(array3D const &A0, array3D const &A1, std::size_t nVS, array3D const *VS, bool innerPolygonPoints, bool polygonEdgeBoxHullIntersection, bool polygonBoxEdgeIntersections, std::vector<array3D> *intrPtr, std::vector<int> *flagPtr, int dim, const double distanceTolerance)
393{
394
395 bool intersect(false);
396 bool addFlag(flagPtr!=nullptr);
397 bool computeIntersection(innerPolygonPoints || polygonEdgeBoxHullIntersection || polygonBoxEdgeIntersections);
398
399 assert( ! (computeIntersection && (intrPtr==nullptr) ) );
400
401 if(computeIntersection){
402 intrPtr->clear();
403 }
404
405 if(addFlag){
406 flagPtr->clear();
407 }
408
409 array3D V0, V1, V2;
410
411 //check if simplex boundig box and box overlap -> necessary condition
412 computeAABBPolygon( nVS, VS, V0, V1);
413 if( !intersectBoxBox( A0, A1, V0, V1, dim, distanceTolerance) ) {
414 return false;
415 }
416
417 std::vector<array3D> partialIntr;
418 std::vector<int> partialFlag;
419
420 // check if triangle vertices lie within box
421 // or if triangles intersect edges of box
422 computeIntersection = innerPolygonPoints || polygonBoxEdgeIntersections;
423 int trianglesCount = polygonSubtriangleCount(nVS, VS);
424 for (int triangle=0; triangle<trianglesCount; ++triangle) {
425 subtriangleOfPolygon(triangle, nVS, VS, V0, V1, V2);
426
427 if( _intersectBoxTriangle( A0, A1, V0, V1, V2, innerPolygonPoints, false, polygonBoxEdgeIntersections, &partialIntr, &partialFlag, dim, distanceTolerance ) ){
428
429 intersect = true;
430 if(!computeIntersection) break;
431
432 int intrCount = partialIntr.size();
433 for( int i=0; i<intrCount; ++i){
434
435 array3D &candidateCoord = partialIntr[i];
436 int candidateFlag = partialFlag[i];
437
438 //prune duplicate points
439 auto PItr = intrPtr->begin();
440 bool iterate = (PItr!=intrPtr->end());
441 while(iterate){
442
443 iterate = !utils::DoubleFloatingEqual()( norm2( *PItr -candidateCoord ), 0., distanceTolerance );
444
445 if(iterate){
446 ++PItr;
447 }
448 iterate &= PItr!=intrPtr->end();
449 }
450
451 if(PItr!=intrPtr->end()){
452 continue;
453 }
454
455 intrPtr->push_back(candidateCoord);
456 if(addFlag){
457 flagPtr->push_back(candidateFlag);
458 }
459 }
460 }
461 }
462
463 // check if edges of polygon intersect box face
464 computeIntersection = polygonEdgeBoxHullIntersection;
465 int edgesCount = polygonEdgesCount(nVS, VS);
466 if(!intersect || polygonEdgeBoxHullIntersection){
467 for (int edge=0; edge<edgesCount; ++edge) {
468 edgeOfPolygon(edge, nVS, VS, V0, V1);
469
470 if( _intersectSegmentBox( V0, V1, A0, A1, false, polygonEdgeBoxHullIntersection, &partialIntr, &partialFlag, dim, distanceTolerance) ){
471
472 intersect = true;
473 if(!computeIntersection) break;
474
475 int intrCount = partialIntr.size();
476 for( int i=0; i<intrCount; ++i){
477
478 array3D &candidateCoord = partialIntr[i];
479 int candidateFlag = partialFlag[i];
480
481 //prune duplicate points
482 auto PItr = intrPtr->begin();
483 bool iterate = (PItr!=intrPtr->end());
484 while(iterate){
485
486 iterate = !utils::DoubleFloatingEqual()( norm2( *PItr -candidateCoord ), 0., distanceTolerance );
487
488 if(iterate){
489 ++PItr;
490 }
491 iterate &= PItr!=intrPtr->end();
492 }
493
494 if(PItr!=intrPtr->end()){
495 continue;
496 }
497
498 intrPtr->push_back(candidateCoord);
499 if(addFlag){
500 flagPtr->push_back(candidateFlag);
501 }
502 }
503 }
504 }
505 }
506
507 return intersect;
508}

552bool validTriangle(const array3D &P0, const array3D &P1, const array3D &P2 )
553{
554
555 array3D v0 = P1 -P0;
556 if( utils::DoubleFloatingEqual()( norm2(v0), 0.) ){
557 return false;
558 }
559
560 array3D v1 = P2 -P1;
561 if( utils::DoubleFloatingEqual()( norm2(v1), 0.) ){
562 return false;
563 }
564
565 array3D v2 = P0 -P2;
566 if( utils::DoubleFloatingEqual()( norm2(v2), 0.) ){
567 return false;
568 }
569
570 array3D n = crossProduct( v0, -1.*v2);
571 if( utils::DoubleFloatingEqual()( norm2(n), 0. ) ){
572 return false;
573 }
574
575 return true;
576}

584bool validBarycentric(double const *lambdaPtr, int n )
585{
586
587 double sum(-1.);
588 double maxValue(0.);
589
590 for(int i=0; i<n; ++i){
591
592 double value = lambdaPtr[i];
593 if( std::abs(value) > std::abs(maxValue) ){
594 sum -= maxValue;
595 maxValue = -value;
596
597 } else {
598 sum += value;
599
600 }
601
602 }
603
604 double tolerance = std::max(1., std::abs(maxValue)) * std::numeric_limits<double>::epsilon();
605 return utils::DoubleFloatingEqual()(sum, maxValue, tolerance);
606}

700int convertBarycentricToFlagPolygon( std::size_t nLambda, double const *lambda, double tolerance)
701{
702
703 assert( validBarycentric(&lambda[0],nLambda) );
704
705 int count = 0;
706 std::size_t lastPositive = std::numeric_limits<std::size_t>::max();
707 for( std::size_t i=0; i<nLambda; ++i){
708 if ( lambda[i] < 0. || utils::DoubleFloatingEqual()( lambda[i], 0., tolerance ) ) {
709 continue;
710 }
711
712 ++count;
713 if (count > 2) {
714 return 0;
715 }
716
717 if (lastPositive != 0 || i == 1) {
718 lastPositive = i;
719 }
720 }
721
722 if( count == 1){
723 count = lastPositive + 1;
724
725 } else if (count==2) {
726 if (lastPositive != 0) {
727 count = - lastPositive;
728 } else {
729 count = - nLambda;
730 }
731
732 }
733
734 return count;
735}

778void computeGeneralizedBarycentric( array3D const &p, std::size_t nVertices, array3D const *vertex, double *lambda)
779{
780 const int MAX_ELEM_VERTICES = 8;
781 BITPIT_CREATE_WORKSPACE(area, double, nVertices, MAX_ELEM_VERTICES);
782
783 for( std::size_t i=0; i<nVertices; ++i){
784 int next = (i +1) %nVertices;
785 area[i] = areaTriangle( vertex[i], vertex[next], p);
786 }
787
788 double sumWeight(0);
789
790 for( std::size_t i=0; i<nVertices; ++i){
791 std::size_t prev = (i +nVertices -1) %nVertices;
792 std::size_t next = (i +1) %nVertices;
793 lambda[i] = areaTriangle(vertex[prev], vertex[i], vertex[next]);
794
795 for( std::size_t j=0; j<nVertices; ++j){
796 if( j==prev || j==i){
797 continue;
798 }
799
800 lambda[i] *= area[j];
801 }
802
803 sumWeight += lambda[i];
804 }
805
806 for( std::size_t i=0; i<nVertices; ++i){
807 lambda[i] /= sumWeight;
808 }
809}

973{
974
975 assert( validSegment(Q0,Q1) );
976
977 array3D n = Q1 -Q0;
978 double t = -dotProduct(n,Q0-P) / dotProduct(n,n);
979
980 // Restrict projection onto the segment
981 t = std::max( std::min( t, 1.), 0. );
982
983 lambda[0] = 1. - t;
984 lambda[1] = t;
985
986 return reconstructPointFromBarycentricSegment( Q0, Q1, lambda);
987}
