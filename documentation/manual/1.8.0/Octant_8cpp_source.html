43bitpit::IBinaryStream& operator>>(bitpit::IBinaryStream &buffer, bitpit::Octant &octant)
44{
45 uint8_t dimensions;
46 buffer >> dimensions;
47
48 uint8_t level;
49 buffer >> level;
50
51 octant.initialize(dimensions, level, true);
52
53 buffer >> octant.m_morton;
54
55 buffer >> octant.m_marker;
56
57 buffer >> octant.m_ghost;
58
59 for(int i = 0; i < bitpit::Octant::INFO_ITEM_COUNT; ++i){
60 bool value;
61 buffer >> value;
62 octant.m_info[i] = value;
63 }
64
65 return buffer;
66}

202void
203Octant::initialize(uint8_t dim, uint8_t level, bool bound) {
204 m_dim = dim;
205 m_level = level;
206
207 // Reset the marker
208 m_marker = 0;
209
210 // Reset the morton
211 m_morton = 0;
212
213 // Initialize octant info
214 m_info.reset();
215 m_info[OctantInfo::INFO_BALANCED] = true;
216 m_ghost = -1;
217
218 // If this is the root octant we need to set the boundary condition bound
219 // for faces
220 if (m_dim >= 2 && m_level == 0) {
221 uint8_t nf = m_dim*2;
222 for (uint8_t i=0; i<nf; i++){
223 m_info[i] = bound;
224 }
225 }
226};

756void Octant::buildChildren(Octant *children) const {
757
758 u32array3 coords = getLogicalCoordinates();
759
760 int nChildren = countChildren();
761 for (int i=0; i<nChildren; ++i){
762 // Octant information
763 uint8_t xf;
764 uint8_t yf;
765 uint8_t zf;
766 uint8_t dx;
767 uint8_t dy;
768 uint8_t dz;
769 switch (i) {
770
771 case 0 :
772 dx = 0;
773 dy = 0;
774 dz = 0;
775
776 xf = 1;
777 yf = 3;
778 zf = 5;
779
780 break;
781
782 case 1 :
783 dx = 1;
784 dy = 0;
785 dz = 0;
786
787 xf = 0;
788 yf = 3;
789 zf = 5;
790
791 break;
792
793 case 2 :
794 dx = 0;
795 dy = 1;
796 dz = 0;
797
798 xf = 1;
799 yf = 2;
800 zf = 5;
801
802 break;
803
804 case 3 :
805 dx = 1;
806 dy = 1;
807 dz = 0;
808
809 xf = 0;
810 yf = 2;
811 zf = 5;
812
813 break;
814
815 case 4 :
816 dx = 0;
817 dy = 0;
818 dz = 1;
819
820 xf = 1;
821 yf = 3;
822 zf = 4;
823
824 break;
825
826 case 5 :
827 dx = 1;
828 dy = 0;
829 dz = 1;
830
831 xf = 0;
832 yf = 3;
833 zf = 4;
834
835 break;
836
837 case 6 :
838 dx = 0;
839 dy = 1;
840 dz = 1;
841
842 xf = 1;
843 yf = 2;
844 zf = 4;
845
846 break;
847
848 case 7 :
849 dx = 1;
850 dy = 1;
851 dz = 1;
852
853 xf = 0;
854 yf = 2;
855 zf = 4;
856
857 break;
858
859 default:
860 BITPIT_UNREACHABLE("The maximum number of children is 8.");
861
862 }
863
864 // Create octant
865 children[i] = Octant(*this);
866 Octant &oct = children[i];
867
868 oct.setMarker(std::max(0, oct.m_marker - 1));
869 oct.setLevel(oct.m_level + 1);
870
871 uint32_t dh = oct.getLogicalSize();
872 oct.m_morton = PABLO::computeMorton(m_dim, coords[0] + dh * dx, coords[1] + dh * dy, coords[2] + dh * dz);
873
874 oct.m_info[OctantInfo::INFO_NEW4REFINEMENT] = true;
875
876 oct.m_info[INFO_BOUNDFACE0 + xf] = false;
877 oct.m_info[INFO_BOUNDFACE0 + yf] = false;
878 oct.m_info[INFO_BOUNDFACE0 + zf] = false;
879
880 oct.m_info[INFO_PBOUNDFACE0 + xf] = false;
881 oct.m_info[INFO_PBOUNDFACE0 + yf] = false;
882 oct.m_info[INFO_PBOUNDFACE0 + zf] = false;
883 }

893void Octant::computeHalfSizeMortons(uint8_t iface, uint32_t *nMortons, std::vector<uint64_t> *mortons) const {
894
895 if (m_info[iface]) {
896 *nMortons = 0;
897 mortons->clear();
898 return;
899 }
900
901 int nchildren = 1<<m_dim;
902 *nMortons = (m_level < sm_treeConstants[m_dim].maxLevel) ? nchildren/2 : 1;
903 mortons->resize(*nMortons);
904
905 u32array3 coords = getLogicalCoordinates();
906
907 uint32_t dh = (m_level < sm_treeConstants[m_dim].maxLevel) ? getLogicalSize()/2 : getLogicalSize();
908 uint32_t dh2 = getLogicalSize();
909 switch (iface) {
910 case 0 :
911 {
912 uint32_t x = coords[0] - dh;
913 for (uint32_t i = 0; i < *nMortons; ++i) {
914 uint32_t y = coords[1] + ((i == 1) || (i == 3)) * dh;
915 uint32_t z = coords[2] + ((m_dim == 3) && ((i == 2) || (i == 3))) * dh;
916
917 (*mortons)[i] = PABLO::computeMorton(m_dim, x, y, z);
918 }
919 }
920 break;
921 case 1 :
922 {
923 uint32_t x = coords[0] + dh2;
924 for (uint32_t i = 0; i < *nMortons; ++i) {
925 uint32_t y = coords[1] + ((i == 1) || (i == 3)) * dh;
926 uint32_t z = coords[2] + ((m_dim == 3) && ((i == 2) || (i == 3))) * dh;
927
928 (*mortons)[i] = PABLO::computeMorton(m_dim, x, y, z);
929 }
930 }
931 break;
932 case 2 :
933 {
934 uint32_t y = coords[1] - dh;
935 for (uint32_t i = 0; i < *nMortons; ++i) {
936 uint32_t x = coords[0] + ((i == 1) || (i == 3)) * dh;
937 uint32_t z = coords[2] + ((m_dim == 3) && ((i == 2) || (i == 3))) * dh;
938
939 (*mortons)[i] = PABLO::computeMorton(m_dim, x, y, z);
940 }
941 }
942 break;
943 case 3 :
944 {
945 uint32_t y = coords[1] + dh2;
946 for (uint32_t i = 0; i < *nMortons; ++i) {
947 uint32_t x = coords[0] + ((i == 1) || (i == 3)) * dh;
948 uint32_t z = coords[2] + ((m_dim == 3) && ((i == 2) || (i == 3))) * dh;
949
950 (*mortons)[i] = PABLO::computeMorton(m_dim, x, y, z);
951 }
952 }
953 break;
954 case 4 :
955 {
956 uint32_t z = coords[2] - dh;
957 for (uint32_t i = 0; i < *nMortons; ++i) {
958 uint32_t x = coords[0] + ((i == 1) || (i == 3)) * dh;
959 uint32_t y = coords[1] + ((i == 2) || (i == 3)) * dh;
960
961 (*mortons)[i] = PABLO::computeMorton(m_dim, x, y, z);
962 }
963 }
964 break;
965 case 5 :
966 {
967 uint32_t z = coords[2] + dh2;
968 for (uint32_t i = 0; i < *nMortons; ++i) {
969 uint32_t x = coords[0] + ((i == 1) || (i == 3)) * dh;
970 uint32_t y = coords[1] + ((i == 2) || (i == 3)) * dh;
971
972 (*mortons)[i] = PABLO::computeMorton(m_dim, x, y, z);
973 }
974 }
975 break;
976 }
977}

987void Octant::computeMinSizeMortons(uint8_t iface, uint8_t maxdepth, uint32_t *nMortons, std::vector<uint64_t> *mortons) const {
988
989 if (m_info[iface]) {
990 *nMortons = 0;
991 mortons->clear();
992 return;
993 }
994
995 *nMortons = (m_level < sm_treeConstants[m_dim].maxLevel) ? uint32_t(1)<<((m_dim-1)*(maxdepth-m_level)) : 1;
996 mortons->resize(*nMortons);
997
998 u32array3 coords = getLogicalCoordinates();
999
1000 uint32_t dh = (m_level < sm_treeConstants[m_dim].maxLevel) ? uint32_t(1)<<(sm_treeConstants[m_dim].maxLevel - maxdepth) : getLogicalSize();
1001 uint32_t dh2 = getLogicalSize();
1002 uint32_t nline = uint32_t(1)<<(maxdepth-m_level);
1003 switch (iface) {
1004 case 0 :
1005 {
1006 uint32_t x = coords[0] - dh;
1007 for (uint32_t i = 0; i < *nMortons; ++i) {
1008 uint32_t y = coords[1] + ((m_dim == 2) * (i % nline) + (m_dim - 2) * (i / nline)) * dh;
1009 uint32_t z = coords[2] + (m_dim - 2) * (i % nline) * dh;
1010
1011 (*mortons)[i] = PABLO::computeMorton(m_dim, x, y, z);
1012 }
1013 }
1014 break;
