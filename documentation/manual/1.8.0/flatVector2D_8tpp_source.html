226template <class T>
227void FlatVector2D<T>::initialize(std::size_t nVectors,
228 const std::size_t *sizes, std::size_t sizesStride,
229 const T *values, std::size_t valuesStride)
230{
231 std::size_t nItems = 0;
232 for (std::size_t i = 0; i < nVectors; ++i) {
233 nItems += *(sizes + i * sizesStride);
234 }
235
236 // Check if the container need reallocation
237 //
238 // If the current number of items is different from the number of items
239 // the container should contain after the initialization, a reallocation
240 // is needed.
241 bool reallocateIndex = (nVectors != size());
242 bool reallocateValues = (nItems != getItemCount());
243
244 // Destroy current data structures
245 //
246 // Index and the storage data will probabily be stored in memory
247 // regions contigous to each other. To reduce memory fragmentation
248 // it's better to deallocate the container before updating its
249 // data structures.
250 if (reallocateIndex || reallocateValues) {
251 destroy(reallocateIndex, reallocateValues);
252 }
253
254 // Initialize the indexes
255 if (reallocateIndex) {
256 m_index.resize(nVectors + 1);
257 }
258
259 m_index[0] = 0;
260 for (std::size_t i = 0; i < nVectors; ++i) {
261 m_index[i+1] = m_index[i] + *(sizes + i * sizesStride);
262 }
263
264 // Initialize the storage
265 if (valuesStride == 0) {
266 if (reallocateValues) {
267 m_v.assign(nItems, *values);
268 } else {
269 for (std::size_t k = 0; k < nItems; ++k) {
270 m_v[k] = *values;
271 }
272 }
273 } else {
274 if (reallocateValues) {
275 m_v.resize(nItems);
276 }
277
278 std::size_t k = 0;
279 for (std::size_t i = 0; i < nVectors; ++i) {
280 std::size_t subArraySize = *(sizes + i * sizesStride);
281 for (std::size_t j = 0; j < subArraySize; ++j) {
282 m_v[k++] = *(values + i * valuesStride + j);
283 }
284 }
285 }
286}

294template <class T>
295void FlatVector2D<T>::initialize(const std::vector<std::vector<T> > &vector2D)
296{
297 std::size_t nVectors = vector2D.size();
298
299 std::size_t nItems = 0;
300 for (std::size_t i = 0; i < nVectors; ++i) {
301 nItems += vector2D[i].size();
302 }
303
304 // Check if the container need reallocation
305 //
306 // If the current number of items is different from the number of items
307 // the container should contain after the initialization, a reallocation
308 // is needed.
309 bool reallocateIndex = (nVectors != size());
310 bool reallocateValues = (nItems != getItemCount());
311
312 // Destroy current data structures
313 //
314 // Index and the storage data will probabily be stored in memory
315 // regions contigous to each other. To reduce memory fragmentation
316 // it's better to deallocate the container before updating its
317 // data structures.
318 if (reallocateIndex || reallocateValues) {
319 destroy(reallocateIndex, reallocateValues);
320 }
321
322 // Initialize the indexes
323 if (reallocateIndex) {
324 m_index.resize(nVectors + 1);
325 }
326
327 m_index[0] = 0;
328 for (std::size_t i = 0; i < nVectors; ++i) {
329 m_index[i+1] = m_index[i] + vector2D[i].size();
330 }
331
332 // Initialize the storage
333 if (reallocateValues) {
334 m_v.resize(nItems);
335 }
336
337 std::size_t k = 0;
338 for (std::size_t i = 0; i < nVectors; ++i) {
339 std::size_t subArraySize = vector2D[i].size();
340 for (std::size_t j = 0; j < subArraySize; ++j) {
341 m_v[k++] = vector2D[i][j];
342 }
343 }

650{
651 std::size_t previousLastIndex = m_index.back();
652 m_index.emplace_back(previousLastIndex + subArraySize);
653
654 m_v.reserve(m_v.size() + subArraySize);
655 for (std::size_t j = 0; j < subArraySize; j++) {
656 m_v.emplace_back();
657 T &storedValue = m_v.back();
658 storedValue = subArray[j];
659 }
