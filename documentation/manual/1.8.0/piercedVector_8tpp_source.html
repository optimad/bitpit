800template<typename value_t, typename id_t>
801void PiercedVector<value_t, id_t>::swap(PiercedVector &other) noexcept
802{
803 // The swap will swap also the slave-master information. This is not what
804 // we want, therefore the two pierced storage will be unregistered and the
805 // registered again after the swap. When the kernel is unset the storage
806 // can't be clear, otherwise its contents will be lost.
807 PiercedVectorStorage<value_t, id_t>::detachKernel();
808 other.PiercedVectorStorage<value_t, id_t>::detachKernel();
809
810 // Swap kernel data
811 PiercedVectorKernel<id_t>::swap(other);
812
813 // Swap storage data
814 PiercedVectorStorage<value_t, id_t>::swap(other);
815
816 // Re-register the storages
817 //
818 // The function that sets a dynamic kernel may throw an exception if the
819 // kernel is already set or if we are trying to set a null kernel. Here
820 // neither of the two cases can happen, because the kernel has been
821 // previously cleared and the kernel we are trying to set is not null.
822 try {
823 PiercedVectorStorage<value_t, id_t>::setDynamicKernel(this, PiercedVectorKernel<id_t>::SYNC_MODE_DISABLED);
824 other.PiercedVectorStorage<value_t, id_t>::setDynamicKernel(&other, PiercedVectorKernel<id_t>::SYNC_MODE_DISABLED);
825 } catch (const std::runtime_error &exception) {
826 assert(false && "Error while swapping the PiercedVector!");
827 std::cout << "Error while swapping the PiercedVector!" << std::endl;
828 exit(0);
829 }
830}

920template<typename value_t, typename id_t>
921typename PiercedVector<value_t, id_t>::iterator PiercedVector<value_t, id_t>::reclaimValue(const FillAction &action)
922{
923 std::size_t pos = action.info[PiercedSyncAction::INFO_POS];
924 switch (static_cast<typename FillAction::FillActionType>(action.type)) {
925
926 case FillAction::TYPE_OVERWRITE:
927 {
928 // Nothing to do
929 break;
930 }
931
932 case FillAction::TYPE_INSERT:
933 {
934 // Since we are increasing the sotrage by an element at the time
935 // calling a reserve will hurt performance badly because this will
936 // prevent the automatic reallocation of the storage.
937 PiercedVectorStorage<value_t, id_t>::rawEmplace(pos);
938 break;
939 }
940
941 case FillAction::TYPE_APPEND:
942 {
943 // Since we are increasing the sotrage by an element at the time
944 // calling a reserve will hurt performance badly because this will
945 // prevent the automatic reallocation of the storage.
946 PiercedVectorStorage<value_t, id_t>::rawEmplaceBack();
947 pos = PiercedVectorKernel<id_t>::getLastUsedPos();
948 break;
949 }
950
951 default:
952 {
953 BITPIT_UNREACHABLE("This action is not handled");
954 break;
955 }
956
957 }
958
959 // Return the iterator to the position where the element was inserted
960 return PiercedVectorStorage<value_t, id_t>::rawFind(pos);
961}

969template<typename value_t, typename id_t>
970typename PiercedVector<value_t, id_t>::iterator PiercedVector<value_t, id_t>::insertValue(const FillAction &action, const value_t &value)
971{
972 std::size_t pos = action.info[PiercedSyncAction::INFO_POS];
973 switch (static_cast<typename FillAction::FillActionType>(action.type)) {
974
975 case FillAction::TYPE_OVERWRITE:
976 {
977 PiercedVectorStorage<value_t, id_t>::rawSet(pos, value);
978 break;
979 }
980
981 case FillAction::TYPE_INSERT:
982 {
983 // Since we are increasing the sotrage by an element at the time
984 // calling a reserve will hurt performance badly because this will
985 // prevent the automatic reallocation of the storage.
986 PiercedVectorStorage<value_t, id_t>::rawInsert(pos, 1, value);
987 break;
988 }
989
990 case FillAction::TYPE_APPEND:
991 {
992 // Since we are increasing the sotrage by an element at the time
993 // calling a reserve will hurt performance badly because this will
994 // prevent the automatic reallocation of the storage.
995 PiercedVectorStorage<value_t, id_t>::rawPushBack(value);
996 pos = PiercedVectorKernel<id_t>::getLastUsedPos();
997 break;
998 }
999
1000 default:
1001 {
1002 BITPIT_UNREACHABLE("This action is not handled");
1003 break;
1004 }
1005
1006 }
1007
1008 // Return the iterator to the position where the element was inserted
1009 return PiercedVectorStorage<value_t, id_t>::rawFind(pos);
1010}
