50// Auxiliary class to write fields in VTK format.
51class FieldStreamer : public VTKBaseStreamer {
52
53public:
54 FieldStreamer(const PatchKernel &patch, const PiercedStorage<double, long> &scalarField, const PiercedStorage<std::array<double, 3>, long> &vectorField)
55 : m_patch(patch), m_scalarField(scalarField), m_vectorField(vectorField)
56 {
57 };
58
59 void flushData(std::fstream &stream, const std::string &name, VTKFormat format)
60 {
61 assert(format == VTKFormat::APPENDED);
62 BITPIT_UNUSED(format);
63
64 if (name == "scalarField") {
65 for (const Cell &cell : m_patch.getVTKCellWriteRange()) {
66 long id = cell.getId();
67 genericIO::flushBINARY(stream, m_scalarField.at(id));
68 }
69 } else if (name == "vectorField") {
70 for (const Cell &cell : m_patch.getVTKCellWriteRange()) {
71 long id = cell.getId();
72 genericIO::flushBINARY(stream, m_vectorField.at(id));
73 }
74 }
75 };
76
77private:
78 const PatchKernel &m_patch;
79 const PiercedStorage<double, long> &m_scalarField;
80 const PiercedStorage<std::array<double, 3>, long> &m_vectorField;
81
82};
83
84// Main program
85int main(int argc, char *argv[])
86{
87 BITPIT_UNUSED(argc);
88 BITPIT_UNUSED(argv);
89
90
91#if BITPIT_ENABLE_MPI == 1
92 // Initialize MPI
93 MPI_Init(&argc, &argv);
94#endif
95
96 //
97 // Domain initialization
98 //
99
100 // Initializing mesh
101 log::cout() << " Initializing mesh..." << std::endl;
102
103 double length = 1.0;
104 std::array<double, 3> center = {{0.0,0.0,0.0}};
105 std::array<double, 3> minimum = center - length / 2.;
106 double dh = length / 16.;
107
108#if BITPIT_ENABLE_MPI
109 VolOctree mesh(3, minimum, length, dh, MPI_COMM_NULL);
110#else
111 VolOctree mesh(3, minimum, length, dh);
112#endif
113 mesh.initializeAdjacencies();
114 mesh.update();
115
116 // Initialize fields
117 log::cout() << "Initializing fields..." << std::endl;
118
119 PiercedStorage<double, long> scalarField;
120 PiercedStorage<std::array<double, 3>, long> vectorField;
121 scalarField.setDynamicKernel(&(mesh.getCells()), PiercedVector<Cell>::SYNC_MODE_JOURNALED);
122 vectorField.setDynamicKernel(&(mesh.getCells()), PiercedVector<Cell>::SYNC_MODE_JOURNALED);
123 for (const Cell &cell : mesh.getCells()) {
124 const long cellId = cell.getId();
125 std::array<double, 3> cellCentroid = mesh.evalCellCentroid(cellId);
126 double r = std::sqrt(cellCentroid[0] * cellCentroid[0] + cellCentroid[1] * cellCentroid[1] + cellCentroid[2] * cellCentroid[2]);
127
128 scalarField.set(cellId, r);
129 vectorField.set(cellId, {{cellCentroid[0] / r, cellCentroid[1] / r, cellCentroid[2] / r}});
130 }
131
132 // Initialize mesh output
133 FieldStreamer dataStreamer(mesh, scalarField, vectorField);
134
135 mesh.getVTK().setName("voloctree_adaptation_example_00001");
136 mesh.getVTK().setCounter();
137 mesh.getVTK().addData<double>("scalarField", VTKFieldType::SCALAR, VTKLocation::CELL, &dataStreamer);
138 mesh.getVTK().addData<std::array<double, 3>>("vectorField", VTKFieldType::VECTOR, VTKLocation::CELL, &dataStreamer);
139
140 // Write mesh
141 mesh.write();
142
143 //
144 // Mesh adaptation
145 //
146 bool trackAdaptation = true;
147 bool squeeshPatchStorage = false;
148 std::vector<adaption::Info> adaptionData;
149
150 // Refinement
151 log::cout() << "Performing refinement..." << std::endl;
152
153 int nRefinements = 2;
154 for (int i = 0; i < nRefinements; ++i) {
155 // Mark cells that need refinement
156 //
157 // All cells inside a circle centered in the origin and with a radius
158 // equal to 0.2 will be refined.
159 log::cout() << " Marking cells that need refinement..." << std::endl;
160
161 for (const Cell &cell : mesh.getCells()) {
162 long cellId = cell.getId();
163 std::array<double, 3> cellCentroid = mesh.evalCellCentroid(cellId);
164
165 double r = std::sqrt(cellCentroid[0] * cellCentroid[0] + cellCentroid[1] * cellCentroid[1] + cellCentroid[2] * cellCentroid[2]);
166 if (r < 0.2) {
167 mesh.markCellForRefinement(cellId);
168 }
169 }
170
171 // Update the mesh
172 //
173 // Adaption data structure contains the information needed for mapping
174 // the fields from the previous mesh to the updated mesh.
175 adaptionData = mesh.adaptionPrepare(trackAdaptation);
176
177 // Save values needed for remapping
178 //
179 // Data of cells no longer in the mesh are moved in a temporary
180 // container to be ued later for remapping.
181 PiercedVector<double, long> previousScalarField;
182 PiercedVector<std::array<double, 3>, long> previousVectorField;
183 for (const adaption::Info &adaptionInfo : adaptionData) {
184 // Consider only cell refinements
185 if (adaptionInfo.entity != adaption::Entity::ENTITY_CELL) {
186 continue;
187 } else if (adaptionInfo.type != adaption::TYPE_REFINEMENT) {
188 continue;
189 }
190
191 // Save parent data
192 for (long previousId : adaptionInfo.previous) {
193 previousScalarField.insert(previousId, std::move(scalarField.at(previousId)));
194 previousVectorField.insert(previousId, std::move(vectorField.at(previousId)));
195 }
196
197 }
198
199 adaptionData = mesh.adaptionAlter(trackAdaptation, squeeshPatchStorage);
200
201 // Map the fields on newly created cells
202 //
203 // On refinement one parent cell has been divided in many children
204 // cells. At every children it will be assigned the value of their
205 // parent.
206 for (const adaption::Info &adaptionInfo : adaptionData) {
207 // Consider only cell refinements
208 if (adaptionInfo.entity != adaption::Entity::ENTITY_CELL) {
209 continue;
210 } else if (adaptionInfo.type != adaption::TYPE_REFINEMENT) {
211 continue;
212 }
213
214 // Assign data to children
215 long parentId = adaptionInfo.previous.front();
216 for (long currentId : adaptionInfo.current) {
217 scalarField.set(currentId, previousScalarField.at(parentId));
218 vectorField.set(currentId, previousVectorField.at(parentId));
219 }
220 }
221
222 mesh.adaptionCleanup();
223
224 // Write mesh
225 mesh.write();
226 }
227
228 // Coarsening
229 int nofCoarsening = 1;
230 for (int i = 0; i < nofCoarsening; ++i) {
231 // Mark cells that need coarsening
232 //
233 // All cells inside a circle centered in the origin and with a radius
234 // equal to 0.15 will be coarsened.
235 for (const Cell &cell : mesh.getCells()) {
236 long cellId = cell.getId();
237 std::array<double, 3> cellCentroid = mesh.evalCellCentroid(cellId);
238
239 double r = std::sqrt(cellCentroid[0] * cellCentroid[0] + cellCentroid[1] * cellCentroid[1] + cellCentroid[2] * cellCentroid[2]);
240 if (r < 0.15) {
241 mesh.markCellForCoarsening(cellId);
242 }
243 }
244
245 adaptionData = mesh.adaptionPrepare(trackAdaptation);
246
247 // Save values needed for remapping
248 //
249 // Data of cells no longer in the mesh are moved in a temporary
250 // container to be ued later for remapping.
251 PiercedVector<double, long> previousScalarField;
252 PiercedVector<std::array<double, 3>, long> previousVectorField;
253 for (const adaption::Info &adaptionInfo : adaptionData) {
254 // Consider only cell coarsenings
255 if (adaptionInfo.entity != adaption::Entity::ENTITY_CELL) {
256 continue;
257 } else if (adaptionInfo.type != adaption::TYPE_COARSENING) {
258 continue;
259 }
260
261 // Save parent data
262 for (long previousId : adaptionInfo.previous) {
263 previousScalarField.insert(previousId, std::move(scalarField.at(previousId)));
264 previousVectorField.insert(previousId, std::move(vectorField.at(previousId)));
265 }
266 }
267
268 // Update the mesh
269 //
270 // Adaption data structure contains the information needed for mapping
271 // the fields from the previous mesh to the updated mesh.
272 adaptionData = mesh.adaptionAlter(trackAdaptation, squeeshPatchStorage);
273
274 // Map the fields on newly created cells
275 //
276 // On coarsening many parent cell have been merged together to create
277 // one child cell. The child fields will be set equal to the average
278 // of the parent fields.
279 for (const adaption::Info &adaptionInfo : adaptionData) {
280 // Consider only cell coarsening
281 if (adaptionInfo.entity != adaption::Entity::ENTITY_CELL) {
282 continue;
283 } else if (adaptionInfo.type != adaption::TYPE_COARSENING) {
284 continue;
285 }
286
287 // Evaluate parent average
288 int nParents = adaptionInfo.previous.size();
289 double scalarParentAverage = 0.;
290 std::array<double, 3> vectorParentAverage = {{0., 0., 0.}};
291 for (long parentId : adaptionInfo.previous) {
292 scalarParentAverage += previousScalarField.at(parentId);
293 vectorParentAverage += previousVectorField.at(parentId);
294 }
295 scalarParentAverage /= nParents;
296 vectorParentAverage /= (double) nParents;
297
298 // Assign data to the child
299 long childId = adaptionInfo.current.front();
300 scalarField.set(childId, scalarParentAverage);
301 vectorField.set(childId, vectorParentAverage);
302 }
303
304 mesh.adaptionCleanup();
305
306 // Write mesh
307 mesh.write();
308 }
309
310#if BITPIT_ENABLE_MPI == 1
311 // Finalize MPI
312 MPI_Finalize();
313#endif
314}
