57 void PatchKernel::initializeCommunicator(MPI_Comm communicator)
58 {
59  // Early return if setting a null communicator
60  if (communicator == MPI_COMM_NULL) {
61  m_communicator = communicator;
62 
63  m_nProcessors = 1;
64  m_rank = 0;
65 
66  return;
67  }
68 
69  // Creat a copy of the user-specified communicator
70  //
71  // No library routine should use MPI_COMM_WORLD as the communicator;
72  // instead, a duplicate of a user-specified communicator should always
73  // be used.
74  MPI_Comm_dup(communicator, &m_communicator);
75 
76  // Get MPI information
77  MPI_Comm_size(m_communicator, &m_nProcessors);
78  MPI_Comm_rank(m_communicator, &m_rank);
79 
80  // Set parallel data for the VTK output
81  if (m_nProcessors > 1) {
82  m_vtk.setParallel(m_nProcessors, m_rank);
83  }
84 }

198 void PatchKernel::updateOwner()
199 {
200  long nInternalCells = getInternalCellCount();
201  long nGlobalInternalCells = nInternalCells;
202  if (isPartitioned()) {
203  MPI_Allreduce(MPI_IN_PLACE, &nGlobalInternalCells, 1, MPI_LONG, MPI_SUM, getCommunicator());
204  }
205 
206  if (nInternalCells == nGlobalInternalCells) {
207  m_owner = getRank();
208  } else {
209  m_owner = -1;
210  }
211 
212  if (isPartitioned()) {
213  MPI_Allreduce(MPI_IN_PLACE, &m_owner, 1, MPI_INT, MPI_MAX, getCommunicator());
214  }

876 PatchKernel::CellIterator PatchKernel::_addGhostCell(ElementType type, std::unique_ptr<long[]> &&connectStorage,
877  int rank, long id)
878 {
879  // Create the cell
880  //
881  // If there are internal cells, the ghost cell should be inserted
882  // after the last internal cell.
883  CellIterator iterator;
884  if (m_lastInternalCellId < 0) {
885  iterator = m_cells.emreclaim(id, id, type, std::move(connectStorage), false, true);
886  } else {
887  iterator = m_cells.emreclaimAfter(m_lastInternalCellId, id, id, type, std::move(connectStorage), false, true);

889  m_nGhostCells++;
890 
891  // Update the id of the first ghost cell
892  if (m_firstGhostCellId < 0) {
893  m_firstGhostCellId = id;
894  } else if (m_cells.rawIndex(m_firstGhostCellId) > m_cells.rawIndex(id)) {
895  m_firstGhostCellId = id;
896  }
897 
898  // Set owner
899  setGhostCellOwner(id, rank);
900 
901  // Set the alteration flags of the cell
902  setAddedCellAlterationFlags(id);
903 
904  return iterator;
