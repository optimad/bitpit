259 void SurfUnstructured::extractEdgeNetwork(SurfUnstructured &net)
260 {
261  // ====================================================================== //
262  // VARIABLES DECLARATION //
263  // ====================================================================== //
264 
265  // Local variables
266  bool check;
267  int n_faces, n_adj;
268  long id;
269  const long *adjacencies;
270 
271  // Counters
272  int i, j;
273  VertexIterator v_, ve_ = vertexEnd();
274  CellIterator c_, ce_ = cellEnd();
275 
276  // ====================================================================== //
277  // INITIALIZE DATA STRUCTURE //
278  // ====================================================================== //
279  net.reserveCells(net.getCellCount() + countFaces());
280  net.reserveVertices(net.getVertexCount() + getVertexCount());
281 
282  // ====================================================================== //
283  // ADD VERTEX TO net //
284  // ====================================================================== //
285  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
286  net.addVertex(v_->getCoords(), v_->getId());
287  } //next v_
288 
289  // ====================================================================== //
290  // ADD EDGES //
291  // ====================================================================== //
292  for (c_ = cellBegin(); c_ != ce_; ++c_) {
293  id = c_->getId();
294  n_faces = c_->getFaceCount();
295  ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
296  for (i = 0; i < n_faces; ++i) {
297  check = true;
298  n_adj = c_->getAdjacencyCount(i);
299  adjacencies = c_->getAdjacencies(i);
300  for (j = 0; j < n_adj; ++j) {
301  check = check && (id > adjacencies[j]);
302  } //next j
303  if (check) {
304  // Get edge type
305  ElementType edgeType = c_->getFaceType(i);
306 
307  // Get edge connect
308  ConstProxyVector<long> faceConnect = c_->getFaceConnect(i);
309  int faceConnectSize = faceConnect.size();
310 
311  std::unique_ptr<long[]> edgeConnect = std::unique_ptr<long[]>(new long[faceConnectSize]);
312  for (int k = 0; k < faceConnectSize; ++k) {
313  edgeConnect[k] = faceConnect[k];
314  }
315 
316  // Add edge
317  net.addCell(edgeType, std::move(edgeConnect));
318  }
319  } //next i
320  } //next c_
321 
322  return;
323 }

399 int SurfUnstructured::importSTL(const std::string &filename, STLReader::Format format,
400  int PIDOffset, bool PIDSquash,
401  std::unordered_map<int, std::string> *PIDNames)
402 {
403  int readerError;
404 
405  // Initialize reader
406  STLReader reader(filename, format);
407  if (format == STLReader::FormatUnknown) {
408  format = reader.getFormat();
409  }
410 
411  // Begin reding the STL file
412  readerError = reader.readBegin();
413  if (readerError != 0) {
414  return readerError;
415  }
416 
417  // Read all the solids in the STL file
418  int pid = PIDOffset;
419 
420  ElementType facetType = ElementType::TRIANGLE;
421  int nFacetVertices = ReferenceElementInfo::getInfo(ElementType::TRIANGLE).nVertices;
422 
423  while (true) {
424  // Read header
425  std::size_t nFacets;
426  std::string name;
427  readerError = reader.readHeader(&name, &nFacets);
428  if (readerError != 0) {
429  if (readerError == -2) {
430  break;
431  } else {
432  return readerError;
433  }
434  }
435 
436  // Generate patch cells from STL facets
437  reserveVertices(getVertexCount() + nFacetVertices * nFacets);
438  reserveCells(getCellCount() + nFacets);
439 
440  for (std::size_t n = 0; n < nFacets; ++n) {
441  // Read facet data
442  std::array<double, 3> coords_0;
443  std::array<double, 3> coords_1;
444  std::array<double, 3> coords_2;
445  std::array<double, 3> normal;
446 
447  readerError = reader.readFacet(&coords_0, &coords_1, &coords_2, &normal);
448  if (readerError != 0) {
449  return readerError;
450  }
451 
452  // Add vertices
453  VertexIterator vertexItr_0 = addVertex(coords_0);
454  VertexIterator vertexItr_1 = addVertex(coords_1);
455  VertexIterator vertexItr_2 = addVertex(coords_2);
456 
457  // Add cell
458  std::unique_ptr<long[]> connectStorage = std::unique_ptr<long[]>(new long[nFacetVertices]);
459  connectStorage[0] = vertexItr_0.getId();
460  connectStorage[1] = vertexItr_1.getId();
461  connectStorage[2] = vertexItr_2.getId();
462 
463  CellIterator cellIterator = addCell(facetType, std::move(connectStorage));
464  cellIterator->setPID(pid);
465  }
466 
467  // Read footer
468  readerError = reader.readFooter(name);
469  if (readerError != 0) {
470  return readerError;
471  }
472 
473  // Assign PID name
474  if (!PIDSquash) {
475  if (PIDNames && !name.empty()) {
476  PIDNames->insert({{pid, name}});
477  }
478  ++pid;
479  }
480 
481  // Multi-body STL files are supported only in ASCII mode
482  if (format == STLReader::FormatBinary) {
483  break;
484  }
485  }
486 
487  // End reading
488  readerError = reader.readEnd();
489  if (readerError != 0) {
490  return readerError;
491  }
492 
493  return 0;
494 }

549 int SurfUnstructured::exportSTLSingle(const std::string &filename, bool isBinary)
550 {
551  int writerError;
552 
553  // Initialize writer
554  STLReader::Format format;
555  if (isBinary) {
556  format = STLReader::FormatBinary;
557  } else {
558  format = STLReader::FormatASCII;
559  }
560 
561  STLWriter writer(filename, format);
562 
563  // Solid name
564  //
565  // An empty solid name will be used.
566  const std::string name = "";
567 
568  // Detect if this is the master writer
569  bool isMasterWriter;
570 #if BITPIT_ENABLE_MPI==1
571  if (isPartitioned()) {
572  isMasterWriter = (getRank() == 0);
573  } else {
574 #else
575  {
576 #endif
577  isMasterWriter = true;
578  }
579 
580  // Detect if the file will be written incrementally
581  bool incrementalWrite;
582 #if BITPIT_ENABLE_MPI==1
583  incrementalWrite = true;
584 #else
585  incrementalWrite = false;
586 #endif
587 
588  // Count the number of facets
589  long nFacets = getInternalCellCount();
590 #if BITPIT_ENABLE_MPI==1
591  if (isPartitioned()) {
592  MPI_Allreduce(MPI_IN_PLACE, &nFacets, 1, MPI_LONG, MPI_SUM, getCommunicator());
593  }
594 #endif
595 
596  // Write header
597  if (isMasterWriter) {
598  // Begin writing
599  writerError = writer.writeBegin(STLWriter::WriteOverwrite, incrementalWrite);
600  if (writerError != 0) {
601  writer.writeEnd();
602 
603  return writerError;
604  }
605 
606  // Write header section
607  writerError = writer.writeHeader(name, nFacets);
608  if (writerError != 0) {
609  writer.writeEnd();
610 
611  return writerError;
612  }
613 
614 #if BITPIT_ENABLE_MPI==1
615  // Close the file to let other process write into it
616  writerError = writer.writeEnd();
617  if (writerError != 0) {
618  return writerError;
619  }
620 #endif
621  }
622 
623  // Write facet data
624 #if BITPIT_ENABLE_MPI==1
625  int nRanks = getProcessorCount();
626  for (int i = 0; i < nRanks; ++i) {
627  if (i == getRank()) {
628  // Begin writing
629  writerError = writer.writeBegin(STLWriter::WriteAppend, incrementalWrite);
630  if (writerError != 0) {
631  writer.writeEnd();
632  return writerError;
633  }
634 
635 #else
636  {
637 #endif
638  // Write facet section
639  CellConstIterator cellBegin = internalCellConstBegin();
640  CellConstIterator cellEnd = internalCellConstEnd();
641  for (CellConstIterator cellItr = cellBegin; cellItr != cellEnd; ++cellItr) {
642  // Get cell
643  const Cell &cell = *cellItr;
644 
645  // Get vertex coordinates
646  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
647  assert(cellVertexIds.size() == 3);
648 
649  const std::array<double, 3> &coords_0 = getVertex(cellVertexIds[0]).getCoords();
650  const std::array<double, 3> &coords_1 = getVertex(cellVertexIds[1]).getCoords();
651  const std::array<double, 3> &coords_2 = getVertex(cellVertexIds[2]).getCoords();
652 
653  // Evaluate normal
654  const std::array<double, 3> normal = evalFacetNormal(cell.getId());
655 
656  // Write data
657  writerError = writer.writeFacet(coords_0, coords_1, coords_2, normal);
658  if (writerError != 0) {
659  writer.writeEnd();
660  return writerError;
661  }
662  }
663 
664 #if BITPIT_ENABLE_MPI==1
665  // Close the file to let other process write into it
666  writerError = writer.writeEnd();
667  if (writerError != 0) {
668  return writerError;
669  }
670  }
671 
672  if (isPartitioned()) {
673  MPI_Barrier(getCommunicator());
674  }
675 #endif
676  }
677 
678  // Write footer
679  if (isMasterWriter) {
680 #if BITPIT_ENABLE_MPI==1
681  // Begin writning
682  //
683  // The file was previously closed to let other process write into it.
684  writerError = writer.writeBegin(STLWriter::WriteAppend, incrementalWrite);
685  if (writerError != 0) {
686  writer.writeEnd();
687  return writerError;
688  }
689 #endif
690 
691  // Write footer section
692  writerError = writer.writeFooter(name);
693  if (writerError != 0) {
694  writer.writeEnd();
695 
696  return writerError;
697  }
698 
699  // End writing
700  writerError = writer.writeEnd();
701  if (writerError != 0) {
702  return writerError;
703  }
704  }
705 
706  return 0;
707 }

720 {
721  int writerError;
722 
723  // Initialize writer
724  STLWriter writer(filename, STLReader::FormatASCII);
725 
726  // Detect if this is the master writer
727  bool isMasterWriter;
728 #if BITPIT_ENABLE_MPI==1
729  if (isPartitioned()) {
730  isMasterWriter = (getRank() == 0);
731  } else {
732 #else
733  {
734 #endif
735  isMasterWriter = true;
736  }
737 
738  // Detect if the file will be written incrementally
739  bool incrementalWrite = true;
740 
741 #if BITPIT_ENABLE_MPI==1
742  // Count number of ranks
743  int nRanks = getProcessorCount();
744 #endif
745 
746  // Get the global list of PID
747  std::set<int> cellPIDs;
748 #if BITPIT_ENABLE_MPI==1
749  if (isPartitioned()) {
750  std::set<int> internalPIDs = getInternalCellPIDs();
751  std::vector<int> localPIDs(internalPIDs.begin(), internalPIDs.end());
752  int nLocalPids = localPIDs.size();
753 
754  std::vector<int> gatherPIDCount(nRanks);
755  MPI_Allgather(&nLocalPids, 1, MPI_INT, gatherPIDCount.data(), 1, MPI_INT, getCommunicator());
756 
757  std::vector<int> gatherPIDDispls(nRanks, 0);
758  for (int i = 1; i < nRanks; ++i) {
759  gatherPIDDispls[i] = gatherPIDDispls[i - 1] + gatherPIDCount[i - 1];
760  }
761 
762  int gatherPIDsSize = gatherPIDDispls.back() + gatherPIDCount.back();
763 
764  std::vector<int> globalPIDs(gatherPIDsSize);
765  MPI_Allgatherv(localPIDs.data(), nLocalPids, MPI_INT, globalPIDs.data(),
766  gatherPIDCount.data(), gatherPIDDispls.data(), MPI_INT, getCommunicator());
767 
768  cellPIDs = std::set<int>(globalPIDs.begin(), globalPIDs.end());
769  } else {
770 #else
771  {
772 #endif
773  cellPIDs = getInternalCellPIDs();
774  }
775 
776  // Export cells

787  }
788 #endif
789 
790  // Solid name
791  std::string name;
792  if (PIDNames && PIDNames->count(pid) > 0) {
793  name = PIDNames->at(pid);
794  } else {
795  name = std::to_string(pid);
796  }
797 
798  // Write header
799  if (isMasterWriter) {
800  // Begin writing
801  STLWriter::WriteMode writeMode;
802  if (firstPID) {
803  writeMode = STLWriter::WriteOverwrite;
804  } else {
805  writeMode = STLWriter::WriteAppend;
806  }
807

809  if (writerError != 0) {
810  writer.writeEnd();
811 
812  return writerError;
813  }
814 
815  // Write header section

819

830  }

836  // Begin writing

929 int SurfUnstructured::importDGF(const std::string &filename, int PIDOffset, bool PIDSquash)
930 {
931  // ====================================================================== //
932  // VARIABLES DECLARATION //
933  // ====================================================================== //
934 
935  // Local variables
936  DGFObj dgf_in(filename);
937  int nV = 0, nS = 0;
938  long vcount, idx;
939  std::vector<std::array<double, 3>> vertex_list;
940  std::vector<std::vector<int>> simplex_list;
941  std::vector<int> simplex_PID;
942  std::vector<long> vertex_map;
943  std::vector<long> connect;
944 
945  // Counters
946  std::vector<std::array<double, 3>>::const_iterator v_, ve_;
947  std::vector<std::vector<int>>::iterator c_, ce_;
948  std::vector<int>::iterator i_, ie_;
949  std::vector<long>::iterator j_, je_;
950 
951  // ====================================================================== //
952  // IMPORT DATA //
953  // ====================================================================== //
954 
955  // Read vertices and cells from DGF file
956  dgf_in.load(nV, nS, vertex_list, simplex_list, simplex_PID);
957 
958  // Add vertices
959  ve_ = vertex_list.cend();
960  vcount = 0;
961  vertex_map.resize(nV);
962  for (v_ = vertex_list.cbegin(); v_ != ve_; ++v_) {
963  idx = addVertex(*v_)->getId();
964  vertex_map[vcount] = idx;
965  ++vcount;
966  } //next v_
967
