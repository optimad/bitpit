221 std::array<double, 3> VolUnstructured::evalInterfaceNormal(long id) const
222 {
223  const Interface &interface = getInterface(id);
224 
225  ConstProxyVector<long> interfaceVertexIds = interface.getVertexIds();
226  std::size_t nInterfaceVertices = interfaceVertexIds.size();
227  BITPIT_CREATE_WORKSPACE(vertexCoordinates, std::array<double BITPIT_COMMA 3>, nInterfaceVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
228  getVertexCoords(nInterfaceVertices, interfaceVertexIds.data(), vertexCoordinates);
229 
230  std::array<double, 3> orientation = {{0., 0., 0.}};
231  if (!isThreeDimensional()) {
232  long ownerId = interface.getOwner();
233  const Cell &owner = getCell(ownerId);
234 
235  ConstProxyVector<long> ownerVertexIds = owner.getVertexIds();
236  const int nOwnerVertices = ownerVertexIds.size();
237 
238  const std::array<double, 3> &V_A = getVertex(ownerVertexIds[0]).getCoords();
239  const std::array<double, 3> &V_B = getVertex(ownerVertexIds[1]).getCoords();
240  const std::array<double, 3> &V_Z = getVertex(ownerVertexIds[nOwnerVertices - 1]).getCoords();
241 
242  orientation = crossProduct(V_B - V_A, V_Z - V_A);
243  }
244 
245  return interface.evalNormal(vertexCoordinates, orientation);
246 }
