215 template <class T>
216 void FlatVector2D<T>::initialize(std::size_t nVectors,
217  const std::size_t *sizes, std::size_t sizesStride,
218  const T *values, std::size_t valuesStride)
219 {
220  std::size_t nItems = 0;
221  for (std::size_t i = 0; i < nVectors; ++i) {
222  nItems += *(sizes + i * sizesStride);
223  }
224 
225  // Check if the container need reallocation
226  //
227  // If the current number of items is different from the number of items
228  // the container should contain after the initialization, a reallocation
229  // is needed.
230  bool reallocateIndex = (nVectors != size());
231  bool reallocateValues = (nItems != getItemCount());
232 
233  // Destroy current data structures
234  //
235  // Index and the storage data will probabily be stored in memory
236  // regions contigous to each other. To reduce memory fragmentation
237  // it's better to deallocate the container before updating its
238  // data structures.
239  if (reallocateIndex || reallocateValues) {
240  destroy(reallocateIndex, reallocateValues);
241  }
242 
243  // Initialize the indexes
244  if (reallocateIndex) {
245  m_index.resize(nVectors + 1);
246  }
247 
248  m_index[0] = 0;
249  for (std::size_t i = 0; i < nVectors; ++i) {
250  m_index[i+1] = m_index[i] + *(sizes + i * sizesStride);
251  }
252 
253  // Initialize the storage
254  if (valuesStride == 0) {
255  if (reallocateValues) {
256  m_v.assign(nItems, *values);
257  } else {
258  for (std::size_t k = 0; k < nItems; ++k) {
259  m_v[k] = *values;
260  }
261  }
262  } else {
263  if (reallocateValues) {
264  m_v.resize(nItems);
265  }
266 
267  std::size_t k = 0;
268  for (std::size_t i = 0; i < nVectors; ++i) {
269  std::size_t subArraySize = *(sizes + i * sizesStride);
270  for (std::size_t j = 0; j < subArraySize; ++j) {
271  m_v[k++] = *(values + i * valuesStride + j);
272  }
273  }
274  }
275 }

295  // If the current number of items is different from the number of items
296  // the container should contain after the initialization, a reallocation

299  bool reallocateValues = (nItems != getItemCount());
300 
301  // Destroy current data structures
302  //
303  // Index and the storage data will probabily be stored in memory
304  // regions contigous to each other. To reduce memory fragmentation
305  // it's better to deallocate the container before updating its
306  // data structures.
307  if (reallocateIndex || reallocateValues) {
308  destroy(reallocateIndex, reallocateValues);
309  }
310 
311  // Initialize the indexes
312  if (reallocateIndex) {
313  m_index.resize(nVectors + 1);
314  }
315 
316  m_index[0] = 0;
317  for (std::size_t i = 0; i < nVectors; ++i) {
318  m_index[i+1] = m_index[i] + vector2D[i].size();

320 
321  // Initialize the storage
322  if (reallocateValues) {
323  m_v.resize(nItems);
324  }
325 
326  std::size_t k = 0;
327  for (std::size_t i = 0; i < nVectors; ++i) {
328  std::size_t subArraySize = vector2D[i].size();
329  for (std::size_t j = 0; j < subArraySize; ++j) {
