91  struct VertexPositionLess
92  {
93  VertexPositionLess(const PatchKernel &patch)
94  : m_patch(patch)
95  {
96  }
97 
98  virtual ~VertexPositionLess() = default;
99 
100  bool operator()(long id_1, long id_2) const
101  {
102  const std::array<double, 3> &coords_1 = m_patch.getVertexCoords(id_1);
103  const std::array<double, 3> &coords_2 = m_patch.getVertexCoords(id_2);
104  for (int k = 0; k < 3; ++k) {
105  if (utils::DoubleFloatingEqual()(coords_1[k], coords_2[k], m_patch.getTol())) {
106  continue;
107  }
108 
109  return coords_1[k] < coords_2[k];
110  }
111 
112  // If we are here the two vertex coordinates coincide. It's not
113  // possible to define an order for the two vertices.
114  std::ostringstream stream;
115  stream << "It was not possible to define an order for vertices " << id_1 << " and " << id_2 << ". ";
116  stream << "The two vertices have the same coordinates.";
117  throw std::runtime_error (stream.str());
118  }
119 
120  const PatchKernel &m_patch;
121  };

146  struct CellPositionLess
147  {
148  CellPositionLess(const PatchKernel &patch, bool native = true)
149  : m_patch(patch), m_native(native)
150  {
151  }
152 
153  virtual ~CellPositionLess() = default;
154 
155  bool operator()(long id_1, long id_2) const
156  {
157  std::array<double, 3> centroid_1;
158  std::array<double, 3> centroid_2;
159  if (m_native) {
160  centroid_1 = m_patch.evalCellCentroid(id_1);
161  centroid_2 = m_patch.evalCellCentroid(id_2);
162  } else {
163  centroid_1 = m_patch.PatchKernel::evalCellCentroid(id_1);
164  centroid_2 = m_patch.PatchKernel::evalCellCentroid(id_2);
165  }
166 
167  for (int k = 0; k < 3; ++k) {
168  if (utils::DoubleFloatingEqual()(centroid_1[k], centroid_2[k], m_patch.getTol())) {
169  continue;
170  }
171 
172  return centroid_1[k] < centroid_2[k];
173  }
174 
175  // If we are here the two cell centroids coincide. It's not
176  // possible to define an order for the two cells.
177  std::ostringstream stream;
178  stream << "It was not possible to define an order for cells " << id_1 << " and " << id_2 << ". ";
179  stream << "The two cells have the same centroid.";
180  throw std::runtime_error (stream.str());
181  }
182 
183  const PatchKernel &m_patch;
184  bool m_native;
185  };

213  struct CellFuzzyPositionLess
214  {
215  CellFuzzyPositionLess(PatchKernel &patch, bool native = true)
216  : m_patch(patch), m_native(native)
217  {
218  }
219 
220  virtual ~CellFuzzyPositionLess() = default;
221 
222  bool operator()(long id_1, long id_2) const
223  {
224  // Select the first vertex of the first cell
225  ConstProxyVector<long> cellVertexIds_1 = m_patch.getCell(id_1).getVertexIds();
226 
227  std::size_t vertexLocalId_1 = 0;
228  long vertexId_1 = cellVertexIds_1[vertexLocalId_1];
229 
230  // The vertex of the second cell is choosen as the first vertex on
231  // that cell not equal to the selected vertex of the first cell.
232  ConstProxyVector<long> cellVertexIds_2 = m_patch.getCell(id_2).getVertexIds();
233  std::size_t nCellVertices_2 = cellVertexIds_2.size();
234 
235  std::size_t vertexLocalId_2 = 0;
236  long vertexId_2 = Vertex::NULL_ID;
237  while (vertexLocalId_2 <= nCellVertices_2) {
238  vertexId_2 = cellVertexIds_2[vertexLocalId_2];
239  if (vertexId_1 != vertexId_2) {
240  break;
241  }
242 
243  ++vertexLocalId_2;
244  }
245 
246  // Compare the two vertices
247  if (vertexId_2 != Vertex::NULL_ID) {
248  const std::array<double, 3> &vertexCoords_1 = m_patch.getVertex(vertexId_1).getCoords();
249  const std::array<double, 3> &vertexCoords_2 = m_patch.getVertex(vertexId_2).getCoords();
250  for (int k = 0; k < 3; ++k) {
251  if (utils::DoubleFloatingEqual()(vertexCoords_1[k], vertexCoords_2[k], m_patch.getTol())) {
252  continue;
253  }
254 
255  return vertexCoords_1[k] < vertexCoords_2[k];
256  }
257  }
258 
259  // If we are here it was not possible to find a vertex on the
260  // second cell for the comparison.
261  std::ostringstream stream;
262  stream << "Unable to fuzzy order cells " << id_1 << " and " << id_2 << ". ";
263  throw std::runtime_error (stream.str());
264  }
265 
266  PatchKernel &m_patch;
267  bool m_native;
268  };
