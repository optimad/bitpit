170 void Element::Tesselation::importPolygon(const std::vector<int> &vertexIds)
171 {
172  int nVertices = vertexIds.size();
173  if (nVertices == 3 || nVertices == 4) {
174  m_nTiles++;
175  m_types.push_back(nVertices == 3 ? ElementType::TRIANGLE : ElementType::QUAD);
176  m_connects.push_back(vertexIds);
177 
178  return;
179  }
180 
181  // Add the centroid
182  std::array<double, 3> centroid = {{0., 0., 0.}};
183  for (int k = 0; k < nVertices; ++k) {
184  centroid += m_coordinates[vertexIds[k]];
185  }
186  centroid = centroid / double(nVertices);
187 
188  int centroidId = importVertexCoordinates(std::move(centroid));
189 
190  // Decompose the polygon in triangles
191  //
192  // Each triangle is composed by the two vertices of a side and by the
193  // centroid.
194  ElementType tileType = ElementType::TRIANGLE;
195  int nTileVertices = ReferenceElementInfo::getInfo(tileType).nVertices;
196  int nSideVertices = ReferenceElementInfo::getInfo(ElementType::LINE).nVertices;
197 
198  int nSides = nVertices;
199  m_types.resize(m_nTiles + nSides, tileType);
200  m_connects.resize(m_nTiles + nSides, std::vector<int>(nTileVertices));
201  for (int i = 0; i < nSides; ++i) {
202  m_nTiles++;
203  for (int k = 0; k < nSideVertices; ++k) {
204  m_connects[m_nTiles - 1][k] = vertexIds[(i + k) % nVertices];
205  }
206  m_connects[m_nTiles - 1][nSideVertices] = centroidId;
207  }
208 }

216 void Element::Tesselation::importPolyhedron(const std::vector<int> &vertexIds, const std::vector<std::vector<int>> &faceVertexIds)
217 {
218  int nFaces = faceVertexIds.size();
219  int nVertices = vertexIds.size();
220 
221  // Generate the tesselation of the surface
222  int nInitialTiles = m_nTiles;
223  for (int i = 0; i < nFaces; ++i) {
224  importPolygon(faceVertexIds[i]);
225  }
226  int nFinalTiles = m_nTiles;
227 
228  // Add the centroid of the element to the tesselation
229  std::array<double, 3> centroid = {{0., 0., 0.}};
230  for (int k = 0; k < nVertices; ++k) {
231  centroid += m_coordinates[vertexIds[k]];
232  }
233  centroid = centroid / double(nVertices);
234 
235  int centroidTesselationId = importVertexCoordinates(std::move(centroid));
236 
237  // Decompose the polyhedron in prisms and pyramids.
238  //
239  // Each prism/pyramid has a tile of the surface tesselation as the
240  // base and the centroid of the element as the apex. Since we have
241  // already generated the surface tesselation, we can "convert" the
242  // two-dimensional tiles of that tesselation in three-dimensional
243  // tiles to obtain the volume tesselation.
244  for (int i = nInitialTiles; i < nFinalTiles; ++i) {
245  // Change the tile type
246  ElementType &tileType = m_types[i];
247  if (tileType == ElementType::TRIANGLE) {
248  tileType = ElementType::TETRA;
249  } else if (tileType == ElementType::QUAD) {
250  tileType = ElementType::PYRAMID;
251  } else {
252  BITPIT_UNREACHABLE("Unsupported tile");
253  throw std::runtime_error ("Unsupported tile");
254  }
255 
256  // Fix the order of the connectivity the match the order of the
257  // three-dimensional element
258  if (tileType == ElementType::TETRA) {
259  std::swap(m_connects[i][0], m_connects[i][2]);
260  } else if (tileType == ElementType::PYRAMID) {
261  std::swap(m_connects[i][1], m_connects[i][3]);
262  }
263 
264  // Add the centroid to the tile connectivity
265  m_connects[i].push_back(centroidTesselationId);
266  }
267 }

758 ConstProxyVector<int> Element::getFaceLocalConnect(int face) const
759 {
760  switch (m_type) {
761 
762  case (ElementType::POLYGON):
763  {
764  int connectSize = getConnectSize();
765 
766  int faceConnectSize = getFaceVertexCount(face);
767  std::vector<int> localFaceConnect(faceConnectSize);
768  for (int i = 0; i < faceConnectSize; ++i) {
769  localFaceConnect[i] = 1 + ((face + i) % (connectSize - 1));
770  }
771 
772  return ConstProxyVector<int>(std::move(localFaceConnect));
773  }
774 
775  case (ElementType::POLYHEDRON):
776  {
777  ElementType faceType = getFaceType(face);
778  ConstProxyVector<long> faceVertexIds = getFaceVertexIds(face);
779  int nFaceVertices = faceVertexIds.size();
780  bool faceHasReferenceInfo = ReferenceElementInfo::hasInfo(faceType);
781 
782  int faceConnectSize = nFaceVertices;
783  int localVertexOffset = 0;
784  if (!faceHasReferenceInfo) {
785  ++faceConnectSize;
786  ++localVertexOffset;
787  }
788 
789  std::vector<int> localFaceConnect(faceConnectSize);
790  if (!faceHasReferenceInfo) {
791  localFaceConnect[0] = nFaceVertices;
792  }
793 
794  // The list of the vertices in sorted, therfore we can search the
795  // vertices using a binary search.
796  ConstProxyVector<long> vertexIds = getVertexIds();
797  for (int k = 0; k < nFaceVertices; ++k) {
798  int vertexId = faceVertexIds[k];
799  auto localVertexIdItr = std::lower_bound(vertexIds.begin(), vertexIds.end(), vertexId);
800  assert(localVertexIdItr != vertexIds.end() && *localVertexIdItr == vertexId);
801  localFaceConnect[localVertexOffset + k] = std::distance(vertexIds.begin(), localVertexIdItr);
802  }
803 
804  return ConstProxyVector<int>(std::move(localFaceConnect));
805  }
806 
807  default:
808  {
809  assert(m_type != ElementType::UNDEFINED);
810 
811  const int localConnectSize = getFaceVertexCount(face);
812  const int *localFaceConnect = getInfo().faceConnectStorage[face].data();
813 
814  return ConstProxyVector<int>(localFaceConnect, localConnectSize);
815  }
816 
817  }
818 }

826 ConstProxyVector<long> Element::getFaceConnect(int face) const
827 {
828  const long *connectivity = getConnect();
829 
830  switch (m_type) {
831 
832  case (ElementType::POLYGON):
833  {
834  int connectSize = getConnectSize();
835 
836  int facePos = 1 + face;
837  int faceConnectSize = getFaceVertexCount(face);
838  if (facePos + faceConnectSize <= connectSize) {
839  return ConstProxyVector<long>(connectivity + facePos, faceConnectSize);
840  } else {
841  std::vector<long> faceConnect(faceConnectSize);
842  for (int i = 0; i < faceConnectSize; ++i) {
843  int position = facePos + i;
844  if (position >= connectSize) {
845  position = position % connectSize + 1;
846  }
847 
848  faceConnect[i] = connectivity[position];
849  }
850 
851  return ConstProxyVector<long>(std::move(faceConnect));
852  }
853  }
854 
855  case (ElementType::POLYHEDRON):
856  {
857  ElementType faceType = getFaceType(face);
858 
859  int facePos = getFaceStreamPosition(connectivity, face);
860  int faceConnectSize = connectivity[facePos];
861  int faceConnectBegin = facePos + 1;
862  if (!ReferenceElementInfo::hasInfo(faceType)) {
863  faceConnectSize++;
864  faceConnectBegin--;
865  }
866 
867  return ConstProxyVector<long>(connectivity + faceConnectBegin, faceConnectSize);
868  }
869 
870  default:
871  {
872  assert(m_type != ElementType::UNDEFINED);
873 
874  // If we are here, the element has a reference element, therefore we
875  // can retrieve the local face connectivity directly form the info
876  // associated the to referenc element.
877  int faceConnectSize = getFaceVertexCount(face);
878  const int *localFaceConnect = getInfo().faceConnectStorage[face].data();
879 
880  std::vector<long> faceConnect(faceConnectSize);
881  for (int k = 0; k < faceConnectSize; ++k) {
882  int localVertexId = localFaceConnect[k];
883  long vertexId = connectivity[localVertexId];
884  faceConnect[k] = vertexId;
885  }
886 
887  return ConstProxyVector<long>(std::move(faceConnect));
888  }
889 
890  }
891 }

1980  }

1999  if (insertResult.second) {

2002  edgeConnectStorage[edgeSet.size() - 1] = ConstProxyVector<long>(std::move(edgeConnect));
2003
