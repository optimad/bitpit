159 void ReferenceElementInfo::initializeFaceEdges(const std::vector<const ReferenceElementInfo *> &facesInfo,
160  const std::vector<const ReferenceElementInfo *> &edgesInfo)
161 {
162  for (int k = 0; k < nFaces; ++k) {
163  const ReferenceElementInfo &faceInfo = *(facesInfo[k]);
164 
165  int nFaceEdges = faceInfo.nFaces;
166  int faceEdgeCounter = 0;
167  for (int i = 0; i < nFaceEdges; ++i) {
168  const ReferenceElementInfo &faceEdgeInfo = *(edgesInfo[i]);
169 
170  // Connectivity of the edge associated to the face
171  const int *localFaceEdgeConnect = faceInfo.faceConnectStorage[i].data();
172 
173  std::set<int> faceEdgeConnect;
174  for (int n = 0; n < faceEdgeInfo.nVertices; ++n) {
175  int localVertexId = localFaceEdgeConnect[n];
176  int vertexId = faceConnectStorage[k][localVertexId];
177 
178  faceEdgeConnect.insert(vertexId);
179  }
180 
181  // Search the edge that has the same connectivity of the face edge
182  for (int j = 0; j < nEdges; ++j) {
183  const ReferenceElementInfo &guessEdgeInfo = *(edgesInfo[j]);
184 
185  // If face edge and the guess edge have a different type, the
186  // two edge cannot be the same.
187  if (guessEdgeInfo.type != faceEdgeInfo.type) {
188  continue;
189  }
190 
191  // If the connecitivity of the face edge and the one of the
192  // guess edge are the same, the two edges coincides.
193  const std::set<int> guessEdgeConnect = std::set<int>(edgeConnectStorage[j].begin(), edgeConnectStorage[j].begin() + guessEdgeInfo.nVertices);
194  if (faceEdgeConnect == guessEdgeConnect) {
195  faceEdgeStorage[k][faceEdgeCounter] = j;
196  ++faceEdgeCounter;
197  }
198  }
199  }
200 
201  assert(faceEdgeCounter == nFaceEdges);
202  }
203 }

288 void Reference3DElementInfo::evalPointProjection(const std::array<double, 3> &point, const std::array<double, 3> *vertexCoords,
289  std::array<double, 3> *projection, double *distance) const
290 {
291  std::array<std::array<double, 3>, MAX_ELEM_VERTICES> faceVertexCoords;
292 
293  *distance = std::numeric_limits<double>::max();

297  for (int n = 0; n < faceInfo.nVertices; ++n) {
298  faceVertexCoords[n] = vertexCoords[faceConnectStorage[i][n]];

303  faceInfo.evalPointProjection(point, faceVertexCoords.data(), &faceProjection, &faceDistance);
304 
305  if (faceDistance < *distance) {
306  *distance = faceDistance;
307  *projection = faceProjection;

715 double ReferenceHexahedronInfo::evalVolume(const std::array<double, 3> *vertexCoords) const
716 {
717  std::array<std::array<double, 3>, MAX_ELEM_VERTICES> pyramidVertexCoords;
718 
719  const ReferenceQuadInfo &quadInfo = static_cast<const ReferenceQuadInfo &>(getInfo(ElementType::QUAD));
720  const ReferencePyramidInfo &pyramidInfo = static_cast<const ReferencePyramidInfo &>(getInfo(ElementType::PYRAMID));
721 
722  // The centroid is the apex of all the pyramids
723  pyramidVertexCoords[4] = vertexCoords[0];
724  for (int i = 1; i < nVertices; ++i) {
725  pyramidVertexCoords[4] += vertexCoords[i];
726  }
727  pyramidVertexCoords[4] = pyramidVertexCoords[4] / double(nVertices);
728 
729  // Sum the volume of the pyramids having a face as the base and the
730  // centroid as the apex.
731  double volume = 0.;
732  for (int i = 0; i < nFaces; ++i) {
733  for (int n = 0; n < quadInfo.nVertices; ++n) {
734  pyramidVertexCoords[quadInfo.nVertices - n - 1] = vertexCoords[faceConnectStorage[i][n]];
735  }
736 
737  volume += pyramidInfo.evalVolume(pyramidVertexCoords.data());
738  }
739 
740  return volume;
741 }
