205 template<typename T>
206 ProxyVector<T>::ProxyVector(const ProxyVector &other)
207  : m_size(other.m_size)
208 {
209  if (other.m_storage) {
210  if (m_storage) {
211  m_storage->assign(other.m_storage->begin(), other.m_storage->end());
212  } else {
213  m_storage = std::unique_ptr<std::vector<T_no_cv>>(new std::vector<T_no_cv>(*(other.m_storage)));
214  }
215 
216  if (other.m_data == other.m_storage->data()) {
217  m_data = m_storage->data();
218  } else {
219  m_data = other.m_data;
220  }
221  } else {
222  m_storage.reset();
223 
224  m_data = other.m_data;
225  }
226 }
