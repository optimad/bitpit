48 template<class T>
49 bitpit::OBinaryStream& operator<<(bitpit::OBinaryStream &buffer, const bitpit::FlatVector2D<T> &vector)
50 {
51  typename std::vector<T>::const_iterator it;
52  typename std::vector<size_t>::const_iterator jt;
53 
54  buffer << vector.m_index.size() << vector.m_v.size();
55  for (jt = vector.m_index.begin(); jt != vector.m_index.end(); ++jt) {
56  buffer << *(jt);
57  }
58  for (it = vector.m_v.begin(); it != vector.m_v.end(); ++it) {
59  buffer << *(it);
60  }
61 
62  return buffer;
63 }

153 template <class T>
154 void FlatVector2D<T>::initialize(const std::vector<int> &sizes, const T &value)
155 {
156  int nVectors = sizes.size();
157 
158  int nItems = 0;
159  for (int i = 0; i < nVectors; ++i) {
160  nItems += sizes[i];
161  }
162 
163  // Check if the container need reallocation
164  //
165  // If the current number of items is different from the number of items
166  // the container should contain after the initialization, a reallocation
167  // is needed.
168  bool reallocateIndex = (nVectors != size());
169  bool reallocateValues = (nItems != getItemCount());
170 
171  // Destroy current data structures
172  //
173  // Index and the storage data will probabily be stored in memory
174  // regions contigous to each other. To reduce memory fragmentation
175  // it's better to deallocate the container before updating its
176  // data structures.
177  if (reallocateIndex || reallocateValues) {
178  destroy(reallocateIndex, reallocateValues);
179  }
180 
181  // Initialize the indexes
182  if (reallocateIndex) {
183  m_index.resize(nVectors + 1);
184  m_index.shrink_to_fit();
185  }
186 
187  for (int i = 0; i < nVectors; ++i) {
188  m_index[i+1] = m_index[i] + sizes[i];
189  }
190 
191  // Initialize the storage
192  if (reallocateValues) {
193  m_v.assign(m_index[nVectors], value);
194  m_v.shrink_to_fit();
195  } else {
196  for (int k = 0; k < nItems; ++k) {
197  m_v[k] = value;
198  }
199  }
200 }
