670 void POD::setSensorMask(const PiercedStorage<bool> & mask, VolumeKernel * mesh)
671 {
672  m_sensorMask.setStaticKernel(&(m_podkernel->getMesh()->getCells()));
673 
674  if (m_staticMesh || mesh == m_podkernel->getMesh()){
675 
676  for (const Cell &cell : m_podkernel->getMesh()->getCells()) {
677  long id = cell.getId();
678  m_sensorMask[id] = mask[id];
679  }
680 
681  }
682  else{
683 
684  if (mesh == nullptr)
685  throw std::runtime_error ("POD: null mesh pointer passed to setSensorMask");
686 
687  //If mapping dirty or not computed, compute mapping of input mesh on pod mesh
688  if (m_podkernel->isMapperDirty())
689  _computeMapper(mesh);
690 
691  std::unordered_set<long> trueCells;
692  for (const Cell & cell : mesh->getCells()){
693  long id = cell.getId();
694  if (mask[id])
695  trueCells.insert(id);
696  }
697 
698 
699  //Map true cells
700  std::unordered_set<long> mappedCells = m_podkernel->mapCellsToPOD(&trueCells);
701 
702  m_podkernel->mapBoolFieldToPOD(mask, mesh, &mappedCells, m_sensorMask);
703 
704  }
705 
706  fillListActiveIDs(m_sensorMask);
707  m_toUpdate = true;
708 
709 }

977 void POD::_evalMeanMesh()
978 {
979 
980  if (m_podkernel->getMesh() == nullptr){
981 
982  //Read first mesh and use as meshPOD
983  VolumeKernel* meshr = m_podkernel->readMesh(m_database[0]);
984  m_podkernel->setMesh(meshr);
985 
986  //Dynamic mesh case
987  if (!m_staticMesh){
988  // Compute meshPOD (starting from inital mesh) and fill filter
989  for (std::size_t i = 1; i < m_nSnapshots; i++) {
990  log::cout() << "pod : evaluation POD mesh - use snapshot " << i+1 << "/" << m_nSnapshots << std::endl;
991  VolumeKernel* readmesh = m_podkernel->readMesh(m_database[i]);
992  m_podkernel->adaptMeshToMesh(m_podkernel->getMesh(), readmesh);
993  m_podkernel->clearMapper();
994  }
995  }
996  }
997 
998  {
999  // Read first snapshot to set n scalar and vector fields
1000  pod::PODField readf;
1001  readSnapshot(m_database[0], readf);
1002  }
1003 
1004  //Compute cells volume
1005  m_podkernel->evalCellsVolume();
1006 
1007  m_filter.setStaticKernel(&(m_podkernel->getMesh()->getCells()));
1008  m_filter.fill(true);
1009 
1010 
1011  if (m_useMean){
1012  m_mean.scalar = std::unique_ptr<pod::ScalarStorage>(new pod::ScalarStorage(m_nScalarFields, &m_podkernel->getMesh()->getCells()));
1013  m_mean.vector = std::unique_ptr<pod::VectorStorage>(new pod::VectorStorage(m_nVectorFields, &m_podkernel->getMesh()->getCells()));
1014  m_mean.scalar->fill(0.0);
1015  m_mean.vector->fill(std::array<double, 3>{{0.0, 0.0, 0.0}});
1016 
1017  // Compute mean (starting from first snapshot) & filter+mask
1018  for (std::size_t i = 0; i < m_nSnapshots; i++) {
1019  log::cout() << "pod : evaluation mean - use snapshot " << i+1 << "/" << m_nSnapshots << std::endl;
1020  pod::PODField readf;
1021  readSnapshot(m_database[i], readf);
1022  if (!m_staticMesh){
1023  //Compute mapping of read mesh on pod mesh
1024  _computeMapper(readf.mesh);
1025  readf = pod::PODField(m_podkernel->mapPODFieldToPOD(readf, nullptr));
1026  m_podkernel->getMeshMapper().clear();
1027  m_podkernel->setMapperDirty(true);
1028  }
1029 
1030  for (const Cell &cell : m_podkernel->getMesh()->getCells()) {
1031  m_filter[cell.getId()] = (m_filter[cell.getId()]) && (readf.mask->at(cell.getId())); //N.B. mask == filter
1032  if (m_filter[cell.getId()]){
1033  for (std::size_t j = 0; j < m_nScalarFields; j++)
1034  m_mean.scalar->at(cell.getId(),j) += readf.scalar->at(cell.getId(),j) / double(m_nSnapshots);
1035  for (std::size_t j = 0; j < m_nVectorFields; j++)
1036  m_mean.vector->at(cell.getId(),j) += readf.vector->at(cell.getId(),j) / double(m_nSnapshots);
1037  }
