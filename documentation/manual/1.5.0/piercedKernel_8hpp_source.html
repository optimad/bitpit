491  struct idLess
492  {
493  const std::vector<id_t> &m_ids;
494 
495  idLess(const std::vector<id_t> &ids)
496  : m_ids(ids)
497  {
498  }
499 
500  inline bool operator() (std::size_t pos_x, std::size_t pos_y)
501  {
502  id_t id_x = m_ids[pos_x];
503  id_t id_y = m_ids[pos_y];
504 
505  if (id_x >= 0 && id_y < 0) {
506  return true;
507  } else if (id_x < 0 && id_y >= 0) {
508  return false;
509  } else if (id_x >= 0) {
510  return (id_x < id_y);
511  } else {
512  return (id_x > id_y);
513  }
514  }
515  };
