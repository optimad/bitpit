205 template<typename T>
206 ProxyVector<T>::ProxyVector(const ProxyVector &other)
207 {
208  // Copy the elements
209  std::vector<T_no_cv> new_storage(*(other.m_storage));
210 
211  // Assign the new memory to the object
212  m_storage = std::unique_ptr<std::vector<T_no_cv>>(new std::vector<T_no_cv>(std::move(new_storage)));
213  if (other.m_data == other.m_storage->data()) {
214  m_data = m_storage->data();
215  } else {
216  m_data = other.m_data;
217  }
218  m_size = other.m_size;
219 }
