85 LoggerBuffer::int_type LoggerBuffer::overflow(int_type character)
86 {
87  if (character != traits_type::eof()) {
88  // Write the charater to the buffer and then increment pptr() by
89  // calling pbump(1). It's always safe to write thecharater to the
90  // buffer because we reserved an extra char at the end of our buffer
91  // in the constructor.
92  assert(std::less_equal<char *>()(pptr(), epptr()));
93  *pptr() = character;
94  pbump(1);
95 
96  // Flush the stream
97  if (sync() == 0) {
98  return character;
99  }
100  }
101 
102  return traits_type::eof();
103 }

122 int LoggerBuffer::flush(bool terminate)
123 {
124  int status = 0;
125  if (pptr() == pbase()) {
126  return status;
127  }
128 
129  // Identify line breakers
130  std::vector<char *> linePointers;
131  linePointers.push_back(pbase());
132  for (char *p = pbase(), *e = pptr() - 1; p != e; ++p) {
133  if (*p == '\n') {
134  linePointers.push_back(p + 1);
135  }
136  }
137  linePointers.push_back(pptr());
138 
139  // Detect if a new line must be added at the end
140  terminate = (terminate && *(linePointers.back() - 1) != '\n');
141 
142  // Move the internal pointer
143  std::ptrdiff_t nCharacters = pptr() - pbase();
144  pbump(- nCharacters);
145 
146  // Write all lines
147  for (unsigned int i = 0; i < linePointers.size() - 1; ++i) {
148  char *firstCharacter = linePointers[i];
149  char *lastCharacter = linePointers[i + 1];
150  std::ptrdiff_t lineSize = lastCharacter - firstCharacter;
151 
152  // Write to the console
153  if (m_console && m_consoleEnabled) {
154  if (!m_consolePrefix.empty()) {
155  *m_console << m_consolePrefix << " :: ";
156  }
157 
158  if (!m_context.empty()) {
159  *m_console << m_context << " :: ";
160  }
161 
162  if (!m_padding.empty()) {
163  *m_console << m_padding;
164  }
165 
166  m_console->write(firstCharacter, lineSize);
167  if ((m_console->rdstate() & std::ifstream::failbit ) != 0) {
168  status = -1;
169  }
170 
171  if (terminate && lastCharacter == linePointers.back()) {
172  *m_console << "\n";
173  }
174  }
175 
176  // Write to file
177  if (m_file && m_fileEnabled && m_file->is_open()) {
178  *m_file << "[" + getTimestamp() + "] ";
179 
180  if (!m_filePrefix.empty()) {
181  *m_file << m_filePrefix << " :: ";
182  }
183 
184  if (!m_context.empty()) {
185  *m_file << m_context + " :: ";
186  }
187 
188  if (!m_padding.empty()) {
189  *m_file << m_padding;
190  }
191 
192  m_file->write(firstCharacter, lineSize);
193  if ((m_file->rdstate() & std::ifstream::failbit ) != 0) {
194  status = -1;
195  }
196 
197  if (terminate && lastCharacter == linePointers.back()) {
198  *m_file << "\n";
199  }
200  }
201  }
202 
203  return status;
204 }

351 const std::string LoggerBuffer::getTimestamp() const
352 {
353  auto currentClock = std::chrono::system_clock::now();
354  std::time_t currentTime = std::chrono::system_clock::to_time_t(currentClock);
355 
356  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(currentClock.time_since_epoch());
357  auto millisecs = std::chrono::duration_cast<std::chrono::milliseconds>(currentClock.time_since_epoch() - seconds);
358  std::ostringstream millisecsConverter;
359  millisecsConverter << (millisecs.count());
360 
361  char buffer[80];
362  strftime(buffer, 80, "%Y-%m-%d %H:%M:%S", localtime(&currentTime));
363  std::string timestamp(buffer);
364  timestamp += "." + millisecsConverter.str();
365 
366  return timestamp;
367 }

957 void LoggerManager::create(const std::string &name, bool reset,
958  const std::string &directory,
959  const int &nProcessors, const int &rank)
960 {
961  // Its not possible to create a log with the default name nor a log
962  // with the same name of an existent logger nor a log with an empty
963  // name
964  if (exists(name)) {
965  m_loggerUsers[name]++;
966 
967  cout(name) << "Detected an attemp to create the logger \"" << name << "\" twice" << std::endl;
968  cout(name) << "The previously created logger will be used." << std::endl;
969  return;
970  } else if (name == m_defaultName) {
971  cout(m_defaultName) << "Detected an attemp to overwrite the default logger" << std::endl;
972  cout(m_defaultName) << "The name of the default logger is \"" << name << "\"" << std::endl;
973  return;
974  } else if (name == "") {
975  cout() << "Detected an attemp to create a logger with an empty name" << std::endl;
976  return;
977  }
978 
979  // Create the logger
980  if (m_mode == log::SEPARATE) {
981  _create(name, reset, directory, nProcessors, rank);
982  } else {
983  _create(name, cout(m_defaultName));
984  }
985 }

998 bool LoggerManager::destroy(const std::string &name, bool force)
999 {
1000  if (!exists(name)) {
1001  return false;
1002  }
1003 
1004  // Decrement the users
1005  int &nUsers = m_loggerUsers[name];
1006  nUsers--;
1007 
1008  // If the logger has no users we can delete it
1009  if (nUsers == 0 || force) {
1010  // Remove the logger from the manager
1011  m_loggers.erase(name);
1012  m_loggerUsers.erase(name);
1013 
1014  // If the logger has it own file stream, delete it
1015  if (m_fileStreams.count(name) > 0) {
1016  // Close the stream
1017  std::ofstream &fileStream = *(m_fileStreams.at(name));
1018  fileStream.close();
1019 
1020  // Remove the stream
1021  m_fileStreams.erase(name);
1022  }
1023 
1024  return true;
1025  } else {
1026  return false;
1027  }
1028 }

1094 void LoggerManager::_create(const std::string &name, bool reset,
1095  const std::string &directory,
1096  const int &nProcessors, const int &rank)
1097 {
1098  // Get the file path
1099  FileHandler fileHandler;
1100  fileHandler.setDirectory(directory);
1101  fileHandler.setName(name);
1102  fileHandler.setAppendix("log");
1103  fileHandler.setParallel(nProcessors > 1);
1104  if (nProcessors > 1) {
1105  fileHandler.setBlock(rank);
1106  }
1107 
1108  std::string filePath = fileHandler.getPath();
1109 
1110  // Create the file stream
1111  std::ios_base::openmode fileMode;
1112  if (reset) {
1113  fileMode = std::ofstream::out;
1114  } else {
1115  fileMode = std::ofstream::app;
1116  }
1117 
1118  m_fileStreams[name] = std::unique_ptr<std::ofstream>(new std::ofstream());
1119  std::ofstream &fileStream = *(m_fileStreams.at(name));
1120  fileStream.rdbuf()->pubsetbuf(0, 0);
1121  fileStream.open(filePath, fileMode);
1122 
1123  // Use cout as console stream
1124  std::ostream &consoleStream = std::cout;
1125 
1126  // Create the logger
1127  m_loggers[name] = std::unique_ptr<Logger>(new Logger(name, &consoleStream, &fileStream, nProcessors, rank));
1128  m_loggerUsers[name] = 1;
1129 }
