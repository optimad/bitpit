171 void LevelSetCachedObject::propagateSign() {
172 
173  VolumeKernel const &mesh = *(m_kernelPtr->getMesh()) ;
174  const PiercedVector<Cell> &cells = mesh.getCells();
175 
176  VolumeKernel::CellConstIterator cellBegin = mesh.cellConstBegin();
177  VolumeKernel::CellConstIterator cellEnd = mesh.cellConstEnd();
178 
179  std::vector<long> seeds;
180  seeds.reserve(mesh.getCellCount());
181 
182  // Evaluate the bounding box of the object
183  std::array<double,3> boxMin;
184  std::array<double,3> boxMax;
185  getBoundingBox(boxMin, boxMax);
186 
187  // Set the initial propagation status of the cells
188  //
189  // We don't need to set the sign of cells associated to a levelset info,
190  // if a cell is associated to a levelset info we know its sign (the value
191  // contained in the levelset info may be a dummy value, but the sign is
192  // the correct one) and the cell can be used as seeds for propagating the
193  // sign to other cells.
194  //
195  // The sign of the cells outise the bounding box of all objects (external
196  // cells) can be either positive or negative depending on the orientation
197  // of the objects. There is no need to propagate the sign into those cells,
198  // once the sign of the external region is know, it can be assigned to all
199  // the cells in the external region. When the propagation reaches the
200  // external region it can be stopped, the sign the sign of the seed from
201  // which the propagation has started that will be the sign of the external
202  // region.
203  PiercedStorage<int, long> propagationStatus(1, &cells);
204  propagationStatus.fill(PROPAGATION_STATUS_WAITING);
205 
206  int externalSign = 0;
207  long nExternal = 0;
208  long nWaiting = mesh.getCellCount();
209  for (auto itr = cellBegin; itr != cellEnd; ++itr) {
210  long cellId = itr.getId();
211 
212  // Process cells associated to a levelset info
213  if (m_ls.find(cellId) != m_ls.end()) {
214  std::size_t cellRawId = itr.getRawIndex();
215  propagationStatus.rawAt(cellRawId) = PROPAGATION_STATUS_REACHED;
216  --nWaiting;
217 
218  seeds.push_back(cellId);
219 
220  continue;
221  }
222 
223  // Process external cells
224  std::array<double, 3> centroid = mesh.evalCellCentroid(cellId);
225 
226  bool isExternal = false;
227  for (int i = 0; i < 3; ++i) {
228  if (centroid[i] < boxMin[i] || centroid[i] > boxMax[i]) {
229  isExternal = true;
230  break;
231  }
232  }
233 
234  if (isExternal) {
235  std::size_t cellRawId = itr.getRawIndex();
236  propagationStatus.rawAt(cellRawId) = PROPAGATION_STATUS_EXTERNAL;
237  --nWaiting;
238  ++nExternal;
239 
240  continue;
241  }
242  }
243 
244  // Use the seeds to propagate the sign
245  propagateSeedSign(seeds, &propagationStatus, &nWaiting, &externalSign);
246 
247 #if BITPIT_ENABLE_MPI
248  // If there are cells with an unknown sign, data communication among
249  // ghost cells is needed. However it is only possibly to have cells with
250  // an unknown sign for partinioned patches.
251  long nGlobalWaiting = nWaiting;
252  if (mesh.isPartitioned()) {
253  MPI_Allreduce(MPI_IN_PLACE, &nGlobalWaiting, 1, MPI_LONG, MPI_SUM, m_kernelPtr->getCommunicator());
254  }
255 
256  if (nGlobalWaiting != 0) {
257  assert(mesh.isPartitioned());
258  assert(mesh.getProcessorCount() != 1);
259 
260  // Initialize the communicator for exchanging the sign of the ghosts
261  DataCommunicator dataCommunicator(m_kernelPtr->getCommunicator());
262 
263  int sign;
264  size_t dataSize = sizeof(sign);
265 
266  // Set the receives
267  for (const auto entry : mesh.getGhostExchangeTargets()) {
268  const int rank = entry.first;
269  const auto &list = entry.second;
270 
271  dataCommunicator.setRecv(rank, list.size() * dataSize);
272  }
273 
274  // Set the sends
275  for (const auto entry : mesh.getGhostExchangeSources()) {
276  const int rank = entry.first;
277  auto &list = entry.second;
278 
279  dataCommunicator.setSend(rank, list.size() * dataSize);
280  }
281 
282  // Communicate the sign among the partitions
283  while (nGlobalWaiting != 0) {
284  // Start the receives
285  for (const auto entry : mesh.getGhostExchangeTargets()) {
286  const int rank = entry.first;
287  dataCommunicator.startRecv(rank);
288  }
289 
290  // Start the sends
291  for (const auto entry : mesh.getGhostExchangeSources()) {
292  const int rank = entry.first;
293  auto &sendIds = entry.second;
294  SendBuffer &buffer = dataCommunicator.getSendBuffer(rank);
295 
296  for (long cellId : sendIds) {
297  int sign = 0;
298  if (propagationStatus.at(cellId) == PROPAGATION_STATUS_REACHED) {
299  sign = getSign(cellId);
300  }
301  buffer << sign;
302  }
303 
304  dataCommunicator.startSend(rank);
305  }
306 
307  // Receive the sign and propagate the sign
308  //
309  // If we discover the sign of a ghost, we can use it as a seed.
310  seeds.clear();
311  int nCompletedRecvs = 0;
312  while (nCompletedRecvs < dataCommunicator.getRecvCount()) {
313  int rank = dataCommunicator.waitAnyRecv();
314  const auto &recvIds = mesh.getGhostExchangeTargets(rank);
315  RecvBuffer &buffer = dataCommunicator.getRecvBuffer(rank);
316 
317  // Receive data and detect new seeds
318  for (long cellId : recvIds) {
319  buffer >> sign;
320 
321  std::size_t cellRawId = cells.getRawIndex(cellId);
322  int &cellPropagationStatus = propagationStatus.rawAt(cellRawId);
323  if (cellPropagationStatus == PROPAGATION_STATUS_WAITING) {
324  if (sign != 0) {
325  setSign(cellId, sign);
326  cellPropagationStatus = PROPAGATION_STATUS_REACHED;
327  --nWaiting;
328  seeds.push_back(cellId);
329  }
330  } else if (cellPropagationStatus == PROPAGATION_STATUS_REACHED) {
331  assert(getSign(cellId) == sign);
332  }
333  }
334 
335  ++nCompletedRecvs;
336  }
337 
338  if (seeds.size() > 0) {
339  propagateSeedSign(seeds, &propagationStatus, &nWaiting, &externalSign);
340  }
341 
342  // Wait to the sends to finish
343  dataCommunicator.waitAllSends();
344 
345  // Update the global counter for cells with an unknow sign
346  nGlobalWaiting = nWaiting;
347  MPI_Allreduce(MPI_IN_PLACE, &nGlobalWaiting, 1, MPI_LONG, MPI_SUM, m_kernelPtr->getCommunicator());
348  }
349  }
350 
351  // Communicate the sign of the external region
352  //
353  // The sign has to be consistent among all the partitions.
354  if (mesh.isPartitioned()) {
355  int minExternalSign = externalSign;
356  MPI_Allreduce(MPI_IN_PLACE, &minExternalSign, 1, MPI_INT, MPI_MIN, m_kernelPtr->getCommunicator());
357 
358  int maxExternalSign = externalSign;
359  MPI_Allreduce(MPI_IN_PLACE, &maxExternalSign, 1, MPI_INT, MPI_MAX, m_kernelPtr->getCommunicator());
360 
361  if (minExternalSign == maxExternalSign) {
362  externalSign = minExternalSign;
363  } else if (minExternalSign == 0) {
364  externalSign = maxExternalSign;
365  } else if (maxExternalSign == 0) {
366  externalSign = minExternalSign;
367  }
368  }
369 #else
370  // Check that the sign has been propagated into all regions
371  assert(nWaiting == 0);
372 #endif
373 
374  // Check that the sign of the external region was correctly identified
375  if (nExternal != 0 && externalSign == 0) {
376  throw std::runtime_error("Sign of external region not properly identified!");
377  }
378 
379  // Assign the sign to the external cells
380  for (auto itr = cellBegin; itr != cellEnd; ++itr) {
381  std::size_t cellRawId = itr.getRawIndex();
382  if (propagationStatus.rawAt(cellRawId) != PROPAGATION_STATUS_EXTERNAL) {
383  continue;
384  }
385 
386  long cellId = itr.getId();
387  setSign(cellId, externalSign);
388  }
389 }

412 void LevelSetCachedObject::propagateSeedSign(const std::vector<long> &seeds,
413  PiercedStorage<int, long> *statuses,
414  long *nWaiting, int *externalSign) {
415 
416  VolumeKernel const &mesh = *(m_kernelPtr->getMesh()) ;
417 
418  std::vector<long> processList;
419 
420  std::size_t seedCursor = seeds.size();
421  while (seedCursor != 0) {
422  // Get a seed
423  --seedCursor;
424  long seedId = seeds[seedCursor];
425 
426  // Get the sign of the seed
427  int seedSign = getSign(seedId);
428 
429  // Initialize the process list with the seed
430  processList.resize(1);
431  processList[0] = seedId;
432 
433  // Propagate the sign
434  while (!processList.empty()) {
435  long cellId = processList.back();
436  processList.resize(processList.size() - 1);
437 
438  // Process non-seed cells
439  bool isSeed = (cellId == seedId);
440  if (!isSeed) {
441  // Consider only cells waiting for the propagation to reach them
442  //
443  // The process list is not unique, it can contain duplicate, so
444  // we need to make sure not to process a cell multiple times.
445  int &cellStatus = statuses->at(cellId);
446  if (cellStatus != PROPAGATION_STATUS_WAITING) {
447  continue;
448  }
449 
450  // Set the sign of the cell
451  setSign(cellId, seedSign);
452  cellStatus = PROPAGATION_STATUS_REACHED;
453  --(*nWaiting);
454 
455  // If there are no more waiting cell we can exit
456  if (*nWaiting == 0) {
457  return;
458  }
459  }
460 
461  // Process cell neighbours
462  //
463  // If a neighbour is waiting for the propagation, add it to the
464  // process list. When the propagation reaches an external cell
465  // the sign of the seed frow which the propagation started will
466  // be the sign of the external region.
467  const Cell &cell = mesh.getCell(cellId);
468  const long *cellNeighs = cell.getAdjacencies() ;
469  int nCellNeighs = cell.getAdjacencyCount() ;
470  for(int n = 0; n < nCellNeighs; ++n){
471  long neighId = cellNeighs[n] ;
472  if (neighId < 0) {
473  continue;
474  }
475 
476  int neighStatus = statuses->at(neighId);
477  if (neighStatus == PROPAGATION_STATUS_WAITING) {
478  processList.push_back(neighId);
479  } else if (neighStatus == PROPAGATION_STATUS_EXTERNAL) {
480  // If the sign of the external region was unknown it can
481  // be assigned, otherwise check if the current sign is
482  // consistent with the previously evaluated sign.
483  if (*externalSign == 0) {
484  *externalSign = seedSign;
485  } else if (*externalSign != seedSign) {
486  throw std::runtime_error("Mismatch in sign of external region!");
487  }
488  }
489  }
490  }
491  }

500 void LevelSetCachedObject::setSign(long id, int sign) {
501 
502  // The sign has to be set only if it is different from the default sign
503  if (sign == levelSetDefaults::SIGN) {
504  return;
505  }
506 
507  // Get the info associated to the id
508  PiercedVector<LevelSetInfo>::iterator infoItr = m_ls.find(id);
509  if (infoItr == m_ls.end()){
510  infoItr = m_ls.emplace(id);
511  }
512 
513  // Update the sign
514  (*infoItr).value = sign * levelSetDefaults::VALUE;
515 }
