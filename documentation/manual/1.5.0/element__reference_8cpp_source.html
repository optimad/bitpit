149 void ReferenceElementInfo::initializeFaceEdges(const std::vector<const ReferenceElementInfo *> &facesInfo)
150 {
151  faceEdges = std::vector<std::vector<int>>(nFaces);
152  for (int k = 0; k < nFaces; ++k) {
153  const ReferenceElementInfo &faceInfo = *(facesInfo[k]);
154 
155  int nFaceEdges = faceInfo.nFaces;
156  for (int i = 0; i < nFaceEdges; ++i) {
157  // Number of vertices of the edge associated to the face
158  std::size_t nFaceEdgeVertices = faceInfo.faceConnect[i].size();
159 
160  // Connectivity of the edge associated to the face
161  const std::vector<int> &localFaceEdgeConnect = faceInfo.faceConnect[i];
162 
163  std::vector<int> faceEdgeConnect(nFaceEdgeVertices);
164  for (std::size_t n = 0; n < nFaceEdgeVertices; ++n) {
165  int localVertexId = localFaceEdgeConnect[n];
166  int vertexId = faceConnect[k][localVertexId];
167 
168  faceEdgeConnect[n] = vertexId;
169  }
170 
171  // Search the edge that has the same connectivity of the face edge
172  for (int j = 0; j < nEdges; ++j) {
173  // If face edge and the guess edge have a different number of
174  // vertices, the two edge cannot be the same.
175  std::size_t nGuessEdgeVertices = edgeConnect[j].size();
176  if (nGuessEdgeVertices != nFaceEdgeVertices) {
177  continue;
178  }
179 
180  // If the connecitivity of the face edge and the one of the
181  // guess edge are the same, the two edges coincides.
182  const std::vector<int> commonVertices = utils::intersectionVector(faceEdgeConnect, edgeConnect[j]);
183  if (commonVertices.size() == nFaceEdgeVertices) {
184  faceEdges[k].push_back(j);
185  }
186  }
187  }
188 
189  assert((int) faceEdges[k].size() == nFaceEdges);
190  }
191 }

225 double Reference3DElementInfo::evalSurfaceArea(const std::array<double, 3> *vertexCoords) const
226 {
227  std::array<std::array<double, 3>, MAX_ELEM_VERTICES> faceVertexCoords;
228 
229  double area = 0;
230  for (int i = 0; i < nFaces; ++i) {
231  for (int n = 0; n < nFaces; ++n) {
232  faceVertexCoords[i] = vertexCoords[faceConnect[i][n]];
233  }
234 
235  const Reference2DElementInfo &faceInfo = static_cast<const Reference2DElementInfo &>(getInfo(face_type[i]));
236  area += faceInfo.evalArea(faceVertexCoords.data());
237  }
238 
239  return area;

273 double Reference3DElementInfo::evalPointDistance(const std::array<double, 3> &point, const std::array<double, 3> *vertexCoords) const
274 {
275  std::array<std::array<double, 3>, MAX_ELEM_VERTICES> faceVertexCoords;
276 
277  double distance = std::numeric_limits<double>::max();
278  for (int i = 0; i < nFaces; ++i) {
279  ElementType faceType = face_type[i];
280  for (int n = 0; n < nFaces; ++n) {
281  faceVertexCoords[i] = vertexCoords[faceConnect[i][n]];
282  }
283 
284  distance = std::min(ReferenceElementInfo::getInfo(faceType).evalPointDistance(point, faceVertexCoords.data()), distance);
285  }
286 
287  return distance;
288 }

656 double ReferenceHexahedronInfo::evalVolume(const std::array<double, 3> *vertexCoords) const
657 {
658  std::array<std::array<double, 3>, MAX_ELEM_VERTICES> pyramidVertexCoords;
659 
660  const ReferenceQuadInfo &quadInfo = static_cast<const ReferenceQuadInfo &>(getInfo(ElementType::QUAD));
661  const ReferencePyramidInfo &pyramidInfo = static_cast<const ReferencePyramidInfo &>(getInfo(ElementType::PYRAMID));
662 
663  // The centroid is the apex of all the pyramids
664  pyramidVertexCoords[4] = vertexCoords[0];
665  for (int i = 1; i < nVertices; ++i) {
666  pyramidVertexCoords[4] += vertexCoords[i];
667  }
668  pyramidVertexCoords[4] = pyramidVertexCoords[4] / double(nVertices);
669 
670  // Sum the volume of the pyramids having a face as the base and the
671  // centroid as the apex.
672  double volume = 0.;
673  for (int i = 0; i < nFaces; ++i) {
674  for (int n = 0; n < quadInfo.nVertices; ++n) {
675  pyramidVertexCoords[quadInfo.nVertices - n - 1] = vertexCoords[faceConnect[i][n]];
676  }
677 
678  volume += pyramidInfo.evalVolume(pyramidVertexCoords.data());
679  }
680 
681  return volume;
682 }

827 double ReferencePyramidInfo::evalVolume(const std::array<double, 3> *vertexCoords) const
828 {
829  const std::array<double, 3> RA = vertexCoords[0] - vertexCoords[4];
830  const std::array<double, 3> DB = vertexCoords[3] - vertexCoords[1];
831  const std::array<double, 3> AC = vertexCoords[2] - vertexCoords[0];
832  const std::array<double, 3> AD = vertexCoords[1] - vertexCoords[0];
833  const std::array<double, 3> AB = vertexCoords[3] - vertexCoords[0];
834 
835  double volume = dotProduct(RA, crossProduct(DB, AC)) / 6. + dotProduct(AC, crossProduct(AD, AB)) / 12.;
836 
837  return volume;
838 }

966 double ReferenceWedgeInfo::evalVolume(const std::array<double, 3> *vertexCoords) const
967 {

969  std::array<int, 3> quadFaces = {{2, 3, 4}};
970 
971  std::array<std::array<double, 3>, MAX_ELEM_VERTICES> pyramidVertexCoords;
972  std::array<std::array<double, 3>, MAX_ELEM_VERTICES> tetraVertexCoords;

978

980  pyramidVertexCoords[4] = vertexCoords[0];
981  for (int i = 1; i < nVertices; ++i) {
982  pyramidVertexCoords[4] += vertexCoords[i];
983  }
984  pyramidVertexCoords[4] = pyramidVertexCoords[4] / double(nVertices);
985 
986  // The centroid is the apex of all the tetras
987  tetraVertexCoords[3] = pyramidVertexCoords[4];
988

990  // centroid as the apex.
991  double volume = 0.;
992  for (int i : triaFaces) {
993  for (int n = 0; n < triaInfo.nVertices; ++n) {
994  tetraVertexCoords[triaInfo.nVertices - n - 1] = vertexCoords[faceConnect[i][n]];
995  }
996

998  }
999 
1000  for (int i : quadFaces) {
1001  for (int n = 0; n < quadInfo.nVertices; ++n) {
1002  pyramidVertexCoords[quadInfo.nVertices - n - 1] = vertexCoords[faceConnect[i][n]];
1003  }
1004

1006  }
