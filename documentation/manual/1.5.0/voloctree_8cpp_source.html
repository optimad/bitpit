420 void VolOctree::setBoundingBox()
421 {
422  std::array<double, 3> minPoint;
423  std::array<double, 3> maxPoint;
424 
425  // Get the bounding box from the tree
426  m_tree->getBoundingBox(minPoint, maxPoint);
427 
428  // The tree is only evaluating the bounding box of the internal octants,
429  // we need to consider also ghosts cells.
430  for (auto ghostItr = ghostBegin(); ghostItr != ghostEnd(); ++ghostItr) {
431  ConstProxyVector<long> ghostVertexIds = ghostItr->getVertexIds();
432  int nGhostVertices = ghostVertexIds.size();
433  for (int i = 0; i < nGhostVertices; ++i) {
434  const std::array<double, 3> coords = m_vertices[ghostVertexIds[i]].getCoords();
435  for (int d = 0; d < 3; ++d) {
436  minPoint[d] = std::min(coords[d], minPoint[d]);
437  maxPoint[d] = std::max(coords[d], maxPoint[d]);
438  }
439  }
440  }
441 
442  // Set the bounding box
443  setBoundingBox(minPoint, maxPoint);
444 }

719 std::vector<adaption::Info> VolOctree::_adaptionPrepare(bool trackAdaption)
720 {
721  BITPIT_UNUSED(trackAdaption);
722 
723  if (getSpawnStatus() == SPAWN_NEEDED) {
724  throw std::runtime_error ("The initial import has not been performed.");
725  }
726 
727  // Call pre-adapt routine
728  m_tree->preadapt();
729 
730  // Track adaption changes
731  adaption::InfoCollection adaptionData;
732  if (trackAdaption) {
733  // Current rank
734  int currentRank = -1;
735 #if BITPIT_ENABLE_MPI==1
736  currentRank = getRank();
737 #endif
738 
739  // Track internal and ghosts octants that will be coarsend/refined
740  std::vector<uint32_t> treeIds;
741  std::vector<int8_t> treeMarkers;
742  std::vector<bool> treeGhosts;
743  m_tree->getPreMapping(treeIds, treeMarkers, treeGhosts);
744 
745  std::size_t n = 0;
746  std::size_t nUpdatedOctants = treeIds.size();
747  while (n < nUpdatedOctants){
748  int8_t marker = treeMarkers[n];
749  int nAdaptionOctants;
750  adaption::Type adaptionType;
751  if (marker > 0) {
752  nAdaptionOctants = 1;
753  adaptionType = adaption::TYPE_REFINEMENT;
754  } else {
755  nAdaptionOctants = pow(2, getDimension());
756  adaptionType = adaption::TYPE_COARSENING;
757  }
758 
759  std::size_t adaptionInfoId = adaptionData.create(adaptionType, adaption::ENTITY_CELL, currentRank);
760  adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
761  adaptionInfo.previous.reserve(nAdaptionOctants);
762  for (int k = 0; k < nAdaptionOctants; ++k) {
763  uint32_t treeId = treeIds[n];
764  bool isghost = treeGhosts[n];
765  OctantInfo octantInfo(treeId, !isghost);
766 
767  adaptionInfo.previous.emplace_back();
768  long &cellId = adaptionInfo.previous.back();
769  cellId = getOctantId(octantInfo);
770 
771  n++;
772  }
773  }
774 
775 #if BITPIT_ENABLE_MPI==1
776  // Ghost cells will be removed
777  if (isPartitioned() && getGhostCount() > 0) {
778  std::size_t adaptionInfoId = adaptionData.create(adaption::TYPE_DELETION, adaption::ENTITY_CELL, currentRank);
779  adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
780  adaptionInfo.previous.reserve(getGhostCount());
781  for (auto itr = ghostBegin(); itr != ghostEnd(); ++itr) {
782  adaptionInfo.previous.emplace_back();
783  long &deletedGhostId = adaptionInfo.previous.back();
784  deletedGhostId = itr.getId();
785  }
786  }
787 #endif
788  }
789 
790  return adaptionData.dump();
791 }

849 std::vector<adaption::Info> VolOctree::sync(bool trackChanges)
850 {
851  log::cout() << ">> Syncing patch..." << std::endl;
852 
853  // Detect if we are in import-from-scratch mode
854  //
855  // In import-from-scratch mode we start form an empty patch and we need
856  // to import all the octants of the tree. If we are importing the tree
857  // from scratch there are no cells to delete/renumber.
858  bool importFromScratch = empty();
859 
860  // Last operation on the tree
861  ParaTree::Operation lastTreeOperation = m_tree->getLastOperation();
862  if (lastTreeOperation == ParaTree::OP_ADAPT_UNMAPPED && !importFromScratch) {
863  throw std::runtime_error ("Unable to sync the patch after an unmapped adaption");
864  }
865 
866  // Info on the tree
867  long nOctants = m_tree->getNumOctants();
868  long nPreviousOctants = m_octantToCell.size();
869 
870  log::cout() << ">> Number of octants : " << nOctants << std::endl;
871 
872  // Info on the tree
873  long nGhostsOctants = m_tree->getNumGhosts();
874  long nPreviousGhosts = m_ghostToCell.size();
875 
876  // Initialize tracking data
877  adaption::InfoCollection adaptionData;
878 
879  // Current rank
880  int currentRank = -1;
881 #if BITPIT_ENABLE_MPI==1
882  currentRank = getRank();
883 #endif
884 
885  // Extract information for transforming the patch
886  //
887  // If there are no cells in the mesh we need to import all
888  // octants.
889  log::cout() << ">> Extract information for transforming the patch...";
890 
891  std::vector<bool> unmappedOctants(nPreviousOctants, true);
892  std::vector<OctantInfo> addedOctants;
893  std::vector<RenumberInfo> renumberedOctants;
894  std::vector<DeleteInfo> deletedOctants;
895 
896  addedOctants.reserve(nOctants + nGhostsOctants);
897  if (!importFromScratch) {
898  renumberedOctants.reserve(nPreviousOctants + nPreviousGhosts);

1735 void VolOctree::updateAdjacencies(const std::vector<long> &cellIds, bool resetAdjacencies)
1736 {

1738  int nCellFaces = 2 * getDimension();

1741

1744  for (long cellId : cellIds) {
1745  m_cells[cellId].resetAdjacencies();

1748

1750  int maxLevel = m_tree->getMaxDepth();

1752  std::vector<std::vector<long>> hierarchicalCellIds(maxLevel + 1);
1753  for (int level = 0; level <= maxLevel; ++level) {
1754  hierarchicalCellIds[level].reserve(averageSize);
1755  }
1756 
1757  for (long cellId : cellIds) {

1759  hierarchicalCellIds[cellLevel].push_back(cellId);
1760  }

1764  processedFaces.reserve(cellIds.size() * getDimension());

1771  FaceInfo currentFaceInfo(cellId, face);

1773  continue;

1778  std::vector<bool> neighGhostFlags;
1779  if (octantInfo.internal) {
1780  m_tree->findNeighbours(octantInfo.id, face, 1, neighTreeIds, neighGhostFlags);
1781  } else {

1783  }

1786  //
1787  // Adjacencies will processed twice, once while processing the
1788  // current cell, and once while processing the neighbour cell.
1789  // However they will be set only once, because the function that
1790  // insert the adjacency in the cell will insert only unique
1791  // adjacencies.
1792  int nNeighs = neighTreeIds.size();

1803  neigh.pushAdjacency(neighFace, cellId);
1804

1806  processedFaces.insert(neighFaceInfo);
1807  }
1808  }
1809  }
