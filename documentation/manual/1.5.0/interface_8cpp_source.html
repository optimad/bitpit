193 std::array<std::array<double, 3>, 3> Interface::evalRotationFromCartesian(std::array<double, 3> &versor)
194 {
195  // The rotation matrix has in its rows the versors that define
196  // the interface coordinate system.
197  //
198  // | [x_int] |
199  // R = | [y_int] |
200  // | [z_int] |
201  //
202  std::array<std::array<double, 3>, 3> R;
203 
204  // x-interface axis
205  for (int k = 0; k < 3; ++k) {
206  R[0][k] = versor[k];
207  }
208 
209  // y-interface axis
210  if (std::abs(std::abs(versor[2]) - 1.) > 1e-8) {
211  std::array<double, 3> z = {{0.0, 0.0, 1.0}};
212  R[1] = crossProduct(z, R[0]);
213  } else {
214  std::array<double, 3> x = {{1.0, 0.0, 0.0}};
215  R[1] = crossProduct(x, R[0]);
216  }
217  R[1] = R[1] / norm2(R[1]);
218 
219  // z-interface axis
220  R[2] = crossProduct(R[0], R[1]);
221  R[2] = R[2] / norm2(R[2]);
222 
223  return R;
224 }

374 void Interface::display(std::ostream &out, unsigned short int indent) const
375 {
376  std::string t_s = std::string(indent, ' ');
377 
378  // If the type is unknown there are no information to display
379  if (getType() == ElementType::UNDEFINED) {
380  out << t_s << "interface type: (unknown)" << std::endl;
381  return;
382  }
383 
384  // General info ----------------------------------------------------- //
385  out << t_s << "interface type: " << getType() << std::endl;
386  out << t_s << "ID: " << getId() << std::endl;
387  out << t_s << "is border: ";
388  if (getNeigh() >= 0) { out << "(false)"; }
389  else { out << "(true)"; }
390  out << std::endl;
391 
392  // Connectivity infos --------------------------------------------------- //
393  int nVertices = getVertexCount();
394  ConstProxyVector<long> cellVertexIds = getVertexIds();
395  out << t_s << "connectivity: [ ";
396  for (int i = 0; i < nVertices - 1; ++i) {
397  out << cellVertexIds[i] << ", ";
398  } //next i
399  out << cellVertexIds[nVertices - 1] << " ]" << std::endl;
400 
401  // Onwer infos ---------------------------------------------------------- //
402  out << t_s << "onwer ID: " << getOwner() << std::endl;
403  out << t_s << "owner face: " << getOwnerFace() << std::endl;
404 
405  // Onwer infos ---------------------------------------------------------- //
406  if (getNeigh() >= 0) {
407  out << t_s << "neighbour ID: " << getNeigh() << std::endl;
408  out << t_s << "neighbour face: " << getNeighFace() << std::endl;
409  }
410 }
