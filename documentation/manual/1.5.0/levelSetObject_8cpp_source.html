204 LevelSetIntersectionStatus LevelSetObject::intersectSurface(const long &i, LevelSetIntersectionMode mode) const{
205 
206  double incircle, circumcircle;
207 
208  switch(mode){
209  case LevelSetIntersectionMode::FAST_GUARANTEE_TRUE:
210  {
211  incircle = m_kernelPtr->computeCellIncircle(i) ;
212  if(std::abs(getLS(i)) <= incircle){
213  return LevelSetIntersectionStatus::TRUE;
214  } else {
215  return LevelSetIntersectionStatus::FALSE;
216  }
217 
218  break;
219  }
220 
221  case LevelSetIntersectionMode::FAST_GUARANTEE_FALSE:
222  {
223  circumcircle = m_kernelPtr->computeCellCircumcircle(i) ;
224  if(std::abs(getLS(i)) > circumcircle){
225  return LevelSetIntersectionStatus::FALSE;
226  } else {
227  return LevelSetIntersectionStatus::TRUE;
228  }
229 
230  break;
231  }
232 
233  case LevelSetIntersectionMode::FAST_FUZZY:
234  {
235  circumcircle = m_kernelPtr->computeCellCircumcircle(i) ;
236  if(std::abs(getLS(i)) > circumcircle){
237  return LevelSetIntersectionStatus::FALSE;
238  }
239 
240  incircle = m_kernelPtr->computeCellIncircle(i) ;
241  if(std::abs(getLS(i)) <= incircle){
242  return LevelSetIntersectionStatus::TRUE;
243  }
244 
245  return LevelSetIntersectionStatus::CLOSE;
246 
247  break;
248  }
249 
250  case LevelSetIntersectionMode::ACCURATE:
251  {
252  circumcircle = m_kernelPtr->computeCellCircumcircle(i) ;
253  if(std::abs(getLS(i)) > circumcircle){
254  return LevelSetIntersectionStatus::FALSE;
255  }
256 
257  incircle = m_kernelPtr->computeCellIncircle(i) ;
258  if(std::abs(getLS(i)) <= incircle){
259  return LevelSetIntersectionStatus::TRUE;
260  }
261 
262  std::array<double,3> root = computeProjectionPoint(i);
263  std::array<double,3> normal = getNormal(i);
264  if( m_kernelPtr->intersectCellPlane(i,root,normal) ){
265  return LevelSetIntersectionStatus::TRUE;
266  } else {
267  return LevelSetIntersectionStatus::FALSE;
268  }
269 
270  break;
271  }
272  }
273 
274  BITPIT_UNREACHABLE("cannot reach");
275 
276 }

384 void LevelSetObject::enableVTKOutput( LevelSetWriteField writeField, bool enable) {
385 
386  std::vector<LevelSetWriteField> fields;
387 
388  if( writeField==LevelSetWriteField::ALL){
389  fields.push_back(LevelSetWriteField::VALUE);
390  fields.push_back(LevelSetWriteField::GRADIENT);
391  fields.push_back(LevelSetWriteField::NORMAL);
392  fields.push_back(LevelSetWriteField::PART);
393 
394  } else if ( writeField==LevelSetWriteField::DEFAULT){
395  fields.push_back(LevelSetWriteField::VALUE);
396  fields.push_back(LevelSetWriteField::GRADIENT);
397 
398  } else {
399  fields.push_back(writeField);
400  }
401 
402  for( LevelSetWriteField &field : fields){
403 
404  stringstream name;
405  name << "ls_";
406  name << getId();
407  name << "_";
408 
409 
410  switch(field){
411  case LevelSetWriteField::VALUE:
412  name << "value";
413  if(enable){
414  m_kernelPtr->getMesh()->getVTK().addData<double>( name.str(), VTKFieldType::SCALAR, VTKLocation::CELL, this);
415  } else {
416  m_kernelPtr->getMesh()->getVTK().removeData( name.str());
417  }
418  break;
419 
420  case LevelSetWriteField::GRADIENT:
421  name << "gradient";
422  if(enable){
423  m_kernelPtr->getMesh()->getVTK().addData<double>( name.str(), VTKFieldType::VECTOR, VTKLocation::CELL, this);
424  } else {
425  m_kernelPtr->getMesh()->getVTK().removeData( name.str());
426  }
427  break;
428 
429  case LevelSetWriteField::NORMAL:
430  name << "normal";
431  if(enable){
432  m_kernelPtr->getMesh()->getVTK().addData<double>( name.str(), VTKFieldType::VECTOR, VTKLocation::CELL, this);
433  } else {
434  m_kernelPtr->getMesh()->getVTK().removeData( name.str());
435  }
436  break;
437 
438  case LevelSetWriteField::PART:
439  name << "partId";
440  if(enable){
441  m_kernelPtr->getMesh()->getVTK().addData<int>( name.str(), VTKFieldType::SCALAR, VTKLocation::CELL, this);
442  } else {
443  m_kernelPtr->getMesh()->getVTK().removeData( name.str());
444  }
445  break;
446 
447  default:
448  throw std::runtime_error ("Unsupported value of field in LevelSetObject::addDataToVTK() ");
449  break;
450 
451  }
452  }

467 
468  if(utils::string::keywordInString(name,"value")){

570 void LevelSetObject::communicate( const std::unordered_map<int,std::vector<long>> &sendList,
571  const std::unordered_map<int,std::vector<long>> &recvList,
572  std::vector<adaption::Info> const *mapper){
573 
574  if (!m_kernelPtr->getMesh()->isPartitioned()) {
575  return;
576  }
577 
578  // Create a data communicator
579  MPI_Comm meshComm = m_kernelPtr->getCommunicator() ;
580  DataCommunicator dataCommunicator(meshComm) ;
581 
582  // Fill the send buffer with the content from the LevelSetObject base
583  // class and the specific derived class.
584  for (const auto entry : sendList) {
585  // Create an empty send
586  int rank = entry.first;
587  dataCommunicator.setSend(rank, 0);
588 
589  // Write data in the buffer
590  SendBuffer &buffer = dataCommunicator.getSendBuffer(rank);
591  writeCommunicationBuffer(entry.second, buffer);
592  }
593 
594  // Discover the receives
595  dataCommunicator.discoverRecvs();
596 
597  // Start the sends
598  dataCommunicator.startAllRecvs();
599 
600  // In case of mesh adaption, the ids of the moved items must be freed
601  // in order to avoid conflicts when ids are recycled
602  if (mapper) {
603  clearAfterMeshAdaption(*mapper);
604  }
605 
606  // Start the sends
607  dataCommunicator.startAllSends();
608 
609  // Check which data communications have arrived. For those which are
610  // available start reading the databuffer into the data structure of
611  // LevelSetObject and its derived classes.
612  int nCompletedRecvs = 0;
613  while (nCompletedRecvs < dataCommunicator.getRecvCount()) {
614  int rank = dataCommunicator.waitAnyRecv();
615 
616  RecvBuffer &dataBuffer = dataCommunicator.getRecvBuffer(rank);
617  readCommunicationBuffer(recvList.at(rank), dataBuffer);
618 
619  ++nCompletedRecvs;
620  }
621 
622  dataCommunicator.waitAllSends();
623  dataCommunicator.finalize();
624 }
