159 void MeshMapper::mappingAdaptionUpdate(const std::vector<adaption::Info> & infoAdapt, bool reference, bool fillInv)
160 {
161 
162  VolOctree* meshAdapted;
163  PiercedStorage<adaption::Info>* mapperAdapted;
164  VolOctree* meshMapped;
165  PiercedStorage<adaption::Info>* mapperMapped;
166  if (reference){
167  meshAdapted = dynamic_cast<VolOctree*>(m_referenceMesh);
168  mapperAdapted = &m_mapper;
169  meshMapped = dynamic_cast<VolOctree*>(m_mappedMesh);
170  mapperMapped = &m_invmapper;
171  }
172  else{
173  meshMapped = dynamic_cast<VolOctree*>(m_referenceMesh);
174  mapperAdapted = &m_invmapper;
175  meshAdapted = dynamic_cast<VolOctree*>(m_mappedMesh);
176  mapperMapped = &m_mapper;
177  }
178 
179  assert(meshAdapted != nullptr);
180  assert(meshMapped != nullptr);
181 
182  for (const adaption::Info & info : infoAdapt){
183  if (info.type == adaption::Type::TYPE_RENUMBERING){
184  long id = info.current[0];
185  (*mapperAdapted)[id].current.clear();
186  (*mapperAdapted)[id].previous.clear();
187 
188  (*mapperAdapted)[id].current.push_back(id);
189  (*mapperAdapted)[id].type = m_previousmapper[info.previous[0]].type;
190  (*mapperAdapted)[id].entity = adaption::Entity::ENTITY_CELL;
191  (*mapperAdapted)[id].previous = m_previousmapper[info.previous[0]].previous;
192 
193  if (fillInv){
194  for (long idp : (*mapperAdapted)[id].previous){
195  std::vector<long>::iterator it = std::find((*mapperMapped)[idp].previous.begin(), (*mapperMapped)[idp].previous.end(), info.previous[0]);
196  assert(it != (*mapperMapped)[idp].previous.end());
197  *it = id;
198  }
199  }
200  }
201  if (info.type == adaption::Type::TYPE_REFINEMENT){
202  std::size_t iprevious = 0;
203  long idprevious = m_previousmapper[info.previous[0]].previous[iprevious];
204  VolOctree::OctantInfo oinfoprev = meshMapped->getCellOctant(idprevious);
205  uint64_t morton = meshMapped->getTree().getMorton(oinfoprev.id);
206  for (const long & id : info.current){
207  (*mapperAdapted)[id].current.clear();
208  (*mapperAdapted)[id].previous.clear();
209  (*mapperAdapted)[id].entity = adaption::Entity::ENTITY_CELL;
210  (*mapperAdapted)[id].current.push_back(id);
211  uint8_t level = meshAdapted->getCellLevel(id);
212  if (level == meshMapped->getCellLevel(idprevious)){
213  (*mapperAdapted)[id].type = adaption::Type::TYPE_RENUMBERING;
214  (*mapperAdapted)[id].previous.push_back(idprevious);
215 
216  if (fillInv){
217  (*mapperMapped)[idprevious].type = adaption::Type::TYPE_RENUMBERING;
218  (*mapperMapped)[idprevious].previous.clear();
219  (*mapperMapped)[idprevious].previous.push_back(id);
220  }
221  }
222  else if (level > meshMapped->getCellLevel(idprevious)){
223  (*mapperAdapted)[id].type = adaption::Type::TYPE_REFINEMENT;
224  (*mapperAdapted)[id].previous.push_back(idprevious);
225 
226  if (fillInv){
227  std::vector<long>::iterator it = std::find((*mapperMapped)[idprevious].previous. begin(), (*mapperMapped)[idprevious].previous.end(), info.previous[0]);
228  if (it != (*mapperMapped)[idprevious].previous.end()){
229  *it = id;
230  }
231  else{
232  (*mapperMapped)[idprevious].previous.push_back(id);
233  }
234  (*mapperMapped)[idprevious].type = adaption::Type::TYPE_COARSENING;
235  }
236  }
237  else if (level < meshMapped->getCellLevel(idprevious)){
238  (*mapperAdapted)[id].type = adaption::Type::TYPE_COARSENING;
239  VolOctree::OctantInfo oinfo = meshAdapted->getCellOctant(id);
240  uint64_t mortonlastdesc = meshAdapted->getTree().getLastDescMorton(oinfo.id);
241  while (morton <= mortonlastdesc && iprevious < m_previousmapper[info.previous[0]].previous.size()){
242 
243  idprevious = m_previousmapper[info.previous[0]].previous[iprevious];
244  oinfoprev = meshMapped->getCellOctant(idprevious);
245  morton = meshMapped->getTree().getMorton(oinfoprev.id);
246  (*mapperAdapted)[id].previous.push_back(idprevious);
247 
248  if (fillInv){
249  (*mapperMapped)[idprevious].type = adaption::Type::TYPE_REFINEMENT;
250  (*mapperMapped)[idprevious].previous[0] = id;
251  }
252 
253  iprevious++;
254  }
255  }
256  }
257  }
258  if (info.type == adaption::Type::TYPE_COARSENING){
259  long id = info.current[0];
260  (*mapperAdapted)[id].current.clear();
261  (*mapperAdapted)[id].previous.clear();
262  (*mapperAdapted)[id].entity = adaption::Entity::ENTITY_CELL;
263  (*mapperAdapted)[id].current.push_back(id);
264  uint8_t level = meshAdapted->getCellLevel(id);
265  std::unordered_set<long> idsprev;
266  for (const long & id_ : info.previous){
267  for (const long & idp : m_previousmapper[id_].previous){
268  idsprev.insert(idp);
269  }
270  }
271  if (idsprev.size() == 1){
272  long idprevious = *idsprev.begin();
273  if (level == meshMapped->getCellLevel(idprevious)){
274  (*mapperAdapted)[id].type = adaption::Type::TYPE_RENUMBERING;
275 
276  if (fillInv){
277  (*mapperMapped)[idprevious].type = adaption::Type::TYPE_RENUMBERING;
278  (*mapperMapped)[idprevious].previous.clear();
279  (*mapperMapped)[idprevious].previous.push_back(id);
280  }
281  }
282  else{
283  (*mapperAdapted)[id].type = adaption::Type::TYPE_REFINEMENT;
284 
285  if (fillInv){
286  std::vector<long>::iterator it = std::find((*mapperMapped)[idprevious].previous. begin(), (*mapperMapped)[idprevious].previous.end(), info.previous[0]);
287  assert(it != (*mapperMapped)[idprevious].previous.end());
288  *it = id;
289  (*mapperMapped)[idprevious].type = adaption::Type::TYPE_COARSENING;
290  }
291 
292  }
293  }
294  else{
295  (*mapperAdapted)[id].type = adaption::Type::TYPE_COARSENING;
296 
297  if (fillInv){
298  for (const long & idprevious : idsprev){
299  (*mapperMapped)[idprevious].type = adaption::Type::TYPE_REFINEMENT;
300  (*mapperMapped)[idprevious].previous[0] = id;
301  }
302  }
303 
304  }
305  for (const long & id_ : idsprev){
306  (*mapperAdapted)[id].previous.push_back(id_);
307  }
308  }
309  }

323 void MeshMapper::_mapMeshes(bitpit::VolOctree * meshReference, bitpit::VolOctree * meshMapped, bool fillInv)
324 {
325 
326  if ( (meshReference->getLength() != meshMapped->getLength()) || (meshReference->getOrigin() != meshMapped->getOrigin()) )
327  throw std::runtime_error ("mesh mapper: different domain of VolOctree meshes not allowed.");
328 
329  m_mapper.setDynamicKernel(&meshReference->getCells(), PiercedSyncMaster::SyncMode::SYNC_MODE_JOURNALED);
330  if (fillInv)
331  m_invmapper.setDynamicKernel(&meshMapped->getCells(), PiercedSyncMaster::SyncMode::SYNC_MODE_JOURNALED);
332  else
333  clearInverseMapping();
334 
335 #if BITPIT_ENABLE_MPI==1
336 
337  if (!(meshReference->isPartitioned())){
338 #endif
339 
340  _mapMeshesSamePartition(meshReference, meshMapped, fillInv);
341 
342 #if BITPIT_ENABLE_MPI==1
343  }
344  else{
345 
346  std::vector<uint64_t> partitionReference = meshReference->getTree().getPartitionLastDesc();
347  std::vector<uint64_t> partitionMapped = meshMapped->getTree().getPartitionLastDesc();
348 
349  bool checkPartition = (partitionReference[m_rank] == partitionMapped[m_rank]);
350  if (m_rank != 0)
351  checkPartition = checkPartition && (partitionReference[m_rank-1] == partitionMapped[m_rank-1]);
352 
353  if (checkPartition){
354  _mapMeshesSamePartition(meshReference, meshMapped, fillInv);
355  }
356  else{
357  throw std::runtime_error("Mapping with different partitions not supported.");
358  }
359 
360  }
361 #endif
362 
363 }

374 void MeshMapper::_mapMeshesSamePartition(bitpit::VolOctree * meshReference, bitpit::VolOctree * meshMapped, bool fillInv)
375 {
376 
377  long nRef = meshReference->getInternalCount();
378  long nMap = meshMapped->getInternalCount();
379 
380  long indRef = 0;
381  long indMap = 0;
382 
383  while (indRef < nRef && indMap < nMap){
384  VolOctree::OctantInfo octinfoRef(indRef, true);
385  long idRef = meshReference->getOctantId(octinfoRef);
386  VolOctree::OctantInfo octinfoMap(indMap, true);
387  long idMap = meshMapped->getOctantId(octinfoMap);
388 
389  m_mapper[idRef].entity = bitpit::adaption::Entity::ENTITY_CELL;
390  m_mapper[idRef].rank = m_rank;
391  if (fillInv){
392  m_invmapper[idMap].entity = bitpit::adaption::Entity::ENTITY_CELL;
393  m_invmapper[idMap].rank = m_rank;
394  }
395 
396  if (meshMapped->getCellLevel(idMap) == meshReference->getCellLevel(idRef)){
397  m_mapper[idRef].current.push_back(idRef);
398  m_mapper[idRef].previous.push_back(idMap);
399  m_mapper[idRef].type = bitpit::adaption::Type::TYPE_RENUMBERING;
400  if (fillInv){
401  m_invmapper[idMap].current.push_back(idMap);
402  m_invmapper[idMap].previous.push_back(idRef);
403  m_invmapper[idMap].type = bitpit::adaption::Type::TYPE_RENUMBERING;
404  }
405  indRef++;
406  indMap++;
407  }
408  else if (meshMapped->getCellLevel(idMap) > meshReference->getCellLevel(idRef)){
409  m_mapper[idRef].current.push_back(idRef);
410  m_mapper[idRef].type = bitpit::adaption::Type::TYPE_COARSENING;
411 
412  uint64_t mortonlastdesc = meshReference->getTree().getLastDescMorton(indRef);
413  uint64_t mortonMap = meshMapped->getTree().getMorton(indMap);
414 
415  while(mortonMap <= mortonlastdesc && indMap < nMap){
416  m_mapper[idRef].previous.push_back(idMap);
417  if (fillInv){
418  m_invmapper[idMap].current.push_back(idMap);
419  m_invmapper[idMap].type = bitpit::adaption::Type::TYPE_REFINEMENT;
420  m_invmapper[idMap].previous.push_back(idRef);
421  }
422  indMap++;
423  octinfoMap = VolOctree::OctantInfo(indMap, true);
424  idMap = meshMapped->getOctantId(octinfoMap);
425  mortonMap = meshMapped->getTree().getMorton(indMap);
426  }
427  indRef++;
428  }
429  else if (meshMapped->getCellLevel(idMap) < meshReference->getCellLevel(idRef)){
430 
431  uint64_t morton= meshReference->getTree().getMorton(indRef);
432  uint64_t mortonlastdescmesh = meshMapped->getTree().getLastDescMorton(indMap);
433 
434  if (fillInv){
435  m_invmapper[idMap].current.push_back(idMap);
436  m_invmapper[idMap].type = bitpit::adaption::Type::TYPE_COARSENING;
437  }
438 
439  while (morton <= mortonlastdescmesh && indRef < nRef){
440  m_mapper[idRef].current.push_back(idRef);
441  m_mapper[idRef].type = bitpit::adaption::Type::TYPE_REFINEMENT;
442  m_mapper[idRef].previous.push_back(idMap);
443  if (fillInv){
444  m_invmapper[idMap].previous.push_back(idRef);
445  }
446  indRef++;
447  morton = meshReference->getTree().getMorton(indRef);
448  octinfoRef = VolOctree::OctantInfo(indRef, true);
449  idRef = meshReference->getOctantId(octinfoRef);
450  }
451  indMap++;
452  }
453  }

464 void MeshMapper::initializeCommunicator(MPI_Comm communicator)
465 {
466  // Communication can be set just once
467  if (isCommunicatorSet())
468  throw std::runtime_error ("MeshMapper communicator can be set just once");
469 
470  // The communicator has to be valid
471  if (communicator == MPI_COMM_NULL)
472  throw std::runtime_error ("MeshMapper communicator is not valid");
473 
474  // Create a copy of the user-specified communicator
475  //
476  // No library routine should use MPI_COMM_WORLD as the communicator;
477  // instead, a duplicate of a user-specified communicator should always
478  // be used.
479  MPI_Comm_dup(communicator, &m_communicator);
480 }
