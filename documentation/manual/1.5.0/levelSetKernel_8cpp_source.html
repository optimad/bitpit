98 void LevelSetKernel::updateGeometryCache( const std::vector<adaption::Info> &mapper ) {
99 
100  // If there are no cells in the mesh we can just delete all the cache
101  if ( m_mesh->getCellCount() == 0) {
102  clearGeometryCache();
103  return;
104  }
105 
106  // Remove the previous cells from the cache
107  for ( auto & map : mapper ){
108  if( map.entity != adaption::Entity::ENTITY_CELL ){
109  continue;
110  }
111 
112  for ( auto & previousId : map.previous){
113  auto centroidItr = m_cellCentroids.find( previousId ) ;
114  if ( centroidItr == m_cellCentroids.end() ) {
115  continue ;
116  }
117 
118  m_cellCentroids.erase( previousId ) ;
119  }
120  }
121 
122 }

149 double LevelSetKernel::computeCellIncircle( long id ) {
150 
151  VolumeKernel *patch = getMesh();
152  Cell &cell = patch->getCell(id);
153 
154  const long* interfaceIds = cell.getInterfaces();
155  int interfaceCount = cell.getInterfaceCount();
156 
157  std::array<double,3> cellCenter = computeCellCentroid(id);
158 
159  double radius = std::numeric_limits<double>::max() ;
160  for (int k = 0; k < interfaceCount; ++k) {
161  long interfaceId = interfaceIds[k];
162  double r = norm2(cellCenter - patch->evalInterfaceCentroid(interfaceId));
163  radius = std::min(radius, r);
164  }
165 
166  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
167  int nCellVertices = cellVertexIds.size();
168  for (int k = 0; k < nCellVertices; ++k) {
169  long vertexId = cellVertexIds[k];
170  double r = norm2(cellCenter - patch->getVertexCoords(vertexId));
171  radius = std::min(radius, r);
172  }
173 
174  return radius;
175 
176 }

230 double LevelSetKernel::isCellInsideBoundingBox( long id, std::array<double, 3> minPoint, std::array<double, 3> maxPoint ){
231 
232  const Cell &cell = m_mesh->getCell(id);
233  double tolerance = m_mesh->getTol();
234 
235  std::array<double, 3> cellMinPoint;
236  std::array<double, 3> cellMaxPoint;
237 
238  cellMinPoint.fill( std::numeric_limits<double>::max() ) ;
239  cellMaxPoint.fill( - std::numeric_limits<double>::max() ) ;
240 
241  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
242  int nCellVertices = cellVertexIds.size();
243  for (int i = 0; i < nCellVertices; ++i) {
244  long vertexId = cellVertexIds[i];
245  std::array<double, 3> vertexCoords = m_mesh->getVertexCoords(vertexId);
246  for (int d = 0; d < 3; ++d) {
247  cellMinPoint[d] = std::min( vertexCoords[d] - tolerance, cellMinPoint[d]) ;
248  cellMaxPoint[d] = std::max( vertexCoords[d] + tolerance, cellMaxPoint[d]) ;
249  }
250  }
251 
252  return CGElem::intersectBoxBox(minPoint, maxPoint, cellMinPoint, cellMaxPoint);
253 }

260 void LevelSetKernel::initializeCommunicator()
261 {
262  // The communicator has to be valid
263  MPI_Comm communicator = m_mesh->getCommunicator();
264  if (communicator == MPI_COMM_NULL) {
265  throw std::runtime_error ("Levelset communicator is not valid");
266  }
267 
268  // Communication can be set just once
269  if (isCommunicatorSet()) {
270  int comparison;
271  MPI_Comm_compare(communicator, m_communicator, &comparison);
272  if (comparison == MPI_IDENT || comparison == MPI_CONGRUENT) {
273  return;
274  } else {
275  throw std::runtime_error ("Levelset communicator cannot be modified.");
276  }
277  }
278 
279  // Create a duplicate of the patch communicator
280  MPI_Comm_dup(communicator, &m_communicator);
281 }
