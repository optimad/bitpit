55 class bubble{
56 public:
57  double c[2];
58  double r;
59 };
60 
66 void run()
67 {
68  int iter = 0;
69 
71  PabloUniform pabloBB(2);
72 
74  pabloBB.setBalanceCodimension(1);
75  uint32_t idx = 0;
76  pabloBB.setBalance(idx,true);
77 
79  for (iter=1; iter<4; iter++){
80  pabloBB.adaptGlobalRefine();
81  }
82 
83 #if BITPIT_ENABLE_MPI==1
84 
85  pabloBB.loadBalance();
86 #endif
87 
89  time_t Time = time(NULL);
90  srand(Time);
91  if(pabloBB.getRank() == 0)
92  cout << "the seed = " << Time << endl;
93 
94 #if BITPIT_ENABLE_MPI==1
95  int nb = 50;
96 #else
97  int nb = 10;
98 #endif
99  vector<bubble> BB;
100  vector<bubble> BB0;
101  vector<double> DY;
102  vector<double> OM;
103  vector<double> AA;
104 
105  for (int i=0; i<nb; i++){
106  double randc[2];
107  randc[0] = 0.8 * double(rand()) / RAND_MAX + 0.1;
108  randc[1] = double(rand()) / RAND_MAX - 0.5;
109  double randr = 0.1 * double(rand()) / RAND_MAX + 0.02;
110  double dy = 0.005 + 0.05 * double(rand()) / RAND_MAX;
111  double omega = 0.5 * double(rand()) / RAND_MAX;
112  double aa = 0.15 * double(rand()) / RAND_MAX;
113  bubble bb;
114  bb.c[0] = randc[0];
115  bb.c[1] = randc[1];
116  bb.r = randr;
117  BB.push_back(bb);
118  BB0.push_back(bb);
119  DY.push_back(dy);
120  OM.push_back(omega);
121  AA.push_back(aa);
122  }
124  double t0 = 0;
125  double t = t0;
126  double Dt = 0.5;
127 
129  int itstart = 1;
130  int iterend = 200;
131 
133  for (iter=itstart; iter<iterend; iter++){
134  if(pabloBB.getRank()==0) cout << "iter " << iter << endl;
135  t += Dt;
136 
138  for (int i=0; i<nb; i++){
139  BB[i].c[0] = BB0[i].c[0] + AA[i]*cos(OM[i]*t);
140  BB[i].c[1] = BB[i].c[1]+ Dt*DY[i];
141  }
142 
144  bool adapt = true;
145  while (adapt){
146  octantIterator it, itend = pabloBB.getInternalOctantsEnd();
147  for (it=pabloBB.getInternalOctantsBegin(); it!=itend; ++it){
148  bool inside = false;
150  vector<array<double,3> > nodes = pabloBB.getNodes((*it));
152  array<double,3> center = pabloBB.getCenter((*it));
153  int ib = 0;
154  while (!inside && ib<nb){
155  double xc = BB[ib].c[0];
156  double yc = BB[ib].c[1];
157  double radius = BB[ib].r;
159  for (int j=0; j<4; j++){
160  double x = nodes[j][0];
161  double y = nodes[j][1];
162  if ( ((!inside) &&
163  (pow((x-xc),2.0)+pow((y-yc),2.0) <= 1.25*pow(radius,2.0) &&
164  pow((x-xc),2.0)+pow((y-yc),2.0) >= 0.75*pow(radius,2.0)))
165  || ((!inside) && (pow((center[0]-xc),2.0)+pow((center[1]-yc),2.0) <= 1.25*pow(radius,2.0) &&
166  pow((center[0]-xc),2.0)+pow((center[1]-yc),2.0) >= 0.75*pow(radius,2.0)))){
167  if (int(pabloBB.getLevel((*it))) < 9){
169  pabloBB.setMarker((*it),1);
170  }
171  else{
172  pabloBB.setMarker((*it),0);
173  }
174  inside = true;
175  }
176  }
177  ib++;
178  }
179  if (int(pabloBB.getLevel((*it))) > 0 && !inside){
181  pabloBB.setMarker((*it),5-pabloBB.getLevel((*it)));
182  }
183  }
184 
185  itend = pabloBB.getPboundOctantsEnd();
186  for (it=pabloBB.getPboundOctantsBegin(); it!=itend; ++it){
187  bool inside = false;
189  vector<array<double,3> > nodes = pabloBB.getNodes((*it));
191  array<double,3> center = pabloBB.getCenter((*it));
192  int ib = 0;
193  while (!inside && ib<nb){
194  double xc = BB[ib].c[0];
195  double yc = BB[ib].c[1];
196  double radius = BB[ib].r;
198  for (int j=0; j<4; j++){
199  double x = nodes[j][0];
200  double y = nodes[j][1];
201  if ( ((!inside) &&
202  (pow((x-xc),2.0)+pow((y-yc),2.0) <= 1.25*pow(radius,2.0) &&
203  pow((x-xc),2.0)+pow((y-yc),2.0) >= 0.75*pow(radius,2.0)))
204  || ((!inside) && (pow((center[0]-xc),2.0)+pow((center[1]-yc),2.0) <= 1.25*pow(radius,2.0) &&
205  pow((center[0]-xc),2.0)+pow((center[1]-yc),2.0) >= 0.75*pow(radius,2.0)))){
206  if (pabloBB.getLevel((*it)) < 9){
208  pabloBB.setMarker((*it),1);
209  }
210  else{
211  pabloBB.setMarker((*it),0);
212  }
213  inside = true;
214  }
215  }
216  ib++;
217  }
218  if (pabloBB.getLevel((*it)) > 0 && !inside){
220  pabloBB.setMarker((*it),5-pabloBB.getLevel((*it)));
221  }
222  }
223 
225  adapt = pabloBB.adapt();
226 
227  }
228 
229 #if BITPIT_ENABLE_MPI==1
230 
231  pabloBB.loadBalance();
232 #endif
233 
235  pabloBB.updateConnectivity();
236  pabloBB.write("PabloBubble_iter"+to_string(static_cast<unsigned long long>(iter)));
237 
238  }
239 }
