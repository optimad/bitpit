227 template<typename id_t>
228 typename PiercedKernel<id_t>::ResizeAction PiercedKernel<id_t>::_resize(std::size_t n)
229 {
230  // If the size of the kernel is already the requested size there is
231  // nothing to do.
232  if (n == size()) {
233  ResizeAction syncAction(ResizeAction::TYPE_NOOP);
234 
235  return syncAction;
236  }
237 
238  // A request for a size equal to 0 is equivalent to a clear.
239  if (n == 0) {
240  _clear();
241 
242  ResizeAction syncAction(ResizeAction::TYPE_CLEAR);
243 
244  return syncAction;
245  }
246 
247  // If the requested size is greater that the current size we may need to
248  // reserve space to allow the kernel to reach the requested size.
249  if (n > size()) {
250  _reserve(n);
251 
252  ResizeAction syncAction(ResizeAction::TYPE_RESERVE);
253  syncAction.info[PiercedSyncAction::INFO_SIZE] = rawSize();
254 
255  return syncAction;
256  }
257 
258  // If the requested size is smaller that the current size
259  // we need to perform a real resize.
260 
261  // Flush holes
262  holesFlush();
263 
264  // Find the id of the last element
265  id_t last_stored_id = getSizeMarker(n - 1);
266 
267  // Find the last position
268  std::size_t last_used_pos = getPos(last_stored_id);
269 
270  // Shrink the kernel
271  shrink(last_used_pos + 1);
272 
273  // Generate the sync action
274  ResizeAction syncAction(ResizeAction::TYPE_RESIZE);
275  syncAction.info[PiercedSyncAction::INFO_SIZE] = rawSize();
276 
277  return syncAction;
278 }

356 template<typename id_t>
357 typename PiercedKernel<id_t>::SortAction PiercedKernel<id_t>::_sort(std::size_t beginPos, std::size_t endPos)
358 {
359  // Get initial kernel size
360  std::size_t initialKernelRawSize = rawSize();
361 
362  // Squeeze the kernel
363  //
364  // After the squeeze there will be no holes in the kernel.
365  SqueezeAction squeezeAction = _squeeze();
366  const std::vector<std::size_t> &squeezePermutations = *(squeezeAction.data);
367  bool squeezed = (squeezeAction.data.get() != nullptr);
368 
369  // Get updated kernel size
370  std::size_t updatedKernelRawSize;
371  if (squeezed) {
372  updatedKernelRawSize = rawSize();
373  } else {
374  updatedKernelRawSize = initialKernelRawSize;
375  }
376 
377  // Update the sort range
378  if (squeezed) {
379  std::size_t updatedBeginPos = initialKernelRawSize;
380  std::size_t updatedEndPos = initialKernelRawSize;
381  for (std::size_t i = 0; i < initialKernelRawSize; ++i) {
382  std::size_t previousPos = squeezePermutations[i];
383  if (previousPos == beginPos) {
384  updatedBeginPos = i;
385  }
386  if (previousPos == endPos) {
387  updatedEndPos = i;
388  }
389 
390  if (updatedBeginPos != initialKernelRawSize && updatedEndPos != initialKernelRawSize) {
391  break;
392  }
393  }
394 
395  if (updatedBeginPos != initialKernelRawSize) {
396  beginPos = updatedBeginPos;
397  } else {
398  beginPos = updatedKernelRawSize;
399  }
400 
401  if (updatedEndPos != initialKernelRawSize) {
402  endPos = updatedEndPos;
403  } else {
404  endPos = updatedKernelRawSize;
405  }
406  }
407 
408  // Evaluates the sort permutations
409  std::vector<std::size_t> sortPermutations(updatedKernelRawSize);
410  for (std::size_t i = 0; i < updatedKernelRawSize; ++i) {
411  sortPermutations[i] = i;
412  }
413 
414  std::sort(sortPermutations.begin() + beginPos, sortPermutations.begin() + endPos, idLess(m_ids));
415 
416  // Create the sync action
417  SortAction syncAction;
418  syncAction.info[PiercedSyncAction::INFO_SIZE] = updatedKernelRawSize;
419  if (squeezed) {
420  syncAction.importData(std::vector<std::size_t>(squeezePermutations));
421  std::vector<std::size_t> &permutations = *(syncAction.data);
422  for (std::size_t i = beginPos; i < endPos; ++i) {
423  permutations[i] = squeezePermutations[sortPermutations[i]];
424  }
425  } else {
426  syncAction.importData(std::vector<std::size_t>(sortPermutations));
427  }
428 
429  // Sort the ids
430  //
431  // NOTE: the reorder function will destroy the permutation vector
432  utils::reorderVector<id_t>(sortPermutations, m_ids, updatedKernelRawSize);
433 
434  // Update the positions
435  m_pos.clear();
436  for (std::size_t i = 0; i < updatedKernelRawSize; ++i) {
437  m_pos[m_ids[i]] = i;
438  }
439 
440  // Return the permutations
441  return syncAction;
442 }

477 template<typename id_t>
478 typename PiercedKernel<id_t>::SqueezeAction PiercedKernel<id_t>::_squeeze()
479 {
480  // Flush changes
481  flush();
482 
483  // Get kernel size
484  std::size_t kernelSize = size();
485  std::size_t kernelRawSize = rawSize();
486 
487  // Initialize the sync action
488  SqueezeAction syncAction;
489  syncAction.info[PiercedSyncAction::INFO_SIZE] = kernelSize;
490 
491  // Compact the kernel
492  std::size_t nHoles = holesCount();
493  if (nHoles != 0) {
494  // Initialize sync data
495  syncAction.importData(std::vector<std::size_t>(kernelRawSize));
496  std::vector<std::size_t> &permutations = *(syncAction.data);
497 
498  // Move the elements
499  std::size_t firstPosToUpdate;
500  if (m_begin_pos == 0) {
501  firstPosToUpdate = m_holes[m_holes_regular_end - 1];
502  } else {
503  firstPosToUpdate = 0;
504  }
505 
506  for (std::size_t pos = 0; pos < firstPosToUpdate; ++pos) {
507  permutations[pos] = pos;
508  }
509 
510  std::size_t offset = 0;
511  for (std::size_t pos = firstPosToUpdate; pos < m_end_pos; pos++) {
512  if (offset < nHoles && m_holes[m_holes_regular_end - offset - 1] == pos) {
513  permutations[m_end_pos - 1 - offset] = pos;
514  ++offset;
515  continue;
516  }
517 
518  id_t id = m_ids[pos];
519  std::size_t updatedPos = pos - offset;
520 
521  setPosId(updatedPos, id);
522  setPosEmptyId(pos, pos + 1);
523  permutations[updatedPos] = pos;
524  }
525 
526  for (std::size_t pos = m_end_pos; pos < rawSize(); ++pos) {
527  permutations[pos] = pos;
528  }
529 
530  // Clear the holes
531  holesClear();
532 
533  // Reset begin and end
534  setBeginPos(0);
535  setEndPos(size());
536 
537  // Shrink the kernel
538  shrink(size(), true);
539  }
540 
541  // Shrink to fit
542  _shrinkToFit();
543 
544  return syncAction;
545 }

874 {

876

878  std::size_t flat = pos;
879 
880  // Subtract pending holes before position
881  if (holesCountPending() > 0) {
882  holes_const_iterator pending_begin_itr = m_holes.cbegin() + m_holes_pending_begin;
883  holes_const_iterator pending_end_itr = m_holes.cbegin() + m_holes_pending_end;

886  for (auto itr = pending_begin_itr; itr != pending_end_itr; ++itr) {
887  if (*itr <= pos) {
888  ++nHolesBefore;
889  }
890  }
891 
892  flat -= nHolesBefore;
893  }
894 
895  // Subtract regular holes before position
896  if (holesCountRegular() > 0) {
897  holes_const_iterator regular_begin_itr = m_holes.cbegin() + m_holes_regular_begin;

903  ++nHolesBefore;
904  }
905  }
906 
907  flat -= nHolesBefore;
908  }

969 template<typename id_t>
970 id_t PiercedKernel<id_t>::getSizeMarker(std::size_t targetSize, const id_t &fallback)
971 {
972  // If the size is zero, we return the first element, if the target
973  // size is equal to the size minus one we return the last element,
974  // if the target size is greater or equal the current kernel size
975  // we return the fallback value.
976  if (targetSize >= size()) {
977  return fallback;
978  } else if (targetSize == 0) {
979  return m_ids[m_begin_pos];
980  } else if (targetSize == (size() - 1)) {
981  return m_ids[m_end_pos - 1];
982  }
983 
984  // Sort the holes
985  holesSortRegular();
986  holesSortPending();
987 
988  // Iterate to find the position before wihch there is the
989  // requeste number of element.
990  holes_iterator regular_begin_itr = m_holes.begin() + m_holes_regular_begin;
991  holes_iterator regular_hole_itr = m_holes.begin() + m_holes_regular_end;
992 
993  holes_iterator pending_begin_itr = m_holes.begin() + m_holes_pending_begin;
994  holes_iterator pending_hole_itr = m_holes.begin() + m_holes_pending_end;
995 
996  std::size_t nEmpties = 0;
997  std::size_t markerPos = targetSize;
998  while (true) {
999  if (isPosEmpty(markerPos)) {
1000  markerPos = findNextUsedPos(markerPos - 1);
1001  }
1002 
1003  // Count the number of holes and pending deletes before the
1004  // current marker position
1005  if (regular_hole_itr != regular_begin_itr) {
1006  holes_iterator itr_previous = regular_hole_itr;

1365

1367  std::size_t pos = m_holes[hole];
1368

1370  setPosId(pos, id);

1925  return;
1926  }
1927

1929  //
1930  // The list of pending holes is sorted, in this way we can iterate
1931  // from the last pending hole in the kernel to the first one.
1932  // We start updating the id of the last pending hole, updating also
1933  // the id of the contiguous holes before that one. Then we advance
1934  // to the next hole, skipping the positions that have already been
1935  // updated.

1939  std::size_t pos = m_end_pos;
