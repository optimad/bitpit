218  std::vector< std::vector< T > > &A,

228

233 // CHECK INPUT //
234 // ========================================================================== //

236 if (m == 0) { return; }
237 n = A[0].size();
238 if (n == 0) { return; }
239 i--; j--;
240 if ((i >= m) || (i < 0)) {
241  return;
242 }
243 if ((j >= n) || (j < 0)) {
244  return;
245 }
246 
247 // ========================================================================== //
248 // EXTRACT COMPLEMENT //
249 // ========================================================================== //
250 
251 // Resize output variables
252 B.resize(m-1);
253 for (i = 0; i < m-1; ++i) {
254  B[i].resize(n-1, 0.0);
255 } //next i
256 
257 // Extract complement
258 for (l = 0; l < i; l++) {
259  for (k = 0; k < j; k++) {
260  B[l][k] = A[l][k];
261  } //next k
262  for (k = j+1; k < n; k++) {
263  B[l][k-1] = A[l][k];
264  } //next k
265 } //next l
266 for (l = i+1; l < m; l++) {
267  for (k = 0; k < j; k++) {
268  B[l-1][k] = A[l][k];
269  } //next k

271  B[l-1][k-1] = A[l][k];
272  } //next k
273 } //next l

275 return; };
