51 template<typename id_t>
52 typename IndexGenerator<id_t>::id_type IndexGenerator<id_t>::generate()
53 {
54  // If the trash is empty generate a new id otherwise recycle the first id
55  // in the trash.
56  if (m_trash.empty()) {
57  if (m_highest == NULL_ID) {
58  m_highest = 0;
59  } else {
60  assert(m_highest < std::numeric_limits<id_type>::max());
61  ++m_highest;
62  }
63  m_latest = m_highest;
64  } else {
65  m_latest = m_trash.front();
66  m_trash.pop_front();
67  }
68 
69  return m_latest;
70 }

99 template<typename id_t>
100 bool IndexGenerator<id_t>::isAssigned(id_type id)
101 {
102  // If the latest id is valid, it is assigned by definition
103  if (m_latest >= 0 && id == m_latest) {
104  return true;
105  }
106 
107  // Ids past the highest one are not assigned
108  if (id > m_highest) {
109  return false;
110  }
111 
112  // The id is assigned only if is not in the trash
113  for (id_type trashedId : m_trash) {
114  if (trashedId == id) {
115  return false;
116  }
117  }
118 
119  return true;
120 }

127 template<typename id_t>
128 void IndexGenerator<id_t>::setAssigned(typename IndexGenerator<id_t>::id_type id)
129 {
130  // If the id is past the highest assigned id we need to trash all the ids
131  // from the highest assigned to this one (the generator only handles
132  // contiguous ids), otherwise look for the id in the trash and, if found,
133  // recycle it (if the id is not in the trash, this means it was already
134  // an assigned id).
135  if (id > m_highest) {
136  for (id_type wasteId = std::max(static_cast<id_t>(0), m_highest + 1); wasteId < id; ++wasteId) {
137  trash(wasteId);
138  }
139 
140  m_highest = id;
141  } else {
142  for (auto trashItr = m_trash.begin(); trashItr != m_trash.end(); ++trashItr) {
143  id_type trashedId = *trashItr;
144  if (trashedId == id) {
145  m_trash.erase(trashItr);
146  break;
147  }
148  }
149  }
150 
151  // This is the latest assigned id
152  m_latest = id;
153 }

162 template<typename id_t>
163 void IndexGenerator<id_t>::trash(typename IndexGenerator<id_t>::id_type id)
164 {
165  m_trash.push_back(id);
166 
167  // We only keep track of the latest assigned id, if we trash that id we
168  // have no information of the previous assigned id.
169  if (id == m_latest) {
170  m_latest = NULL_ID;
171  }
172 }
