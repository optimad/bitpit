138 void SurfUnstructured::_dump(std::ostream &stream) const
139 {
140 #if BITPIT_ENABLE_MPI==1
141  // Dump works only for serial calculations
142  if (getProcessorCount() != 1) {
143  throw std::runtime_error ("Dump of surfunstructured is implemented only for serial calculations.");
144  }
145 #endif
146 
147  // Space dimension
148  utils::binary::write(stream, getSpaceDimension());
149 
150  // Save the vertices
151  utils::binary::write(stream, getVertexCount());
152 
153  for (const Vertex &vertex : m_vertices) {
154  utils::binary::write(stream, vertex.getId());
155 
156  std::array<double, 3> coords = vertex.getCoords();
157  utils::binary::write(stream, coords[0]);
158  utils::binary::write(stream, coords[1]);
159  utils::binary::write(stream, coords[2]);
160  }
161 
162  // Save the cells
163  utils::binary::write(stream, getInternalCount());
164  utils::binary::write(stream, getGhostCount());
165 
166  for (const Cell &cell: m_cells) {
167  const ReferenceElementInfo &cellInfo = cell.getInfo();
168 
169  utils::binary::write(stream, cell.getId());
170  utils::binary::write(stream, cell.getPID());
171  utils::binary::write(stream, cellInfo.type);
172 
173  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
174  int nCellVertices = cellVertexIds.size();
175  for (int i = 0; i < nCellVertices; ++i) {
176  utils::binary::write(stream, cellVertexIds[i]);
177  }
178  }
179 }

186 void SurfUnstructured::_restore(std::istream &stream)
187 {
188 #if BITPIT_ENABLE_MPI==1
189  // Restore works only for serial calculations
190  if (getProcessorCount() != 1) {
191  throw std::runtime_error ("Restore of surfunstructured is implemented only for serial calculations.");
192  }
193 #endif
194 
195  // Space dimension
196  int spaceDimension;
197  utils::binary::read(stream, spaceDimension);
198  setSpaceDimension(spaceDimension);
199 
200  // Restore the vertices
201  long nVertices;
202  utils::binary::read(stream, nVertices);
203 
204  reserveVertices(nVertices);
205  for (long i = 0; i < nVertices; ++i) {
206  long id;
207  utils::binary::read(stream, id);
208 
209  std::array<double, 3> coords;
210  utils::binary::read(stream, coords[0]);
211  utils::binary::read(stream, coords[1]);
212  utils::binary::read(stream, coords[2]);
213 
214  addVertex(coords, id);
215  }
216 
217  // Restore the cells
218  long nInternals;
219  utils::binary::read(stream, nInternals);
220 
221  long nGhosts;
222  utils::binary::read(stream, nGhosts);
223 
224  long nCells = nInternals + nGhosts;
225 
226  reserveCells(nCells);
227  for (long i = 0; i < nCells; ++i) {
228  long id;
229  utils::binary::read(stream, id);
230 
231  int PID;
232  utils::binary::read(stream, PID);
233 
234  ElementType type;
235  utils::binary::read(stream, type);
236  const ReferenceElementInfo &cellInfo = ReferenceElementInfo::getInfo(type);
237 
238  int nCellVertices = cellInfo.nVertices;
239  std::vector<long> connect(nCellVertices, Vertex::NULL_ID);
240  for (int k = 0; k < nCellVertices; ++k) {
241  utils::binary::read(stream, connect[k]);
242  }
243 
244  CellIterator cellIterator = addCell(type, true, connect, id);
245  cellIterator->setPID(PID);
246  }
247 
248  // Build ghost exchange info
249 #if BITPIT_ENABLE_MPI==1
250  if (getProcessorCount()) {
251  buildGhostExchangeInfo();
252  }
253 #endif
254 }

286 void SurfUnstructured::extractEdgeNetwork(SurfUnstructured &net)
287 {
288  // ====================================================================== //
289  // VARIABLES DECLARATION //
290  // ====================================================================== //
291 
292  // Local variables
293  bool check;
294  int n_faces, n_adj;
295  long id;
296 
297  // Counters
298  int i, j;
299  vector<int>::const_iterator i_;
300  vector<long>::iterator j_;
301  VertexIterator v_, ve_ = vertexEnd();
302  CellIterator c_, ce_ = cellEnd();
303 
304  // ====================================================================== //
305  // INITIALIZE DATA STRUCTURE //
306  // ====================================================================== //
307  net.reserveCells(net.getCellCount() + countFaces());
308  net.reserveVertices(net.getVertexCount() + getVertexCount());
309 
310  // ====================================================================== //
311  // ADD VERTEX TO net //
312  // ====================================================================== //
313  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
314  net.addVertex(v_->getCoords(), v_->getId());
315  } //next v_
316 
317  // ====================================================================== //
318  // ADD EDGES //
319  // ====================================================================== //
320  for (c_ = cellBegin(); c_ != ce_; ++c_) {
321  id = c_->getId();
322  n_faces = c_->getFaceCount();
323  ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
324  for (i = 0; i < n_faces; ++i) {
325  check = true;
326  n_adj = c_->getAdjacencyCount(i);
327  for (j = 0; j < n_adj; ++j) {
328  check = check && (id > c_->getAdjacency(i, j));
329  } //next j
330  if (check) {
331  // Get edge type
332  ElementType edgeType = c_->getFaceType(i);
333 
334  // Get edge connect
335  ConstProxyVector<long> faceConnect = c_->getFaceConnect(i);
336  int faceConnectSize = faceConnect.size();
337 
338  std::unique_ptr<long[]> edgeConnect = std::unique_ptr<long[]>(new long[faceConnectSize]);
339  for (int k = 0; k < faceConnectSize; ++k) {
340  edgeConnect[k] = faceConnect[k];
341  }
342 
343  // Add edge
344  net.addCell(edgeType, true, std::move(edgeConnect));
345  }
346  } //next i
347  } //next c_
348 
349  return;
350 }

430 unsigned short SurfUnstructured::importSTL(STLObj &STL, int PIDOffset, bool PIDSquash,
431  std::unordered_map<int, std::string> *PIDNames)
432 {
433  // ====================================================================== //
434  // OPEN STL FILE //
435  // ====================================================================== //
436  STL.open("in");
437  if (STL.err != 0) {
438  return STL.err;
439  }
440 
441  // ====================================================================== //
442  // LOAD ALL SOLID FROM THE STL FILE //
443  // ====================================================================== //
444  int pid = PIDOffset;
445  if (!PIDSquash) {
446  --pid;
447  }
448 
449  while (true) {
450  // ====================================================================== //
451  // LOAD SOLID FROM THE STL FILE //
452  // ====================================================================== //
453  int nVertex = 0;
454  int nSimplex = 0;
455  std::vector<std::array<double, 3>> vertexList;
456  std::vector<std::array<double, 3>> normalList;
457  std::vector<std::array<int, 3>> connectivityList;
458  std::string name = "";
459 
460  STL.loadSolid(nVertex, nSimplex, vertexList, normalList, connectivityList, name);
461  if (nVertex == 0) {
462  break;
463  }
464 
465  // ====================================================================== //
466  // PID AND NAME OF THE SOLID //
467  // ====================================================================== //
468  if (!PIDSquash) {
469  ++pid;
470  if (PIDNames && !name.empty()) {
471  PIDNames->insert({{pid, name}});
472  }
473  }
474 
475  // ====================================================================== //
476  // PREPARE MESH FOR DATA IMPORT //
477  // ====================================================================== //
478  reserveVertices(getVertexCount() + nVertex);
479  reserveCells(m_nInternals + m_nGhosts + nSimplex);
480 
481  // ====================================================================== //
482  // ADD VERTICES TO MESH //
483  // ====================================================================== //
484  vector<array<double, 3>>::const_iterator v_, ve_;
485 
486  std::vector<long> vertexMap;
487  vertexMap.reserve(nVertex);
488 
489  long v_counter = 0;
490  ve_ = vertexList.cend();
491  for (v_ = vertexList.cbegin(); v_ != ve_; ++v_) {
492  VertexIterator i_ = addVertex(*v_);
493  vertexMap[v_counter] = i_->getId();
494  ++v_counter;
495  } //next v_
496 
497  // ====================================================================== //
498  // ADD CELLS TO MESH //
499  // ====================================================================== //
500  vector<array<int,3>>::const_iterator c_, ce_;
501  array<int,3>::const_iterator w_, we_;
502 
503  ce_ = connectivityList.cend();
504  for (c_ = connectivityList.cbegin(); c_ != ce_; ++c_) {
505  // Remap STL connectivity
506  int n_v = c_->size();
507  std::vector<long> connect(n_v, Vertex::NULL_ID);
508  we_ = c_->cend();
509  int i = 0;
510  for (w_ = c_->cbegin(); w_ < we_; ++w_) {
511  connect[i] = vertexMap[*w_];
512  ++i;
513  } //next w_
514 
515  // Add cell
516  CellIterator cellIterator = addCell(getSTLFacetType(n_v), true, connect);
517  cellIterator->setPID(pid);
518  } //next c_
519 
520  // ====================================================================== //
521  // Multi-body STL files are supported only in ASCII mode //
522  // ====================================================================== //
523  if (STL.stl_type) {
524  break;
525  }
526  }
527 
528  // ====================================================================== //
529  // CLOSE STL FILE //
530  // ====================================================================== //
531  STL.close("in");
532 
533  return 0;
534 }

597 unsigned short SurfUnstructured::exportSTLSingle(const std::string &name, const bool &isBinary, bool exportInternalsOnly)
598 {
599  // ====================================================================== //
600  // VARIABLES DECLARATION //
601  // ====================================================================== //
602 
603  // Local variables
604  int nVertex;
605  int nSimplex;
606  vector<array<double, 3>> vertexList;
607  vector<array<double, 3>> normalList;
608  vector<array<int,3>> connectivityList;
609  unordered_map<long, long> vertexMap;
610  array<int,3> dummyIntArray;
611 
612  // Counters
613  int v_count ,j;
614  vector<array<double, 3>>::iterator i_;
615  vector<array<int,3>>::iterator j_;
616  array<int,3>::iterator k_, ke_;
617  VertexIterator v_, ve_;
618  CellIterator c_, cb_, ce_;
619 
620  // ====================================================================== //
621  // INITIALIZE DATA STRUCTURE //
622  // ====================================================================== //
623  dummyIntArray.fill(0) ;
624 
625  nSimplex = m_nInternals;
626  if (!exportInternalsOnly) nSimplex += m_nGhosts;
627  vertexList.resize(getVertexCount());
628  normalList.resize(nSimplex);
629  connectivityList.resize(nSimplex, dummyIntArray);
630 
631  // ====================================================================== //
632  // CREATE VERTEX LIST //
633  // ====================================================================== //
634  i_ = vertexList.begin();
635  ve_ = vertexEnd();
636  v_count = 0;
637  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
638 
639  // Store vertex coordinates
640  *i_ = v_->getCoords();
641  vertexMap[v_->getId()] = v_count;
642 
643  // Update counters
644  ++v_count;
645  ++i_;
646 
647  } //next v_
648  nVertex = getVertexCount();
649 
650  // ====================================================================== //
651  // CREATE CONNECTIVITY //
652  // ====================================================================== //
653  if (exportInternalsOnly) {
654  cb_ = internalBegin();
655  ce_ = internalEnd();
656  }
657  else {
658  cb_ = cellBegin();
659  ce_ = cellEnd();
660  }
661  i_ = normalList.begin();
662  j_ = connectivityList.begin();
663  for (c_ = cb_; c_ != ce_; ++c_) {
664 
665  // Build normals
666  *i_ = evalFacetNormal(c_->getId());
667 
668  // Build connectivity
669  ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
670 
671  ke_ = j_->end();
672  j = 0;
673  for (k_ = j_->begin(); k_ != ke_; ++k_) {
674  *k_ = vertexMap[cellVertexIds[j]];
675  ++j;
676  } //next k_
677 
678  // Update counters
679  ++j_;
680  ++i_;
681  } //next c_
682 
683  // ====================================================================== //
684  // EXPORT STL DATA //
685  // ====================================================================== //
686  STLObj STL(name, isBinary);
687  STL.save("", nVertex, nSimplex, vertexList, normalList, connectivityList);
688 
689  return 0;
690 }

704 unsigned short SurfUnstructured::exportSTLMulti(const std::string &name, bool exportInternalsOnly,
705  std::unordered_map<int, std::string> *PIDNames)
706 {
707  int nTotVertex;
708  vector<array<double, 3>> totVertexList;
709  unordered_map<long, long> vertexMap;
710 
711  int nLocSimplex;
712  vector<array<double, 3>> normalLocList;
713  vector<array<int,3>> connectivityLocList;
714 
715  STLObj STL(name, false);
716 
717  STL.open("out");
718  if (STL.err != 0) {
719  return STL.err;
720  }
721 
722  // Create the vertex map
723  nTotVertex = getVertexCount();
724  totVertexList.resize(nTotVertex);
725  unsigned int count_v = 0;
726  for(const Vertex &v: getVertices()){
727  totVertexList[count_v] = v.getCoords();
728  vertexMap[v.getId()] = count_v;
729  ++count_v;
730  }
731 
732  // Export the internal cells
733  for(int pid : getInternalPIDs()){
734  std::vector<long> cells = getInternalsByPID(pid);
735 
736  nLocSimplex = (int) cells.size();
737  connectivityLocList.resize(nLocSimplex);
738  normalLocList.resize(nLocSimplex);
739 
740  vector<array<int,3>>::iterator itC = connectivityLocList.begin();
741  vector<array<double, 3>>::iterator itN = normalLocList.begin();
742 
743  // Fill local connectivity and normals structures
744  for (auto id : cells) {
745  // Fill connectivity
746  const Cell &cell = getCell(id);
747  for (int iloc = 0; iloc<3; ++iloc) {
748  (*itC)[iloc] = vertexMap[cell.getConnect()[iloc]];
749  }
750 
751  // Fill normal
752  *itN = evalFacetNormal(cell.getId());
753 
754  // Increment iterators
755  ++itC;
756  ++itN;
757  }
758 
759  // Write the solid associated to the current PID
760  std::string name;
761  if (PIDNames && PIDNames->count(pid) > 0) {
762  name = PIDNames->at(pid);
763  } else {
764  name = std::to_string(pid);
765  }
766 
767  STL.saveSolid(name, nTotVertex, nLocSimplex, totVertexList, normalLocList, connectivityLocList);
768  }
769 
770  // Export ghost cells
771  long nGhosts = getGhostCount();
772  if (!exportInternalsOnly && nGhosts > 0) {
773  nLocSimplex = nGhosts;
774  connectivityLocList.resize(nLocSimplex);
775  normalLocList.resize(nLocSimplex);
776 
777  vector<array<int,3>>::iterator itC = connectivityLocList.begin();
778  vector<array<double, 3>>::iterator itN = normalLocList.begin();
779 
780  // Fill local connectivity and normals structures
781  CellConstIterator endItr = ghostConstEnd();
782  for (CellConstIterator itr = ghostConstBegin(); itr != endItr; ++itr) {
783  // Fill connectivity
784  for (int iloc = 0; iloc<3; ++iloc) {
785  (*itC)[iloc] = vertexMap[itr->getConnect()[iloc]];
786  }
787 
788  // Fill normal
789  *itN = evalFacetNormal(itr.getId());
790 
791  // Increment iterators
792  ++itC;
793  ++itN;
794  }
795 
796  // Write the solid associated to the ghosts
797  STL.saveSolid("ghosts", nTotVertex, nLocSimplex, totVertexList, normalLocList, connectivityLocList);
798  }
799 
800  STL.close("out");
801 
802  return 0;
803 }

844 unsigned short SurfUnstructured::importDGF(const std::string &dgf_name, int PIDOffset, bool PIDSquash)
845 {
846  // ====================================================================== //
847  // VARIABLES DECLARATION //
848  // ====================================================================== //
849 
850  // Local variables
851  DGFObj dgf_in(dgf_name);
852  int nV = 0, nS = 0;
853  long vcount, idx;
854  std::vector<std::array<double, 3>> vertex_list;
855  std::vector<std::vector<int>> simplex_list;
856  std::vector<int> simplex_PID;
857  std::vector<long> vertex_map;
858  std::vector<long> connect;
859 
860  // Counters
861  std::vector<std::array<double, 3>>::const_iterator v_, ve_;
862  std::vector<std::vector<int>>::iterator c_, ce_;
863  std::vector<int>::iterator i_, ie_;
864  std::vector<long>::iterator j_, je_;
865 
866  // ====================================================================== //
867  // IMPORT DATA //
868  // ====================================================================== //
869 
870  // Read vertices and cells from DGF file
871  dgf_in.load(nV, nS, vertex_list, simplex_list, simplex_PID);
872 
873  // Add vertices
874  ve_ = vertex_list.cend();
875  vcount = 0;
876  vertex_map.resize(nV);
877  for (v_ = vertex_list.cbegin(); v_ != ve_; ++v_) {
878  idx = addVertex(*v_)->getId();
879  vertex_map[vcount] = idx;
880  ++vcount;
881  } //next v_
882 
883  // Update connectivity infos
884  ce_ = simplex_list.end();
885  for (c_ = simplex_list.begin(); c_ != ce_; ++c_) {
886  ie_ = c_->end();
887  for (i_ = c_->begin(); i_ != ie_; ++i_) {
888  *i_ = vertex_map[*i_];
889  } //next i_
890  } //next c_
891 
892  // Add cells
893  int k;
894  for (c_ = simplex_list.begin(), k = 0; c_ != ce_; ++c_, ++k) {
895  // Create cell
896  i_ = c_->begin();
897  connect.resize(c_->size(), Vertex::NULL_ID);
898  je_ = connect.end();
899  for (j_ = connect.begin(); j_ != je_; ++j_) {
900  *j_ = *i_;
901  ++i_;
902  } //next j_
903  CellIterator cellIterator = addCell(getDGFFacetType(c_->size()), true, connect);
904 
905  // Set cell PID
906  int cellPID = PIDOffset;
907  if (!PIDSquash) {
908  cellPID += simplex_PID[k];
909  }
910 
911  cellIterator->setPID(cellPID);
912  } //next c_
913 
914  return 0;
915 }
