892 unsigned int dgf::scanData(
893  std::ifstream &file_handle,
894  int &n
895 ) {
896 
897 // ========================================================================== //
898 // VARIABLES DECLARATION //
899 // ========================================================================== //
900 
901 // Local variables
902 long int current_pos;
903 string line, word;
904 stringstream sline;
905 
906 // Counters
907 // none
908 
909 // ========================================================================== //
910 // CHECK STREAM STATUS //
911 // ========================================================================== //
912 if (!file_handle.good()) { return(1); }
913 
914 // ========================================================================== //
915 // SCAN DATA SET //
916 // ========================================================================== //
917 
918 // Get current line
919 current_pos = file_handle.tellg();
920 word = "begin";
921 
922 // Loop until end of data block
923 while (!file_handle.eof()
924  && ((word.compare("#") != 0)
925  && (word.compare("VERTEX") != 0)
926  && (word.compare("SIMPLEX") != 0)
927  && (word.compare("VERTEXDATA") != 0)
928  && (word.compare("SIMPLEXDATA") != 0))) {
929 
930  // Get next line
931  current_pos = file_handle.tellg();
932  getline(file_handle, line);
933  line = utils::string::trim(line);
934  sline.clear();
935  sline.str(line);
936 
937  // Check if data exist
938  if (sline >> word) {
939  n++;
940  }
941 
942 } //next line
943 if ((word.compare("#") == 0)
944  || (word.compare("VERTEX") == 0)
945  || (word.compare("SIMPLEX") == 0)
946  || (word.compare("VERTEXDATA") == 0)
947  || (word.compare("SIMPLEXDATA") == 0)){
948  file_handle.clear();
949  file_handle.seekg(current_pos);
950  n--;
951 }
952 
953 return(0); }

955 // -------------------------------------------------------------------------- //
971 unsigned int dgf::scan(
972  std::ifstream &file_handle,
973  int &nV,
974  int &nS,
975  std::vector<std::string> &sV_data,
976  std::vector<std::string> &sS_data,
977  std::vector<int> &nV_data,
978  std::vector<int> &nS_data
979 ) {
980 
981 // ========================================================================== //
982 // VARIABLES DECLARATION //
983 // ========================================================================== //
984 
985 // Local variables
986 unsigned int err = 0;
987 long int start_pos;
988 string line, word;
989 stringstream sline;
990 
991 // Counters
992 int n;
993 
994 // ========================================================================== //
995 // CHECK STREAM STATUS //
996 // ========================================================================== //
997 if (!file_handle.good()) { return(1); }
998 
999 // ========================================================================== //
1000 // RESET INPUT DATA //
1001 // ========================================================================== //
1002 nV = 0;
1003 nS = 0;
1004 sV_data.resize(0);
1005 sS_data.resize(0);
1006 nV_data.resize(0);
1007 nS_data.resize(0);
1008 
1009 // ========================================================================== //
1010 // RESET CURSOR POSITION TO FILE BEGIN //
1011 // ========================================================================== //
1012 start_pos = file_handle.tellg();
1013 file_handle.clear();
1014 file_handle.seekg(0);
1015 
1016 // ========================================================================== //
1017 // SCAN DGF FILE //
1018 // ========================================================================== //
1019 word = "begin";
1020 while (!file_handle.eof()) {
1021 
1022  // Get current line
1023  getline(file_handle, line);
1024  line = utils::string::trim(line);
1025  sline.clear();
1026  sline.str(line);
1027 
1028  // Look for relevant keyword
1029  if (sline >> word) {
1030  if (word.compare("VERTEX") == 0) {
1031 
1032  // Get number of vertices
1033  err = dgf::scanData(file_handle, nV);
1034  }
1035  else if (word.compare("SIMPLEX") == 0) {
1036 
1037  // Get number of simplicies
1038  err = dgf::scanData(file_handle, nS);
1039  }
1040  else if (word.compare("VERTEXDATA") == 0) {
