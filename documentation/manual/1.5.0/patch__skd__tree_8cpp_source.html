58 void SkdPatchInfo::buildCache()
59 {
60  const PiercedVector<Cell> &cells = m_patch->getCells();
61 
62  m_cellBoxes = std::unique_ptr<BoxCache>(new BoxCache(2, &(m_patch->getCells())));
63  m_cellCentroids = std::unique_ptr<BoxCache>(new CentroidCache(1, &(m_patch->getCells())));
64  for (auto itr = m_patch->internalConstBegin(); itr != m_patch->internalConstEnd(); ++itr) {
65  std::size_t rawCellId = itr.getRawIndex();
66 
67  // Cell info
68  const Cell &cell = cells.rawAt(rawCellId);
69  int nCellVertices = cell.getVertexCount();
70  ConstProxyVector<long> cellConnect = cell.getVertexIds();
71 
72  // Bounding box
73  std::array<double, 3> &cellBoxMin = m_cellBoxes->rawAt(rawCellId, 0);
74  std::array<double, 3> &cellBoxMax = m_cellBoxes->rawAt(rawCellId, 1);
75  std::array<double, 3> &cellCentroid = m_cellCentroids->rawAt(rawCellId, 0);
76 
77  cellBoxMin = m_patch->getVertexCoords(cellConnect[0]);
78  cellBoxMax = cellBoxMin;
79  cellCentroid = cellBoxMin;
80  for (int i = 1; i < nCellVertices; ++i) {
81  const std::array<double, 3> &coords = m_patch->getVertexCoords(cellConnect[i]);
82  for (int d = 0; d < 3; ++d) {
83  cellBoxMin[d] = std::min(coords[d], cellBoxMin[d]);
84  cellBoxMax[d] = std::max(coords[d], cellBoxMax[d]);
85  cellCentroid[d] += coords[d];
86  }
87  }
88  cellCentroid /= double(nCellVertices);
89  }
90 }

218 void SkdNode::initializeBoundingBox()
219 {
220  const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
221 
222  // Evaluate the bounding box
223  m_boxMin = m_patchInfo->getCachedBoxMin(cellRawIds[m_cellRangeBegin]);
224  m_boxMax = m_patchInfo->getCachedBoxMax(cellRawIds[m_cellRangeBegin]);
225  for (std::size_t n = m_cellRangeBegin + 1; n < m_cellRangeEnd; n++) {
226  const std::size_t rawCellId = cellRawIds[n];
227  const std::array<double, 3> &cellBoxMin = m_patchInfo->getCachedBoxMin(rawCellId);
228  const std::array<double, 3> &cellBoxMax = m_patchInfo->getCachedBoxMax(rawCellId);
229  for (int d = 0; d < 3; ++d) {
230  m_boxMin[d] = std::min(cellBoxMin[d], m_boxMin[d]);
231  m_boxMax[d] = std::max(cellBoxMax[d], m_boxMax[d]);
232  }
233  }
234 
235  // Inlfate the bounding box by a small epsilon
236  //
237  // The small inflation allows us to test intersection of a query
238  // against a safety box and be assured that non-intersection of
239  // with the box implies non-intersection with all the contents
240  // in the box, even when computations are susceptible to round-off
241  // error.
242  double tolerance = m_patchInfo->getPatch().getTol();
243 
244  m_boxMin -= tolerance;
245  m_boxMax += tolerance;
246 }

448 void SkdNode::findPointClosestCell(const std::array<double, 3> &point,
449  long *id, double *distance) const
450 {
451  *id = Cell::NULL_ID;
452  *distance = std::numeric_limits<double>::max();
453  if (getCellCount() == 0) {
454  return;
455  }
456 
457  const PatchKernel &patch = m_patchInfo->getPatch();
458  const PiercedVector<Cell> &cells = patch.getCells();
459  const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
460  std::vector<std::array<double, 3>> cellVertexCoordinates(ReferenceElementInfo::MAX_ELEM_VERTICES);
461 
462  for (std::size_t n = m_cellRangeBegin; n < m_cellRangeEnd; n++) {
463  std::size_t cellRawId = cellRawIds[n];
464  const Cell &cell = cells.rawAt(cellRawId);
465 
466  // Get the vertices ids
467  ConstProxyVector<long> elementVertexIds = cell.getVertexIds();
468  const int nElementVertices = elementVertexIds.size();
469 
470  // Get vertex coordinates
471  cellVertexCoordinates.resize(nElementVertices);
472  for (int i = 0; i < nElementVertices; ++i) {
473  cellVertexCoordinates[i] = patch.getVertex(elementVertexIds[i]).getCoords();
474  }
475 
476  // Evaluate the distance from the cell
477  double cellDistance = cell.evalPointDistance(point, cellVertexCoordinates.data());
478 
479  // Update the distance
480  if (cellDistance < *distance) {
481  *distance = cellDistance;
482  *id = cell.getId();
483  }
484  }
485 }

498 {

501  return false;

521 bool SkdNode::boxIntersectsSphere(const std::array<double, 3> &center, double radius) const
522 {
523  // If the box contains the center it will also intersect the sphere
524  if (boxContainsPoint(center, 0.)) {
525  return true;
526  }
527 
528  // Check if the distance between the center and its projection on
529  // the bounding box is smaller than the radius of the sphere.
530  std::array<double, 3> delta;
531  for (int d = 0; d < 3; d++) {
532  delta[d] = std::min(std::max(center[d], m_boxMin[d]), m_boxMax[d]) - center[d];
533  }
534  double distance = dotProduct(delta, delta);
535 
536  return (distance < radius * radius);
537 }

629 void PatchSkdTree::build(int leafCapacity)
630 {
631  const PatchKernel &patch = m_patchInfo.getPatch();
632 
633  // Clear existing tree
634  clear();
635 
636  // Set leaf size
637  setLeafCapacity(leafCapacity);
638 
639  // Initialize list of cell raw ids
640  std::size_t nCells = patch.getInternalCount();
641  if (m_cellRawIds.size() != nCells) {
642  m_cellRawIds.resize(nCells);
643  }
644 
645  std::size_t k = 0;
646  for (auto itr = patch.internalConstBegin(); itr != patch.internalConstEnd(); ++itr) {
647  m_cellRawIds[k++] = itr.getRawIndex();
648  }
649 
650  // Build patch cache
651  m_patchInfo.buildCache();
652 
653  // Initialize node list
654  m_nodes.reserve(1.5 * nCells / getLeafCapacity());
655 
656  // Create the root
657  m_nodes.emplace_back(&m_patchInfo, 0, nCells);
658 
659  // Create the tree
660  std::queue<std::size_t> nodeStack;
661  nodeStack.push(0);
662  while (!nodeStack.empty()) {
663  std::size_t nodeId = nodeStack.front();
664  nodeStack.pop();
665 
666  // Create the children of the node
667  createChildren(nodeId);
668 
669  // Add the newly created childrend to the stack
670  const SkdNode &node = getNode(nodeId);
671  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
672  std::size_t childId = node.getChildId(static_cast<SkdNode::ChildLocation>(i));
673  if (childId != SkdNode::NULL_ID) {
674  nodeStack.push(childId);
675  }
676  }
677  }
678 
679  // Patch cache is no longer needed
680  m_patchInfo.destroyCache();
681 }

784 void PatchSkdTree::createChildren(std::size_t parentId)
785 {
786  SkdNode &parent = _getNode(parentId);
787 
788  // Check if the parent is a leaf.
789  if (parent.getCellCount() <= m_leafCapacity) {
790  ++m_nLeafs;
791  return;
792  }
793 
794  // Evaluate the preferred direction along which elements will be split.
795  //
796  // The elements will be split along a plane normal to the direction
797  // for which the bounding box has the maximum length.
798  const std::array<double, 3> &parentBoxMin = parent.getBoxMin();
799  const std::array<double, 3> &parentBoxMax = parent.getBoxMax();
800 
801  int largerDirection = 0;
802  double boxMaximumLength = parentBoxMax[largerDirection] - parentBoxMin[largerDirection];
803  for (int d = 1; d < 3; ++d) {
804  double length = parentBoxMax[d] - parentBoxMin[d];
805  if (length > boxMaximumLength) {
806  largerDirection = d;
807  boxMaximumLength = length;
808  }
809  }
810 
811  // Split the elements.
812  //
813  // Preferred direction is tried first, if the split along this direction
814  // will produce and empty child (e.g., all the cell's box centroids have
815  // the same coordinate along the split direction and therefore all the
816  // cells would be assigned to the left chiled) the split will be performed
817  // along one of the other directions.
818  std::array<double, 3> parentWeightedCentroid = parent.evalBoxWeightedMean();
819  for (int d = 0; d < 3; ++d) {
820  // Update the split direction
821  int splitDirection = (largerDirection + d) % 3;
822 
823  // Get the threshold for the split
824  double splitThreshold = parentWeightedCentroid[splitDirection];
825 
826  // Order the elements
827  //
828  // All the elements with a centroid coordinate less or equal than the
829  // threshold will be assigned to the left child, the others will be
830  // assigned to the right child.
831  std::size_t leftBegin = parent.m_cellRangeBegin;
832  std::size_t leftEnd = parent.m_cellRangeEnd;
833  std::size_t rightBegin = parent.m_cellRangeBegin;
834  std::size_t rightEnd = parent.m_cellRangeEnd;
835  while (true) {
836  // Update the right begin
837  while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[rightBegin])[splitDirection] <= splitThreshold) {
838  rightBegin++;
839  }
840 
841  // Update the left end
842  while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[leftEnd - 1])[splitDirection] > splitThreshold) {
843  leftEnd--;
844  }
845 
846  // If all the elements are in the right position we can exit
847  if (rightBegin == leftEnd) {
848  break;
849  }
850 
851  // If left end and and right begin are not equal, that the two ids
852  // point to misplaced elements. Swap the elements, advance the ids
853  // and continue iterating.
854  std::iter_swap(m_cellRawIds.begin() + (leftEnd - 1), m_cellRawIds.begin() + rightBegin);
855  }
856 
857  if (leftEnd <= leftBegin || rightEnd <= rightBegin) {
858  continue;
859  }
860 
861  // Create the left child
862  long leftId = m_nodes.size();
863  m_nodes.emplace_back(&m_patchInfo, leftBegin, leftEnd);
864  m_nodes[parentId].m_children[static_cast<std::size_t>(SkdNode::CHILD_LEFT)] = leftId;
865 
866  // Create the right child
867  long rightId = m_nodes.size();
868  m_nodes.emplace_back(&m_patchInfo, rightBegin, rightEnd);
869  m_nodes[parentId].m_children[static_cast<std::size_t>(SkdNode::CHILD_RIGHT)] = rightId;
870 
871  // Done
872  return;
873  }
874 
875  throw std::runtime_error("Unable to create the children.");
876 }
