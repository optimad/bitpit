381 void LevelSet::addProcessingOrder( int objectId ) {
382 
383  bool primary = m_objects.at(objectId)->isPrimary() ;
384 
385  if(primary){
386  std::vector<int>::iterator orderItr = m_order.begin() ;
387  bool iterate( orderItr != m_order.end()) ;
388 
389  while(iterate){
390  int id = *orderItr ;
391  if( m_objects.at(id)->isPrimary() ){
392  ++orderItr ;
393  iterate = orderItr != m_order.end() ;
394  } else {
395  iterate = false;
396  }
397  }
398 
399  m_order.insert(orderItr,objectId) ;
400 
401  } else {
402  m_order.push_back(objectId) ;
403 
404  }
405 
406 }

414 bool LevelSet::removeProcessingOrder(int objectId){
415 
416  std::vector<int>::iterator orderItr;
417 
418  for(orderItr=m_order.begin(); orderItr!=m_order.end(); ++orderItr){
419  if(*orderItr==objectId){
420  m_order.erase(orderItr);
421  return true;
422  }
423  }
424 
425  return false ;

564 void LevelSet::partition( const std::vector<adaption::Info> &mapper ){
565 
566  assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::partition()");
567 
568  // Set the communicator
569  if (!m_kernel->isCommunicatorSet()) {
570  m_kernel->initializeCommunicator();
571  }
572 
573  // Udate the cache of the kernel
574  m_kernel->updateGeometryCache( mapper ) ;
575 
576  // Compile send and receive lists
577  std::unordered_map<int,std::vector<long>> sendList, recvList ;
578  for( const auto &event : mapper){
579  if( event.entity != adaption::Entity::ENTITY_CELL){
580  continue;
581  }
582 
583  if( event.type == adaption::Type::TYPE_PARTITION_SEND){
584  sendList.insert({{event.rank,event.previous}}) ;
585  } else if( event.type == adaption::Type::TYPE_PARTITION_RECV){
586  recvList.insert({{event.rank,event.current}}) ;
587  }
588  }
589 
590  // Communicate according to new partitioning
591  for( int objectId : m_order){
592  auto &visitor = *(m_objects.at(objectId)) ;
593 
594  visitor.communicate( sendList, recvList, &mapper ) ;
595  }
596 
597  for( int objectId : m_order){
598  auto &visitor = *(m_objects.at(objectId)) ;
599 
600  visitor.exchangeGhosts();
601  }
602 
603 }
