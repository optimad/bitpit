104 
105  double *datamappedS = mappedField.scalar->data(id);
106  std::array<double,3> *datamappedV = mappedField.vector->data(id);
107 
108  if (m_mapper[id].type == adaption::Type::TYPE_RENUMBERING){
109  bool dataB = field.mask->at(m_mapper[id].previous[0]);

113  for (std::size_t i = 0; i < nsf; i++){

121  std::array<double,3> *datamappedVi = datamappedV + i;
122  (*datamappedVi) = (*dataVi);
123  }
124  }

129  double *datamappedSi = datamappedS + i;
130  (*datamappedSi) = 0.0;
