796 void computeGeneralizedBarycentric( array3D const &p, std::size_t nVertices, array3D const *vertex, std::vector<double> &lambda)
797 {
798  lambda.resize(nVertices);
799 
800  std::vector<double> area(nVertices);
801  for( std::size_t i=0; i<nVertices; ++i){
802  int next = (i +1) %nVertices;
803  area[i] = areaTriangle( vertex[i], vertex[next], p);
804  }
805 
806  double sumWeight(0);
807 
808  for( std::size_t i=0; i<nVertices; ++i){
809  std::size_t prev = (i +nVertices -1) %nVertices;
810  std::size_t next = (i +1) %nVertices;
811  lambda[i] = areaTriangle(vertex[prev], vertex[i], vertex[next]);
812 
813  for( std::size_t j=0; j<nVertices; ++j){
814  if( j==prev || j==i){
815  continue;
816  }
817 
818  lambda[i] *= area[j];
819  }
820 
821  sumWeight += lambda[i];
822  }
823 
824  lambda /= sumWeight;
825 }
