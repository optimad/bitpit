177 double SurfaceKernel::evalCellArea(const long &id) const
178 {
179  // ====================================================================== //
180  // VARIABLES DECLARATION //
181  // ====================================================================== //
182 
183  // Local variables
184  const Cell *cell_ = &m_cells[id];
185  ConstProxyVector<long> cellVertexIds = cell_->getVertexIds();
186 
187  // Counters
188  // none
189 
190  // ====================================================================== //
191  // EVALUATE FACET AREA //
192  // ====================================================================== //
193  if ((cell_->getType() == ElementType::UNDEFINED)
194  || (cell_->getType() == ElementType::VERTEX)) return 0.0;
195  if (cell_->getType() == ElementType::LINE) {
196  return ( norm2(m_vertices[cellVertexIds[0]].getCoords()
197  - m_vertices[cellVertexIds[1]].getCoords()) );
198  }
199  array<double, 3> d1, d2;
200  if (cell_->getType() == ElementType::TRIANGLE) {
201  d1 = m_vertices[cellVertexIds[1]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
202  d2 = m_vertices[cellVertexIds[2]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
203  return (0.5*norm2(crossProduct(d1, d2)));
204  }
205  else {
206  int nvert = cell_->getVertexCount();
207  int next, prev;
208  double coeff = 0.25;
209  double area = 0.0;
210  for (int i = 0; i < nvert; ++i) {
211  next = (i + 1) % nvert;
212  prev = (nvert + i - 1) % nvert;
213  d1 = m_vertices[cellVertexIds[next]].getCoords() - m_vertices[cellVertexIds[i]].getCoords();
214  d2 = m_vertices[cellVertexIds[prev]].getCoords() - m_vertices[cellVertexIds[i]].getCoords();
215  area += coeff*norm2(crossProduct(d1, d2));
216  } //next i
217  return(area);
218  }
219 }

364 double SurfaceKernel::evalAngleAtVertex(const long &id, const int &vertex_id) const
365 {
366  // ====================================================================== //
367  // VARIABLES DECLARATION //
368  // ====================================================================== //
369 
370  // Local variables
371  const Cell *cell_ = &m_cells[id];
372  ConstProxyVector<long> cellVertexIds = cell_->getVertexIds();
373 
374  // Counters
375 
376  // ====================================================================== //
377  // EVALUATE ANGLE AT SPECIFIED VERTEX //
378  // ====================================================================== //
379  if ((cell_->getType() == ElementType::UNDEFINED)
380  || (cell_->getType() == ElementType::VERTEX)) return 0.0;
381  if (cell_->getType() == ElementType::LINE) {
382  if (m_spaceDim - getDimension() == 1) return BITPIT_PI;
383  else return 0.0;
384  }
385 
386  int n_vert = cell_->getVertexCount();
387  int prev = (n_vert + vertex_id - 1) % n_vert;
388  int next = (vertex_id + 1) % n_vert;
389  double angle;
390  array<double, 3> d1, d2;
391 
392  d1 = m_vertices[cellVertexIds[next]].getCoords() - m_vertices[cellVertexIds[vertex_id]].getCoords();
393  d2 = m_vertices[cellVertexIds[prev]].getCoords() - m_vertices[cellVertexIds[vertex_id]].getCoords();
394  d1 = d1/norm2(d1);
395  d2 = d2/norm2(d2);
396  angle = acos( min(1.0, max(-1.0, dotProduct(d1, d2) ) ) );
397 
398  return(angle);
399 }

537 array<double, 3> SurfaceKernel::evalFacetNormal(const long &id) const
538 {
539  // ====================================================================== //
540  // VARIABLES DECLARATION //
541  // ====================================================================== //
542 
543  // Local variables
544  array<double, 3> normal = {{0.0, 0.0, 0.0}};
545  const Cell *cell_ = &m_cells[id];
546  ConstProxyVector<long> cellVertexIds = cell_->getVertexIds();
547 
548  // Counters
549  // none
550 
551  // ====================================================================== //
552  // COMPUTE NORMAL //
553  // ====================================================================== //
554  if ((cell_->getType() == ElementType::UNDEFINED)
555  || (cell_->getType() == ElementType::VERTEX)) return normal;
556 
557  if (cell_->getType() == ElementType::LINE) {
558  if (m_spaceDim - getDimension() == 1) {
559  std::array<double, 3> z = {{0.0, 0.0, 1.0}};
560  normal = m_vertices[cellVertexIds[1]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
561  normal = crossProduct(normal, z);
562  }
563  else return normal;
564  }
565 
566  if (cell_->getType() == ElementType::TRIANGLE) {
567  array<double, 3> d1, d2;
568  d1 = m_vertices[cellVertexIds[1]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
569  d2 = m_vertices[cellVertexIds[2]].getCoords() - m_vertices[cellVertexIds[0]].getCoords();
570  normal = crossProduct(d1, d2);
571  }
572  else {
573  array<double, 3> d1, d2;
574  int next, prev, i, nvert = cell_->getVertexCount();
575  double coeff = 1.0/double(nvert);
576  for (i = 0; i < nvert; ++i) {
577  next = (i+1) % nvert;
578  prev = (nvert + i - 1) % nvert;
579  d1 = m_vertices[cellVertexIds[next]].getCoords() - m_vertices[cellVertexIds[i]].getCoords();
580  d2 = m_vertices[cellVertexIds[prev]].getCoords() - m_vertices[cellVertexIds[i]].getCoords();
581  normal += coeff*crossProduct(d1, d2);
582  } //next i
583  }
584  normal = normal/norm2(normal);
585 
586  return(normal);
587 
588 }

655 std::array<double, 3> SurfaceKernel::evalLimitedVertexNormal(const long &id, const int &vertex, const double &limit) const
656 {
657  // Get vertex id
658  const Cell &cell = m_cells[id];
659  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
660  long vertexId = cellVertexIds[vertex];
661 
662  // Cell normal
663  std::array<double, 3> cellNormal = evalFacetNormal(id);
664 
665  // Cell angle at vertex
666  double cellVertexAngle = evalAngleAtVertex(id, cell.findVertex(vertexId));
667 
668  // Compute cell vertex neighs
669  std::vector<long> vertexNeighs = findCellVertexNeighs(id, vertex);
670 
671  // Compute non-normalized normal
672  std::array<double, 3> normal = cellVertexAngle * cellNormal;
673  for (long facetId : vertexNeighs) {
674  // Eval facet normal
675  std::array<double, 3> facetNormal = evalFacetNormal(facetId);
676 
677  // Discard facets with a misalignment greater than the specified limit
678  //
679  // The argument of the acos function has to be in the range [-1, 1].
680  // Rounding errors may lead to a dot product slightly outside this
681  // range. Since the arguments of the dot product are unit vectors,
682  // we can safetly clamp the dot product result to be between -1 and
683  // 1.
684  double misalignment = std::acos(std::min(1.0, std::max(-1.0, dotProduct(facetNormal, cellNormal))));
685  if (misalignment > std::abs(limit)) {
686  continue;
687  }
688 
689  // Eval facet angle used as normalization weight
690  const Cell &facet = m_cells[facetId];
691  double facetVertexAngle = evalAngleAtVertex(facetId, facet.findVertex(vertexId));
692 
693  // Add facet contribution to normal
694  normal += facetVertexAngle * facetNormal;
695  }
696 
697  // Normalization
698  normal = normal / norm2(normal);
699 
700  return normal;
701 }

746 bool SurfaceKernel::adjustCellOrientation(const long &seed, const bool &invert)
747 {
748 #if BITPIT_ENABLE_MPI==1
749  // Check if the communications are needed
750  bool ghostExchangeNeeded = isPartitioned();
751 
752  // Initialize ghost communications
753  std::unordered_set<long> flipped;
754  std::unique_ptr<DataCommunicator> ghostComm;
755  if (ghostExchangeNeeded) {
756  ghostComm = std::unique_ptr<DataCommunicator>(new DataCommunicator(getCommunicator()));
757 
758  for (auto &entry : getGhostExchangeSources()) {
759  const int rank = entry.first;
760  const std::vector<long> sources = entry.second;
761  ghostComm->setSend(rank, sources.size() * sizeof(bool));
762  }
763 
764  for (auto &entry : getGhostExchangeTargets()) {
765  const int rank = entry.first;
766  const std::vector<long> targets = entry.second;
767  ghostComm->setRecv(rank, targets.size() * sizeof(bool));
768  }
769  }
770 #endif
771 
772  // Initialize the seed
773  std::set<long> toVisit;
774  if (seed != Element::NULL_ID) {
775  assert(getCells().exists(seed));
776  toVisit.insert(seed);
777  if (invert) {
778  flipCellOrientation(seed);
779 #if BITPIT_ENABLE_MPI==1
780  if (ghostExchangeNeeded) {
781  flipped.insert(seed);
782  }
783 #endif
784  }
785  }
786 
787  // Orient the surface
788  std::unordered_set<long> visited;
789 
790  bool orientable = true;
791  bool completed = false;
792  while (!completed) {
793  while (orientable && !toVisit.empty()) {
794  auto toVisitBegin = toVisit.begin();
795  long cellId = *toVisitBegin;
796  toVisit.erase(toVisitBegin);
797 
798  visited.insert(cellId);
799 
800  const auto &cell = getCell(cellId);
801  const long *adjacencyIds = cell.getAdjacencies();
802  const long *interfaceIds = cell.getInterfaces();
803  const int nInterfaces = cell.getInterfaceCount();
804 
805  for (int i = 0; i < nInterfaces; ++i) {
806  const long neighId = adjacencyIds[i];
807  if (neighId < 0) {
808  continue;
809  }
810 
811 #if BITPIT_ENABLE_MPI==1
812  if (ghostExchangeNeeded) {
813  const auto &neigh = getCell(neighId);
814  if (!neigh.isInterior()) {
815  continue;
816  }
817  }
818 #endif
819 
820  bool isNeighOriented = sameOrientationAtInterface(interfaceIds[i]);
821  if (visited.count(neighId) == 0) {
822  if (!isNeighOriented) {
823  flipCellOrientation(neighId);
824 #if BITPIT_ENABLE_MPI==1
825  if (ghostExchangeNeeded) {
826  flipped.insert(neighId);
827  }
828 #endif
829  }
830 
831  toVisit.insert(neighId);
832  } else if (!isNeighOriented) {
833  orientable = false;
834  break;
835  }
836  }
837  }
838 
839 #if BITPIT_ENABLE_MPI==1
840  // Comunicate the orientable flag among the partitions
841  if (ghostExchangeNeeded) {
842  MPI_Allreduce(MPI_IN_PLACE, &orientable, 1, MPI_C_BOOL, MPI_LAND, getCommunicator());
843  }
844 #endif
845 
846  // If the surface is not orientable we can exit
847  if (!orientable) {
848  break;
849  }
850 
851 #if BITPIT_ENABLE_MPI==1
852  if (ghostExchangeNeeded) {
853  // Add seeds from other partitions
854  ghostComm->startAllRecvs();
855 
856  for(auto &entry : getGhostExchangeSources()) {
857  int rank = entry.first;
858  SendBuffer &buffer = ghostComm->getSendBuffer(rank);
859  for (long cellId : entry.second) {
860  if (flipped.count(cellId) > 0) {
861  buffer << true;
862  } else {
863  buffer << false;
864  }
865  }
866  ghostComm->startSend(rank);
867  }
868 
869  int nPendingRecvs = ghostComm->getRecvCount();
870  while (nPendingRecvs != 0) {
871  int rank = ghostComm->waitAnyRecv();
872  RecvBuffer buffer = ghostComm->getRecvBuffer(rank);
873 
874  for (long cellId : getGhostExchangeTargets(rank)) {
875  bool ghostFlipped;
876  buffer >> ghostFlipped;
877  if (ghostFlipped) {
878  toVisit.insert(cellId);
879  flipCellOrientation(cellId);
880  }
881  }
882 
883  --nPendingRecvs;
884  }
885 
886  ghostComm->waitAllSends();
887 
888  // Clear list of flipped cells
889  flipped.clear();
890  }
891 #endif
892 
893  // Detect if the orientation is completed
894  completed = (toVisit.size() == 0);
895 #if BITPIT_ENABLE_MPI==1
896  if (ghostExchangeNeeded) {
897  MPI_Allreduce(MPI_IN_PLACE, &completed, 1, MPI_C_BOOL, MPI_LAND, getCommunicator());
898  }
899 #endif
900  }
901 
902  return orientable;
903 }

915 bool SurfaceKernel::sameOrientationAtInterface(const long &id)
916 {
917  const auto &face = getInterface(id);
918  if (face.isBorder()) {
919  return true;
920  }
921 
922  const long ownerId = face.getOwner();
923  const auto &owner = getCell(ownerId);
924 
925  const long neighId = face.getNeigh();
926  const auto &neigh = getCell(neighId);
927 
928  if (face.getType() == ElementType::VERTEX) {
929  ConstProxyVector<long> ownerVertexIds = owner.getVertexIds();

956 void SurfaceKernel::flipCellOrientation(const long &id)
957 {
958  int top, end;
959  auto &cell = getCell(id);
960 
961  //
962  // In order to flip the orientation of cell,
963  // the connectivity will be inverted
964  //
965  int nCellVertices = cell.getVertexCount();
966  const long *cellConnect = cell.getConnect();
967  std::unique_ptr<long[]> newCellConnect = std::unique_ptr<long[]>(new long[nCellVertices]);
968  for (int j = 0; j < nCellVertices; ++j) {
969  newCellConnect[j] = cellConnect[j];
970  }
971 
972  top = 0;
973  end = nCellVertices - 1;
974  for (int j = 0; j < (int) std::floor(nCellVertices/2); ++j) {
975  std::swap(newCellConnect[top], newCellConnect[end]);
976  ++top;
977  --end;
978  }
979 
980  //
981  // The numbering of the adjacencies and interfaces must be
982  // changed according the new connectivity. This implies that
983  // all numbering must be inverted, but the last entry of the
984  // adjacencies and interfaces. Since both adjacencies and
985  // interfaces are grouped by faces, in a second step each
986  // ordering within a face must be inverted
987  //
988  int faceCount = cell.getFaceCount();
989  std::vector<std::vector<long>> newAdjacency, newInterface;
990  newAdjacency.resize(faceCount);
991  newInterface.resize(faceCount);
992 
993  // Copy original ordering in newAdjacency and newInterface
994  //
995  for (int f = 0; f < faceCount; ++f) {
996  std::vector<long> &locAdj = newAdjacency[f];
997  std::vector<long> &locInt = newInterface[f];
998 
999  int nCellAdjacencies = cell.getAdjacencyCount(f);
1000  const long *adjacency = cell.getAdjacencies(f);
1001  const long *interface = cell.getInterfaces(f);
1002 
1003  locAdj.resize(nCellAdjacencies);
1004  locInt.resize(nCellAdjacencies);
1005  for (int j = 0; j < nCellAdjacencies; ++j) {
1006  locAdj[j] = adjacency[j];
1007  locInt[j] = interface[j];
1008  }
1009 
1010  }
1011 
1012  //
1013  // Invert entire face based ordering but tha last entry
1014  //
1015  top = 0;
1016  end = faceCount - 2;
1017  for (int f = 0; f < std::max(1, (int) std::floor((faceCount - 1) / 2)); ++f) {
1018  std::swap(newAdjacency[top], newAdjacency[end]);
1019  std::swap(newInterface[top], newInterface[end]);
1020  ++top;
1021  --end;
1022  }
1023 
1024  //
1025  // Invert all enties within one face
1026  //
1027  for (int f = 0; f < faceCount; ++f) {
1028  std::vector<long> &locAdj =newAdjacency[f];
1029  std::vector<long> &locInt =newInterface[f];
1030  int nFaceAdjacencies = locAdj.size();
1031 
1032  top = 0;
1033  end = nFaceAdjacencies - 1;
1034  for (int j = 0; j<(int) std::floor(nFaceAdjacencies / 2); ++j) {
1035  std::swap(locAdj[top], locAdj[end]);
1036  std::swap(locInt[top], locInt[end]);
1037  ++top;
1038  --end;
1039  }
1040  }
1041 
1042  cell.setConnect(std::move(newCellConnect));
1043 
1044  if (newAdjacency[0].size() != 0) {
1045  cell.setAdjacencies(newAdjacency);
1046  }
1047 
1048  if (newInterface[0].size() != 0) {
1049  cell.setInterfaces(newInterface);
1050 
1051  //
1052  // For the interfaces, we need to correct either
1053  // ownerFace or neighFace, respectivly.
1054  //
1055  const long* intPtr = cell.getInterfaces();
1056  int interfaceCount = cell.getInterfaceCount();
1057  for (int i=0; i<interfaceCount; ++i) {
1058  long interfaceId = intPtr[i];
1059  if (interfaceId < 0) {
1060  continue;
1061  }
1062 
1063  Interface &interface=getInterface(interfaceId);
1064  long owner = interface.getOwner();
1065  if (owner == id) {
1066  int ownerFace = interface.getOwnerFace();
1067  if (ownerFace != faceCount - 1) {
1068  ownerFace = faceCount - 2 - ownerFace;
1069  interface.setOwner(id,ownerFace);
1070  }
1071  } else {
1072  int neighFace = interface.getNeighFace();
1073  if (neighFace != faceCount - 1) {
1074  neighFace = faceCount - 2 - neighFace;
1075  interface.setNeigh(id,neighFace);
1076  }
1077  }
1078  }
1079  }
