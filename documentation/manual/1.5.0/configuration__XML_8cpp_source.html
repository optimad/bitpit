50 void readNode(xmlNodePtr root, Config *config)
51 {
52  for (xmlNode *node = root; node; node = node->next) {
53  if (node->type != XML_ELEMENT_NODE) {
54  readNode(node->children, config);
55  continue;
56  }
57 
58  bool isSection = (xmlChildElementCount(node) != 0);
59  std::string key(reinterpret_cast<const char*>(node->name));
60  if (isSection) {
61  Config::Section *section;
62  if (!config->isMultiSectionsEnabled() && config->hasSection(key)) {
63  section = &(config->getSection(key));
64  } else {
65  section = &(config->addSection(key));
66  }
67  readNode(node->children, section);
68  } else {
69  std::string value(reinterpret_cast<const char*>(xmlNodeGetContent(node)));
70  config->set(key, value);
71  }
72  }
73 }

82 void writeNode(xmlTextWriterPtr writer, const Config *config, const std::string &encoding)
83 {
84  int status;
85 
86  // Write the options
87  for (auto &entry : config->getOptions()) {
88  std::string key = entry.first;
89  std::string value = entry.second;
90 
91  xmlChar *elementName = encodeString(key, encoding);
92  xmlChar *elementText = encodeString(value, encoding);
93  status = xmlTextWriterWriteFormatElement(writer, BAD_CAST elementName, "%s", elementText);
94  if (status < 0) {
95  throw std::runtime_error("Error at xmlTextWriterWriteFormatElement");
96  }
97  }
98 
99  // Write the sections
100  for (auto &entry : config->getSections()) {
101  std::string key = entry.first;
102  const Config::Section *section = entry.second.get();
103 
104  // Start the section
105  xmlChar *elementName = encodeString(key, encoding);
106  int status = xmlTextWriterStartElement(writer, BAD_CAST elementName);
107  if (status < 0) {
108  throw std::runtime_error("Error at xmlTextWriterStartElement");
109  }
110 
111  // Write the section
112  writeNode(writer, section, encoding);
113 
114  // End section
115  status = xmlTextWriterEndElement(writer);
116  if (status < 0) {
117  throw std::runtime_error("Error at xmlTextWriterEndElement");
118  }
119  }
120 }

129 xmlChar * encodeString(const std::string &in, const std::string &encoding)
130 {
131  if (in.empty()) {
132  return nullptr;
133  }
134 
135  xmlCharEncodingHandlerPtr handler = xmlFindCharEncodingHandler(encoding.c_str());
136  if (!handler) {
137  return nullptr;
138  }
139 
140  int size = (int) in.length() + 1;
141  int out_size = size * 2 - 1;
142  xmlChar *out = (unsigned char *) xmlMalloc((size_t) out_size);
143  if (out == nullptr) {
144  return nullptr;
145  }
146 
147  int temp = size - 1;
148  int ret = handler->input(out, &out_size, (const xmlChar *) in.c_str(), &temp);
149  if ((ret < 0) || (temp - size + 1)) {
150  xmlFree(out);
151  out = 0;
152  } else {
153  out = (unsigned char *) xmlRealloc(out, out_size + 1);
154  out[out_size] = 0; /*null terminating out */
155  }
156 
157  return out;
158 }
