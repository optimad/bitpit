122 std::unique_ptr<SurfUnstructured> LevelSetMask::extractCellEnvelope(const std::unordered_set<long> &mask, const VolumeKernel &mesh, std::unordered_map<long,long> &meshToEnvelope){
123 
124  std::vector<long> list;
125 
126  for( const long &cellIndex : mask){
127  const auto &cell = mesh.getCell(cellIndex);
128  const long *adjacencies = cell.getAdjacencies();
129  const long *interfaceIndex = cell.getInterfaces();
130 
131  int interfaceCount= cell.getInterfaceCount() ;
132 
133  for(int i=0; i<interfaceCount; i++){
134  const long &neigh = adjacencies[i];
135 
136  if(mask.count(neigh)==0){
137  list.push_back(interfaceIndex[i]);
138  }
139  }
140  }
141 
142  return extractFaceEnvelope(list,mesh,meshToEnvelope);

153 std::unique_ptr<SurfUnstructured> LevelSetMask::extractFaceEnvelope(const std::vector<long> &list, const VolumeKernel &mesh, std::unordered_map<long,long> &meshToEnvelope){
154 
155  std::unique_ptr<SurfUnstructured> envelope = std::unique_ptr<SurfUnstructured>(new SurfUnstructured(mesh.getDimension()-1,mesh.getDimension()));
156 
157  // ====================================================================== //
158  // RESIZE DATA STRUCTURES //
159  // ====================================================================== //
160  long nVertices(0);
161  long nCells(list.size());
162 
163  for( const long &faceIndex : list){
164  auto const &interface = mesh.getInterface(faceIndex);
165  nVertices += interface.getVertexCount() ;
166  }
167 
168  envelope->reserveVertices(nVertices);
169  envelope->reserveCells(nCells);
170 
171  // ====================================================================== //
172  // LOOP OVER CELLS //
173  // ====================================================================== //
174  std::unordered_map<long,long> vertexMap;
175 
176  for( const long &faceIndex : list){
177  auto const &interface = mesh.getInterface(faceIndex);
178  ConstProxyVector<long> faceVertexIds = interface.getVertexIds();
179  int nFaceVertices = faceVertexIds.size();
180 
181  // Add face vertices to the envelope and get face
182  // connectivity in the envelope
183  std::unique_ptr<long[]> faceEnvelopeConnect = std::unique_ptr<long[]>(new long[nFaceVertices]);
184  for (int j = 0; j < nFaceVertices; ++j) {
185  long vertexId = faceVertexIds[j];
186 
187  // If the vertex is not yet in the envelope
188  // add it.
189  if (vertexMap.count(vertexId) == 0) {
190  const Vertex &vertex = mesh.getVertex(vertexId);
191  auto envelopeVertex = envelope->addVertex(vertex);
192  vertexMap[vertexId] = envelopeVertex->getId();
193  }
194 
195  // Update face connectivity in the envelope
196  faceEnvelopeConnect[j] = vertexMap.at(vertexId);
197  }
198 
199  // Add face to envelope
200  ElementType faceType = interface.getType();
201  PatchKernel::CellIterator cellItr = envelope->addCell(faceType, true, std::move(faceEnvelopeConnect));
202  meshToEnvelope.insert({{faceIndex,cellItr->getId()}});
203  }
204 
205  envelope->squeeze();
206  envelope->buildAdjacencies();
207  envelope->buildInterfaces();
208 
209  envelope->getVTK().setName("geometry_002") ;
210  envelope->write() ;
211 
212  return envelope;
213 
214 }
