164 void PatchNumberingInfo::_extract()
165 {
166  long consecutiveId;
167 
168 #if BITPIT_ENABLE_MPI==1
169  // Initialize communications
170  size_t exchangeDataSize = sizeof(consecutiveId);
171  std::unique_ptr<DataCommunicator> dataCommunicator;
172 
173  if (m_patch->getProcessorCount() > 1) {
174  // Create the data communicator
175  dataCommunicator = std::unique_ptr<DataCommunicator>(new DataCommunicator(m_patch->getCommunicator()));
176 
177  // Set and start the receives
178  for (const auto entry : m_patch->getGhostExchangeTargets()) {
179  const int rank = entry.first;
180  const auto &list = entry.second;
181 
182  dataCommunicator->setRecv(rank, list.size() * exchangeDataSize);
183  dataCommunicator->startRecv(rank);
184  }
185  }
186 #endif
187 
188 #if BITPIT_ENABLE_MPI==1
189  // Get the internal count of all the partitions
190  m_nGlobalInternals.resize(m_patch->getProcessorCount());
191  if (m_patch->getProcessorCount() > 1) {
192  long nLocalInternals = m_patch->getInternalCount();
193  MPI_Allgather(&nLocalInternals, 1, MPI_LONG, m_nGlobalInternals.data(), 1, MPI_LONG, m_patch->getCommunicator());
194  } else {
195  m_nGlobalInternals[0] = m_patch->getInternalCount();
196  }
197 #endif
198 
199  // Evalaute the consecutive id of the internal cells
200  if (m_patch->getInternalCount() > 0) {
201  std::map<long,long> nativeIds;
202  for (auto itr = m_patch->internalConstBegin(); itr != m_patch->internalConstEnd(); ++itr) {
203  long id = itr.getId();
204  long nativeId = m_patch->_getCellNativeIndex(id);
205 
206  nativeIds.insert({nativeId, id});
207  }
208 
209 #if BITPIT_ENABLE_MPI==1
210  consecutiveId = getCellGlobalCountOffset();
211 #else
212  consecutiveId = 0;
213 #endif
214  for (auto itr = nativeIds.begin(); itr != nativeIds.end(); ++itr) {
215  m_cellLocalToConsecutiveMap.insert({itr->second, consecutiveId++});
216  }
217 
218  std::map<long,long>().swap(nativeIds);
219  }
220 
221 #if BITPIT_ENABLE_MPI==1
222  // Communicate the consecutive id of the ghost cells
223  if (m_patch->getProcessorCount() > 1) {
224  // Set and start the sends
225  for (const auto entry : m_patch->getGhostExchangeSources()) {
226  const int rank = entry.first;
227  auto &list = entry.second;
228 
229  dataCommunicator->setSend(rank, list.size() * exchangeDataSize);
230  SendBuffer &buffer = dataCommunicator->getSendBuffer(rank);
231  for (long id : list) {
232  buffer << m_cellLocalToConsecutiveMap.at(id);
233  }
234  dataCommunicator->startSend(rank);
235  }
236 
237  // Receive the consecutive ids of the ghosts
238  int nCompletedRecvs = 0;
239  while (nCompletedRecvs < dataCommunicator->getRecvCount()) {
240  int rank = dataCommunicator->waitAnyRecv();
241  const auto &list = m_patch->getGhostExchangeTargets(rank);
242 
243  RecvBuffer &buffer = dataCommunicator->getRecvBuffer(rank);
244  for (long id : list) {
245  buffer >> consecutiveId;
246  m_cellLocalToConsecutiveMap.insert({id, consecutiveId});
247  }
248 
249  ++nCompletedRecvs;
250  }
251 
252  // Wait for the sends to finish
253  dataCommunicator->waitAllSends();
254  }
255 #endif
256 }

368 int PatchNumberingInfo::getCellRankFromConsecutive(long id) const
369 {
370  long offset = 0;
371  for (int k = 0; k < m_patch->getProcessorCount(); ++k) {
372  offset += m_nGlobalInternals[k];
373  if (id < offset) {

376  }
377 
378  return -1;
