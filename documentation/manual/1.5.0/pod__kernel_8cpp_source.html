262 void PODKernel::initializeCommunicator(MPI_Comm communicator)
263 {
264  // Communication can be set just once
265  if (isCommunicatorSet())
266  throw std::runtime_error ("PODKernel communicator can be set just once");
267 
268  // The communicator has to be valid
269  if (communicator == MPI_COMM_NULL)
270  throw std::runtime_error ("PODKernel communicator is not valid");
271 
272  // Create a copy of the user-specified communicator
273  //
274  // No library routine should use MPI_COMM_WORLD as the communicator;
275  // instead, a duplicate of a user-specified communicator should always
276  // be used.
277  MPI_Comm_dup(communicator, &m_communicator);
278 }
