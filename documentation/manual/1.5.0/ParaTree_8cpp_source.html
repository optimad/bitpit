281  void
282  ParaTree::_initializePartitions() {
283 #if BITPIT_ENABLE_MPI==1
284  // Set MPI information
285  if (isCommSet()) {
286  MPI_Comm_size(m_comm, &m_nproc);
287  MPI_Comm_rank(m_comm, &m_rank);
288  } else {
289  m_rank = 0;
290  m_nproc = 1;

292 #else
293  // Set dummy MPI information
294  m_rank = 0;
295  m_nproc = 1;
296 #endif

298  // Create the data structures for storing partion information
299  m_partitionFirstDesc.resize(m_nproc);
300  m_partitionLastDesc.resize(m_nproc);
301  m_partitionRangeGlobalIdx.resize(m_nproc);
302  m_partitionRangeGlobalIdx0.resize(m_nproc);
303  uint64_t lastDescMorton = m_octree.getLastDescMorton();
304  uint64_t firstDescMorton = m_octree.getFirstDescMorton();
305  for(int p = 0; p < m_nproc; ++p){
306  m_partitionRangeGlobalIdx0[p] = 0;
307  m_partitionRangeGlobalIdx[p] = m_globalNumOctants - 1;
308  m_partitionLastDesc[p] = lastDescMorton;
309  m_partitionFirstDesc[p] = firstDescMorton;
310  }
311  }

459  void ParaTree::dump(std::ostream &stream, bool full)
460  {
461  // Version
462  utils::binary::write(stream, getDumpVersion());
463 
464  // Tree data
465  utils::binary::write(stream, getNproc());
466 
467  utils::binary::write(stream, getDim());
468 
469  utils::binary::write(stream, getSerial());
470  utils::binary::write(stream, getNofGhostLayers());
471  utils::binary::write(stream, getMaxDepth());
472  utils::binary::write(stream, getStatus());
473  utils::binary::write(stream, getBalanceCodimension());
474 
475  for (int i = 0; i < m_global.m_nfaces; i++) {
476  utils::binary::write(stream, getPeriodic(i));
477  }
478 
479  // Octant data
480  uint32_t nOctants = getNumOctants();
481  utils::binary::write(stream, nOctants);
482 
483  uint32_t nGlobalOctants = getGlobalNumOctants();
484  utils::binary::write(stream, nGlobalOctants);
485 
486  for (uint32_t i = 0; i < nOctants; i++) {
487  const Octant &octant = m_octree.m_octants[i];
488 
489  utils::binary::write(stream, octant.getLevel());
490  utils::binary::write(stream, octant.getX());
491  utils::binary::write(stream, octant.getY());
492  utils::binary::write(stream, octant.getZ());
493  utils::binary::write(stream, octant.getGhostLayer());
494 
495  for (size_t k = 0; k < Octant::INFO_ITEM_COUNT; ++k) {
496  utils::binary::write(stream, (bool) octant.m_info[k]);
497  }
498 
499  utils::binary::write(stream, octant.getBalance());
500  utils::binary::write(stream, octant.getMarker());
501  }
502 
503  // Information about partitioning
504  for (int k = 0; k < m_nproc; ++k) {
505  utils::binary::write(stream, m_partitionFirstDesc[k]);
506  }
507 
508  for (int k = 0; k < m_nproc; ++k) {
509  utils::binary::write(stream, m_partitionLastDesc[k]);
510  }
511 
512  for (int k = 0; k < m_nproc; ++k) {
513  utils::binary::write(stream, m_partitionRangeGlobalIdx[k]);
514  }
515 
516  // Extended information (mapping, ...)
517  utils::binary::write(stream, full);
518  if (full) {
519  utils::binary::write(stream, m_lastOp);
520  if (m_lastOp == OP_ADAPT_MAPPED){
521  for (auto idx : m_mapIdx) {
522  utils::binary::write(stream, idx);
523  }
524 
525  utils::binary::write(stream, m_octree.m_lastGhostBros.size());
526  for (auto lastGhostBrother : m_octree.m_lastGhostBros) {
527  utils::binary::write(stream, lastGhostBrother);
528  }
529  }
530  else if (m_lastOp == OP_LOADBALANCE || m_lastOp == OP_LOADBALANCE_FIRST){
531  for (int i = 0; i < m_nproc; ++i) {
532  utils::binary::write(stream, m_partitionRangeGlobalIdx0[i]);
533  }
534  }
535  }
536 
537  }

545  void ParaTree::restore(std::istream &stream)
546  {
547  // Version
548  int version;
549  utils::binary::read(stream, version);
550  if (version != getDumpVersion()) {
551  throw std::runtime_error ("The version of the file does not match the required version");
552  }
553 
554  // Check if the number of processors matches
555  int nProcs;
556  utils::binary::read(stream, nProcs);
557  if (nProcs != m_nproc) {
558  throw std::runtime_error ("The restart was saved with a different number of processors.");
559  }
560 
561  // Initialize the tree
562  uint8_t dimension;
563  utils::binary::read(stream, dimension);
564 
565  m_octree.initialize(dimension);
566  m_trans.initialize(dimension);
567  reinitialize(dimension, m_log->getName());
568  reset(false);
569 
570  // Set tree properties
571  utils::binary::read(stream, m_serial);
572  utils::binary::read(stream, m_nofGhostLayers);
573  utils::binary::read(stream, m_maxDepth);
574  utils::binary::read(stream, m_status);
575 
576  bool balanceCodimension;
577  utils::binary::read(stream, balanceCodimension);
578  setBalanceCodimension(balanceCodimension);
579 
580  for (int i = 0; i < m_global.m_nfaces; i++) {
581  bool periodicBorder;
582  utils::binary::read(stream, periodicBorder);
583  if (periodicBorder){
584  setPeriodic(i);
585  }
586  }
587 
588  // Restore octants
589  uint32_t nOctants;
590  utils::binary::read(stream, nOctants);
591  m_octree.m_sizeOctants = nOctants;
592 
593  uint32_t nGlobalOctants;
594  utils::binary::read(stream, nGlobalOctants);
595  m_globalNumOctants = nGlobalOctants;
596 
597  m_octree.m_octants.clear();
598  m_octree.m_octants.reserve(nOctants);
599  for (uint32_t i = 0; i < nOctants; i++) {
600  // Create octant
601  uint8_t level;
602  utils::binary::read(stream, level);
603 
604  uint32_t x;
605  utils::binary::read(stream, x);
606 
607  uint32_t y;
608  utils::binary::read(stream, y);
609 
610  uint32_t z;
611  utils::binary::read(stream, z);
612 
613  Octant octant(false, m_dim, level, x, y, z);
614 
615  int ghost;
616  utils::binary::read(stream, ghost);
617  octant.setGhostLayer(ghost);
618 
619  // Set octant info
620  for (size_t k = 0; k < Octant::INFO_ITEM_COUNT; ++k) {
621  bool bit;
622  utils::binary::read(stream, bit);
623  octant.m_info.set(k, bit);
624  }
625 
626  // Set octant 2:1 balance
627  bool balance21;
628  utils::binary::read(stream, balance21);
629  octant.setBalance(balance21);
630 
631  // Set marker
632  int8_t marker;
633  utils::binary::read(stream, marker);
634  octant.setMarker(marker);
635 
636  // Add octant to the list
637  m_octree.m_octants.push_back(std::move(octant));
638  }
639 
640  m_octree.updateLocalMaxDepth();
641 
642  // Set first last descendant
643  m_partitionFirstDesc.resize(m_nproc);
644  for (int k = 0; k < m_nproc; ++k) {
645  uint64_t descendant;
646  utils::binary::read(stream, descendant);
647  m_partitionFirstDesc[k] = descendant;
648  }
649  m_octree.m_firstDescMorton = m_partitionFirstDesc[m_rank];
650 
651  m_partitionLastDesc.resize(m_nproc);
652  for (int k = 0; k < m_nproc; ++k) {
653  uint64_t descendant;
654  utils::binary::read(stream, descendant);
655  m_partitionLastDesc[k] = descendant;
656  }
657  m_octree.m_lastDescMorton = m_partitionLastDesc[m_rank];
658 
659  // Set partitions and parallel information
660  m_partitionRangeGlobalIdx.resize(m_nproc);
661  for (int k = 0; k < m_nproc; ++k) {
662  uint64_t rangeGlobalIdx;
663  utils::binary::read(stream, rangeGlobalIdx);
664  m_partitionRangeGlobalIdx[k] = rangeGlobalIdx;
665  }
666 
667 #if BITPIT_ENABLE_MPI==1
668  if (!m_serial) {
669  computeGhostHalo();
670  }
671 #endif
672 
673  // Full restore (i.e. restore with mapper of last operation)
674  m_mapIdx.clear();
675 
676  for (int i = 0; i < m_nproc; ++i) {
677  m_partitionRangeGlobalIdx0[i] = 0;
678  }
679 
680  bool full;
681  utils::binary::read(stream, full);
682  if (full) {
683  utils::binary::read(stream, m_lastOp);
684  if (m_lastOp == OP_ADAPT_MAPPED) {
685  m_mapIdx.resize(m_octree.m_octants.size());
686  for (size_t i = 0; i < m_octree.m_octants.size(); ++i) {
687  utils::binary::read(stream, m_mapIdx[i]);
688  }
689 
690  size_t lastGhostBrosSize;
691  utils::binary::read(stream, lastGhostBrosSize);
692  m_octree.m_lastGhostBros.resize(lastGhostBrosSize);
693  for (size_t i = 0; i < lastGhostBrosSize; ++i) {
694  utils::binary::read(stream, m_octree.m_lastGhostBros[i]);
695  }
696  }
697  else if (m_lastOp == OP_LOADBALANCE || m_lastOp == OP_LOADBALANCE_FIRST){
698  for (int i = 0; i < m_nproc; ++i) {
699  utils::binary::read(stream, m_partitionRangeGlobalIdx0[i]);
700  }
701  }
702  }
703  else {
704  m_lastOp = OP_INIT;
705  }
706  }

815  void
816  ParaTree::setComm(MPI_Comm communicator)
817  {
818  // Communication can be set just once
819  if (isCommSet()) {
820  throw std::runtime_error ("PABLO communicator can be set just once");
821  }
822 
823  // The communicator has to be valid
824  if (communicator == MPI_COMM_NULL) {
825  throw std::runtime_error ("PABLO communicator is not valid");
826  }
827 
828  // Create a copy of the user-specified communicator
829  //
830  // No library routine should use MPI_COMM_WORLD as the communicator;
831  // instead, a duplicate of a user-specified communicator should always
832  // be used.
833  MPI_Comm_dup(communicator, &m_comm);
834 
835  // Initialize partition data
836  _initializePartitions();
837  }

854  void
855  ParaTree::replaceComm(MPI_Comm communicator, MPI_Comm *previousCommunicator)
856  {
857  // The communicator has to be already set
858  if (!isCommSet()) {
859  throw std::runtime_error ("PABLO communicator is not set");
860  }
861 
862  // Check if the communicator is valid
863  //
864  // The communicator should be equaivalent to the one currently set,
865  // i.e., the rank of the processes have to be the same in both
866  // communicators.
867  int updatedRank;
868  MPI_Comm_rank(communicator, &updatedRank);
869  int currentRank = getRank();
870 
871  int isValid = (updatedRank == currentRank) ? 1 : 0;
872  MPI_Allreduce(MPI_IN_PLACE, &isValid, 1, MPI_INT, MPI_LAND, m_comm);
873  if (!isValid) {
874  throw std::runtime_error ("New communicator is not valid");
875  }
876 
877  // Handle previous communicator
878  if (previousCommunicator) {
879  *previousCommunicator = m_comm;
880  m_comm = MPI_COMM_NULL;
881  } else {
882  freeComm();
883  }
884 
885  // Set the communicator
886  setComm(communicator);
887  }
