139  m_tolerance(other.m_tolerance),
140  m_rank(other.m_rank),
141  m_nProcessors(other.m_nProcessors)
142 #if BITPIT_ENABLE_MPI==1
143  , m_communicator(MPI_COMM_NULL),
144  m_partitioned(other.m_partitioned),
145  m_partitioningStatus(other.m_partitioningStatus),
146  m_haloSize(other.m_haloSize),
147  m_ghostOwners(other.m_ghostOwners),
148  m_ghostExchangeTargets(other.m_ghostExchangeTargets),
149  m_ghostExchangeSources(other.m_ghostExchangeSources)
150 #endif
151 {
152  // Register the patch
153  patch::manager().registerPatch(this);
154 
155  // Update the VTK streamer
156  //
157  // The pointer to VTK streamers are copied, if there are pointer to the
158  // original object they have to be replace with a pointer to this object.
159  std::vector<std::string> streamedGeomFields;
160  streamedGeomFields.reserve(m_vtk.getGeomDataCount());
161  for (auto itr = m_vtk.getGeomDataBegin(); itr != m_vtk.getGeomDataEnd(); ++itr) {
162  const VTKField &field = *itr;
163  if (&field.getStreamer() != &other) {
164  continue;
165  }
166 
167  streamedGeomFields.push_back(field.getName());
168  }
169 
170  for (const std::string &name : streamedGeomFields) {
171  const VTKField &field = *(m_vtk.findGeomData(name));
172  VTKField updatedField(field);
173  updatedField.setStreamer(*this);
174 
175  m_vtk.setGeomData(std::move(updatedField));
176  }
177 
178  std::vector<std::string> streamedDataFields;
179  streamedDataFields.reserve(m_vtk.getDataCount());
180  for (auto itr = m_vtk.getDataBegin(); itr != m_vtk.getDataEnd(); ++itr) {
181  const VTKField &field = *itr;
182  if (&field.getStreamer() != &other) {
183  continue;
184  }
185 
186  streamedDataFields.push_back(field.getName());
187  }
188 
189  for (const std::string &name : streamedDataFields) {
190  const VTKField &field = *(m_vtk.findData(name));
191  VTKField updatedField(field);
192  updatedField.setStreamer(*this);
193 
194  m_vtk.removeData(field.getName());
195  m_vtk.addData(std::move(updatedField));
196  }
197 
198 #if BITPIT_ENABLE_MPI==1
199  // Set the communicator
200  MPI_Comm communicator = other.getCommunicator();
201  if (communicator != MPI_COMM_NULL) {
202  setCommunicator(communicator);
203  }
204 #endif
205 }

315 std::vector<adaption::Info> PatchKernel::update(bool trackAdaption, bool squeezeStorage)
316 {
317  std::vector<adaption::Info> updateInfo;
318 
319  // Check if there are pending changes
320  bool spawnNeeed = (getSpawnStatus() == SPAWN_NEEDED);
321  bool adaptionDirty = (getAdaptionStatus(true) == ADAPTION_DIRTY);
322  bool boundingBoxDirty = isBoundingBoxDirty();
323 
324  bool pendingChanges = (spawnNeeed || adaptionDirty || boundingBoxDirty);
325  if (!pendingChanges) {
326  return updateInfo;
327  }
328 
329  // Spawn
330  if (spawnNeeed) {
331  mergeAdaptionInfo(spawn(trackAdaption), updateInfo);
332  }
333 
334  // Adaption
335  if (adaptionDirty) {
336  mergeAdaptionInfo(adaption(trackAdaption, squeezeStorage), updateInfo);
337  }
338 
339  // Update bounding box
340  //
341  // Previous updates may already have updated the bounding box, before
342  // doing the update re-check if it is still dirty.
343  boundingBoxDirty = isBoundingBoxDirty();
344  if (boundingBoxDirty) {
345  updateBoundingBox();
346  }
347 
348  return updateInfo;
349 }

358 std::vector<adaption::Info> PatchKernel::spawn(bool trackSpawn)
359 {
360  std::vector<adaption::Info> spawnInfo;
361 
362 #if BITPIT_ENABLE_MPI==1
363  // This is a collevtive operation and should be called by all processes
364  if (isCommunicatorSet()) {
365  const auto &communicator = getCommunicator();
366  MPI_Barrier(communicator);
367  }
368 #endif
369 
370  // Check spawn status
371  SpawnStatus spawnStatus = getSpawnStatus();
372  if (spawnStatus == SPAWN_UNNEEDED || spawnStatus == SPAWN_DONE) {
373  return spawnInfo;
374  }
375 
376  // Begin patch alteration
377  beginAlteration();
378 
379  // Alter patch
380  spawnInfo = _spawn(trackSpawn);
381 
382  // End patch alteration
383  endAlteration(true);
384 
385  // Spwan is done
386  setSpawnStatus(SPAWN_DONE);
387 
388  // Done
389  return spawnInfo;
390 }

505 {
506  AdaptionStatus adaptionStatus = getAdaptionStatus();
507  if (adaptionStatus == ADAPTION_UNSUPPORTED || adaptionStatus == ADAPTION_CLEAN) {
508  return;

516  _adaptionCleanup();
517 
518  // Update the status

788 void PatchKernel::write(VTKWriteMode mode)
789 {
790  // Get VTK cell count
791  long vtkCellCount = 0;
792  if (m_vtkWriteTarget == WRITE_TARGET_CELLS_ALL) {
793  vtkCellCount = getCellCount();
794 #if BITPIT_ENABLE_MPI==1
795  } else if (m_vtkWriteTarget == WRITE_TARGET_CELLS_INTERNAL) {
796  vtkCellCount = getInternalCount();
797 #endif
798  }
799 
800  // Set the dimensinos of the mesh
801  PiercedStorage<long, long> vertexWriteFlag(1, &m_vertices);
802  vertexWriteFlag.fill(false);
803 
804  bool vtkFaceStreamNeeded = false;
805  for (const Cell &cell : m_cells) {
806  if (cell.getDimension() > 2 && !cell.hasInfo()) {
807  vtkFaceStreamNeeded = true;
808  break;
809  }
810  }
811 
812  long vtkConnectSize = 0;
813  long vtkFaceStreamSize = 0;
814  for (const Cell &cell : getVTKCellWriteRange()) {
815  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
816  const int nCellVertices = cellVertexIds.size();
817  for (int k = 0; k < nCellVertices; ++k) {
818  long vertexId = cellVertexIds[k];
819  vertexWriteFlag.at(vertexId) = true;
820  }
821 
822  vtkConnectSize += nCellVertices;
823  if (vtkFaceStreamNeeded) {
824  if (cell.getDimension() <= 2 || cell.hasInfo()) {
825  vtkFaceStreamSize += 1;
826  } else {
827  vtkFaceStreamSize += cell.getFaceStreamSize();
828  }
829  }
830  }
831 
832  int vtkVertexCount = 0;
833  m_vtkVertexMap.unsetKernel(true);
834  m_vtkVertexMap.setStaticKernel(&m_vertices);
835  for (VertexConstIterator itr = vertexConstBegin(); itr != vertexConstEnd(); ++itr) {
836  std::size_t vertexRawId = itr.getRawIndex();
837  if (vertexWriteFlag.rawAt(vertexRawId)) {
838  m_vtkVertexMap.rawAt(vertexRawId) = vtkVertexCount++;
839  } else {
840  m_vtkVertexMap.rawAt(vertexRawId) = Vertex::NULL_ID;
841  }
842  }
843 
844  m_vtk.setDimensions(vtkCellCount, vtkVertexCount, vtkConnectSize, vtkFaceStreamSize);
845 
846  // Write the mesh
847  m_vtk.write(mode);
848 }
