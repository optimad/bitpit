225 void PiercedSyncMaster::registerSlave(PiercedSyncSlave *slave, SyncMode syncMode)
226 {
227  if (m_slaves.count(slave) > 0) {
228  throw std::out_of_range("Slave is already registered");
229  }
230 
231  // Register the slave
232  m_slaves.insert({slave, syncMode});
233 
234  // Add the slave to the proper synchronization group
235  SyncGroup &syncGroup = m_syncGroups.at(syncMode);
236  syncGroup.push_back(slave);
237 
238  // If needed enable synchronization
239  if (!isSyncEnabled() && syncMode != SYNC_MODE_DISABLED) {
240  setSyncEnabled(true);
241  }
242 }

249 void PiercedSyncMaster::processSyncAction(const PiercedSyncAction &action)
250 {
251  if (!isSyncEnabled()) {
252  return;
253  }
254 
255  // Synchronize concurrent slaves
256  for (PiercedSyncSlave *slave : m_syncGroups.at(PiercedSyncMaster::SYNC_MODE_CONCURRENT)) {
257  commitSyncAction(slave, action);
258  }
259 
260  // If there are journaled slaves the synchronization action needs to be
261  // journaled
262  if (m_syncGroups.at(PiercedSyncMaster::SYNC_MODE_JOURNALED).size() > 0) {
263  journalSyncAction(action);
264  }
265 }

283 void PiercedSyncMaster::journalSyncAction(const PiercedSyncAction &action)
284 {
285  switch (action.type) {
286 
287  case PiercedSyncAction::TYPE_CLEAR:
288  m_syncJournal.resize(1);
289  m_syncJournal[0] = action;
290  break;
291 
292  case PiercedSyncAction::TYPE_NOOP:
293  break;
294 
295  default:
296  m_syncJournal.push_back(action);

307 void PiercedSyncMaster::unregisterSlave(const PiercedSyncSlave *slave)
308 {
309  // Check if the slave was actually registered
310  if (!isSlaveRegistered(slave)) {
311  return;
312  }
313 
314  // Remove the slave from the synchronization group
315  SyncMode syncMode = getSlaveSyncMode(slave);
316  SyncGroup &syncGroup = m_syncGroups.at(syncMode);
317  for (auto itr = syncGroup.begin(); itr != syncGroup.end(); ++itr) {
318  PiercedSyncSlave *item = *itr;
319  if (item == slave) {
320  syncGroup.erase(itr);
321  break;
322  }
323  }
324 
325  // Unregister the slave
326  m_slaves.erase(const_cast<PiercedSyncSlave *>(slave));
327 }

353 void PiercedSyncMaster::sync()
354 {
355  // Only journaled slaved need to be synchronized
356  for (PiercedSyncSlave *slave : m_syncGroups.at(SYNC_MODE_JOURNALED)) {
357  for (const PiercedSyncAction &action : m_syncJournal) {
358  commitSyncAction(slave, action);
359  }
360  }
361 
362  // Clear the sync journal
363  m_syncJournal.clear();
364  m_syncJournal.shrink_to_fit();
365 }
