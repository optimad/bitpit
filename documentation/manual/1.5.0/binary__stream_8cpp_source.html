221 bool BinaryStream::seekg(std::streamoff offset, std::ios_base::seekdir way)
222 {
223  if ((way == std::ios_base::beg) && (offset < (long) getSize())) {
224  m_pos = offset;
225  } else if ((way == std::ios_base::cur) && (m_pos + offset < getSize())) {
226  m_pos += offset;
227  } else if ((way == std::ios_base::end) && ((long) getSize() - offset >= 0)) {
228  m_pos = getSize() - offset;
229  } else {
230  return false;
231  }
232 
233  return true;
234 }

270 void BinaryStream::setCapacity(std::size_t capacity)
271 {
272  m_chunkSize = 1;
273  std::size_t nChunks = capacity;
274  while (nChunks > std::numeric_limits<int>::max()) {
275  m_chunkSize = 2 * m_chunkSize;
276  nChunks = (1 + ((capacity - 1) / m_chunkSize));
277  }
278 
279  m_buffer.resize(nChunks * m_chunkSize);
280 }

523 void OBinaryStream::write(const char *data, std::size_t size)
524 {
525  if (getSize() - m_pos < size) {
526  // If the stream is not expandable, the request for a new size
527  // will throw an exception.
528  std::size_t bufferSize = getSize() + size;
529  setSize(bufferSize);
530  }
531 
532  for (std::size_t i = 0; i < size; ++i) {
533  m_buffer[m_pos] = data[i];
534  ++m_pos;
535  }
536 }
