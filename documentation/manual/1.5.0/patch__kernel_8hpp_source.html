91  struct CellPositionLess
92  {
93  CellPositionLess(PatchKernel &patch, bool native = true)
94  : m_patch(patch), m_native(native)
95  {
96  }
97 
98  bool operator()(const long &id_1, const long &id_2) const
99  {
100  std::array<double, 3> centroid_1;
101  std::array<double, 3> centroid_2;
102  if (m_native) {
103  centroid_1 = m_patch.evalCellCentroid(id_1);
104  centroid_2 = m_patch.evalCellCentroid(id_2);
105  } else {
106  centroid_1 = m_patch.PatchKernel::evalCellCentroid(id_1);
107  centroid_2 = m_patch.PatchKernel::evalCellCentroid(id_2);
108  }
109 
110  for (int k = 0; k < 3; ++k) {
111  if (std::abs(centroid_1[k] - centroid_2[k]) <= m_patch.getTol()) {
112  continue;
113  }
114 
115  return centroid_1[k] < centroid_2[k];
116  }
117 
118  // If we are here the two cell centroids coincide. It's not
119  // possible to define an order for the two cells.
120  std::ostringstream stream;
121  stream << "It was not possible to define an order for cells " << id_1 << " and " << id_2 << ". ";
122  stream << "The two cells have the same centroid.";
123  throw std::runtime_error (stream.str());
124  }
125 
126  PatchKernel &m_patch;
127  bool m_native;
128  };
