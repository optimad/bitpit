54 template<typename item_t, typename id_t>
55 std::unordered_map<id_t, id_t> PatchKernel::consecutiveItemRenumbering(PiercedVector<item_t, id_t> &container, long offset)
56 {
57  // Build renumber map
58  std::unordered_map<id_t, id_t> renumberMap;
59 
60  id_t counter = offset;
61  for(const item_t &item : container) {
62  id_t originalId = item.getId();
63  id_t finalId = counter++;
64 
65  renumberMap.insert({originalId, finalId});
66  }
67 
68  // Renumber items
69  mappedItemRenumbering(container, renumberMap);
70 
71  return renumberMap;
72 }

80 template<typename item_t, typename id_t>
81 void PatchKernel::mappedItemRenumbering(PiercedVector<item_t, id_t> &container,
82  const std::unordered_map<id_t, id_t> &renumberMap)
83 {
84  // Find an unused id
85  id_t unusedId = std::numeric_limits<id_t>::max();
86  while (container.exists(unusedId)) {
87  --unusedId;
88  if (unusedId < 0) {
89  break;
90  }
91  }
92 
93  // Renumber the items
94  std::unordered_map<id_t, id_t> conflictMap;
95 
96  for(const item_t &item : container) {
97  // Original id of the item
98  //
99  // To get the original id of the item we use the conflict map, once
100  // we get the id of an item we can remove it from the conflict map.
101  id_t currentId = item.getId();
102 
103  id_t originalId;
104  auto conflictMapItr = conflictMap.find(currentId);
105  if (conflictMapItr != conflictMap.end()) {
106  originalId = conflictMapItr->second;
107  conflictMap.erase(conflictMapItr);
108  } else {
109  originalId = currentId;
110  }
111 
112  // Final id of the item
113  id_t finalId = renumberMap.at(originalId);
114 
115  // Nothing else to do if the item has already the correct id, otherwise
116  // we need to renumber the item.
117  if (currentId == finalId) {
118  continue;
119  }
120 
121  // Check if the final id is already in the container.
122  //
123  // If there is a conflict, the item that holds the final id will
124  // be temporarly renumbered with the unused id find before, the current
125  // item will be renumberd with is final id and eventually the item with
126  // the temporary id will be assoicated with the, now avilable, id of
127  // the current item.
128  id_t oldConflictId;
129  id_t newConflicdId;
130  id_t tmpConflictId;
131 
132  bool conflict = container.exists(finalId);
133  if (conflict) {
134  if (unusedId < 0) {
135  throw std::runtime_error("Renumbering requires an unused id, but all ids are used.");
136  }
137 
138  oldConflictId = finalId;
139  newConflicdId = currentId;
140  tmpConflictId = unusedId;
141  }
142 
143  // Temporary renumber of the conflicting item
144  //
145  // There is no need to change the id of the element, it is sufficient
146  // update the container. The id will be updated later.
147  if (conflict) {
148  container.updateId(oldConflictId, tmpConflictId);
149  }
150 
151  // Renumber of the current element
152  container[currentId].setId(finalId);
153  container.updateId(currentId, finalId);
154 
155  // Renumber the conflicting element with the previous id of the item
156  if (conflict) {
157  container.updateId(tmpConflictId, newConflicdId);
158  container[newConflicdId].setId(newConflicdId);
159 
160  // Update conflic map
161  auto conflictMapItr = conflictMap.find(oldConflictId);
162  if (conflictMapItr == conflictMap.end()) {
163  conflictMap.insert({newConflicdId, oldConflictId});
164  } else {
165  conflictMap.insert({newConflicdId, conflictMapItr->second});
166  conflictMap.erase(conflictMapItr);
167  }
168  }
169  }
170 }
171 
172 }
