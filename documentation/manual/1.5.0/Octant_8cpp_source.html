214 Octant::initialize(uint8_t dim, uint8_t level, bool bound) {
215  m_dim = dim;
216  m_level = level;
217 
218  // Reset the marker
219  m_marker = 0;
220 
221  // Set the coordinates
222  m_x = 0;
223  m_y = 0;
224  m_z = 0;
225 
226  // Initialize octant info
227  m_info.reset();
228  m_info[OctantInfo::INFO_BALANCED] = true;
229  m_ghost = -1;
230 
231  // If this is the root octant we need to set the boundary condition bound
232  // for faces
233  if (m_dim >= 2 && m_level == 0) {
234  uint8_t nf = m_dim*2;
235  for (uint8_t i=0; i<nf; i++){
236  m_info[i] = bound;
237  }
238  }
239 };

842  uint32_t dh,dh2;
843  uint32_t nneigh;
844  uint32_t i,cx,cy,cz;
845  int nchildren = 1<<m_dim;

847  nneigh = (m_level < Global::getMaxLevel()) ? nchildren/2 : 1;

849  dh2 = getSize();
