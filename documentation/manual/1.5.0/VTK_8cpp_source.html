581 void VTK::calcAppendedOffsets(){
582 
583  uint64_t offset(0) ;
584  uint64_t HeaderByte(0) ;
585 
586  if( getHeaderType() == "UInt32"){
587  HeaderByte = sizeof(uint32_t) ;
588  }
589 
590  else if( getHeaderType() == "UInt64") {
591  HeaderByte = sizeof(uint64_t) ;
592  }
593 
594  for( auto & field : m_data ){
595  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::POINT ) {
596  field.setOffset( offset) ;
597  offset += HeaderByte + calcFieldSize(field) ;
598  }
599  }
600 
601  for( auto & field : m_data ){
602  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::CELL) {
603  field.setOffset( offset) ;
604  offset += HeaderByte + calcFieldSize(field) ;
605  }
606  }
607 
608  for( auto & field : m_geometry ){
609  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED ) {
610  field.setOffset( offset) ;
611  offset += HeaderByte + calcFieldSize(field) ;
612  }
613  }
614 
615 }

700 void VTK::writeData( ){
701 
702  std::fstream str ;
703  std::fstream::pos_type position_insert, position_eof ;
704 
705  int length;
706  char* buffer ;
707 
708  str.open( m_fh.getPath( ), std::ios::in | std::ios::out ) ;
709 
710  { // Write Ascii
711 
712  position_insert = str.tellg();
713 
714  //Writing first point data then cell data
715  for( auto &field : m_data ){
716  if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII && field.getLocation() == VTKLocation::POINT ) {
717  str.seekg( position_insert);
718  readDataArray( str, field ) ;
719 
720  str.seekg( field.getPosition() ) ;
721  genericIO::copyUntilEOFInString( str, buffer, length );
722 
723  field.write( str ) ;
724  position_insert = str.tellg();
725 
726  if(position_insert==field.getPosition()){
727  log::cout() << "Error VTK: No data has been written for field " << field.getName() << std::endl;
728  assert(false);
729  }
730 
731  str << std::endl ;
732  genericIO::flushBINARY( str, buffer, length) ;
733 
734  delete [] buffer ;
735 
736  }
737  }
738 
739  for( auto &field : m_data ){
740  if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII && field.getLocation() == VTKLocation::CELL ) {
741  str.seekg( position_insert);
742  readDataArray( str, field ) ;
743 
744  str.seekg( field.getPosition() ) ;
745  genericIO::copyUntilEOFInString( str, buffer, length );
746 
747  field.write( str ) ;
748  position_insert = str.tellg();
749 
750  if(position_insert==field.getPosition()){
751  log::cout() << "Error VTK: No data has been written for field " << field.getName() << std::endl;
752  assert(false);
753  }
754 
755  str << std::endl ;
756  genericIO::flushBINARY( str, buffer, length) ;
757 
758  delete [] buffer ;
759  }
760  }
761 
762  for( auto &field : m_geometry ){
763  if( field.isEnabled() && field.getCodification() == VTKFormat::ASCII ) {
764  str.seekg( position_insert);
765  readDataArray( str, field ) ;
766 
767  str.seekg( field.getPosition() ) ;
768  genericIO::copyUntilEOFInString( str, buffer, length );
769 
770  field.write( str ) ;
771  position_insert = str.tellg();
772 
773  if(position_insert==field.getPosition()){
774  log::cout() << "Error VTK: No data has been written for field " << field.getName() << std::endl;
775  assert(false);
776  }
777 
778  str << std::endl ;
779  genericIO::flushBINARY( str, buffer, length) ;
780 
781  delete [] buffer ;
782  }
783  }
784 
785  // Position the stream at the beginning of the stream
786  str.seekg(0, std::ios::beg);
787 
788  }
789 
790  { // Write Appended
791 
792  char c_;
793  std::string line ;
794  std::fstream::pos_type position_appended ;
795  std::fstream::pos_type position_before_write ;
796 
797  //Go to the initial position of the appended section
798  while( getline(str, line) && (! bitpit::utils::string::keywordInString( line, "<AppendedData")) ){}
799 
800  str >> c_;
801  while( c_ != '_') str >> c_;
802 
803  position_insert = str.tellg();
804  genericIO::copyUntilEOFInString( str, buffer, length );
805 
806  str.close();
807  str.clear();
808 
809 
810  //Reopening in binary mode
811  str.open( m_fh.getPath( ), std::ios::out | std::ios::in | std::ios::binary);
812  str.seekg( position_insert) ;
813 
814  //str.open( "data.dat", std::ios::out | std::ios::binary);
815 
816  //Writing first point data then cell data
817  for( auto &field : m_data ){
818  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::POINT ) {
819  if( getHeaderType() == "UInt32"){
820  uint32_t nbytes = calcFieldSize(field) ;
821  genericIO::flushBINARY(str, nbytes) ;
822  }
823 
824  else{
825  uint64_t nbytes = calcFieldSize(field) ;
826  genericIO::flushBINARY(str, nbytes) ;
827  }
828 
829  position_before_write = str.tellg();
830  field.write(str) ;
831  if( (uint64_t) str.tellg()-position_before_write != calcFieldSize(field) ){
832  log::cout() << "Error VTK: Data written do not corrispond to size of field " << field.getName() << std::endl;
833  assert(false);
834  }
835  }
836  }
837 
838  for( auto &field : m_data ){
839  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED && field.getLocation() == VTKLocation::CELL ) {
840 
841  if( getHeaderType() == "UInt32"){
842  uint32_t nbytes = calcFieldSize(field) ;
843  genericIO::flushBINARY(str, nbytes) ;
844  }
845 
846  else{
847  uint64_t nbytes = calcFieldSize(field) ;
848  genericIO::flushBINARY(str, nbytes) ;
849  }
850 
851  position_before_write = str.tellg();

855  assert(false);
856  }
857 
858  }
859  }
860 
861  //Writing Geometry Data
862  for( auto &field : m_geometry ){
863  if( field.isEnabled() && field.getCodification() == VTKFormat::APPENDED ) {
864  if( getHeaderType() == "UInt32"){

878  assert(false);
879  }
880  }
881  }
882 
883  genericIO::flushBINARY( str, buffer, length) ;
884 
885  delete [] buffer ;

890

898 void VTK::writeDataHeader( std::fstream &str, bool parallel ){
899 
900  VTKLocation location ;
901  std::stringstream scalars, vectors ;
902 
903  for( int j=0; j<2; j++){
904 
905  if( j==0 ) location = VTKLocation::POINT ;
906  if( j==1 ) location = VTKLocation::CELL ;
907 
908  scalars.str("");
909  vectors.str("");
910 
911  //Creating Scalar and Vector Lists
912  scalars << "\"" ;
913  vectors << "\"" ;
914 
915  for( auto &field : m_data ){
916  if( field.isEnabled() && field.getLocation() == location){
917  if( field.getFieldType() == VTKFieldType::SCALAR ) scalars << field.getName() << " " ;
918  else if( field.getFieldType() == VTKFieldType::VECTOR ) vectors << field.getName() << " " ;
919  }
920 
921  }
922 
923  scalars << "\"" ;
924  vectors << "\"" ;
925 
926  if( location == VTKLocation::POINT) {
927  str << " <" ;
928  if( parallel ) str << "P" ;
929  str << "PointData " ;
930  }
931 
932  else if( location == VTKLocation::CELL ) {
933  str << " <" ;
934  if( parallel ) str << "P" ;
935  str << "CellData " ;
936  }
937 
938  str << " Scalars=" << scalars.str()
939  << " Vectors=" << vectors.str()
940  << ">" << std::endl;
941 
942  //Writing DataArray
943  for( auto &field : m_data ){
944  if( field.isEnabled() && field.getLocation() == location && !parallel) writeDataArray( str, field ) ;
945  if( field.isEnabled() && field.getLocation() == location && parallel) writePDataArray( str, field );
946  }
947 
948  str << " </" ;
949  if( parallel ) str << "P" ;
950 
951  if( location == VTKLocation::POINT) str << "PointData> " << std::endl;
952  if( location == VTKLocation::CELL) str << "CellData> " << std::endl;
953 
954  }
955
