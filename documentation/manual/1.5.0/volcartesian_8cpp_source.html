217 void VolCartesian::initialize()
218 {
219  // Normals
220  int i = 0;
221  for (int n = 0; n < 3; n++) {
222  for (int k = -1; k <= 1; k += 2) {
223  std::array<double, 3> normal = {{0.0, 0.0, 0.0}};
224  normal[n] = k;
225 
226  m_normals[i++] = normal;
227  }
228  }
229 
230  // Deltas for the evaluation of the vertex neighbours
231  m_vertexNeighDeltas = std::vector<std::array<int, 3>>(8);
232  m_vertexNeighDeltas[0] = {{ 0, 0, 0}};
233  m_vertexNeighDeltas[1] = {{-1, 0, 0}};
234  m_vertexNeighDeltas[2] = {{ 0, -1, 0}};
235  m_vertexNeighDeltas[3] = {{-1, -1, 0}};
236  m_vertexNeighDeltas[4] = {{ 0, 0, -1}};
237  m_vertexNeighDeltas[5] = {{-1, 0, -1}};
238  m_vertexNeighDeltas[6] = {{ 0, -1, -1}};
239  m_vertexNeighDeltas[7] = {{-1, -1, -1}};
240 
241  // Deltas for the evaluation of the edge neighbours
242  m_edgeNeighDeltas = std::vector<std::array<int, 3>>(12);
243  m_edgeNeighDeltas[ 0] = {{-1, 0, -1}};
244  m_edgeNeighDeltas[ 1] = {{ 1, 0, -1}};
245  m_edgeNeighDeltas[ 2] = {{ 0, -1, -1}};
246  m_edgeNeighDeltas[ 3] = {{ 0, 1, -1}};
247  m_edgeNeighDeltas[ 4] = {{-1, -1, 0}};
248  m_edgeNeighDeltas[ 5] = {{ 1, -1, 0}};
249  m_edgeNeighDeltas[ 6] = {{-1, 1, 0}};
250  m_edgeNeighDeltas[ 7] = {{ 1, 1, 0}};

314 void VolCartesian::setDiscretization(const std::array<int, 3> &nCells)
315 {
316  // Spacing
317  for (int n = 0; n < getDimension(); ++n) {
318  // Initialize cells
319  m_nCells1D[n] = nCells[n];
320  m_cellSpacings[n] = (m_maxCoords[n] - m_minCoords[n]) / m_nCells1D[n];
321 
322  m_cellCenters[n].resize(m_nCells1D[n]);
323  for (int i = 0; i < m_nCells1D[n]; i++) {
324  m_cellCenters[n][i] = m_minCoords[n] + (0.5 + i) * m_cellSpacings[n];
325  }
326 
327  log::cout() << " - Cell count along direction " << n << " : " << m_nCells1D[n] << "\n";
328 
329  // Initialize vertices
330  m_nVertices1D[n] = m_nCells1D[n] + 1;
331 
332  m_vertexCoords[n].resize(m_nVertices1D[n]);
333  for (int i = 0; i < m_nVertices1D[n]; i++) {
334  m_vertexCoords[n][i] = m_minCoords[n] + i * m_cellSpacings[n];
335  }
336  }
337 
338  if (!isThreeDimensional()) {
339  m_nCells1D[Vertex::COORD_Z] = 0;
340  m_cellSpacings[Vertex::COORD_Z] = 0.;
341 
342  m_nVertices1D[Vertex::COORD_Z] = 0;
343  }
344 
345  log::cout() << std::endl;
346 
347  // Count the total number of vertices
348  m_nVertices = 1;
349  for (int n = 0; n < getDimension(); ++n) {
350  m_nVertices *= m_nVertices1D[n];
351  }
352  log::cout() << " - Total vertex count: " << m_nVertices << "\n";
353 
354  // Count the total number of cells
355  m_nCells = 1;
356  for (int n = 0; n < getDimension(); ++n) {
357  m_nCells *= m_nCells1D[n];
358  }
359  log::cout() << " - Total cell count: " << m_nCells << "\n";
360 
361  // Count the total number of interfaces
362  m_nInterfaces = 0;
363  for (int n = 0; n < getDimension(); n++) {
364  std::array<int, 3> interfaceCount1D = getInterfaceCountDirection(n);
365 
366  int nDirectionInterfaces = 1;
367  for (int n = 0; n < getDimension(); n++) {
368  nDirectionInterfaces *= interfaceCount1D[n];
369  }
370  m_nInterfaces += nDirectionInterfaces;
371  }
372  log::cout() << " - Total interface count: " << m_nInterfaces << "\n";
373 
374  // Cell volume
375  initializeCellVolume();
376 
377  // Interface area
378  initializeInterfaceArea();
379 
380  // Update patch data structures
381  if (getMemoryMode() == MemoryMode::MEMORY_NORMAL) {
382  // Switch to light mode to reset patchdata structures
383  switchMemoryMode(MemoryMode::MEMORY_LIGHT);
384 
385  // Swich back to normal mode to rebuild patchdata structures
386  switchMemoryMode(MemoryMode::MEMORY_NORMAL);
387  }
388 }

544 void VolCartesian::switchMemoryMode(MemoryMode mode)
545 {
546  if (mode == getMemoryMode()) {
547  return;
548  }
549 
550  // Update the data structures
551  switch (mode) {
552 
553  case MemoryMode::MEMORY_NORMAL:
554  // Spawn the patch to activate normal memory mode
555  spawn(false);
556 
557  break;
558 
559  case MemoryMode::MEMORY_LIGHT:
560  // To put the patch in memory mode we need to reset the generic data
561  // of the patch, therefore we can call the 'reset' implementation of
562  // the kernel.
563  VolumeKernel::reset();
564 
565  // Now the patch needs to be spawn
566  setSpawnStatus(SPAWN_NEEDED);
567 
568  // Set the light memory mode
569  setMemoryMode(mode);
570 
571  break;
572 
573  }
574 }

621 std::vector<adaption::Info> VolCartesian::_spawn(bool trackSpawn)
622 {
623  std::vector<adaption::Info> updateInfo;
624 
625  // If the patch is in 'normal' mode there is nothing to do.
626  if (getMemoryMode() == MEMORY_NORMAL) {
627  return updateInfo;
628  }
629 
630  // Enable advanced editing
631  setExpert(true);
632 
633  // Definition of the mesh
634  addVertices();
635  addCells();
636  addInterfaces();
637 
638  // Disable advanced editing
639  setExpert(false);
640 
641  // Adaption info
642  if (trackSpawn) {
643  updateInfo.emplace_back();
644  adaption::Info &adaptionCellInfo = updateInfo.back();
645  adaptionCellInfo.type = adaption::TYPE_CREATION;
646  adaptionCellInfo.entity = adaption::ENTITY_CELL;
647  adaptionCellInfo.current.reserve(m_cells.size());
648  for (auto &cell : m_cells) {
649  adaptionCellInfo.current.emplace_back();
650  long &cellId = adaptionCellInfo.current.back();
651  cellId = cell.getId();
652  }
653 
654  updateInfo.emplace_back();
655  adaption::Info &adaptionInterfaceInfo = updateInfo.back();
656  adaptionInterfaceInfo.type = adaption::TYPE_CREATION;
657  adaptionInterfaceInfo.entity = adaption::ENTITY_INTERFACE;
658  adaptionInterfaceInfo.current.reserve(m_interfaces.size());
659  for (auto &interface : m_interfaces) {
660  adaptionInterfaceInfo.current.emplace_back();
661  long &interfaceId = adaptionInterfaceInfo.current.back();
662  interfaceId = interface.getId();
663  }
664  } else {
665  updateInfo.emplace_back();
666  }
667 
668  // Updating the adaption brings the patch is in normal memory mode
669  setMemoryMode(MemoryMode::MEMORY_NORMAL);
670 
671  // Done
672  return updateInfo;
673 }

798 void VolCartesian::addInterfacesDirection(const int &direction)
799 {
800  log::cout() << " >> Creating interfaces normal to direction " << direction << "\n";
801 
802  // Info on the interfaces
803  ElementType interfaceType = getInterfaceType();
804 
805  const ReferenceElementInfo &interfaceTypeInfo = ReferenceElementInfo::getInfo(interfaceType);
806  const int nInterfaceVertices = interfaceTypeInfo.nVertices;
807  std::array<int, 3> interfaceCount1D = getInterfaceCountDirection(direction);
808 
809  // Counters
810  std::array<int, 3> counters = {{0, 0, 0}};
811  int &i = counters[Vertex::COORD_X];
812  int &j = counters[Vertex::COORD_Y];
813  int &k = counters[Vertex::COORD_Z];
814 
815  // Creation of the interfaces
816  for (k = 0; (isThreeDimensional()) ? (k < interfaceCount1D[Vertex::COORD_Z]) : (k <= 0); k++) {
817  for (j = 0; j < interfaceCount1D[Vertex::COORD_Y]; j++) {
818  for (i = 0; i < interfaceCount1D[Vertex::COORD_X]; i++) {
819  InterfaceIterator interfaceIterator = VolumeKernel::addInterface(interfaceType);
820  Interface &interface = *interfaceIterator;
821 
822  // Owner id
823  std::array<int, 3> ownerIJK(counters);
824  if (counters[direction] == (interfaceCount1D[direction] - 1)) {
825  ownerIJK[direction] -= 1;
826  }
827 
828  long ownerId = getCellLinearId(ownerIJK);
829 
830  // Neighbour id
831  long neighId;
832  if (counters[direction] != 0 && counters[direction] != interfaceCount1D[direction] - 1) {
833  std::array<int, 3> neighIJK(counters);
834  neighIJK[direction] -= 1;
835 
836  neighId = getCellLinearId(neighIJK);
837  } else {
838  neighId = Element::NULL_ID;
839  }
840 
841  // Owner data
842  Cell &owner = m_cells[ownerId];
843 
844  int ownerFace = 2 * direction;
845  if (counters[direction] == interfaceCount1D[direction] - 1) {
846  ownerFace++;
847  }
848 
849  interface.setOwner(owner.getId(), ownerFace);
850  owner.setInterface(ownerFace, 0, interface.getId());
851  owner.setAdjacency(ownerFace, 0, neighId);
852 
853  // Neighbour data
854  if (counters[direction] != 0 && counters[direction] != interfaceCount1D[direction] - 1) {
855  Cell &neigh = m_cells[neighId];
856 
857  int neighFace = 2 * direction + 1;
858 
859  interface.setNeigh(neigh.getId(), neighFace);
860  neigh.setInterface(neighFace, 0, interface.getId());
861  neigh.setAdjacency(neighFace, 0, ownerId);
862  } else {
863  interface.unsetNeigh();
864  }
865 
866  // Connectivity
867  std::unique_ptr<long[]> connect = std::unique_ptr<long[]>(new long[nInterfaceVertices]);
868  if (direction == Vertex::COORD_X) {
869  connect[0] = getVertexLinearId(i, j, k);
870  connect[1] = getVertexLinearId(i, j + 1, k);
871  if (isThreeDimensional()) {
872  connect[2] = getVertexLinearId(i, j + 1, k + 1);
873  connect[3] = getVertexLinearId(i, j, k + 1);
874  }
875  } else if (direction == Vertex::COORD_Y) {
876  connect[0] = getVertexLinearId(i, j, k);
877  connect[1] = getVertexLinearId(i + 1, j, k);
878  if (isThreeDimensional()) {
879  connect[2] = getVertexLinearId(i + 1, j, k + 1);
880  connect[3] = getVertexLinearId(i, j, k + 1);
