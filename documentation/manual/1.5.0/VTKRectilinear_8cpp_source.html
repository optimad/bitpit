172 void VTKRectilinearGrid::readMetaInformation( ){
173 
174  std::fstream str;
175  std::string line, temp;
176 
177  std::fstream::pos_type position;
178  std::array<int,6> extensions ;
179 
180 
181  str.open( m_fh.getPath( ), std::ios::in ) ;
182 
183  getline( str, line);
184  while( ! bitpit::utils::string::keywordInString( line, "<VTKFile")){
185  getline(str, line);
186  }
187 
188  if( bitpit::utils::string::getAfterKeyword( line, "header_type", '\"', temp) ){
189  setHeaderType( temp) ;
190  }
191 
192  while( ! bitpit::utils::string::keywordInString( line, "<Piece")){
193  getline(str, line);
194  }
195 
196  bitpit::utils::string::getAfterKeyword( line, "Extent", '\"', temp) ;
197  bitpit::utils::string::convertString( temp, extensions );
198 
199  m_localIndex[0][0] = extensions[0] ;
200  m_localIndex[0][1] = extensions[1] ;
201  m_localIndex[1][0] = extensions[2] ;
202  m_localIndex[1][1] = extensions[3] ;
203  m_localIndex[2][0] = extensions[4] ;
204  m_localIndex[2][1] = extensions[5] ;
205 
206  position = str.tellg() ;
207 
208  readDataHeader( str ) ;
209 
210  // Read metadata information
211  for( auto &field : m_geometry ){ //int i=0; i<geometry.size(); ++i){
212  str.seekg( position) ;
213  if( ! readDataArray( str, field ) ) {
214 #if BITPIT_ENABLE_DEBUG
215  log::cout() <<"Geometry field " << field.getName() << " not found, it will be disabled" << std::endl ;
216 #endif
217  field.disable();
218  }
219  }
220 
221 
222  setDimensions( m_localIndex[0][0], m_localIndex[0][1], m_localIndex[1][0], m_localIndex[1][1], m_localIndex[2][0], m_localIndex[2][1] ) ;
223  str.close() ;
224

352 void VTKRectilinearGrid::setDimensions( int n1, int n2, int m1, int m2, int l1, int l2 ){
353 
354  m_dimensions = ( l1 == l2 ) ? 2 :3 ;
355 
356  m_localIndex[0][0] = n1 ;
357  m_localIndex[0][1] = n2 ;
358  m_localIndex[1][0] = m1 ;
359  m_localIndex[1][1] = m2 ;
360  m_localIndex[2][0] = l1 ;
361  m_localIndex[2][1] = l2 ;
362 
363  if(m_procs==1)
364  m_globalIndex = m_localIndex ;
365 
366  m_cells = 1 ;
367  m_points = 1 ;
368 
369  for(int d=0; d<m_dimensions; ++d){
370  m_cells = m_cells * ( m_localIndex[d][1] -m_localIndex[d][0] ) ;
371  m_points = m_points * ( m_localIndex[d][1] -m_localIndex[d][0] +1 ) ;
372  }
373 
374 }

513 uint64_t VTKRectilinearGrid::calcFieldEntries( const VTKField &field ){
514 
515  uint64_t entries(0) ;
516  std::string name( field.getName() ) ;
517 
518  if( name == "x_Coord" ){
519  entries = m_localIndex[0][1] -m_localIndex[0][0] +1 ;
520 
521  } else if( name == "y_Coord" ){
522  entries = m_localIndex[1][1] -m_localIndex[1][0] +1 ;
523 
524  } else if( name == "z_Coord" ){
525  entries = m_localIndex[2][1] -m_localIndex[2][0] +1 ;
526 
527  } else{
528 
529  VTKLocation location( field.getLocation() ) ;
530  assert( location != VTKLocation::UNDEFINED) ;
531 
532  if( location == VTKLocation::CELL ){
533  entries = m_cells ;
534 
535  } else if( location == VTKLocation::POINT ){
536  entries = m_points ;
537 
538  }
539 
540  VTKFieldType fieldType( field.getFieldType() ) ;
541  assert( fieldType != VTKFieldType::UNDEFINED) ;
542 
543  entries *= static_cast<uint64_t>(fieldType) ;
544 
545  }
546 
547  return entries ;
548 
549 }
