94 struct DoubleFloatingEqual
95 {
106  bool operator()(const double &x, const double &y, double errorFactor = 1.0) const
107  {
108  const double ABS_MAX_DIFF = 1e-15;
109  const double REL_MAX_DIFF = std::numeric_limits<double>::epsilon();
110 
111  // Check if the numbers are really close (needed when comparing
112  // numbers near zero).
113  double diff = std::abs(x - y);
114  if (diff <= errorFactor * ABS_MAX_DIFF) {
115  return true;
116  }
117 
118  // Check if the numbers have the same sign
119  if ((x < 0 && y > 0) || (x > 0 && y < 0)) {
120  return false;
121  }
122 
123  // Compare using a relative difference
124  double abs_x = std::abs(x);
125  double abs_y = std::abs(y);
126  double largest = (abs_y > abs_x) ? abs_y : abs_x;
127 
128  return (diff <= errorFactor * largest * REL_MAX_DIFF);
129  }
130 };
