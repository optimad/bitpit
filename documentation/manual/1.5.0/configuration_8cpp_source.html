135 void ConfigParser::read(const std::string &filename, bool append)
136 {
137  // Macro to check API for match with the DLL we are using
138  LIBXML_TEST_VERSION
139 
140  // Read the XML file
141  xmlDoc *doc = xmlReadFile(filename.c_str(), NULL, 0);
142  if (doc == nullptr) {
143  throw std::runtime_error("Could not parse configuration file \"" + filename + "\"");
144  }
145 
146  // Get the root element
147  xmlNode * rootElement = xmlDocGetRootElement(doc);
148 
149  // check if the root name is the requeste one
150  std::string rootName(reinterpret_cast<const char*>(rootElement->name));
151  if (rootName != m_root) {
152  throw std::runtime_error("The name of the root element is not \"" + m_root + "\"");
153  }
154 
155  // Check if the version is supported
156  const xmlChar *versionAttributeName = reinterpret_cast<const xmlChar *>("version");
157  if (m_checkVersion && xmlHasProp(rootElement, versionAttributeName)) {
158  xmlChar *versionValue = xmlGetProp(rootElement, versionAttributeName);
159  std::string versionString((char *) versionValue);
160  xmlFree(versionValue);
161 
162  int version;
163  std::istringstream(versionString) >> version;
164 
165  if (version != m_version) {
166  throw std::runtime_error("The version ofthe file is not not \"" + std::to_string(m_version) + "\"");
167  }
168  }
169 
170  // Load the options in the configuration file
171  if (!append) {
172  clear();
173  }
174  config::XML::readNode(rootElement->children, this);
175 
176  // Clean-up
177  xmlFreeDoc(doc);
178  xmlCleanupParser();
179 }

186 void ConfigParser::write(const std::string &filename) const
187 {
188  int status;
189 
190  // Create a new XmlWriter for DOM tree, with no compression
191  xmlTextWriterPtr writer = xmlNewTextWriterFilename(filename.c_str(), 0);
192  if (writer == NULL) {
193  throw std::runtime_error("Error creating the xml writer");
194  }
195 
196  xmlTextWriterSetIndent(writer, 1);
197 
198  // Start the document
199  status = xmlTextWriterStartDocument(writer, NULL, config::XML::DEFAULT_ENCODING.c_str(), NULL);
200  if (status < 0) {
201  throw std::runtime_error("Error at xmlTextWriterStartDocument");
202  }
203 
204  // Start the root element
205  xmlChar *elementName = config::XML::encodeString(m_root, config::XML::DEFAULT_ENCODING);
206  status = xmlTextWriterStartElement(writer, BAD_CAST elementName);
207  if (status < 0) {
208  throw std::runtime_error("Error at xmlTextWriterStartElement");
209  }
210 
211  // Add an attribute with version
212  std::ostringstream versionStream;
213  versionStream << m_version;
214 
215  xmlChar *versionAttr = config::XML::encodeString(versionStream.str(), config::XML::DEFAULT_ENCODING);
216  status = xmlTextWriterWriteAttribute(writer, BAD_CAST "version", BAD_CAST versionAttr);
217  if (status < 0) {
218  throw std::runtime_error("Error at xmlTextWriterWriteAttribute");
219  }
220 
221  // Write the configuration
222  config::XML::writeNode(writer, this, config::XML::DEFAULT_ENCODING);
223 
224  // End section
225  status = xmlTextWriterEndElement(writer);
226  if (status < 0) {
227  throw std::runtime_error("Error at xmlTextWriterEndElement");
228  }
229 
230  // Close the document
231  status = xmlTextWriterEndDocument(writer);
232  if (status < 0) {
233  throw std::runtime_error("Error at xmlTextWriterEndDocument");
234  }
235 
236  // Write the XML
237  xmlFreeTextWriter(writer);
238 }
