202 {
203  if (!kernel) {

336 template<typename value_t, typename id_t>
337 void PiercedStorage<value_t, id_t>::commitSyncAction(const PiercedSyncAction &action)
338 {
339  switch (action.type) {
340 
341  case PiercedSyncAction::TYPE_CLEAR:
342  {
343  rawClear(true);
344  break;
345  }
346 
347  case PiercedSyncAction::TYPE_RESERVE:
348  {
349  rawReserve(action.info[PiercedSyncAction::INFO_SIZE]);
350  break;
351  }
352 
353  case PiercedSyncAction::TYPE_RESIZE:
354  {
355  rawResize(action.info[PiercedSyncAction::INFO_SIZE]);
356  break;
357  }
358 
359  case PiercedSyncAction::TYPE_SHRINK_TO_FIT:
360  {
361  rawShrinkToFit();
362  break;
363  }
364 
365  case PiercedSyncAction::TYPE_REORDER:
366  {
367  if (action.data && action.data->size() > 0) {
368  rawReorder(*action.data);
369  }
370  rawResize(action.info[PiercedSyncAction::INFO_SIZE]);
371  rawShrinkToFit();
372  break;
373  }
374 
375  case PiercedSyncAction::TYPE_APPEND:
376  {
377  // Since we are increasing the sotrage by an element at the time
378  // calling a reserve will hurt performance badly because this will
379  // prevent the automatic reallocation of the storage.
380  rawEmplaceBack();
381  break;
382  }
383 
384  case PiercedSyncAction::TYPE_INSERT:
385  {
386  // Since we are increasing the sotrage by an element at the time
387  // calling a reserve will hurt performance badly because this will
388  // prevent the automatic reallocation of the storage.
389  rawEmplace(action.info[PiercedSyncAction::INFO_POS]);
390  break;
391  }
392 
393  case PiercedSyncAction::TYPE_OVERWRITE:
394  {
395  break;
396  }
397 
398  case PiercedSyncAction::TYPE_MOVE_APPEND:
399  {
400  // Since we are increasing the sotrage by an element at the time
401  // calling a reserve will hurt performance badly because this will
402  // prevent the automatic reallocation of the storage.
403  rawPushBack(std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
404  rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
405  break;
406  }
407 
408  case PiercedSyncAction::TYPE_MOVE_INSERT:
409  {
410  // Since we are increasing the sotrage by an element at the time
411  // calling a reserve will hurt performance badly because this will
412  // prevent the automatic reallocation of the storage.
413  rawInsert(action.info[PiercedSyncAction::INFO_POS_SECOND], 1, std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
414  rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
415  break;
416  }
417 
418  case PiercedSyncAction::TYPE_MOVE_OVERWRITE:
419  {
420  rawSet(action.info[PiercedSyncAction::INFO_POS_SECOND], std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
421  rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
422  break;
423  }
424 
425  case PiercedSyncAction::TYPE_PIERCE:
426  {
427  // Nothing to do. To improve performance the element will not be
428  // cleared.
429  break;
430  }
431 
432  case PiercedSyncAction::TYPE_SWAP:
433  {
434  rawSwap(action.info[PiercedSyncAction::INFO_POS_FIRST], action.info[PiercedSyncAction::INFO_POS_SECOND]);
435  break;
436  }
437 
438  case PiercedSyncAction::TYPE_NOOP:
439  {
440  break;
441  }
442 
443  default:
444  {
445  throw std::runtime_error("Undefined synchronization action.");
446  break;
447  }
448 
449  }
450 }

530 template<typename value_t, typename id_t>
531 void PiercedStorage<value_t, id_t>::rawReorder(const std::vector<std::size_t> &permutations)
532 {
533  std::size_t storageRawSize = rawSize();
534  assert(permutations.size() == storageRawSize);
535 
536  // Evaluate the permutation of the fields
537  std::vector<std::size_t> fieldPermutations(storageRawSize * m_nFields);
538  for (std::size_t pos = 0; pos < storageRawSize; ++pos) {
539  std::size_t posOffset = pos * m_nFields;
540  std::size_t permutationOffset = permutations[pos] * m_nFields;
541  for (std::size_t k = 0; k < m_nFields; ++k) {
542  fieldPermutations[posOffset + k] = permutationOffset + k;
543  }
544  }
545 
546  // Sort the fields
547  utils::reorderVector<value_t>(fieldPermutations, m_fields, storageRawSize * m_nFields);
548 }

709 template<typename value_t, typename id_t>
710 void PiercedStorage<value_t, id_t>::swap(PiercedStorage &x) noexcept
711 {
712  // It is only possible to swap two storages with the same number of field.
713  // If this condition is not fulfilled we can not continue. However, we
714  // cannot throw an exception because the function is declared snoexcept.
715  if (x.getFieldCount() != getFieldCount()) {
716  std::cout << "It is only possible to swap storages with the same number of fields." << std::endl;
717  assert(false);
718  exit(EXIT_FAILURE);
719  }
720 
721  PiercedSyncSlave::swap(x);
722  std::swap(x.m_nFields, m_nFields);
723  std::swap(x.m_fields, m_fields);
724  std::swap(x.m_kernel, m_kernel);
725  std::swap(x.m_const_kernel, m_const_kernel);
726 }
