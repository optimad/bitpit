66 double LevelSetCartesian::computeCellIncircle(long id) {
67 
68  BITPIT_UNUSED(id);
69 
70  int dim = m_cartesian->getDimension();
71  std::array<double,3> spacing = m_cartesian->getSpacing();
72  double minSpacing = std::numeric_limits<double>::max() ;
