688  void
689  PabloUniform::getBoundingBox(darray3 & P0, darray3 & P1) const {
690  // If there are no octants the bounding box is empty
691  uint32_t nocts = ParaTree::getNumOctants();
692  if (nocts == 0) {
693  P0 = getOrigin();
694  P1 = P0;
695 
696  return;
697  }
698 
699  // If the octree is serial we can evaluate the bounding box easily
700  // otherwise we need to scan all the octants
701  if (getSerial()) {
702  P0 = getOrigin();
703  P1 = P0;
704  for (int i=0; i<ParaTree::getDim(); i++){
705  P1[i] += getL();
706  }
707 
708  return;
709  }
710 
711  // If the octree is parallel we need to scan all the octants
712  darray3 cnode0, cnode1;
713 
714  uint32_t id = 0;
715  uint8_t nnodes = ParaTree::getNnodes();
716 
717  P0 = getNode(id, 0);
718  P1 = getNode(nocts-1, nnodes-1);
719 
720  for (id=0; id<nocts; id++){
721  cnode0 = getNode(id, 0);
722  cnode1 = getNode(id, nnodes-1);
723  for (int i=0; i<ParaTree::getDim(); i++){
724  P0[i] = min(P0[i], cnode0[i]);
725  P1[i] = max(P1[i], cnode1[i]);
726  }
727  }
728  };
