376 void LevelSet::addProcessingOrder( int objectId ) {
377 
378  bool primary = m_objects.at(objectId)->isPrimary() ;
379 
380  if(primary){
381  std::vector<int>::iterator orderItr = m_order.begin() ;
382  bool iterate( orderItr != m_order.end()) ;
383 
384  while(iterate){
385  int id = *orderItr ;
386  if( m_objects.at(id)->isPrimary() ){
387  ++orderItr ;
388  iterate = orderItr != m_order.end() ;
389  } else {
390  iterate = false;
391  }
392  }
393 
394  m_order.insert(orderItr,objectId) ;
395 
396  } else {
397  m_order.push_back(objectId) ;
398 
399  }
400 
401 }

512 void LevelSet::compute(){
513 
514  assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::compute()");
515 
516  double RSearch ;
517 
518  for( int objectId : m_order){
519  auto &visitor = *(m_objects.at(objectId)) ;
520  if( !m_userRSearch){
521  RSearch = visitor.computeSizeNarrowBand() ;
522  visitor.setSizeNarrowBand(RSearch) ;
523  }
524  }
525 
526 
527  for( int objectId : m_order){
528  auto &visitor = *(m_objects.at(objectId)) ;
529  RSearch = visitor.getSizeNarrowBand();
530  visitor.computeLSInNarrowBand( RSearch, m_signedDF) ;
531  if( m_propagateS ) visitor.propagateSign() ;
532  }
533

540 void LevelSet::update( const std::vector<adaption::Info> &mapper ){
541 
542  assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::update()");
543 
544  // Udate the cache of the kernel
545  m_kernel->updateGeometryCache( mapper ) ;
546 
547  // Check the mapper to detect the operations to perform
548  bool updateNarrowBand = false;
549  std::unordered_map<int,std::vector<long>> sendList, recvList ;
550 
551  for( const auto &event : mapper){
552  if( event.entity != adaption::Entity::ENTITY_CELL){
553  continue;
554  }
555 
556  if( event.type == adaption::Type::TYPE_PARTITION_SEND){
557  sendList.insert({{event.rank,event.previous}}) ;
558  } else if( event.type == adaption::Type::TYPE_PARTITION_RECV){
559  recvList.insert({{event.rank,event.current}}) ;
560  } else if( event.type != adaption::Type::TYPE_PARTITION_NOTICE){
561  updateNarrowBand = true ;
562  }
563  }
564 
565  // Evaluate new narrow band size
566  double newRSearch ;
567  if (updateNarrowBand && !m_userRSearch) {
568 
569  for( int objectId : m_order){
570  auto &visitor = *(m_objects.at(objectId)) ;
571  newRSearch = visitor.updateSizeNarrowBand(mapper) ;
572  visitor.setSizeNarrowBand(newRSearch) ;
573  }
574  }
575 
576  // Update ls in narrow band
577  for( int objectId : m_order){
578  auto &visitor = *(m_objects.at(objectId)) ;
579 
580  if (updateNarrowBand) {
581  newRSearch = visitor.getSizeNarrowBand() ;
582  visitor.updateLSInNarrowBand( mapper, newRSearch, m_signedDF ) ;
583 
584  } else {
585  visitor.clearAfterMeshAdaption(mapper) ;
586 
587  }
588 
589 #if BITPIT_ENABLE_MPI
590  // Parallel communications
591  if (sendList.size() > 0 || recvList.size() > 0) {
592  visitor.communicate( sendList, recvList, &mapper ) ;
593  }
594 
595  visitor.exchangeGhosts() ;
596 #endif
597  if (updateNarrowBand && m_propagateS) {
598  visitor.propagateSign() ;
599  }
600  }
601 
602  // Finish narrow band update
603  if (updateNarrowBand) {
604 
605  for( int objectId : m_order){
606  auto &visitor = *(m_objects.at(objectId)) ;
607  newRSearch = m_kernel->computeSizeNarrowBandFromLS( &visitor, m_signedDF );
608  visitor.setSizeNarrowBand(newRSearch) ;
609  }
610 
611  for( int objectId : m_order){
612  auto &visitor = *(m_objects.at(objectId)) ;
613  newRSearch = visitor.getSizeNarrowBand();
614  visitor.filterOutsideNarrowBand(newRSearch) ;
615  }
616  }
617 }
