39 bitpit::IBinaryStream& operator>>(bitpit::IBinaryStream &buffer, bitpit::Element &element)
40 {
41  buffer >> element.m_type;
42  buffer >> element.m_id;
43  element._initialize(element.m_type);
44  int nVertices = element.getVertexCount();
45  for (int i = 0; i < nVertices; ++i) {
46  buffer >> element.m_connect[i];
47  }
48 
49  return buffer;
50 }

968 void ElementInfo::initializeFaceEdges(const std::vector<ElementInfo *> &facesInfo)
969 {
970  faceEdges = std::vector<std::vector<int>>(nFaces);
971  for (int k = 0; k < nFaces; ++k) {
972  const ElementInfo &faceInfo = *(facesInfo[k]);
973 
974  int nFaceEdges = faceInfo.nFaces;
975  for (int i = 0; i < nFaceEdges; ++i) {
976  // Number of vertices of the edge associated to the face
977  std::size_t nFaceEdgeVertices = faceInfo.faceConnect[i].size();
978 
979  // Connectivity of the edge associated to the face
980  const std::vector<int> &localFaceEdgeConnect = faceInfo.faceConnect[i];
981 
982  std::vector<int> faceEdgeConnect(nFaceEdgeVertices);
983  for (std::size_t n = 0; n < nFaceEdgeVertices; ++n) {
984  int localVertexId = localFaceEdgeConnect[n];
985  int vertexId = faceConnect[k][localVertexId];
986 
987  faceEdgeConnect[n] = vertexId;
988  }
989 
990  // Search the edge that has the same connectivity of the face edge
991  for (int j = 0; j < nEdges; ++j) {
992  // If face edge and the guess edge have a different number of
993  // vertices, the two edge cannot be the same.
994  std::size_t nGuessEdgeVertices = edgeConnect[j].size();
995  if (nGuessEdgeVertices != nFaceEdgeVertices) {
996  continue;
997  }
998 
999  // If the connecitivity of the face edge and the one of the
1000  // guess edge are the same, the two edges coincides.
1001  const std::vector<int> commonVertices = utils::intersectionVector(faceEdgeConnect, edgeConnect[j]);
1002  if (commonVertices.size() == nFaceEdgeVertices) {
1003  faceEdges[k].push_back(j);
1004  }
1005  }
1006  }
1007 
1008  assert(faceEdges[k].size() == nFaceEdges);
1009  }
1010 }
