48 template<class T>
49 bitpit::OBinaryStream& operator<<(bitpit::OBinaryStream &buffer, const bitpit::FlatVector2D<T> &vector)
50 {
51  typename std::vector<T>::const_iterator it;
52  typename std::vector<size_t>::const_iterator jt;
53 
54  buffer << vector.m_index.size() << vector.m_v.size();
55  for (jt = vector.m_index.begin(); jt != vector.m_index.end(); ++jt) {
56  buffer << *(jt);
57  }
58  for (it = vector.m_v.begin(); it != vector.m_v.end(); ++it) {
59  buffer << *(it);
60  }
61 
62  return buffer;
63 }

75 {
76  size_t size_m_v, size_m_index;

79  buffer >> size_m_index;
80  buffer >> size_m_v;
81 
82  vector.m_index.resize(size_m_index, 0);
83  vector.m_v.resize(size_m_v);
84 
85  for (i = 0; i < size_m_index; ++i) {
86  buffer >> vector.m_index[i];
87  }
88 
89  for (i = 0; i < size_m_v; ++i) {
90  buffer >> vector.m_v[i];
91  }
92 
93  return buffer;
94 }

190 template <class T>
191 void FlatVector2D<T>::initialize(const std::vector<int> &sizes, const T &value)
192 {
193  int nVectors = sizes.size();
194 
195  // Destroy the container
196  //
197  // Index and the storage data will probabily be stored in memory
198  // regions contigous to each other. To reduce memory fragmentation
199  // it's better to deallocate the container before updating its
200  // data structures.
201  destroy();
202 
203  // Initialize the indexes
204  //
205  // Clear the vector before the resize to reduce memory fragmentation
206  m_index.resize(nVectors + 1);
207  m_index.shrink_to_fit();
208  for (int i = 0; i < nVectors; ++i) {
209  m_index[i+1] = m_index[i] + sizes[i];
210  }
211 
212  // Initialize the storage
213  //
214  // Clear the vector before the resize to reduce memory fragmentation
215  m_v.assign(m_index[nVectors], value);
216  m_v.shrink_to_fit();
217 }
