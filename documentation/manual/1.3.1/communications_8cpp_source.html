158 void DataCommunicator::discoverSends(int discoverTag)
159 {
160  // Cancel current sends
161  clearAllSends();
162 
163  // Send the size of the messages that the processors want to receive
164  //
165  // We need to use buffered sends to be sure that the data to transfer
166  // will not be overwritten during the loop over the receive list.
167  int commBufferSize = m_recvIds.size() * (sizeof(long) + MPI_BSEND_OVERHEAD);
168  std::vector<char> commBuffer(commBufferSize);
169  MPI_Buffer_attach(commBuffer.data(), commBufferSize);
170 
171  for (auto &entry : m_recvIds) {
172  int rank = entry.first;
173  RecvBuffer &buffer = getRecvBuffer(rank);
174  long dataSize = buffer.capacity();
175 
176  MPI_Request dataSizeRequest;
177  MPI_Ibsend(&dataSize, 1, MPI_LONG, rank, discoverTag, m_communicator, &dataSizeRequest);
178 
179  // MPI_Isend initiates an asynchronous (background) data transfer.
180  // The actual data transfer might not happen unless one of the
181  // MPI_Wait* or MPI_Test* calls has been made on the request.
182  int completeFlag;
183  MPI_Test(&dataSizeRequest, &completeFlag, MPI_STATUS_IGNORE);
184  }
185 
186  // Raise a barrier to make sure that all the sends starts
187  MPI_Barrier(m_communicator);
188 
189  // Receive the data size of the messages that the processors want to receive
190  std::unordered_map<int, long> dataSizes;
191  while (true) {
192  // Probe for messages
193  int messageAvailable;
194  MPI_Status status;
195 
196  MPI_Iprobe(MPI_ANY_SOURCE, discoverTag, m_communicator, &messageAvailable, &status);
197  if (!messageAvailable) {
198  break;
199  }
200 
201  // Receive the data size that will be received from the source
202  long dataSize;
203  MPI_Recv(&dataSize, 1, MPI_LONG, status.MPI_SOURCE, discoverTag, m_communicator, MPI_STATUS_IGNORE);
204  dataSizes[status.MPI_SOURCE] = dataSize;
205  }
206 
207  // Wait that all processors correctly receive the data to communicate.
208  //
209  // Without the barrier some processors may start the receives while other
210  // are still waiting to receive the data sizes. Since the probe for the
211  // data size will accept message for all sources, if a processor start
212  // receiveing data it will intefere with the other processors still
213  // receiving the data sizes.
214  if (discoverTag == m_tag) {
215  MPI_Barrier(m_communicator);
216  }
217 
218  // Deatach the buffer
219  MPI_Buffer_detach(commBuffer.data(), &commBufferSize);
220 
221  // Set the sends
222  for (auto &entry : dataSizes) {
223  setSend(entry.first, entry.second);
224  }
225 }

318 void DataCommunicator::clearSend(int rank)
319 {
320  if (m_sendIds.count(rank) == 0) {
321  return;
322  }
323 
324  // Cancel waiting send
325  cancelSend(rank);
326 
327  // Remove the send associated to the specified rank
328  int id = m_sendIds[rank];
329  m_sendIds.erase(rank);
330  for (auto &entry : m_sendIds) {
331  if (entry.second > id) {
332  entry.second--;
333  }
334  }
335 
336  m_sendRanks.erase(m_sendRanks.begin() + id);
337  m_sendRequests.erase(m_sendRequests.begin() + id);
338  m_sendBuffers.erase(m_sendBuffers.begin() + id);
339 }

655 int DataCommunicator::waitAnySend(const std::vector<int> &blackList)
656 {
657  // Exclude blackListed ranks
658  std::vector<MPI_Request> requestList(m_sendRequests);
659  for (const int rank : blackList) {
660  int id = m_sendIds.at(rank);
661  requestList[id] = MPI_REQUEST_NULL;
662  }
663 
664  // Wait for a send to complete
665  int id;
666  MPI_Waitany(requestList.size(), requestList.data(), &id, MPI_STATUS_IGNORE);
667  if (id == MPI_UNDEFINED) {
668  return MPI_UNDEFINED;
669  }
670 
671  m_sendRequests[id] = requestList[id];
672 
673  // Reset the position of the buffer
674  m_sendBuffers[id].seekg(0);
675 
676  // Return the rank associated to the completed send
677  return m_sendRanks[id];
678 }
