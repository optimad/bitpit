289  void
290  ParaTree::setComm(MPI_Comm communicator)
291  {
292  // Communication can be set just once
293  if (isCommSet()) {
294  throw std::runtime_error ("PABLO communicator can be set just once");
295  }
296 
297  // The communicator has to be valid
298  if (communicator == MPI_COMM_NULL) {
299  throw std::runtime_error ("PABLO communicator is not valid");
300  }
301 
302  // Create a copy of the user-specified communicator
303  //
304  // No library routine should use MPI_COMM_WORLD as the communicator;
305  // instead, a duplicate of a user-specified communicator should always
306  // be used.
307  MPI_Comm_dup(communicator, &m_comm);
308 
309  // Get MPI information
310  MPI_Comm_size(m_comm, &m_nproc);
311  MPI_Comm_rank(m_comm, &m_rank);
312 
313  // Initialize partition data
314  createPartitionInfo();
315  }
