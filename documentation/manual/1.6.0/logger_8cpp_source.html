85 LoggerBuffer::int_type LoggerBuffer::overflow(int_type character)
86 {
87  if (character != traits_type::eof()) {
88  // Write the charater to the buffer and then increment pptr() by
89  // calling pbump(1). It's always safe to write thecharater to the
90  // buffer because we reserved an extra char at the end of our buffer
91  // in the constructor.
92  assert(std::less_equal<char *>()(pptr(), epptr()));
93  *pptr() = character;
94  pbump(1);
95 
96  // Flush the stream
97  if (sync() == 0) {
98  return character;
99  }
100  }
101 
102  return traits_type::eof();
103 }

122 int LoggerBuffer::flush(bool terminate)
123 {
124  int status = 0;
125  if (pptr() == pbase()) {
126  return status;
127  }
128 
129  // Identify line breakers
130  std::vector<char *> linePointers;
131  linePointers.push_back(pbase());
132  for (char *p = pbase(), *e = pptr() - 1; p != e; ++p) {
133  if (*p == '\n') {
134  linePointers.push_back(p + 1);
135  }
136  }
137  linePointers.push_back(pptr());
138 
139  // Detect if a new line must be added at the end
140  terminate = (terminate && *(linePointers.back() - 1) != '\n');
141 
142  // Move the internal pointer
143  std::ptrdiff_t nCharacters = pptr() - pbase();
144  pbump(- nCharacters);
145 
146  // Write all lines
147  for (unsigned int i = 0; i < linePointers.size() - 1; ++i) {
148  char *firstCharacter = linePointers[i];
149  char *lastCharacter = linePointers[i + 1];
150  std::ptrdiff_t lineSize = lastCharacter - firstCharacter;
151 
152  // Write to the console
153  if (m_console && m_consoleEnabled) {
154  if (m_consoleTimestampEnabled) {
155  *m_console << "[" + getTimestamp() + "] ";
156  }
157 
158  if (!m_consolePrefix.empty()) {
159  *m_console << m_consolePrefix << " :: ";
160  }
161 
162  if (!m_context.empty()) {
163  *m_console << m_context << " :: ";
164  }
165 
166  if (!m_padding.empty()) {
167  *m_console << m_padding;
168  }
169 
170  m_console->write(firstCharacter, lineSize);
171  if ((m_console->rdstate() & std::ifstream::failbit ) != 0) {
172  status = -1;
173  }
174 
175  if (terminate && lastCharacter == linePointers.back()) {
176  *m_console << "\n";
177  }
178  }
179 
180  // Write to file
181  if (m_file && m_fileEnabled && m_file->is_open()) {
182  if (m_fileTimestampEnabled) {
183  *m_file << "[" + getTimestamp() + "] ";
184  }
185 
186  if (!m_filePrefix.empty()) {
187  *m_file << m_filePrefix << " :: ";
188  }
189 
190  if (!m_context.empty()) {
191  *m_file << m_context + " :: ";
192  }
193 
194  if (!m_padding.empty()) {
195  *m_file << m_padding;
196  }
197 
198  m_file->write(firstCharacter, lineSize);
199  if ((m_file->rdstate() & std::ifstream::failbit ) != 0) {
200  status = -1;
201  }
202 
203  if (terminate && lastCharacter == linePointers.back()) {
204  *m_file << "\n";
205  }
206  }
207  }
208 
209  return status;
210 }

399 const std::string LoggerBuffer::getTimestamp() const
400 {
401  auto currentClock = std::chrono::system_clock::now();
402  std::time_t currentTime = std::chrono::system_clock::to_time_t(currentClock);
403 
404  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(currentClock.time_since_epoch());
405  auto millisecs = std::chrono::duration_cast<std::chrono::milliseconds>(currentClock.time_since_epoch() - seconds);
406  std::ostringstream millisecsConverter;
407  millisecsConverter << (millisecs.count());
408 
409  char buffer[80];
410  strftime(buffer, 80, "%Y-%m-%d %H:%M:%S", localtime(&currentTime));
411  std::string timestamp(buffer);
412  timestamp += "." + millisecsConverter.str();
413 
414  return timestamp;
415 }

1059 void LoggerManager::create(const std::string &name, bool reset,
1060  const std::string &directory,
1061  int nProcessors, int rank)
1062 {
1063  // Its not possible to create a log with the default name nor a log
1064  // with the same name of an existent logger nor a log with an empty
1065  // name
1066  if (exists(name)) {
1067  m_loggerUsers[name]++;
1068 
1069  cout(name) << "Detected an attemp to create the logger \"" << name << "\" twice" << std::endl;
1070  cout(name) << "The previously created logger will be used." << std::endl;
1071  return;
1072  } else if (name == m_defaultName) {
1073  cout(m_defaultName) << "Detected an attemp to overwrite the default logger" << std::endl;
1074  cout(m_defaultName) << "The name of the default logger is \"" << name << "\"" << std::endl;
1075  return;
1076  } else if (name == "") {
1077  cout() << "Detected an attemp to create a logger with an empty name" << std::endl;
1078  return;
1079  }
1080 
1081  // Create the logger
1082  if (m_mode == log::SEPARATE) {
1083  _create(name, reset, directory, nProcessors, rank);
1084  } else {
1085  _create(name, cout(m_defaultName));
1086  }
1087 }

1100 bool LoggerManager::destroy(const std::string &name, bool force)
1101 {
1102  if (!exists(name)) {
1103  return false;
1104  }
1105 
1106  // Decrement the users
1107  int &nUsers = m_loggerUsers[name];
1108  nUsers--;
1109 
1110  // If the logger has no users we can delete it
1111  if (nUsers == 0 || force) {
1112  // Remove the logger from the manager
1113  m_loggers.erase(name);
1114  m_loggerUsers.erase(name);
1115 
1116  // If the logger has it own file stream, delete it
1117  if (m_fileStreams.count(name) > 0) {
1118  // Close the stream
1119  std::ofstream &fileStream = *(m_fileStreams.at(name));
1120  fileStream.close();
1121 
1122  // Remove the stream
1123  m_fileStreams.erase(name);
1124  }
1125 
1126  return true;
1127  } else {
1128  return false;
1129  }
1130 }

1196 void LoggerManager::_create(const std::string &name, bool reset,
1197  const std::string &directory,
1198  int nProcessors, int rank)
1199 {
1200  // Get the file path
1201  FileHandler fileHandler;
1202  fileHandler.setDirectory(directory);
1203  fileHandler.setName(name);
1204  fileHandler.setAppendix("log");
1205  fileHandler.setParallel(nProcessors > 1);
1206  if (nProcessors > 1) {
1207  fileHandler.setBlock(rank);
1208  }
1209 
1210  std::string filePath = fileHandler.getPath();
1211 
1212  // Create the file stream
1213  std::ios_base::openmode fileMode;
1214  if (reset) {
1215  fileMode = std::ofstream::out;
1216  } else {
1217  fileMode = std::ofstream::app;
1218  }
1219 
1220  m_fileStreams[name] = std::unique_ptr<std::ofstream>(new std::ofstream());
1221  std::ofstream &fileStream = *(m_fileStreams.at(name));
1222  fileStream.rdbuf()->pubsetbuf(0, 0);
1223  fileStream.open(filePath, fileMode);
1224 
1225  // Use cout as console stream
1226  std::ostream &consoleStream = std::cout;
1227 
1228  // Create the logger
1229  m_loggers[name] = std::unique_ptr<Logger>(new Logger(name, &consoleStream, &fileStream, nProcessors, rank));
1230  m_loggerUsers[name] = 1;
1231 }
