174 void Element::Tesselation::importPolygon(const std::vector<int> &vertexIds)
175 {
176  int nVertices = vertexIds.size();
177  if (nVertices == 3 || nVertices == 4) {
178  m_nTiles++;
179  m_types.push_back(nVertices == 3 ? ElementType::TRIANGLE : ElementType::QUAD);
180  m_connects.push_back(vertexIds);
181 
182  return;
183  }
184 
185  // Add the centroid
186  std::array<double, 3> centroid = {{0., 0., 0.}};
187  for (int k = 0; k < nVertices; ++k) {
188  centroid += m_coordinates[vertexIds[k]];
189  }
190  centroid = centroid / double(nVertices);
191 
192  int centroidId = importVertexCoordinates(std::move(centroid));
193 
194  // Decompose the polygon in triangles
195  //
196  // Each triangle is composed by the two vertices of a side and by the
197  // centroid.
198  ElementType tileType = ElementType::TRIANGLE;
199  int nTileVertices = ReferenceElementInfo::getInfo(tileType).nVertices;
200  int nSideVertices = ReferenceElementInfo::getInfo(ElementType::LINE).nVertices;
201 
202  int nSides = nVertices;
203  m_types.resize(m_nTiles + nSides, tileType);
204  m_connects.resize(m_nTiles + nSides, std::vector<int>(nTileVertices));
205  for (int i = 0; i < nSides; ++i) {
206  m_nTiles++;
207  for (int k = 0; k < nSideVertices; ++k) {
208  m_connects[m_nTiles - 1][k] = vertexIds[(i + k) % nVertices];
209  }
210  m_connects[m_nTiles - 1][nSideVertices] = centroidId;
211  }
212 }

220 void Element::Tesselation::importPolyhedron(const std::vector<int> &vertexIds, const std::vector<std::vector<int>> &faceVertexIds)
221 {
222  int nFaces = faceVertexIds.size();
223  int nVertices = vertexIds.size();
224 
225  // Generate the tesselation of the surface
226  int nInitialTiles = m_nTiles;
227  for (int i = 0; i < nFaces; ++i) {
228  importPolygon(faceVertexIds[i]);
229  }
230  int nFinalTiles = m_nTiles;
231 
232  // Add the centroid of the element to the tesselation
233  std::array<double, 3> centroid = {{0., 0., 0.}};
234  for (int k = 0; k < nVertices; ++k) {
235  centroid += m_coordinates[vertexIds[k]];
236  }
237  centroid = centroid / double(nVertices);
238 
239  int centroidTesselationId = importVertexCoordinates(std::move(centroid));
240 
241  // Decompose the polyhedron in prisms and pyramids.
242  //
243  // Each prism/pyramid has a tile of the surface tesselation as the
244  // base and the centroid of the element as the apex. Since we have
245  // already generated the surface tesselation, we can "convert" the
246  // two-dimensional tiles of that tesselation in three-dimensional
247  // tiles to obtain the volume tesselation.
248  for (int i = nInitialTiles; i < nFinalTiles; ++i) {
249  // Change the tile type
250  ElementType &tileType = m_types[i];
251  if (tileType == ElementType::TRIANGLE) {
252  tileType = ElementType::TETRA;
253  } else if (tileType == ElementType::QUAD) {
254  tileType = ElementType::PYRAMID;
255  } else {
256  BITPIT_UNREACHABLE("Unsupported tile");
257  throw std::runtime_error ("Unsupported tile");
258  }
259 
260  // Fix the order of the connectivity the match the order of the
261  // three-dimensional element
262  if (tileType == ElementType::TETRA) {
263  std::swap(m_connects[i][0], m_connects[i][2]);
264  } else if (tileType == ElementType::PYRAMID) {
265  std::swap(m_connects[i][1], m_connects[i][3]);
266  }
267 
268  // Add the centroid to the tile connectivity
269  m_connects[i].push_back(centroidTesselationId);
270  }
271 }

762 ConstProxyVector<int> Element::getFaceLocalConnect(int face) const
763 {
764  switch (m_type) {
765 
766  case (ElementType::POLYGON):
767  {
768  int connectSize = getConnectSize();
769 
770  int faceConnectSize = getFaceVertexCount(face);
771  std::vector<int> localFaceConnect(faceConnectSize);
772  for (int i = 0; i < faceConnectSize; ++i) {
773  int localVertexId = (face + i) % connectSize;
774 
775  localFaceConnect[i] = localVertexId;
776  }
777 
778  return ConstProxyVector<int>(std::move(localFaceConnect));
779  }
780 
781  case (ElementType::POLYHEDRON):
782  {
783  ElementType faceType = getFaceType(face);
784  ConstProxyVector<long> faceVertexIds = getFaceVertexIds(face);
785  int nFaceVertices = faceVertexIds.size();
786  bool faceHasReferenceInfo = ReferenceElementInfo::hasInfo(faceType);
787 
788  int faceConnectSize = nFaceVertices;
789  int localVertexOffset = 0;
790  if (!faceHasReferenceInfo) {
791  ++faceConnectSize;
792  ++localVertexOffset;
793  }
794 
795  std::vector<int> localFaceConnect(faceConnectSize);
796  if (!faceHasReferenceInfo) {
797  localFaceConnect[0] = nFaceVertices;
798  }
799 
800  // The list of the vertices in sorted, therfore we can search the
801  // vertices using a binary search.
802  ConstProxyVector<long> vertexIds = getVertexIds();
803  for (int k = 0; k < nFaceVertices; ++k) {
804  int vertexId = faceVertexIds[k];
805  auto localVertexIdItr = std::lower_bound(vertexIds.begin(), vertexIds.end(), vertexId);
806  assert(localVertexIdItr != vertexIds.end() && *localVertexIdItr == vertexId);
807  localFaceConnect[localVertexOffset + k] = std::distance(vertexIds.begin(), localVertexIdItr);
808  }
809 
810  return ConstProxyVector<int>(std::move(localFaceConnect));
811  }
812 
813  default:
814  {
815  assert(m_type != ElementType::UNDEFINED);
816 
817  const int localConnectSize = getFaceVertexCount(face);
818  const int *localFaceConnect = getInfo().faceConnectStorage[face].data();
819 
820  return ConstProxyVector<int>(localFaceConnect, localConnectSize);
821  }
822 
823  }
824 }

832 ConstProxyVector<long> Element::getFaceConnect(int face) const
833 {
834  const long *connectivity = getConnect();
835 
836  switch (m_type) {
837 
838  case (ElementType::POLYGON):
839  {
840  int connectSize = getConnectSize();
841 
842  int facePos = 1 + face;
843  int faceConnectSize = getFaceVertexCount(face);
844  if (facePos + faceConnectSize <= connectSize) {
845  return ConstProxyVector<long>(connectivity + facePos, faceConnectSize);
846  } else {
847  std::vector<long> faceConnect(faceConnectSize);
848  for (int i = 0; i < faceConnectSize; ++i) {
849  int position = facePos + i;
850  if (position >= connectSize) {
851  position = position % connectSize + 1;
852  }
853 
854  faceConnect[i] = connectivity[position];
855  }
856 
857  return ConstProxyVector<long>(std::move(faceConnect));
858  }
859  }
860 
861  case (ElementType::POLYHEDRON):
862  {
863  ElementType faceType = getFaceType(face);
864 
865  int facePos = getFaceStreamPosition(connectivity, face);
866  int faceConnectSize = connectivity[facePos];
867  int faceConnectBegin = facePos + 1;
868  if (!ReferenceElementInfo::hasInfo(faceType)) {
869  faceConnectSize++;
870  faceConnectBegin--;
871  }
872 
873  return ConstProxyVector<long>(connectivity + faceConnectBegin, faceConnectSize);
874  }
875 
876  default:
877  {
878  assert(m_type != ElementType::UNDEFINED);
879 
880  // If we are here, the element has a reference element, therefore we
881  // can retrieve the local face connectivity directly form the info
882  // associated the to referenc element.
883  int faceConnectSize = getFaceVertexCount(face);
884  const int *localFaceConnect = getInfo().faceConnectStorage[face].data();
885 
886  std::vector<long> faceConnect(faceConnectSize);
887  for (int k = 0; k < faceConnectSize; ++k) {
888  int localVertexId = localFaceConnect[k];
889  long vertexId = connectivity[localVertexId];
890  faceConnect[k] = vertexId;
891  }
892 
893  return ConstProxyVector<long>(std::move(faceConnect));
894  }
895 
896  }
