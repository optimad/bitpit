421 void VolOctree::setBoundingBox()
422 {
423  std::array<double, 3> minPoint;
424  std::array<double, 3> maxPoint;
425 
426  // Get the bounding box from the tree
427  m_tree->getBoundingBox(minPoint, maxPoint);
428 
429 #if BITPIT_ENABLE_MPI==1
430  // The tree is only evaluating the bounding box of the internal octants,
431  // we need to consider also ghosts cells.
432  for (auto ghostItr = ghostBegin(); ghostItr != ghostEnd(); ++ghostItr) {
433  ConstProxyVector<long> ghostVertexIds = ghostItr->getVertexIds();
434  int nGhostVertices = ghostVertexIds.size();
435  for (int i = 0; i < nGhostVertices; ++i) {
436  const std::array<double, 3> coords = m_vertices[ghostVertexIds[i]].getCoords();
437  for (int d = 0; d < 3; ++d) {
438  minPoint[d] = std::min(coords[d], minPoint[d]);
439  maxPoint[d] = std::max(coords[d], maxPoint[d]);
440  }
441  }
442  }
443 #endif
444 
445  // Set the bounding box
446  setBoundingBox(minPoint, maxPoint);
447 }

459 void VolOctree::simulateCellUpdate(long id, adaption::Marker marker, std::vector<Cell> *virtualCells, PiercedVector<Vertex, long> *virtualVertices) const
460 {
461  // Get virtual post-update octants
462  OctantInfo octantInfo = getCellOctant(id);
463  const Octant *octant = getOctantPointer(octantInfo);
464 
465  int8_t markerValue;
466  switch (marker) {
467 
468  case adaption::MARKER_COARSEN:
469  markerValue = -1;
470  break;
471 
472  case adaption::MARKER_REFINE:
473  markerValue = 1;
474  break;
475 
476  default:
477  markerValue = 0;
478 
479  }
480 
481  int nMaxVirtualOctants;
482  if (getDimension() == 3) {
483  nMaxVirtualOctants = 8;
484  } else {
485  nMaxVirtualOctants = 4;
486  }
487 
488  std::vector<Octant> virtualOctants;
489  virtualOctants.reserve(nMaxVirtualOctants);
490  m_tree->expectedOctantAdapt(octant, markerValue, &virtualOctants);
491  std::size_t nVirtualOctants = virtualOctants.size();
492 
493  // Create virtual post-update cells
494  int nCellVertices = m_cellTypeInfo->nVertices;
495 
496  virtualVertices->clear();
497  virtualVertices->reserve(nCellVertices * nVirtualOctants);
498 
499  virtualCells->clear();
500  virtualCells->resize(nVirtualOctants);
501 
502  for (std::size_t k = 0; k < nVirtualOctants; ++k) {
503  const Octant *virtualOctant = &(virtualOctants[k]);
504 
505  std::unique_ptr<long[]> connectStorage = std::unique_ptr<long[]>(new long[nCellVertices]);
506  for (int i = 0; i < nCellVertices; ++i) {
507  long vertexId = k * nCellVertices + i;
508  connectStorage[i] = vertexId;
509  virtualVertices->emplaceBack(vertexId, vertexId, m_tree->getNode(virtualOctant, i));
510  }
511 
512  virtualCells->emplace_back(id, m_cellTypeInfo->type, std::move(connectStorage), true, false);
513  }
514 }

803 std::vector<adaption::Info> VolOctree::_adaptionPrepare(bool trackAdaption)
804 {
805  BITPIT_UNUSED(trackAdaption);
806 
807  if (getSpawnStatus() == SPAWN_NEEDED) {
808  throw std::runtime_error ("The initial import has not been performed.");
809  }
810 
811  // Call pre-adapt routine
812  m_tree->preadapt();
813 
814  // Track adaption changes
815  adaption::InfoCollection adaptionData;
816  if (trackAdaption) {
817  // Current rank
818  int currentRank = -1;
819 #if BITPIT_ENABLE_MPI==1
820  currentRank = getRank();
821 #endif
822 
823  // Track internal and ghosts octants that will be coarsend/refined
824  std::vector<uint32_t> treeIds;
825  std::vector<int8_t> treeMarkers;
826  std::vector<bool> treeGhosts;
827  m_tree->getPreMapping(treeIds, treeMarkers, treeGhosts);
828 
829  std::size_t n = 0;
830  std::size_t nUpdatedOctants = treeIds.size();
831  while (n < nUpdatedOctants){
832  int8_t marker = treeMarkers[n];
833  int nAdaptionOctants;
834  adaption::Type adaptionType;
835  if (marker > 0) {
836  nAdaptionOctants = 1;
837  adaptionType = adaption::TYPE_REFINEMENT;
838  } else {
839  nAdaptionOctants = pow(2, getDimension());
840  adaptionType = adaption::TYPE_COARSENING;
841  }
842 
843  std::size_t adaptionInfoId = adaptionData.create(adaptionType, adaption::ENTITY_CELL, currentRank);
844  adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
845  adaptionInfo.previous.reserve(nAdaptionOctants);
846  for (int k = 0; k < nAdaptionOctants; ++k) {
847  uint32_t treeId = treeIds[n];
848  bool isghost = treeGhosts[n];
849  OctantInfo octantInfo(treeId, !isghost);
850 
851  adaptionInfo.previous.emplace_back();
852  long &cellId = adaptionInfo.previous.back();
853  cellId = getOctantId(octantInfo);
854 
855  n++;
856  }
857  }
858 
859 #if BITPIT_ENABLE_MPI==1
860  // Ghost cells will be removed
861  if (isPartitioned() && getGhostCount() > 0) {
862  std::size_t adaptionInfoId = adaptionData.create(adaption::TYPE_DELETION, adaption::ENTITY_CELL, currentRank);
863  adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
864  adaptionInfo.previous.reserve(getGhostCount());
865  for (auto itr = ghostBegin(); itr != ghostEnd(); ++itr) {
866  adaptionInfo.previous.emplace_back();
867  long &deletedGhostId = adaptionInfo.previous.back();
868  deletedGhostId = itr.getId();
869  }
870  }
871 #endif
872  }
873 
874  return adaptionData.dump();
875 }

1049  // whereas cells associated to previous octants

1816 {

1821

1823  int maxLevel = m_tree->getMaxDepth();

1825  std::vector<std::vector<long>> hierarchicalCellIds(maxLevel + 1);
1826  for (int level = 0; level <= maxLevel; ++level) {
1827  hierarchicalCellIds[level].reserve(averageSize);
1828  }
1829 
1830  for (long cellId : cellIds) {
1831  int cellLevel = getCellLevel(cellId);
1832  hierarchicalCellIds[cellLevel].push_back(cellId);
1833  }
1834 
1835  // Update the adjacencies
1836  FaceInfoSet processedFaces;
1837  processedFaces.reserve(cellIds.size() * getDimension());
1838 
1839  for (int level = 0; level <= maxLevel; ++level) {
1840  for (long cellId : hierarchicalCellIds[level]) {
1841  Cell &cell = m_cells[cellId];
1842  OctantInfo octantInfo = getCellOctant(cellId);
1843  for (int face = 0; face < nCellFaces; ++face) {
1844  FaceInfo currentFaceInfo(cellId, face);
1845  if (processedFaces.count(currentFaceInfo) > 0) {
1846  continue;
1847  }
1848 
1849  // Find cell neighbours
1850  std::vector<uint32_t> neighTreeIds;
1851  std::vector<bool> neighGhostFlags;
1852  if (octantInfo.internal) {
1853  m_tree->findNeighbours(octantInfo.id, face, 1, neighTreeIds, neighGhostFlags);
1854  } else {
1855  m_tree->findGhostNeighbours(octantInfo.id, face, 1, neighTreeIds, neighGhostFlags);
1856  }

1859  //
1860  // Adjacencies will processed twice, once while processing the
1861  // current cell, and once while processing the neighbour cell.
1862  // However they will be set only once, because the function that
1863  // insert the adjacency in the cell will insert only unique
1864  // adjacencies.
1865  int nNeighs = neighTreeIds.size();
1866  for (int k = 0; k < nNeighs; ++k) {
1867  OctantInfo neighOctantInfo(neighTreeIds[k], !neighGhostFlags[k]);
1868  long neighId = getOctantId(neighOctantInfo);

1871  cell.pushAdjacency(face, neighId);
1872 
1873  // Set neighbour data
1874  int neighFace = oppositeFace[face];
1875  Cell &neigh = m_cells[neighId];
1876  neigh.pushAdjacency(neighFace, cellId);
1877 
1878  FaceInfo neighFaceInfo(neighId, neighFace);
1879  processedFaces.insert(neighFaceInfo);
1880  }
1881  }
1882  }
1883  }

1886  setAdjacenciesBuildStrategy(ADJACENCIES_AUTOMATIC);
