76 int CommunicationTags::generate(MPI_Comm communicator)
77 {
78  // Early return if the communicator is not valid
79  if (communicator == MPI_COMM_NULL) {
80  return IndexGenerator<int>::generate();
81  }
82 
83  // The master generates a unique tag
84  int tag;
85 
86  int rank;
87  MPI_Comm_rank(communicator, &rank);
88  if (rank == 0) {
89  tag = IndexGenerator<int>::generate();
90  }
91 
92  // Share the tag among the processes
93  MPI_Bcast(&tag, 1, MPI_INT, 0, communicator);
94 
95  return tag;
96 }

105 bool CommunicationTags::isAssigned(int tag, MPI_Comm communicator)
106 {
107  // The master checks if the tag is assigned
108  bool assigned;
109 
110  int rank;
111  MPI_Comm_rank(communicator, &rank);
112  if (rank == 0) {
113  assigned = IndexGenerator<int>::isAssigned(tag);
114  }
115 
116  // Share the result among the processes
117  MPI_Bcast(&assigned, 1, MPI_C_BOOL, 0, communicator);
118 
119  return assigned;
120 }

146 void CommunicationTags::trash(int tag, MPI_Comm communicator)
147 {
148  // Wait all the processes
149  //
150  // We need to avoid that a tag is trashed while some processes are still
151  // using it.
152  MPI_Barrier(communicator);
153 
154  // Only the master needs to trash the tag
155  int rank;
156  MPI_Comm_rank(communicator, &rank);
157  if (rank == 0) {
158  IndexGenerator<int>::trash(tag);
159  }
160 }
