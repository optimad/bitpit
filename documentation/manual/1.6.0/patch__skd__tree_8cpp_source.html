70 void SkdPatchInfo::buildCache(const PatchKernel::CellConstRange &cellRange)
71 {
72  m_cellBoxes = std::unique_ptr<BoxCache>(new BoxCache(2, &(m_patch->getCells())));
73  m_cellCentroids = std::unique_ptr<BoxCache>(new CentroidCache(1, &(m_patch->getCells())));
74  for (auto itr = cellRange.cbegin(); itr != cellRange.cend(); ++itr) {
75  std::size_t rawCellId = itr.getRawIndex();
76 
77  // Cell info
78  const Cell &cell = *itr;
79  int nCellVertices = cell.getVertexCount();
80  ConstProxyVector<long> cellConnect = cell.getVertexIds();
81 
82  // Bounding box
83  std::array<double, 3> &cellBoxMin = m_cellBoxes->rawAt(rawCellId, 0);
84  std::array<double, 3> &cellBoxMax = m_cellBoxes->rawAt(rawCellId, 1);
85  std::array<double, 3> &cellCentroid = m_cellCentroids->rawAt(rawCellId, 0);
86 
87  cellBoxMin = m_patch->getVertexCoords(cellConnect[0]);
88  cellBoxMax = cellBoxMin;
89  cellCentroid = cellBoxMin;
90  for (int i = 1; i < nCellVertices; ++i) {
91  const std::array<double, 3> &coords = m_patch->getVertexCoords(cellConnect[i]);
92  for (int d = 0; d < 3; ++d) {
93  cellBoxMin[d] = std::min(coords[d], cellBoxMin[d]);
94  cellBoxMax[d] = std::max(coords[d], cellBoxMax[d]);
95  cellCentroid[d] += coords[d];
96  }
97  }
98  cellCentroid /= double(nCellVertices);
99  }
100 }

228 void SkdNode::initializeBoundingBox()
229 {
230  const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
231 
232  // Evaluate the bounding box
233  m_boxMin = m_patchInfo->getCachedBoxMin(cellRawIds[m_cellRangeBegin]);
234  m_boxMax = m_patchInfo->getCachedBoxMax(cellRawIds[m_cellRangeBegin]);
235  for (std::size_t n = m_cellRangeBegin + 1; n < m_cellRangeEnd; n++) {
236  const std::size_t rawCellId = cellRawIds[n];
237  const std::array<double, 3> &cellBoxMin = m_patchInfo->getCachedBoxMin(rawCellId);
238  const std::array<double, 3> &cellBoxMax = m_patchInfo->getCachedBoxMax(rawCellId);
239  for (int d = 0; d < 3; ++d) {
240  m_boxMin[d] = std::min(cellBoxMin[d], m_boxMin[d]);
241  m_boxMax[d] = std::max(cellBoxMax[d], m_boxMax[d]);
242  }
243  }
244 
245  // Inlfate the bounding box by a small epsilon
246  //
247  // The small inflation allows us to test intersection of a query
248  // against a safety box and be assured that non-intersection of
249  // with the box implies non-intersection with all the contents
250  // in the box, even when computations are susceptible to round-off
251  // error.
252  double tolerance = m_patchInfo->getPatch().getTol();
253 
254  m_boxMin -= tolerance;
255  m_boxMax += tolerance;
256 }

482 void SkdNode::updatePointClosestCell(const std::array<double, 3> &point,
483  long *id, double *distance) const
484 {
485  if (getCellCount() == 0) {
486  return;
487  }
488 
489  const PatchKernel &patch = m_patchInfo->getPatch();
490  const PiercedVector<Cell> &cells = patch.getCells();
491  const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
492  std::vector<std::array<double, 3>> cellVertexCoordinates(ReferenceElementInfo::MAX_ELEM_VERTICES);
493 
494  for (std::size_t n = m_cellRangeBegin; n < m_cellRangeEnd; n++) {
495  std::size_t cellRawId = cellRawIds[n];
496  const Cell &cell = cells.rawAt(cellRawId);
497 
498  // Get the vertices ids
499  ConstProxyVector<long> elementVertexIds = cell.getVertexIds();
500  const int nElementVertices = elementVertexIds.size();
501 
502  // Get vertex coordinates
503  cellVertexCoordinates.resize(nElementVertices);
504  for (int i = 0; i < nElementVertices; ++i) {
505  cellVertexCoordinates[i] = patch.getVertex(elementVertexIds[i]).getCoords();
506  }
507 
508  // Evaluate the distance from the cell
509  double cellDistance = cell.evalPointDistance(point, cellVertexCoordinates.data());
510 
511  // Update closest distance
512  updateClosestCellInfo(point, cell.getId(), cellDistance, id, distance);
513  }

533 void SkdNode::updateClosestCellInfo(const std::array<double, 3> &point,
534  long cellId, double cellDistance,
535  long *closestId, double *closestDistance) const
536 {
537  // Detect if the specified cell is closer than the current closest cell
538  const int DISTANCE_CLOSER = - 1;
539  const int DISTANCE_EQUAL = 0;
540  const int DISTANCE_FARTHER = 1;
541 
542  int distanceFlag = DISTANCE_FARTHER;
543  if (utils::DoubleFloatingEqual()(cellDistance, *closestDistance)) {
544  distanceFlag = DISTANCE_EQUAL;
545  } else if (cellDistance < *closestDistance) {
546  distanceFlag = DISTANCE_CLOSER;
547  }
548 
549  // Consider the case where no closest cell is defined
550  //
551  // Even if the id of the closest cell is null, we may have an
552  // estimated of the closest cell distance. We need to update
553  // the closest cell information only if the current cells is
554  // closer than the estimate.
555  if (*closestId == Cell::NULL_ID) {
556  if (distanceFlag != DISTANCE_FARTHER) {
557  *closestId = cellId;
558  *closestDistance = cellDistance;
559  }
560 
561  return;
562  }
563 
564  // Update closest cell information accordingly to the distance flag:
565  // - if the specified cell is closer the specified cell will become the
566  // closest cell
567  // - if the two cells have the same distance, the closest cell will be
568  // chosen using the normal of the cells.
569  // - if the specified cell is farther there is nothing to do;
570  switch (distanceFlag) {
571 
572  case DISTANCE_CLOSER:
573  {
574  *closestId = cellId;
575  *closestDistance = cellDistance;
576 
577  break;
578  }
579 
580  case DISTANCE_EQUAL:
581  {
582  const PatchKernel &patch = m_patchInfo->getPatch();
583 
584  // Project point ont the specified cell
585  const Cell &cell = patch.getCell(cellId);
586 
587  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
588  const int nCellVertices = cellVertexIds.size();
589 
590  std::vector<std::array<double, 3>> cellVertexCoordinates(nCellVertices);
591  for (int i = 0; i < nCellVertices; ++i) {
592  cellVertexCoordinates[i] = patch.getVertex(cellVertexIds[i]).getCoords();
593  }
594 
595  double cellProjectionDistance;
596  std::array<double, 3> cellProjection;
597  cell.evalPointProjection(point, cellVertexCoordinates.data(), &cellProjection, &cellProjectionDistance);
598 
599  // Project point ont the closest cell
600  const Cell &closest = patch.getCell(*closestId);
601 
602  ConstProxyVector<long> closestVertexIds = closest.getVertexIds();
603  const int nClosestVertices = closestVertexIds.size();
604 
605  std::vector<std::array<double, 3>> closestVertexCoordinates(nClosestVertices);
606  for (int i = 0; i < nClosestVertices; ++i) {
607  closestVertexCoordinates[i] = patch.getVertex(closestVertexIds[i]).getCoords();
608  }
609 
610  double closestProjectionDistance;
611  std::array<double, 3> closestProjection;
612  closest.evalPointProjection(point, closestVertexCoordinates.data(), &closestProjection, &closestProjectionDistance);
613 
614  // Find normal of the cells at the centroids
615  //
616  // To be more precise, the normals should beevaluated on projection
617  // points.
618  std::array<double, 3> cellNormal = cell.evalNormal(cellVertexCoordinates.data());
619  std::array<double, 3> closestNormal = closest.evalNormal(closestVertexCoordinates.data());
620 
621  // Find cell more aligned with tespect to the normal
622  double cellAligement = std::abs(dotProduct(cellNormal, point - cellProjection));
623  double closestAligement = std::abs(dotProduct(closestNormal, point - closestProjection));
624 
625  // If the specified cell is more aligned than the closest cell, update
626  // the closest info.
627  if (cellAligement > closestAligement) {
628  *closestId = cellId;
629  *closestDistance = cellDistance;
630  }
631 
632  break;
633  }
634 
635  default:
636  {
637  // Nothing to do
638  }
639 
640  }
641 }

677 bool SkdNode::boxIntersectsSphere(const std::array<double, 3> &center, double radius) const
678 {
679  // If the box contains the center it will also intersect the sphere
680  if (boxContainsPoint(center, 0.)) {
681  return true;
682  }
683 
684  // Check if the distance between the center and its projection on
685  // the bounding box is smaller than the radius of the sphere.
686  std::array<double, 3> delta;
687  for (int d = 0; d < 3; d++) {
688  delta[d] = std::min(std::max(center[d], m_boxMin[d]), m_boxMax[d]) - center[d];
689  }
690  double distance = dotProduct(delta, delta);
691 
692  return (distance < radius * radius);
693 }

789 void PatchSkdTree::build(int leafCapacity)
790 {
791  const PatchKernel &patch = m_patchInfo.getPatch();
792 
793  // Clear existing tree
794  clear();
795 
796  // Set leaf size
797  setLeafCapacity(leafCapacity);
798 
799  // Initialize list of cell raw ids
800  std::size_t nCells;
801  PatchKernel::CellConstRange cellRange;
802  if (m_includeGhosts) {
803  nCells = patch.getCellCount();
804  cellRange.initialize(patch.cellConstBegin(), patch.cellConstEnd());
805  } else {
806  nCells = patch.getInternalCount();
807  cellRange.initialize(patch.internalConstBegin(), patch.internalConstEnd());
808  }
809 
810  if (m_cellRawIds.size() != nCells) {
811  m_cellRawIds.resize(nCells);
812  }
813 
814  std::size_t k = 0;
815  for (auto itr = cellRange.begin(); itr != cellRange.end(); ++itr) {
816  m_cellRawIds[k++] = itr.getRawIndex();
817  }
818 
819  // Build patch cache
820  m_patchInfo.buildCache(cellRange);
821 
822  // Initialize node list
823  m_nodes.reserve(1.5 * nCells / getLeafCapacity());
824 
825  // Create the root
826  m_nodes.emplace_back(&m_patchInfo, 0, nCells);
827 
828  // Create the tree
829  std::queue<std::size_t> nodeStack;
830  nodeStack.push(0);
831  while (!nodeStack.empty()) {
832  std::size_t nodeId = nodeStack.front();
833  nodeStack.pop();
834 
835  // Create the children of the node
836  createChildren(nodeId);
837 
838  // Add the newly created childrend to the stack
839  const SkdNode &node = getNode(nodeId);
840  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
841  std::size_t childId = node.getChildId(static_cast<SkdNode::ChildLocation>(i));
842  if (childId != SkdNode::NULL_ID) {
843  nodeStack.push(childId);
844  }
845  }
846  }
847 
848  // Patch cache is no longer needed
849  m_patchInfo.destroyCache();
850 }

953 void PatchSkdTree::createChildren(std::size_t parentId)
954 {
955  SkdNode &parent = _getNode(parentId);
956 
957  // Check if the parent is a leaf.
958  if (parent.getCellCount() <= m_leafCapacity) {
959  ++m_nLeafs;
960  return;
961  }
962 
963  // Evaluate the preferred direction along which elements will be split.
964  //
965  // The elements will be split along a plane normal to the direction
966  // for which the bounding box has the maximum length.
967  const std::array<double, 3> &parentBoxMin = parent.getBoxMin();
968  const std::array<double, 3> &parentBoxMax = parent.getBoxMax();
969 
970  int largerDirection = 0;
971  double boxMaximumLength = parentBoxMax[largerDirection] - parentBoxMin[largerDirection];
972  for (int d = 1; d < 3; ++d) {
973  double length = parentBoxMax[d] - parentBoxMin[d];
974  if (length > boxMaximumLength) {
975  largerDirection = d;
976  boxMaximumLength = length;
977  }
978  }
979 
980  // Split the elements.
981  //
982  // Preferred direction is tried first, if the split along this direction
983  // will produce and empty child (e.g., all the cell's box centroids have
984  // the same coordinate along the split direction and therefore all the
985  // cells would be assigned to the left chiled) the split will be performed
986  // along one of the other directions.
987  std::array<double, 3> parentWeightedCentroid = parent.evalBoxWeightedMean();
988  for (int d = 0; d < 3; ++d) {
989  // Update the split direction
990  int splitDirection = (largerDirection + d) % 3;
991 
992  // Get the threshold for the split
993  double splitThreshold = parentWeightedCentroid[splitDirection];
994 
995  // Order the elements
996  //
997  // All the elements with a centroid coordinate less or equal than the
998  // threshold will be assigned to the left child, the others will be
999  // assigned to the right child.
1000  std::size_t leftBegin = parent.m_cellRangeBegin;
1001  std::size_t leftEnd = parent.m_cellRangeEnd;
1002  std::size_t rightBegin = parent.m_cellRangeBegin;
1003  std::size_t rightEnd = parent.m_cellRangeEnd;
1004  while (true) {
1005  // Update the right begin
1006  while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[rightBegin])[splitDirection] <= splitThreshold) {
1007  rightBegin++;
1008  }
1009 
1010  // Update the left end
1011  while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[leftEnd - 1])[splitDirection] > splitThreshold) {
1012  leftEnd--;
1013  }
1014 
1015  // If all the elements are in the right position we can exit
1016  if (rightBegin == leftEnd) {
1017  break;
1018  }
1019 
1020  // If left end and and right begin are not equal, that the two ids
1021  // point to misplaced elements. Swap the elements, advance the ids
1022  // and continue iterating.
1023  std::iter_swap(m_cellRawIds.begin() + (leftEnd - 1), m_cellRawIds.begin() + rightBegin);
1024  }
1025 
1026  if (leftEnd <= leftBegin || rightEnd <= rightBegin) {
1027  continue;
1028  }
1029 
1030  // Create the left child
1031  long leftId = m_nodes.size();
1032  m_nodes.emplace_back(&m_patchInfo, leftBegin, leftEnd);
1033  m_nodes[parentId].m_children[static_cast<std::size_t>(SkdNode::CHILD_LEFT)] = leftId;
1034 
1035  // Create the right child
1036  long rightId = m_nodes.size();
1037  m_nodes.emplace_back(&m_patchInfo, rightBegin, rightEnd);
1038  m_nodes[parentId].m_children[static_cast<std::size_t>(SkdNode::CHILD_RIGHT)] = rightId;
1039 
1040  // Done
1041  return;
1042  }
1043 
1044  throw std::runtime_error("Unable to create the children.");
1045 }
