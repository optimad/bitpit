91  struct CellPositionLess
92  {
93  CellPositionLess(const PatchKernel &patch, bool native = true)
94  : m_patch(patch), m_native(native)
95  {
96  }
97 
98  virtual ~CellPositionLess() = default;
99 
100  bool operator()(long id_1, long id_2) const
101  {
102  std::array<double, 3> centroid_1;
103  std::array<double, 3> centroid_2;
104  if (m_native) {
105  centroid_1 = m_patch.evalCellCentroid(id_1);
106  centroid_2 = m_patch.evalCellCentroid(id_2);
107  } else {
108  centroid_1 = m_patch.PatchKernel::evalCellCentroid(id_1);
109  centroid_2 = m_patch.PatchKernel::evalCellCentroid(id_2);
110  }
111 
112  for (int k = 0; k < 3; ++k) {
113  if (utils::DoubleFloatingEqual()(centroid_1[k], centroid_2[k], m_patch.getTol())) {
114  continue;
115  }
116 
117  return centroid_1[k] < centroid_2[k];
118  }
119 
120  // If we are here the two cell centroids coincide. It's not
121  // possible to define an order for the two cells.
122  std::ostringstream stream;
123  stream << "It was not possible to define an order for cells " << id_1 << " and " << id_2 << ". ";
124  stream << "The two cells have the same centroid.";
125  throw std::runtime_error (stream.str());
126  }
127 
128  const PatchKernel &m_patch;
129  bool m_native;
130  };

158  struct CellFuzzyPositionLess
159  {
160  CellFuzzyPositionLess(PatchKernel &patch, bool native = true)
161  : m_patch(patch), m_native(native)
162  {
163  }
164 
165  virtual ~CellFuzzyPositionLess() = default;
166 
167  bool operator()(long id_1, long id_2) const
168  {
169  // Select the first vertex of the first cell
170  ConstProxyVector<long> cellVertexIds_1 = m_patch.getCell(id_1).getVertexIds();
171 
172  std::size_t vertexLocalId_1 = 0;
173  long vertexId_1 = cellVertexIds_1[vertexLocalId_1];
174 
175  // The vertex of the second cell is choosen as the first vertex on
176  // that cell not equal to the selected vertex of the first cell.
177  ConstProxyVector<long> cellVertexIds_2 = m_patch.getCell(id_2).getVertexIds();
178  std::size_t nCellVertices_2 = cellVertexIds_2.size();
179 
180  std::size_t vertexLocalId_2 = 0;
181  long vertexId_2 = Vertex::NULL_ID;
182  while (vertexLocalId_2 <= nCellVertices_2) {
183  vertexId_2 = cellVertexIds_2[vertexLocalId_2];
184  if (vertexId_1 != vertexId_2) {
185  break;
186  }
187 
188  ++vertexLocalId_2;
189  }
190 
191  // Compare the two vertices
192  if (vertexId_2 != Vertex::NULL_ID) {
193  const std::array<double, 3> &vertexCoords_1 = m_patch.getVertex(vertexId_1).getCoords();
194  const std::array<double, 3> &vertexCoords_2 = m_patch.getVertex(vertexId_2).getCoords();
195  for (int k = 0; k < 3; ++k) {
196  if (utils::DoubleFloatingEqual()(vertexCoords_1[k], vertexCoords_2[k], m_patch.getTol())) {
197  continue;
198  }
199 
200  return vertexCoords_1[k] < vertexCoords_2[k];
201  }
202  }
203 
204  // If we are here it was not possible to find a vertex on the
205  // second cell for the comparison.
206  std::ostringstream stream;
207  stream << "Unable to fuzzy order cells " << id_1 << " and " << id_2 << ". ";
208  throw std::runtime_error (stream.str());
209  }
210 
211  PatchKernel &m_patch;
212  bool m_native;
213  };
