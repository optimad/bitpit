106 SystemSolver::SystemSolver(bool debug)
107  : m_assembled(false), m_pivotType(PIVOT_NONE),
108 #if BITPIT_ENABLE_MPI==1
109  m_communicator(MPI_COMM_SELF), m_partitioned(false),
110 #endif
111  m_rowGlobalOffset(0), m_colGlobalOffset(0),
112  m_A(nullptr), m_rhs(nullptr), m_solution(nullptr),
113  m_rpivot(nullptr), m_cpivot(nullptr), m_KSP(nullptr)
114 {
115  // Add debug options
116  if (debug) {
117  addInitOption("-log_view");
118  addInitOption("-ksp_monitor_true_residual");
119  addInitOption("-ksp_converged_reason");
120  addInitOption("-ksp_monitor_singular_value");
121  }
122 
123  // Initialize Petsc
124  if (m_nInstances == 0) {
125  // Generate command line arguments
126  //
127  // The first argument is the executable name and it is set to a
128  // dummy value.
129  std::string help = "None";
130  std::string programName = "bitpit_system_solver";
131 
132  int argc = 1 + m_options.size();
133  char **argv = new char*[argc + 1];
134  argv[0] = strdup(programName.data());
135  for (std::size_t i = 0; i < m_options.size(); i++) {
136  argv[1 + i] = strdup(m_options[i].data());
137  }
138  argv[argc] = nullptr;
139 
140  // Call initialization
141  PetscInitialize(&argc, &argv, 0, help.data());
142 
143  // Clean-up
144  for (int i = 0; i < argc; ++i) {
145  free(argv[i]);
146  }
147 
148  delete[] argv;
149  }
150 
151  // Increase the number of instances
152  ++m_nInstances;
153 }

205 void SystemSolver::assembly(const SparseMatrix &matrix, PivotType pivotType)
206 {
207  // Check if the matrix is assembled
208  if (!matrix.isAssembled()) {
209  throw std::runtime_error("Unable to assembly the system. The matrix is not yet assembled.");
210  }
211 
212  // Clear the system
213  clear();
214 
215 #if BITPIT_ENABLE_MPI == 1
216  // Set the communicator
217  setCommunicator(matrix.getCommunicator());
218 
219  // Detect if the system is partitioned
220  m_partitioned = matrix.isPartitioned();
221 #endif
222 
223  // Initialize matrix
224  matrixInit(matrix);
225  matrixFill(matrix);
226 
227  // Initialize pivot
228  pivotInit(pivotType);
229  if (getPivotType() != PIVOT_NONE) {
230  matrixReorder();
231  }
232 
233  // Initialize RHS and solution vectors
234 #if BITPIT_ENABLE_MPI == 1
235  vectorsInit(matrix.extractGhostGlobalCols());
236 #else
237  vectorsInit();
238 #endif
239 
240  // Initialize Krylov solver
241  KSPInit();
242 
243  // The system is now assembled
244  m_assembled = true;
245 }

366 void SystemSolver::solve()
367 {
368  // Check if the system is assembled
369  if (!isAssembled()) {
370  throw std::runtime_error("Unable to solve the system. The system is not yet assembled.");
371  }
372 
373  // Reorder the vectors
374  if (getPivotType() != PIVOT_NONE) {
375  vectorsReorder(PETSC_FALSE);
376  }
377 
378  // Solve the system
379  m_KSPStatus.error = KSPSolve(m_KSP, m_rhs, m_solution);
380 
381  // Set solver info
382  if (m_KSPStatus.error == 0) {
383  KSPGetIterationNumber(m_KSP, &m_KSPStatus.its);
384  KSPGetConvergedReason(m_KSP, &m_KSPStatus.convergence);
385  } else {
386  m_KSPStatus.its = -1;
387  m_KSPStatus.convergence = KSP_DIVERGED_BREAKDOWN;
388  }
389 
390  // Reorder the vectors
391  if (getPivotType() != PIVOT_NONE) {
392  vectorsReorder(PETSC_TRUE);
393  }
394 }

420 void SystemSolver::matrixInit(const SparseMatrix &matrix)
421 {
422  long nRows = matrix.getRowCount();
423  long nCols = matrix.getColCount();
424 
425  // Set row and column global offset
426 #if BITPIT_ENABLE_MPI == 1
427  m_rowGlobalOffset = matrix.getRowGlobalOffset();
428  m_colGlobalOffset = matrix.getColGlobalOffset();
429 #else
430  m_rowGlobalOffset = 0;
431  m_colGlobalOffset = 0;
432 #endif
433 
434 #if BITPIT_ENABLE_MPI == 1
435  // Evaluate the number of non-zero elements
436  //
437  // For each row we count the number of local non-zero elements (d_nnz) and
438  // the number of non-zero elements that belong to other processors (o_nnz)
439  long nGlobalCols = matrix.getColGlobalCount();
440  long nOffDiagonalCols = nGlobalCols - nCols;
441  long firstColGlobalId = matrix.getColGlobalOffset();
442  long lastColGlobalId = firstColGlobalId + nCols - 1;
443 
444  std::vector<int> d_nnz(nRows, 0);
445  std::vector<int> o_nnz(nRows, 0);
446  if (nOffDiagonalCols > 0) {
447  for (long row = 0; row < nRows; ++row) {
448  ConstProxyVector<long> rowPattern = matrix.getRowPattern(row);
449  int nRowNZ = rowPattern.size();
450  for (int k = 0; k < nRowNZ; ++k) {
451  long columnGlobalId = rowPattern[k];
452  if (columnGlobalId < firstColGlobalId || columnGlobalId > lastColGlobalId) {
453  ++o_nnz[row];
454  } else {
455  ++d_nnz[row];
456  }
457  }
458  }
459  } else {
460  for (long row = 0; row < nRows; ++row) {
461  ConstProxyVector<long> rowPattern = matrix.getRowPattern(row);
462  d_nnz[row] = rowPattern.size();
463  }
464  }
465 
466  // Create the matrix
467  MatCreateAIJ(m_communicator, nRows, nCols, PETSC_DETERMINE, PETSC_DETERMINE, 0, d_nnz.data(), 0, o_nnz.data(), &m_A);
468 #else
469  // Evaluate the number of non-zero elements
470  std::vector<int> d_nnz(nCols);
471 
472  for (long row = 0; row < nRows; ++row) {
473  ConstProxyVector<long> rowPattern = matrix.getRowPattern(row);
474  d_nnz[row] = rowPattern.size();
475  }
476 
477  // Create the matrix
478  MatCreateSeqAIJ(PETSC_COMM_SELF, nRows, nCols, 0, d_nnz.data(), &m_A);
479 #endif
480 }

487 void SystemSolver::matrixFill(const SparseMatrix &matrix)
488 {
489  const long nRows = matrix.getRowCount();
490  const long maxRowNZ = matrix.getMaxRowNZCount();
491 
492  // Create the matrix
493  if (maxRowNZ > 0) {
494  std::vector<PetscInt> rowNZGlobalIds(maxRowNZ);
495  std::vector<PetscScalar> rowNZValues(maxRowNZ);
496 
497  for (long row = 0; row < nRows; ++row) {
498  ConstProxyVector<long> rowPattern = matrix.getRowPattern(row);
499  ConstProxyVector<double> rowValues = matrix.getRowValues(row);
500 
501  const int nRowNZ = rowPattern.size();
502  const PetscInt globalRow = m_rowGlobalOffset + row;
503  for (int k = 0; k < nRowNZ; ++k) {
504  rowNZGlobalIds[k] = rowPattern[k];
505  rowNZValues[k] = rowValues[k];
506  }
507 
508  MatSetValues(m_A, 1, &globalRow, nRowNZ, rowNZGlobalIds.data(), rowNZValues.data(), INSERT_VALUES);
509  }
510  }
511 
512  // Let petsc build the matrix
513  MatAssemblyBegin(m_A, MAT_FINAL_ASSEMBLY);
514  MatAssemblyEnd(m_A, MAT_FINAL_ASSEMBLY);
515 }

526 void SystemSolver::matrixUpdate(const std::vector<long> &rows, const SparseMatrix &elements)
527 {
528  const long maxRowElements = std::max(elements.getMaxRowNZCount(), 0L);
529 
530  // Check if element columns are already in the pattern
531  std::unordered_set<PetscInt> currentRowPattern;
532 
533  for (std::size_t n = 0; n < rows.size(); ++n) {
534  ConstProxyVector<long> rowPattern = elements.getRowPattern(n);
535  const int nRowElements = rowPattern.size();
536  if (nRowElements == 0) {
537  continue;
538  }
539 
540  // Get global row
541  long row = rows[n];
542  const PetscInt globalRow = m_rowGlobalOffset + row;
543 
544  // Get current row pattern
545  PetscInt nCurrentRowElements = 0;
546  const PetscInt *rawCurrentRowPattern = nullptr;
547  MatGetRow(m_A, globalRow, &nCurrentRowElements, &rawCurrentRowPattern, NULL);
548  assert(nCurrentRowElements != 0);
549  assert(rawCurrentRowPattern != nullptr);
550 
551  currentRowPattern.clear();
552  for (PetscInt k = 0; k < nCurrentRowElements; ++k) {
553  currentRowPattern.insert(rawCurrentRowPattern[k]);
554  }
555 
556  // Check if element columns are already in the pattern
557  for (int k = 0; k < nRowElements; ++k) {
558  if (currentRowPattern.count(rowPattern[k]) == 0) {
559  throw std::runtime_error("The element is not in the matrix.");
560  }
561  }
562 
563  // Restore row
564  MatRestoreRow(m_A, globalRow, &nCurrentRowElements, &rawCurrentRowPattern, NULL);
565  }
566 
567  // Update element values
568  std::vector<PetscInt> rawRowPattern(maxRowElements);
569  std::vector<PetscScalar> rawRowValues(maxRowElements);
570 
571  for (std::size_t n = 0; n < rows.size(); ++n) {
572  ConstProxyVector<double> rowValues = elements.getRowValues(n);
573  const int nRowElements = rowValues.size();
574  if (nRowElements == 0) {
575  continue;
576  }
577 
578  // Get global row
579  long row = rows[n];
580  const PetscInt globalRow = m_rowGlobalOffset + row;
581 
582  // Get pattern
583  ConstProxyVector<long> rowPattern = elements.getRowPattern(n);
584 
585  // Update values
586  for (int k = 0; k < nRowElements; ++k) {
587  rawRowPattern[k] = rowPattern[k];;
588  rawRowValues[k] = rowValues[k];
589  }
590 
591  MatSetValues(m_A, 1, &globalRow, nRowElements, rawRowPattern.data(), rawRowValues.data(), INSERT_VALUES);
592  }
593 
594  // Let petsc assembly the matrix after the update
595  MatAssemblyBegin(m_A, MAT_FINAL_ASSEMBLY);
596  MatAssemblyEnd(m_A, MAT_FINAL_ASSEMBLY);
597 }

602 void SystemSolver::matrixReorder()
603 {
604  Mat B;
605  MatPermute(m_A, m_rpivot, m_cpivot, &B);
606  MatDestroy(&m_A);
607  MatDuplicate(B, MAT_COPY_VALUES, &m_A);
608  MatDestroy(&B);

644  VecCreateSeq(PETSC_COMM_SELF, nRows, &m_rhs);

888 void SystemSolver::pivotInit(PivotType pivotType)
889 {
890  m_pivotType = pivotType;
891 
892  MatOrderingType petscPivotType;
893  switch (m_pivotType) {
894 
895  case (PIVOT_ND):
896  petscPivotType = MATORDERINGNATURAL;
897  break;
898 
899  case (PIVOT_1WD):
900  petscPivotType = MATORDERING1WD;
901  break;
902 
903  case (PIVOT_RCM):
904  petscPivotType = MATORDERINGRCM;
905  break;
906 
907  case (PIVOT_MD):
908  petscPivotType = MATORDERINGQMD;
909  break;
910 
911  default:
912  return;
913 
914  }
915 
916  MatGetOrdering(m_A, petscPivotType, &m_rpivot, &m_cpivot);
917  ISSetPermutation(m_rpivot);
918  ISSetPermutation(m_cpivot);

963 #endif

965 
966  KSPSetType(m_KSP, KSPFGMRES);
967  if (m_KSPOptions.restart != PETSC_DEFAULT) {
968  KSPGMRESSetRestart(m_KSP, m_KSPOptions.restart);
969  }
970  if (m_KSPOptions.rtol != PETSC_DEFAULT || m_KSPOptions.maxits != PETSC_DEFAULT) {
971  KSPSetTolerances(m_KSP, m_KSPOptions.rtol, PETSC_DEFAULT, PETSC_DEFAULT, m_KSPOptions.maxits);
972  }
973  KSPSetInitialGuessNonzero(m_KSP, PETSC_TRUE);
974 
975  PCType preconditionerType;
976 #if BITPIT_ENABLE_MPI == 1
977  if (isPartitioned()) {
978  preconditionerType = PCASM;
979  } else {
980  preconditionerType = PCILU;
981  }
982 #else
983  preconditionerType = PCILU;
984 #endif
985 
986  PC preconditioner;

988  PCSetType(preconditioner, preconditionerType);
989  if (strcmp(preconditionerType, PCASM) == 0) {
990  if (m_KSPOptions.overlap != PETSC_DEFAULT) {
991  PCASMSetOverlap(preconditioner, m_KSPOptions.overlap);
992  }
993  } else if (strcmp(preconditionerType, PCILU) == 0) {
994  if (m_KSPOptions.levels != PETSC_DEFAULT) {

996  }
997  }

999  KSPSetUp(m_KSP);
1000

1003  KSP *subksp;

1011  PCFactorSetLevels(subpc, m_KSPOptions.sublevels);
1012  }
1013  if (m_KSPOptions.subrtol != PETSC_DEFAULT) {
1014  KSPSetTolerances(subksp[i], m_KSPOptions.subrtol, PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT);
1015  }
1016  }
1017  }
