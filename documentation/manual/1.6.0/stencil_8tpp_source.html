775 template<typename weight_t>
776 void DiscreteStencil<weight_t>::optimize(double tolerance)
777 {
778  int nBuckets = getBucketCount();
779  for (int i = 0; i < nBuckets; ++i) {
780  std::size_t nBucketItems = size(i);
781  for (std::size_t j = 0; j < nBucketItems; ++j) {
782  if (isWeightNeglibile(i, j, tolerance)) {
783  m_pattern.eraseItem(i, j);
784  m_weights.eraseItem(i, j);
785  --j;
786  --nBucketItems;
787  }
788  }
789  }
790 }
