209 void SurfUnstructured::extractEdgeNetwork(SurfUnstructured &net)
210 {
211  // ====================================================================== //
212  // VARIABLES DECLARATION //
213  // ====================================================================== //
214 
215  // Local variables
216  bool check;
217  int n_faces, n_adj;
218  long id;
219  const long *adjacencies;
220 
221  // Counters
222  int i, j;
223  VertexIterator v_, ve_ = vertexEnd();
224  CellIterator c_, ce_ = cellEnd();
225 
226  // ====================================================================== //
227  // INITIALIZE DATA STRUCTURE //
228  // ====================================================================== //
229  net.reserveCells(net.getCellCount() + countFaces());
230  net.reserveVertices(net.getVertexCount() + getVertexCount());
231 
232  // ====================================================================== //
233  // ADD VERTEX TO net //
234  // ====================================================================== //
235  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
236  net.addVertex(v_->getCoords(), v_->getId());
237  } //next v_
238 
239  // ====================================================================== //
240  // ADD EDGES //
241  // ====================================================================== //
242  for (c_ = cellBegin(); c_ != ce_; ++c_) {
243  id = c_->getId();
244  n_faces = c_->getFaceCount();
245  ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
246  for (i = 0; i < n_faces; ++i) {
247  check = true;
248  n_adj = c_->getAdjacencyCount(i);
249  adjacencies = c_->getAdjacencies(i);
250  for (j = 0; j < n_adj; ++j) {
251  check = check && (id > adjacencies[j]);
252  } //next j
253  if (check) {
254  // Get edge type
255  ElementType edgeType = c_->getFaceType(i);
256 
257  // Get edge connect
258  ConstProxyVector<long> faceConnect = c_->getFaceConnect(i);
259  int faceConnectSize = faceConnect.size();
260 
261  std::unique_ptr<long[]> edgeConnect = std::unique_ptr<long[]>(new long[faceConnectSize]);
262  for (int k = 0; k < faceConnectSize; ++k) {
263  edgeConnect[k] = faceConnect[k];
264  }
265 
266  // Add edge
267  net.addCell(edgeType, std::move(edgeConnect));
268  }
269  } //next i
270  } //next c_
271 
272  return;
273 }

353 unsigned short SurfUnstructured::importSTL(STLObj &STL, int PIDOffset, bool PIDSquash,
354  std::unordered_map<int, std::string> *PIDNames)
355 {
356  // ====================================================================== //
357  // OPEN STL FILE //
358  // ====================================================================== //
359  STL.open("in");
360  if (STL.err != 0) {
361  return STL.err;
362  }
363 
364  // ====================================================================== //
365  // LOAD ALL SOLID FROM THE STL FILE //
366  // ====================================================================== //
367  int pid = PIDOffset;
368  if (!PIDSquash) {
369  --pid;
370  }
371 
372  while (true) {
373  // ====================================================================== //
374  // LOAD SOLID FROM THE STL FILE //
375  // ====================================================================== //
376  int nVertex = 0;
377  int nSimplex = 0;
378  std::vector<std::array<double, 3>> vertexList;
379  std::vector<std::array<double, 3>> normalList;
380  std::vector<std::array<int, 3>> connectivityList;
381  std::string name = "";
382 
383  STL.loadSolid(nVertex, nSimplex, vertexList, normalList, connectivityList, name);
384  if (nVertex == 0) {
385  break;
386  }
387 
388  // ====================================================================== //
389  // PID AND NAME OF THE SOLID //
390  // ====================================================================== //
391  if (!PIDSquash) {
392  ++pid;
393  if (PIDNames && !name.empty()) {
394  PIDNames->insert({{pid, name}});
395  }
396  }
397 
398  // ====================================================================== //
399  // PREPARE MESH FOR DATA IMPORT //
400  // ====================================================================== //
401  reserveVertices(getVertexCount() + nVertex);
402  reserveCells(getCellCount() + nSimplex);
403 
404  // ====================================================================== //
405  // ADD VERTICES TO MESH //
406  // ====================================================================== //
407  std::vector<std::array<double, 3>>::const_iterator v_, ve_;
408 
409  std::vector<long> vertexMap;
410  vertexMap.reserve(nVertex);
411 
412  long v_counter = 0;
413  ve_ = vertexList.cend();
414  for (v_ = vertexList.cbegin(); v_ != ve_; ++v_) {
415  VertexIterator i_ = addVertex(*v_);
416  vertexMap[v_counter] = i_->getId();
417  ++v_counter;
418  } //next v_
419 
420  // ====================================================================== //
421  // ADD CELLS TO MESH //
422  // ====================================================================== //
423  std::vector<std::array<int,3>>::const_iterator c_, ce_;
424  std::array<int,3>::const_iterator w_, we_;
425 
426  ce_ = connectivityList.cend();
427  for (c_ = connectivityList.cbegin(); c_ != ce_; ++c_) {
428  // Remap STL connectivity
429  int n_v = c_->size();
430  std::vector<long> connect(n_v, Vertex::NULL_ID);
431  we_ = c_->cend();
432  int i = 0;
433  for (w_ = c_->cbegin(); w_ < we_; ++w_) {
434  connect[i] = vertexMap[*w_];
435  ++i;
436  } //next w_
437 
438  // Add cell
439  CellIterator cellIterator = addCell(getSTLFacetType(n_v), connect);
440  cellIterator->setPID(pid);
441  } //next c_
442 
443  // ====================================================================== //
444  // Multi-body STL files are supported only in ASCII mode //
445  // ====================================================================== //
446  if (STL.stl_type) {
447  break;
448  }
449  }
450 
451  // ====================================================================== //
452  // CLOSE STL FILE //
453  // ====================================================================== //
454  STL.close("in");
455 
456  return 0;
457 }

520 unsigned short SurfUnstructured::exportSTLSingle(const std::string &name, const bool &isBinary, bool exportInternalsOnly)
521 {
522  // ====================================================================== //
523  // VARIABLES DECLARATION //
524  // ====================================================================== //
525 
526  // Local variables
527  int nVertex;
528  int nSimplex;
529  std::vector<std::array<double, 3>> vertexList;
530  std::vector<std::array<double, 3>> normalList;
531  std::vector<std::array<int,3>> connectivityList;
532  std::unordered_map<long, long> vertexMap;
533  std::array<int,3> dummyIntArray;
534 
535  // Counters
536  int v_count ,j;
537  std::vector<std::array<double, 3>>::iterator i_;
538  std::vector<std::array<int,3>>::iterator j_;
539  std::array<int,3>::iterator k_, ke_;
540  VertexIterator v_, ve_;
541  CellIterator c_, cb_, ce_;
542 
543  // ====================================================================== //
544  // INITIALIZE DATA STRUCTURE //
545  // ====================================================================== //
546  dummyIntArray.fill(0) ;
547 
548  nSimplex = getInternalCount();
549 #if BITPIT_ENABLE_MPI==1
550  if (!exportInternalsOnly) {
551  nSimplex += getGhostCount();
552  }
553 #endif
554 
555  vertexList.resize(getVertexCount());
556  normalList.resize(nSimplex);
557  connectivityList.resize(nSimplex, dummyIntArray);
558 
559  // ====================================================================== //
560  // CREATE VERTEX LIST //
561  // ====================================================================== //
562  i_ = vertexList.begin();
563  ve_ = vertexEnd();
564  v_count = 0;
565  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
566 
567  // Store vertex coordinates
568  *i_ = v_->getCoords();
569  vertexMap[v_->getId()] = v_count;
570 
571  // Update counters
572  ++v_count;
573  ++i_;
574 
575  } //next v_
576  nVertex = getVertexCount();
577 
578  // ====================================================================== //
579  // CREATE CONNECTIVITY //
580  // ====================================================================== //
581  if (exportInternalsOnly) {
582  cb_ = internalBegin();
583  ce_ = internalEnd();
584  }
585  else {
586  cb_ = cellBegin();
587  ce_ = cellEnd();
588  }
589  i_ = normalList.begin();
590  j_ = connectivityList.begin();
591  for (c_ = cb_; c_ != ce_; ++c_) {
592 
593  // Build normals
594  *i_ = evalFacetNormal(c_->getId());
595 
596  // Build connectivity
597  ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
598 
599  ke_ = j_->end();
600  j = 0;
601  for (k_ = j_->begin(); k_ != ke_; ++k_) {
602  *k_ = vertexMap[cellVertexIds[j]];
603  ++j;
604  } //next k_
605 
606  // Update counters
607  ++j_;
608  ++i_;
609  } //next c_
610 
611  // ====================================================================== //
612  // EXPORT STL DATA //
613  // ====================================================================== //
614  STLObj STL(name, isBinary);
615  STL.save("", nVertex, nSimplex, vertexList, normalList, connectivityList);
616 
617  return 0;
618 }

632 unsigned short SurfUnstructured::exportSTLMulti(const std::string &name, bool exportInternalsOnly,
633  std::unordered_map<int, std::string> *PIDNames)
634 {
635 #if not BITPIT_ENABLE_MPI==1
636  BITPIT_UNUSED(exportInternalsOnly);
637 #endif
638 
639  int nTotVertex;
640  std::vector<std::array<double, 3>> totVertexList;
641  std::unordered_map<long, long> vertexMap;
642 
643  int nLocSimplex;
644  std::vector<std::array<double, 3>> normalLocList;
645  std::vector<std::array<int,3>> connectivityLocList;
646 
647  STLObj STL(name, false);
648 
649  STL.open("out");
650  if (STL.err != 0) {
651  return STL.err;
652  }
653 
654  // Create the vertex map
655  nTotVertex = getVertexCount();
656  totVertexList.resize(nTotVertex);
657  unsigned int count_v = 0;
658  for(const Vertex &v: getVertices()){
659  totVertexList[count_v] = v.getCoords();
660  vertexMap[v.getId()] = count_v;
661  ++count_v;
662  }
663 
664  // Export the internal cells
665  for(int pid : getInternalPIDs()){
666  std::vector<long> cells = getInternalsByPID(pid);
667 
668  nLocSimplex = (int) cells.size();
669  connectivityLocList.resize(nLocSimplex);
670  normalLocList.resize(nLocSimplex);
671 
672  std::vector<std::array<int,3>>::iterator itC = connectivityLocList.begin();
673  std::vector<std::array<double, 3>>::iterator itN = normalLocList.begin();
674 
675  // Fill local connectivity and normals structures
676  for (auto id : cells) {
677  // Fill connectivity
678  const Cell &cell = getCell(id);
679  for (int iloc = 0; iloc<3; ++iloc) {
680  (*itC)[iloc] = vertexMap[cell.getConnect()[iloc]];
681  }
682 
683  // Fill normal
684  *itN = evalFacetNormal(cell.getId());
685 
686  // Increment iterators
687  ++itC;
688  ++itN;
689  }
690 
691  // Write the solid associated to the current PID
692  std::string name;
693  if (PIDNames && PIDNames->count(pid) > 0) {
694  name = PIDNames->at(pid);
695  } else {
696  name = std::to_string(pid);
697  }
698 
699  STL.saveSolid(name, nTotVertex, nLocSimplex, totVertexList, normalLocList, connectivityLocList);
700  }
701 
702 #if BITPIT_ENABLE_MPI==1
703  // Export ghost cells
704  long nGhosts = getGhostCount();
705  if (!exportInternalsOnly && nGhosts > 0) {
706  nLocSimplex = nGhosts;
707  connectivityLocList.resize(nLocSimplex);
708  normalLocList.resize(nLocSimplex);
709 
710  std::vector<std::array<int,3>>::iterator itC = connectivityLocList.begin();
711  std::vector<std::array<double, 3>>::iterator itN = normalLocList.begin();
712 
713  // Fill local connectivity and normals structures
714  CellConstIterator endItr = ghostConstEnd();
715  for (CellConstIterator itr = ghostConstBegin(); itr != endItr; ++itr) {
716  // Fill connectivity
717  for (int iloc = 0; iloc<3; ++iloc) {
718  (*itC)[iloc] = vertexMap[itr->getConnect()[iloc]];
719  }
720 
721  // Fill normal
722  *itN = evalFacetNormal(itr.getId());
723 
724  // Increment iterators
725  ++itC;
726  ++itN;
727  }
728 
729  // Write the solid associated to the ghosts
730  STL.saveSolid("ghosts", nTotVertex, nLocSimplex, totVertexList, normalLocList, connectivityLocList);
731  }
732 #endif
733 
734  STL.close("out");
735 
736  return 0;
737 }

779 {

781  // VARIABLES DECLARATION //
782  // ====================================================================== //
783 
784  // Local variables
785  DGFObj dgf_in(dgf_name);
786  int nV = 0, nS = 0;
787  long vcount, idx;
788  std::vector<std::array<double, 3>> vertex_list;
789  std::vector<std::vector<int>> simplex_list;
790  std::vector<int> simplex_PID;
791  std::vector<long> vertex_map;

793 
794  // Counters
795  std::vector<std::array<double, 3>>::const_iterator v_, ve_;
796  std::vector<std::vector<int>>::iterator c_, ce_;
797  std::vector<int>::iterator i_, ie_;
798  std::vector<long>::iterator j_, je_;
799 
800  // ====================================================================== //
801  // IMPORT DATA //
802  // ====================================================================== //
803 
804  // Read vertices and cells from DGF file
805  dgf_in.load(nV, nS, vertex_list, simplex_list, simplex_PID);
806 
807  // Add vertices
808  ve_ = vertex_list.cend();
809  vcount = 0;
810  vertex_map.resize(nV);
811  for (v_ = vertex_list.cbegin(); v_ != ve_; ++v_) {
812  idx = addVertex(*v_)->getId();
813  vertex_map[vcount] = idx;
814  ++vcount;
815  } //next v_
816 
817  // Update connectivity infos
818  ce_ = simplex_list.end();
819  for (c_ = simplex_list.begin(); c_ != ce_; ++c_) {
820  ie_ = c_->end();
821  for (i_ = c_->begin(); i_ != ie_; ++i_) {
822  *i_ = vertex_map[*i_];
823  } //next i_
824  } //next c_
825 
826  // Add cells
827  int k;
828  for (c_ = simplex_list.begin(), k = 0; c_ != ce_; ++c_, ++k) {
829  // Create cell
830  i_ = c_->begin();
831  connect.resize(c_->size(), Vertex::NULL_ID);
832  je_ = connect.end();
833  for (j_ = connect.begin(); j_ != je_; ++j_) {
834  *j_ = *i_;
835  ++i_;
836  } //next j_
837  CellIterator cellIterator = addCell(getDGFFacetType(c_->size()), connect);
838 
839  // Set cell PID
840  int cellPID = PIDOffset;
841  if (!PIDSquash) {
842  cellPID += simplex_PID[k];
843  }
844 
845  cellIterator->setPID(cellPID);
846  } //next c_
847 
848  return 0;
849 }
