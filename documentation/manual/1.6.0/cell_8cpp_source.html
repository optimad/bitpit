99 bitpit::FlatVector2D<long> Cell::createNeighbourhoodStorage(bool storeNeighbourhood)
100 {
101  ElementType type = getType();

256 void Cell::_initialize(bool interior, bool initializeNeighbourhood, bool storeNeighbourhood)
257 {
258  setInterior(interior);
259 
260  // Neighbourhood
261  if (initializeNeighbourhood) {
262  // To reduce memory fragmentation, destroy both interfaces/adjacencies
263  // before resetting the interfaces/adjacencies
264  int reallocateInterfaces = (m_interfaces.getItemCount() != getFaceCount());
265  if (reallocateInterfaces) {
266  m_interfaces.destroy();
267  }
268 
269  int reallocateAdjacencies = (m_adjacencies.getItemCount() != getFaceCount());
270  if (reallocateAdjacencies) {
271  m_adjacencies.destroy();
272  }
273 
274  // Reset the interfaces/adjacencies
275  resetInterfaces(storeNeighbourhood);
276  resetAdjacencies(storeNeighbourhood);
277  }
278 }

774 void Cell::display(std::ostream &out, unsigned short int indent) const
775 {
776  // ====================================================================== //
777  // VARIABLES DECLARATION //
778  // ====================================================================== //
779 
780  // Local variables
781  std::string t_s(indent, ' ');
782 
783  // Counters
784  int i, j;
785  int nn;
786 
787  // ====================================================================== //
788  // DISPLAY INFOS //
789  // ====================================================================== //
790  if (getType() == ElementType::UNDEFINED) {
791  out << t_s << "cell type: (unknown)" << std::endl;
792  return;
793  }
794 
795  // Scope variables -------------------------------------------------- //
796  int nv = getVertexCount();
797  int nf = getFaceCount();
798 
799  ConstProxyVector<long> cellVertexIds = getVertexIds();
800 
801  // General info ----------------------------------------------------- //
802  out << t_s << "cell type: " << getType() << std::endl;
803  out << t_s << "ID: " << getId() << std::endl;
804  out << t_s << "is ghost: ";
805  if (m_interior) { out << "(false)"; }
806  else { out << "(true)"; }
807  out << std::endl;
808 
809  // Connectivity infos --------------------------------------------------- //
810  out << t_s << "connectivity: [ ";
811  for (i = 0; i < nv-1; ++i) {
812  if (cellVertexIds[i] == Vertex::NULL_ID) out << "n.a. ";
813  else out << cellVertexIds[i] << ", ";
814  } //next i
815  if (cellVertexIds[nv-1] == Vertex::NULL_ID) out << "n.a. ";
816  else out << cellVertexIds[nv-1] << " ]" << std::endl;
817 
818  // neighbors infos ------------------------------------------------------ //
819  if (m_adjacencies.size() > 0) {
820  out << t_s << "neighbors: [ ";
821  for (i = 0; i < nf; ++i) {
822  nn = getAdjacencyCount(i);
823  out << "[ ";
824  if (nn == 0) {
825  out << "n.a. ";
826  } else {
827  for (j = 0; j < nn; ++j) {
828  out << getAdjacency(i, j);
829  if (j != (nn - 1)) {
830  out << ",";
831  }
832  out << " ";
833  }
834  }
835  out << "]";
836  if (i != (nf - 1)) {
837  out << ",";
838  }
839  out << " ";
840  }
841  out << "]" << std::endl;
842  }
843 
844  // interface infos ------------------------------------------------------ //
845  if (m_interfaces.size() > 0) {
846  out << t_s << "interfaces: [ ";
847  for (i = 0; i < nf; ++i) {
848  nn = getInterfaceCount(i);
849  out << "[ ";
850  if (nn == 0) {
851  out << "n.a. ";
852  } else {
853  for (j = 0; j < nn; ++j) {
854  out << getInterface(i, j);
855  if (j != (nn - 1)) {
856  out << ",";
857  }
858  out << " ";
859  }
860  }
861  out << "]";
862  if (i != (nf - 1)) {
863  out << ",";
864  }
865  out << " ";
866  }
867  out << "]" << std::endl;
868  }
869 
870 }
