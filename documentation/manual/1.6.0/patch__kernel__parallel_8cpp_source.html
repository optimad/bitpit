57 void PatchKernel::setCommunicator(MPI_Comm communicator)
58 {
59  // Communication can be set just once
60  if (isCommunicatorSet()) {
61  throw std::runtime_error ("Patch communicator can be set just once");
62  }
63 
64  // The communicator has to be valid
65  if (communicator == MPI_COMM_NULL) {
66  throw std::runtime_error ("Patch communicator is not valid");
67  }
68 
69  // Creat a copy of the user-specified communicator
70  //
71  // No library routine should use MPI_COMM_WORLD as the communicator;
72  // instead, a duplicate of a user-specified communicator should always
73  // be used.
74  MPI_Comm_dup(communicator, &m_communicator);
75 
76  // Get MPI information
77  MPI_Comm_size(m_communicator, &m_nProcessors);
78  MPI_Comm_rank(m_communicator, &m_rank);
79 
80  // Set parallel data for the VTK output
81  if (m_nProcessors > 1) {
82  m_vtk.setParallel(m_nProcessors, m_rank);
83  }
84 }

468 PatchKernel::CellIterator PatchKernel::_addGhost(ElementType type, std::unique_ptr<long[]> &&connectStorage,
469  int rank, long id)
470 {
471  // Create the cell
472  //
473  // If there are internal cells, the ghost cell should be inserted
474  // after the last internal cell.
475  PiercedVector<Cell>::iterator iterator;
476  if (m_lastInternalId < 0) {
477  iterator = m_cells.emreclaim(id, id, type, std::move(connectStorage), false, true);
478  } else {
479  iterator = m_cells.emreclaimAfter(m_lastInternalId, id, id, type, std::move(connectStorage), false, true);
480  }
481  m_nGhosts++;
482 
483  // Update the id of the first ghost cell
484  if (m_firstGhostId < 0) {
485  m_firstGhostId = id;
486  } else if (m_cells.rawIndex(m_firstGhostId) > m_cells.rawIndex(id)) {
487  m_firstGhostId = id;
488  }
489 
490  // Set owner
491  setGhostOwner(id, rank);
492 
493  return iterator;
494 }

817 std::vector<adaption::Info> PatchKernel::partitioningPrepare(const std::unordered_map<long, int> &cellRanks, bool trackPartitioning)
818 {
819  // Patch needs to support partitioning
820  if (!isPartitioningSupported()) {
821  throw std::runtime_error ("The patch does not support partitioning!");
822  }
823 
824  // Communicator has to be set
825  if (!isCommunicatorSet()) {
826  throw std::runtime_error ("There is no communicator set for the patch.");
827  }
828 
829  // Check partitioning status
830  PartitioningStatus partitioningStatus = getPartitioningStatus(true);
831  if (partitioningStatus != PARTITIONING_CLEAN) {
832  throw std::runtime_error ("A partitioning is already in progress.");
833  }
834 
835  // Fill partitioning ranks
836  int patchRank = getRank();
837 
838  std::set<int> recvRanks;
839  m_partitioningOutgoings.clear();
840  for (auto &entry : cellRanks) {
841  int recvRank = entry.second;
842  if (recvRank == patchRank) {
843  continue;
844  }
845 
846  long cellId = entry.first;
847  if (m_ghostOwners.count(cellId) > 0) {
848  continue;
849  }
850 
851  m_partitioningOutgoings.insert(entry);
852  recvRanks.insert(recvRank);
853  }
854 
855  // Identify exchange entries
856  int nRanks = getProcessorCount();
857 
858  int nExchanges = recvRanks.size();
859  std::vector<std::pair<int, int>> exchanges;
860  exchanges.reserve(nExchanges);
861  for (int recvRank : recvRanks) {
862  exchanges.emplace_back(patchRank, recvRank);
863  }
864 
865  int exchangesGatherCount = 2 * nExchanges;
866  std::vector<int> exchangeGatherCount(nRanks);
867  MPI_Allgather(&exchangesGatherCount, 1, MPI_INT, exchangeGatherCount.data(), 1, MPI_INT, m_communicator);
868 
869  std::vector<int> exchangesGatherDispls(nRanks, 0);
870  for (int i = 1; i < nRanks; ++i) {
871  exchangesGatherDispls[i] = exchangesGatherDispls[i - 1] + exchangeGatherCount[i - 1];
872  }
873 
874  int nGlobalExchanges = nExchanges;
875  MPI_Allreduce(MPI_IN_PLACE, &nGlobalExchanges, 1, MPI_INT, MPI_SUM, m_communicator);
876 
877  m_partitioningGlobalExchanges.resize(nGlobalExchanges);
878  MPI_Allgatherv(exchanges.data(), exchangesGatherCount, MPI_INT, m_partitioningGlobalExchanges.data(),
879  exchangeGatherCount.data(), exchangesGatherDispls.data(), MPI_INT, m_communicator);
880 
881  std::sort(m_partitioningGlobalExchanges.begin(), m_partitioningGlobalExchanges.end(), greater<std::pair<int,int>>());
882 
883  // Get global list of ranks that will send data
884  std::unordered_set<int> globalSendRanks;
885  for (const std::pair<int, int> &entry : m_partitioningGlobalExchanges) {
886  globalSendRanks.insert(entry.first);
887  }
888 
889  // Get global list of ranks that will receive data
890  std::unordered_set<int> globalRecvRanks;
891  for (const std::pair<int, int> &entry : m_partitioningGlobalExchanges) {
892  globalRecvRanks.insert(entry.second);
893  }
894 
895  // Identify if this is a serialization or a normal partitioning
896  //
897  // We are serializing the patch if all the processes are sending all
898  // their cells to the same rank.
899  m_partitioningSerialization = (globalRecvRanks.size() == 1);
900  if (m_partitioningSerialization) {
901  int receiverRank = *(globalRecvRanks.begin());
902  if (patchRank != receiverRank) {
903  if (m_partitioningOutgoings.size() != (std::size_t) getInternalCount()) {
904  m_partitioningSerialization = false;
905  }
906  }
907 
908  MPI_Allreduce(MPI_IN_PLACE, &m_partitioningSerialization, 1, MPI_C_BOOL, MPI_LAND, m_communicator);
909  }
910 
911  // Build the information on the cells that will be sent
912  std::vector<adaption::Info> partitioningData;
913  if (trackPartitioning) {
914  for (int recvRank : globalSendRanks) {
915  partitioningData.emplace_back();
916  adaption::Info &partitioningInfo = partitioningData.back();
917  partitioningInfo.entity = adaption::ENTITY_CELL;
918  partitioningInfo.type = adaption::TYPE_PARTITION_SEND;
919  partitioningInfo.rank = recvRank;
920 
921  for (const auto &entry : m_partitioningOutgoings) {
922  int cellRank = entry.second;
923  if (cellRank == recvRank) {
924  continue;
925  }
926 
927  long cellId = entry.first;
928  partitioningInfo.previous.push_back(cellId);
929  }
930  }
931  }
932 
933  // Update the status
934  setPartitioningStatus(PARTITIONING_PREPARED);
935 
936  return partitioningData;
937 }

1090 void PatchKernel::partitioningCleanup()
1091 {
1092  if (!isPartitioningSupported()) {
1093  return;
1094  }
1095 
1096  PartitioningStatus partitioningStatus = getPartitioningStatus();
1097  if (partitioningStatus == PARTITIONING_CLEAN) {
1098  return;
1099  } else if (partitioningStatus == PARTITIONING_PREPARED) {

1206 {

1208  return 0.;
1209  }
1210

1212  double partitionWeight;
1213  if (!cellWeights.empty()) {
1214  partitionWeight = 0.;
1215  for (auto cellItr = internalConstBegin(); cellItr != internalConstEnd(); ++cellItr) {
1216  long cellId = cellItr.getId();
1217 
1218  double cellWeight;
1219  auto weightItr = cellWeights.find(cellId);
1220  if (weightItr != cellWeights.end()) {
1221  cellWeight = weightItr->second;
1222  } else {
1223  cellWeight = DEFAULT_PARTITIONING_WEIGTH;
1224  }
1225 
1226  partitionWeight += cellWeight;
1227  }
1228  } else {
1229  partitionWeight = DEFAULT_PARTITIONING_WEIGTH * getInternalCount();
1230  }
1231

1233  double totalPartitionWeight;
1234  MPI_Allreduce(&partitionWeight, &totalPartitionWeight, 1, MPI_DOUBLE, MPI_SUM, getCommunicator());
1235

1284  //

1286  //

1300  int patchRank = getRank();
1301

1303  m_partitioningVerticesTag = communications::tags().generate(m_communicator);
1304

1306  std::unordered_set<int> batchRecvRanks;

1374 {

1378  //
1379  // The communications will exchange the ranks on ghosts cells.

1381  size_t notificationDataSize = sizeof(patchRank);
1382

1389  notificationCommunicator.startRecv(rank);
1390  }
1391

1393  //
1394  // Data buffer is filled with the ranks that will own source cells after
1395  // the partitioning.

1397  const int rank = entry.first;
1398  auto &sourceCells = entry.second;

1405  finalOwner = m_partitioningOutgoings.at(id);
1406  } else {
1407  finalOwner = patchRank;
1408  }
1409 
1410  buffer << finalOwner;
1411  }

1420

1425

1430 
1431  finalGhostOwners[id] = finalOwner;
1432  }
1433 
1434  ++nCompletedRecvs;
1435  }
1436

1500

1502  // Send data to the receivers
1503  //

1540  //
1541  outgoingCells.clear();
1542  for (const auto &rankEntry : m_partitioningOutgoings) {
1543  int rank = rankEntry.second;
1544  if (rank != recvRank) {

1553  // Identify frame and halo
1554  //
1555  // Along with the cells explicitly marked for sending, we need to send
1556  // also a halo of surrounding cells. Those cells will be used by the
1557  // receiver to connect the cells it receives to the existing cells and
1558  // to generate the ghost cells.
1559  //
1560  // The faces that tell apart cells explicitly marked for sending from
1561  // halo cells are called "frontier faces". We can identify frontier
1562  // faces looping through the outgoing cells an looking for faces with
1563  // a neighbour not explicitly marked for sending.
1564  //
1565  // Frame is made up by cells explicitly marked for sending that have
1566  // a face, an edge or a vertex on a frontier face.
1567  //
1568  // Halo is made up by cells not explicitly marked for sending that

1572  // be ghosts for the current rank. These are all non-outgonig cells
1573  // that have a face/edge/vertex on the inner frontier. A face/edge/
1574  // vertex is on the inner frontier if its on the frontier and one of
1575  // the cells that contain that item is not an outgoing cell.
1576  //

1578  //
1579  // NOTE: for three-dimensional unstructured non-conformal meshes we
1580  // need to explicitly search cells that have an edge on the frontier,
1581  // searching for cells that have a vertex on the frontier is not
1582  // enough (for unstructured meshes we may have a cell that touches the
1583  // frontier only through an edge).

1629  //
