229 void VolCartesian::buildInterfaces()
230 {
231  // Reset interfaces
232  if (getInterfacesBuildStrategy() != INTERFACES_NONE) {
233  clearInterfaces();
234  }
235 
236  // Count the total number of interfaces
237  m_nInterfaces = 0;
238  for (int n = 0; n < getDimension(); n++) {
239  std::array<int, 3> interfaceCount1D = getInterfaceCountDirection(n);
240 
241  int nDirectionInterfaces = 1;
242  for (int n = 0; n < getDimension(); n++) {
243  nDirectionInterfaces *= interfaceCount1D[n];
244  }
245  m_nInterfaces += nDirectionInterfaces;
246  }
247 
248  // Build interfaces
249  if (getMemoryMode() == MemoryMode::MEMORY_NORMAL) {
250  // Enable advanced editing
251  setExpert(true);
252 
253  // Update interfaces
254  addInterfaces();
255 
256  // Disable advanced editing
257  setExpert(false);
258  } else {
259  // Set interfaces build strategy
260  setInterfacesBuildStrategy(INTERFACES_AUTOMATIC);
261  }
262 }

383 void VolCartesian::setDiscretization(const std::array<int, 3> &nCells)
384 {
385  // Spacing
386  for (int n = 0; n < getDimension(); ++n) {
387  // Initialize cells
388  m_nCells1D[n] = nCells[n];
389  m_cellSpacings[n] = (m_maxCoords[n] - m_minCoords[n]) / m_nCells1D[n];
390 
391  m_cellCenters[n].resize(m_nCells1D[n]);
392  for (int i = 0; i < m_nCells1D[n]; i++) {
393  m_cellCenters[n][i] = m_minCoords[n] + (0.5 + i) * m_cellSpacings[n];
394  }
395 
396  log::cout() << " - Cell count along direction " << n << " : " << m_nCells1D[n] << "\n";
397 
398  // Initialize vertices
399  m_nVertices1D[n] = m_nCells1D[n] + 1;
400 
401  m_vertexCoords[n].resize(m_nVertices1D[n]);
402  for (int i = 0; i < m_nVertices1D[n]; i++) {
403  m_vertexCoords[n][i] = m_minCoords[n] + i * m_cellSpacings[n];
404  }
405  }
406 
407  if (!isThreeDimensional()) {
408  m_nCells1D[Vertex::COORD_Z] = 0;
409  m_cellSpacings[Vertex::COORD_Z] = 0.;
410 
411  m_nVertices1D[Vertex::COORD_Z] = 0;
412  }
413 
414  log::cout() << std::endl;
415 
416  // Count the total number of vertices
417  m_nVertices = 1;
418  for (int n = 0; n < getDimension(); ++n) {
419  m_nVertices *= m_nVertices1D[n];
420  }
421  log::cout() << " - Total vertex count: " << m_nVertices << "\n";
422 
423  // Count the total number of cells
424  m_nCells = 1;
425  for (int n = 0; n < getDimension(); ++n) {
426  m_nCells *= m_nCells1D[n];
427  }
428  log::cout() << " - Total cell count: " << m_nCells << "\n";
429 
430  // Cell volume
431  initializeCellVolume();
432 
433  // Interface area
434  initializeInterfaceArea();
435 
436  // Update patch data structures
437  if (getMemoryMode() == MemoryMode::MEMORY_NORMAL) {
438  // Switch to light mode to reset patchdata structures
439  switchMemoryMode(MemoryMode::MEMORY_LIGHT);
440 
441  // Swich back to normal mode to rebuild patchdata structures
442  switchMemoryMode(MemoryMode::MEMORY_NORMAL);
443  }
444 }

600 void VolCartesian::switchMemoryMode(MemoryMode mode)
601 {
602  if (mode == getMemoryMode()) {
603  return;
604  }
605 
606  // Update the data structures
607  switch (mode) {
608 
609  case MemoryMode::MEMORY_NORMAL:
610  // Spawn the patch to activate normal memory mode
611  spawn(false);
612 
613  break;
614 
615  case MemoryMode::MEMORY_LIGHT:
616  // To put the patch in memory mode we need to reset the generic data
617  // of the patch, therefore we can call the 'reset' implementation of
618  // the kernel.
619  VolumeKernel::reset();
620 
621  // Now the patch needs to be spawn
622  setSpawnStatus(SPAWN_NEEDED);
623 
624  // Set the light memory mode
625  setMemoryMode(mode);
626 
627  break;
628 
629  }
630 }

677 std::vector<adaption::Info> VolCartesian::_spawn(bool trackSpawn)
678 {
679  std::vector<adaption::Info> updateInfo;
680 
681  // If the patch is in 'normal' mode there is nothing to do.
682  if (getMemoryMode() == MEMORY_NORMAL) {
683  return updateInfo;
684  }
685 
686  // Enable advanced editing
687  setExpert(true);
688 
689  // Definition of the mesh
690  addVertices();
691  addCells();
692  if (getInterfacesBuildStrategy() == INTERFACES_AUTOMATIC) {
693  addInterfaces();
694  }
695 
696  // Disable advanced editing
697  setExpert(false);
698 
699  // Adaption info
700  if (trackSpawn) {
701  updateInfo.emplace_back();
702  adaption::Info &adaptionCellInfo = updateInfo.back();
703  adaptionCellInfo.type = adaption::TYPE_CREATION;
704  adaptionCellInfo.entity = adaption::ENTITY_CELL;
705  adaptionCellInfo.current.reserve(m_cells.size());
706  for (auto &cell : m_cells) {
707  adaptionCellInfo.current.emplace_back();
708  long &cellId = adaptionCellInfo.current.back();
709  cellId = cell.getId();
710  }
711 
712  updateInfo.emplace_back();
713  adaption::Info &adaptionInterfaceInfo = updateInfo.back();
714  adaptionInterfaceInfo.type = adaption::TYPE_CREATION;
715  adaptionInterfaceInfo.entity = adaption::ENTITY_INTERFACE;
716  adaptionInterfaceInfo.current.reserve(m_interfaces.size());
717  for (auto &interface : m_interfaces) {
718  adaptionInterfaceInfo.current.emplace_back();
719  long &interfaceId = adaptionInterfaceInfo.current.back();
720  interfaceId = interface.getId();
721  }
722  } else {
723  updateInfo.emplace_back();
724  }
725 
726  // Updating the adaption brings the patch is in normal memory mode
727  setMemoryMode(MemoryMode::MEMORY_NORMAL);
728 
729  // Done
730  return updateInfo;
731 }

860 void VolCartesian::addInterfacesDirection(int direction)
861 {
862  log::cout() << " >> Creating interfaces normal to direction " << direction << "\n";
863 
864  // Info on the interfaces
865  ElementType interfaceType = getInterfaceType();
866 
867  const ReferenceElementInfo &interfaceTypeInfo = ReferenceElementInfo::getInfo(interfaceType);
868  const int nInterfaceVertices = interfaceTypeInfo.nVertices;
869  std::array<int, 3> interfaceCount1D = getInterfaceCountDirection(direction);
870 
871  // Counters
872  std::array<int, 3> counters = {{0, 0, 0}};
873  int &i = counters[Vertex::COORD_X];
874  int &j = counters[Vertex::COORD_Y];
875  int &k = counters[Vertex::COORD_Z];
876 
877  // Creation of the interfaces
878  for (k = 0; (isThreeDimensional()) ? (k < interfaceCount1D[Vertex::COORD_Z]) : (k <= 0); k++) {
879  for (j = 0; j < interfaceCount1D[Vertex::COORD_Y]; j++) {
880  for (i = 0; i < interfaceCount1D[Vertex::COORD_X]; i++) {
881  InterfaceIterator interfaceIterator = VolumeKernel::addInterface(interfaceType);
882  Interface &interface = *interfaceIterator;
883 
884  // Owner id
885  std::array<int, 3> ownerIJK(counters);
886  if (counters[direction] == (interfaceCount1D[direction] - 1)) {
887  ownerIJK[direction] -= 1;
888  }
889 
890  long ownerId = getCellLinearId(ownerIJK);
891 
892  // Neighbour id
893  long neighId;
894  if (counters[direction] != 0 && counters[direction] != interfaceCount1D[direction] - 1) {
895  std::array<int, 3> neighIJK(counters);
896  neighIJK[direction] -= 1;
897 
898  neighId = getCellLinearId(neighIJK);
899  } else {
900  neighId = Element::NULL_ID;
901  }
902 
903  // Owner data
904  Cell &owner = m_cells[ownerId];
905 
906  int ownerFace = 2 * direction;
907  if (counters[direction] == interfaceCount1D[direction] - 1) {
908  ownerFace++;
909  }
910 
911  interface.setOwner(owner.getId(), ownerFace);
912  owner.pushInterface(ownerFace, interface.getId());
913  owner.pushAdjacency(ownerFace, neighId);
914 
915  // Neighbour data
916  if (counters[direction] != 0 && counters[direction] != interfaceCount1D[direction] - 1) {
917  Cell &neigh = m_cells[neighId];
918
