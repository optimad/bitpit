104 std::vector<adaption::Info> VolOctree::_partitioningPrepare(const std::unordered_map<long, double> &cellWeights, double defaultWeight, bool trackPartitioning)
105 {
106  // Eval partitioning weights
107  computePartitioningOctantWeights(cellWeights, defaultWeight);
108 
109  // Generate partitioning information
110  std::vector<adaption::Info> partitioningData;
111  if (trackPartitioning) {
112  int currentRank = getRank();
113  PabloUniform::LoadBalanceRanges loadBalanceRanges = m_tree->evalLoadBalanceRanges(m_partitioningOctantWeights.get());
114  for (const auto &entry : loadBalanceRanges.sendRanges) {
115  int receiver = entry.first;
116  if (receiver == currentRank) {
117  continue;
118  }
119 
120  adaption::Type adaptionType;
121  if (loadBalanceRanges.sendAction == PabloUniform::LoadBalanceRanges::ACTION_DELETE) {
122  adaptionType = adaption::TYPE_DELETION;
123  } else {
124  adaptionType = adaption::TYPE_PARTITION_SEND;
125  }
126 
127  partitioningData.emplace_back();
128  adaption::Info &partitioningInfo = partitioningData.back();
129  partitioningInfo.entity = adaption::ENTITY_CELL;
130  partitioningInfo.type = adaptionType;
131  partitioningInfo.rank = receiver;
132 
133  const std::array<uint32_t, 2> &sendRange = entry.second;
134  uint32_t beginTreeId = sendRange[0];
135  uint32_t endTreeId = sendRange[1];
136  partitioningInfo.previous.reserve(endTreeId - beginTreeId);
137  for (uint32_t treeId = beginTreeId; treeId < endTreeId; ++treeId) {
138  OctantInfo octantInfo(treeId, true);
139  partitioningInfo.previous.emplace_back();
140  long &cellId = partitioningInfo.previous.back();
141  cellId = getOctantId(octantInfo);
142  }
143  }
144  }
145 
146  return partitioningData;
147 }

213 void VolOctree::computePartitioningOctantWeights(const std::unordered_map<long, double> &cellWeights, double defaultWeight)
214 {
215  bool useWeights = !cellWeights.empty();
216  MPI_Allreduce(MPI_IN_PLACE, &useWeights, 1, MPI_C_BOOL, MPI_LOR, getCommunicator());
217  if (!useWeights) {
218  clearPartitioningOctantWeights();
219  return;
220  }
221 
222  std::size_t nOctants = m_tree->getNumOctants();
223  m_partitioningOctantWeights = std::unique_ptr<std::vector<double>>(new std::vector<double>(nOctants, defaultWeight));
224  for (auto cellItr = internalConstBegin(); cellItr != internalConstEnd(); ++cellItr) {
225  long cellId = cellItr.getId();
226  auto weightItr = cellWeights.find(cellId);
227  if (weightItr != cellWeights.end()) {
228  OctantInfo octantInfo = getCellOctant(cellId);
229  uint32_t treeId = octantInfo.id;
230  (*m_partitioningOctantWeights)[treeId] = weightItr->second;
231  }
232  }
233 }
