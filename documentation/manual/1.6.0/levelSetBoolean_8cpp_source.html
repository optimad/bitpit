263 /*
264  * Determines the relevant object which determines the levelset value in the cell
265  * Taken from http://www.iue.tuwien.ac.at/phd/ertl/node57.html
266  * @param[in] id cell index
267  * @param[in,out] factor if is not null, on output it will contain the multiplier
268  * of the levelset function of the competent primary object according the boolean
269  * operation. In particular, for substractions the relevant levelset could be
270  * -1.0*levelset_primary_object.
271  * @return pointer to competent LevelSetObject
272  */
273 LevelSetObject* LevelSetBoolean::getCompetentObject( long id, double *factor) const{
274 
275  if(m_objPtr.size()==0){
276  return nullptr;
277  }
278 
279  double result, second;
280  LevelSetObject *resPtr, *secPtr;
281 
282  result = m_objPtr[0]->getLS(id);
283  resPtr = m_objPtr[0];
284 
285  for( size_t n=1; n<m_objPtr.size(); ++n){
286  second = m_objPtr[n]->getLS(id) ;
287  secPtr = m_objPtr[n];
288 
289  if( getBooleanOperation() == LevelSetBooleanOperation::UNION){
290  if(result>second) {
291  result = second;
292  resPtr = secPtr;
293  }
294 
295  } else if ( getBooleanOperation() == LevelSetBooleanOperation::INTERSECTION){
296  if(result<second) {
297  result = second;
298  resPtr = secPtr;
299  }
300 
301  } else if ( getBooleanOperation() == LevelSetBooleanOperation::SUBTRACTION){
302  if(result<-1.*second) {
303  result = -1.*second;
304  resPtr = secPtr;
305  }
306  }
307  }
308 
309  if(factor){
310  *factor = (utils::DoubleFloatingEqual()(resPtr->getLS(id),result)) ? 1. : -1.;
311  }
312 
313  return resPtr;
314 }

343 LevelSetInfo LevelSetBoolean::booleanOperation(long id) const{
344 
345  if(m_objPtr.empty()){
346  return LevelSetInfo();
347  }
348 
349  LevelSetInfo result = m_objPtr[0]->getLevelSetInfo(id);
350  for( size_t n=1; n<m_objPtr.size(); ++n){
351  double value = m_objPtr[n]->getLS(id) ;
352 
353  if( getBooleanOperation() == LevelSetBooleanOperation::UNION){
354  if(result.value>value) {
355  result = m_objPtr[n]->getLevelSetInfo(id);
356  }
357 
358  } else if ( getBooleanOperation() == LevelSetBooleanOperation::INTERSECTION){
359  if(result.value<value) {
360  result = m_objPtr[n]->getLevelSetInfo(id);
361  }
362 
363  } else if ( getBooleanOperation() == LevelSetBooleanOperation::SUBTRACTION){
364  if(result.value<-value) {
365  result = m_objPtr[n]->getLevelSetInfo(id);
366  result.value *= -1.;
367  result.gradient *= -1.;
368  }
369  }
370  }
371 
372  return result;
373 }

381 LevelSetInfo LevelSetBoolean::booleanOperation(const std::array<double,3> &coords) const{
382 
383  if(m_objPtr.empty()){
384  return LevelSetInfo();
385  }
386 
387  LevelSetInfo result = m_objPtr[0]->computeLevelSetInfo(coords);
388  for( size_t n=1; n<m_objPtr.size(); ++n){
389  LevelSetInfo second = m_objPtr[n]->computeLevelSetInfo(coords);
390  double value = second.value;
391 
392  if( getBooleanOperation() == LevelSetBooleanOperation::UNION){
393  if(result.value>value) {
394  result = second;
395  }
396 
397  } else if ( getBooleanOperation() == LevelSetBooleanOperation::INTERSECTION){
398  if(result.value<value) {
399  result = second;
400  }
401 
402  } else if ( getBooleanOperation() == LevelSetBooleanOperation::SUBTRACTION){
403  if(result.value<-value) {
404  result = second;
405  result.value *= -1.;
406  result.gradient *= -1.;
407  }
408  }
409  }
