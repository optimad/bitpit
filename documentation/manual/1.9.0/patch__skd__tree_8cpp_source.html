70void SkdPatchInfo::buildCache(const PatchKernel::CellConstRange &cellRange)
71{
72 m_cellBoxes = std::unique_ptr<BoxCache>(new BoxCache(1, &(m_patch->getCells())));
73 for (auto itr = cellRange.cbegin(); itr != cellRange.cend(); ++itr) {
74 std::size_t rawCellId = itr.getRawIndex();
75
76 // Bounding box
77 std::array<std::array<double, 3>, 2> *cellBox = m_cellBoxes->rawData(rawCellId);
78 m_patch->evalElementBoundingBox(*itr, cellBox->data(), cellBox->data() + 1);
79 }
80}

383bool SkdBox::boxContainsPoint(const std::array<double, 3> &point, double offset) const
384{
385 if (isEmpty()) {
386 return false;
387 }
388
389 for (int d = 0; d < 3; d++) {
390 if (point[d] < (m_boxMin[d] - offset)) {
391 return false;
392 }
393
394 if (point[d] > (m_boxMax[d] + offset)) {
395 return false;
396 }
397 }
398
399 return true;
400}

410bool SkdBox::boxIntersectsSphere(const std::array<double, 3> &center, double radius) const
411{
412 // If the box contains the center it will also intersect the sphere
413 if (boxContainsPoint(center, 0.)) {
414 return true;
415 }
416
417 // Check if the distance between the center and its projection on
418 // the bounding box is smaller than the radius of the sphere.
419 std::array<double, 3> delta;
420 for (int d = 0; d < 3; d++) {
421 delta[d] = std::min(std::max(center[d], m_boxMin[d]), m_boxMax[d]) - center[d];
422 }
423 double distance = dotProduct(delta, delta);
424
425 return (distance < radius * radius);
426}

469void SkdNode::initializeBoundingBox()
470{
471 // Early return if there are no cells
472 if (m_cellRangeBegin == m_cellRangeEnd){
473 m_boxMin.fill( std::numeric_limits<double>::max());
474 m_boxMax.fill(- std::numeric_limits<double>::max());
475
476 return;
477 }
478
479 // Evaluate the bounding box
480 const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
481
482 const std::array<std::array<double, 3>, 2> &firstCellBox = m_patchInfo->getCachedBox(cellRawIds[m_cellRangeBegin]);
483 m_boxMin = firstCellBox[0];
484 m_boxMax = firstCellBox[1];
485 for (std::size_t n = m_cellRangeBegin + 1; n < m_cellRangeEnd; n++) {
486 const std::array<std::array<double, 3>, 2> &cellBox = m_patchInfo->getCachedBox(cellRawIds[n]);
487 for (int d = 0; d < 3; ++d) {
488 m_boxMin[d] = std::min(cellBox[0][d], m_boxMin[d]);
489 m_boxMax[d] = std::max(cellBox[1][d], m_boxMax[d]);
490 }
491 }
492
493 // Inlfate the bounding box by a small epsilon
494 //
495 // The small inflation allows us to test intersection of a query
496 // against a safety box and be assured that non-intersection of
497 // with the box implies non-intersection with all the contents
498 // in the box, even when computations are susceptible to round-off
499 // error.
500 double tolerance = m_patchInfo->getPatch().getTol();
501
502 m_boxMin -= tolerance;
503 m_boxMax += tolerance;
504}

709void SkdNode::updatePointClosestCell(const std::array<double, 3> &point, bool interiorCellsOnly,
710 long *closestId, double *closestDistance) const
711{
712 if (getCellCount() == 0) {
713 return;
714 }
715
716 const PatchKernel &patch = m_patchInfo->getPatch();
717 const PiercedVector<Cell> &cells = patch.getCells();
718 const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
719
720 for (std::size_t n = m_cellRangeBegin; n < m_cellRangeEnd; n++) {
721 std::size_t cellRawId = cellRawIds[n];
722 const Cell &cell = cells.rawAt(cellRawId);
723 if (interiorCellsOnly && !cell.isInterior()) {
724 continue;
725 }
726
727
728 updatePointClosestCell(point, cell, closestId, closestDistance);
729 }
730}

748void SkdNode::updatePointClosestCell(const std::array<double, 3> &point, const Cell &cell,
749 long *closestId, double *closestDistance) const
750{
751 const PatchKernel &patch = m_patchInfo->getPatch();
752
753 // Cell id
754 long cellId = cell.getId();
755
756 // Project the point on the cell
757 int nCellVertices = cell.getVertexCount();
758 BITPIT_CREATE_WORKSPACE(cellVertexCoordinates, std::array<double BITPIT_COMMA 3>, nCellVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
759 patch.getElementVertexCoordinates(cell, cellVertexCoordinates);
760
761 double cellDistance;
762 std::array<double, 3> cellProjection;
763 cell.evalPointProjection(point, cellVertexCoordinates, &cellProjection, &cellDistance);
764
765 // Detect if the specified cell is closer than the current closest cell
766 const int DISTANCE_CLOSER = - 1;
767 const int DISTANCE_EQUAL = 0;
768 const int DISTANCE_FARTHER = 1;
769
770 int distanceFlag = DISTANCE_FARTHER;
771 if (utils::DoubleFloatingEqual()(cellDistance, *closestDistance, patch.getTol(), patch.getTol())) {
772 distanceFlag = DISTANCE_EQUAL;
773 } else if (cellDistance < *closestDistance) {
774 distanceFlag = DISTANCE_CLOSER;
775 }
776
777 // Consider the case where no closest cell is defined
778 //
779 // Even if the id of the closest cell is null, we may have an
780 // estimated of the closest cell distance. We need to update
781 // the closest cell information only if the current cells is
782 // closer than the estimate.
783 if (*closestId == Cell::NULL_ID) {
784 if (distanceFlag != DISTANCE_FARTHER) {
785 *closestId = cellId;
786 *closestDistance = cellDistance;
787 }
788
789 return;
790 }
791
792 // Update closest cell information accordingly to the distance flag:
793 // - if the specified cell is closer the specified cell will become the
794 // closest cell
795 // - if the two cells have the same distance, the closest cell will be
796 // chosen using the normal of the cells.
797 // - if the specified cell is farther there is nothing to do;
798 switch (distanceFlag) {
799
800 case DISTANCE_CLOSER:
801 {
802 *closestId = cellId;
803 *closestDistance = cellDistance;
804
805 break;
806 }
807
808 case DISTANCE_EQUAL:
809 {
810 // Project point ont the closest cell
811 const Cell &closest = patch.getCell(*closestId);
812
813 int nClosestCellVertices = closest.getVertexCount();
814 BITPIT_CREATE_WORKSPACE(closestCellVertexCoordinates, std::array<double BITPIT_COMMA 3>, nClosestCellVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
815 patch.getElementVertexCoordinates(closest, closestCellVertexCoordinates);
816
817 double closestCellDistance;
818 std::array<double, 3> closestCellProjection;
819 closest.evalPointProjection(point, closestCellVertexCoordinates, &closestCellProjection, &closestCellDistance);
820
821 // Find normal of the cells at the centroids
822 //
823 // To be more precise, the normals should beevaluated on projection
824 // points.
825 std::array<double, 3> cellNormal = cell.evalNormal(cellVertexCoordinates);
826 std::array<double, 3> closestNormal = closest.evalNormal(closestCellVertexCoordinates);
827
828 // Find cell more aligned with respect to the point
829 //
830 // We want to chhose the cell whose normal is most aligned towards
831 // the point.
832 double cellAligement = dotProduct(cellNormal, point - cellProjection);
833 double closestAligement = dotProduct(closestNormal, point - closestCellProjection);
834
835 // If the specified cell is more aligned than the closest cell, update
836 // the closest info.
837 if (cellAligement > closestAligement) {
838 *closestId = cellId;
839 *closestDistance = cellDistance;
840 }
841
842 break;
843 }
844
845 default:
846 {
847 // Nothing to do
848 }
849
850 }
851}

909void SkdGlobalCellDistance::executeMPIMinOperation(SkdGlobalCellDistance *in, SkdGlobalCellDistance *inout, int *len, MPI_Datatype *datatype)
910{
911 BITPIT_UNUSED(datatype);
912
913 int nDistances = *len;
914 for (int i = 0; i < nDistances; ++i) {
915 bool updateDistance = false;
916 if (in[i].m_id != Cell::NULL_ID) {
917 if (inout[i].m_id == Cell::NULL_ID) {
918 updateDistance = true;
919 } else if (std::abs(in[i].m_distance) < std::abs(inout[i].m_distance)) {
920 updateDistance = true;
921 }
922 }
923
924 if (updateDistance) {
925 inout[i] = in[i];
926 }
927 }

1088void PatchSkdTree::build(std::size_t leafThreshold, bool squeezeStorage)
1089{
1090 const PatchKernel &patch = m_patchInfo.getPatch();
1091
1092 // Clear existing tree
1093 clear();
1094
1095 // Initialize list of cell raw ids
1096 std::size_t nCells;
1097 PatchKernel::CellConstRange cellRange;
1098 if (m_interiorCellsOnly) {
1099 nCells = patch.getInternalCellCount();
1100 cellRange.initialize(patch.internalCellConstBegin(), patch.internalCellConstEnd());
1101 } else {
1102 nCells = patch.getCellCount();
1103 cellRange.initialize(patch.cellConstBegin(), patch.cellConstEnd());
1104 }
1105
1106 if (m_cellRawIds.size() != nCells) {
1107 m_cellRawIds.resize(nCells);
1108 }
1109
1110 std::size_t k = 0;
1111 for (auto itr = cellRange.begin(); itr != cellRange.end(); ++itr) {
1112 m_cellRawIds[k++] = itr.getRawIndex();
1113 }
1114
1115 // Build patch cache
1116 m_patchInfo.buildCache(cellRange);
1117
1118 // Initialize node list
1119 std::size_t nodesCount = std::max(1, int(std::ceil(2. * nCells / leafThreshold - 1.)));
1120 m_nodes.reserve(nodesCount);
1121
1122 // Create the root
1123 m_nodes.emplace_back(&m_patchInfo, 0, nCells);
1124
1125 // Create the tree
1126 std::queue<std::size_t> nodeStack;
1127 nodeStack.push(0);
1128 while (!nodeStack.empty()) {
1129 std::size_t nodeId = nodeStack.front();
1130 nodeStack.pop();
1131
1132 // Create the children of the node
1133 //
1134 // This function may add new nodes to the tree, this may invalidate
1135 // any reference or pointer to the nodes. To avoid potential problems
1136 // we pass to the function a node id.
1137 createChildren(nodeId, leafThreshold);
1138
1139 // Add the newly created childrend to the stack
1140 const SkdNode &node = getNode(nodeId);
1141 for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
1142 std::size_t childId = node.getChildId(static_cast<SkdNode::ChildLocation>(i));
1143 if (childId != SkdNode::NULL_ID) {
1144 nodeStack.push(childId);
1145 }
1146 }
1147 }
1148
1149 // Squeeze storage
1150 if (squeezeStorage) {
1151 m_nodes.shrink_to_fit();
1152 }
1153
1154 // Patch cache is no longer needed
1155 m_patchInfo.destroyCache();
1156
1157#if BITPIT_ENABLE_MPI
1158 // Set partition information
1159 if (patch.isPartitioned()){
1160 // Set communicator
1161 setCommunicator(patch.getCommunicator());
1162
1163 // Build partition info with partition boxes if the patch is partitioned
1164 buildPartitionBoxes();
1165 } else {

1167 m_nProcessors = 1;

1285void PatchSkdTree::createChildren(std::size_t parentId, std::size_t leafThreshold)
1286{
1287 const SkdNode &parent = getNode(parentId);
1288
1289 // Check if the parent is a leaf
1290 std::size_t parentCellCount = parent.getCellCount();
1291 if (parentCellCount <= leafThreshold) {
1292 createLeaf(parentId);
1293 return;
1294 }
1295
1296 // Evaluate the preferred direction along which elements will be split.
1297 //
1298 // The elements will be split along a plane normal to the direction
1299 // for which the bounding box has the maximum length.
1300 const std::array<double, 3> &parentBoxMin = parent.getBoxMin();
1301 const std::array<double, 3> &parentBoxMax = parent.getBoxMax();
1302
1303 int largerDirection = 0;
1304 double boxMaximumLength = parentBoxMax[largerDirection] - parentBoxMin[largerDirection];
1305 for (int d = 1; d < 3; ++d) {
1306 double length = parentBoxMax[d] - parentBoxMin[d];
1307 if (length > boxMaximumLength) {
1308 largerDirection = d;
1309 boxMaximumLength = length;
1310 }
1311 }
1312
1313 // Split the elements.
1314 //
1315 // Preferred direction is tried first, if the split along this direction
1316 // will produce and empty child (e.g., all the cell's box centroids have
1317 // the same coordinate along the split direction and therefore all the
1318 // cells would be assigned to the left chiled) the split will be performed
1319 // along one of the other directions.
1320 for (int d = 0; d < 3; ++d) {
1321 // Update the split direction
1322 int splitDirection = (largerDirection + d) % 3;
1323
1324 // Get the threshold for the split
1325 double splitThreshold = parent.evalBoxWeightedMean(splitDirection);
1326
1327 // Order the elements
1328 //
1329 // All the elements with a centroid coordinate less or equal than the
1330 // threshold will be assigned to the left child, the others will be
1331 // assigned to the right child.
1332 std::size_t leftBegin = parent.m_cellRangeBegin;
1333 std::size_t leftEnd = parent.m_cellRangeEnd;
1334 std::size_t rightBegin = parent.m_cellRangeBegin;
1335 std::size_t rightEnd = parent.m_cellRangeEnd;
1336 while (true) {
1337 // Update the right begin
1338 while (rightBegin != leftEnd && m_patchInfo.evalCachedBoxMean(m_cellRawIds[rightBegin], splitDirection) <= splitThreshold) {
1339 rightBegin++;
1340 }
1341
1342 // Update the left end
1343 while (rightBegin != leftEnd && m_patchInfo.evalCachedBoxMean(m_cellRawIds[leftEnd - 1], splitDirection) > splitThreshold) {
1344 leftEnd--;
1345 }
1346
1347 // If all the elements are in the right position we can exit
1348 if (rightBegin == leftEnd) {
1349 break;
1350 }
1351
1352 // If left end and and right begin are not equal, that the two ids
1353 // point to misplaced elements. Swap the elements, advance the ids
1354 // and continue iterating.
1355 std::iter_swap(m_cellRawIds.begin() + (leftEnd - 1), m_cellRawIds.begin() + rightBegin);
1356 }
1357
1358 if (leftEnd <= leftBegin || rightEnd <= rightBegin) {
1359 continue;
1360 }
1361
1362 // Create the left child
1363 //
1364 // Adding new nodes may invalidate any pointer and reference to the
1365 // nodes, we cannot store a reference to the parent node.
1366 long leftId = m_nodes.size();
1367 m_nodes.emplace_back(&m_patchInfo, leftBegin, leftEnd);
1368 _getNode(parentId).m_children[static_cast<std::size_t>(SkdNode::CHILD_LEFT)] = leftId;
1369
1370 // Create the right child
1371 //
1372 // Adding new nodes may invalidate any pointer and reference to the
1373 // nodes, we cannot store a reference to the parent node.
1374 long rightId = m_nodes.size();
1375 m_nodes.emplace_back(&m_patchInfo, rightBegin, rightEnd);
1376 _getNode(parentId).m_children[static_cast<std::size_t>(SkdNode::CHILD_RIGHT)] = rightId;
1377
1378 // Done
1379 return;
1380 }
1381
1382 // It was not possible to split the elements. They have all the same
1383 // characteristic position and therefore they will be clustered together
1384 // in a leaf node.
1385 createLeaf(parentId);
1386}

1499void PatchSkdTree::buildPartitionBoxes()
1500{
1501 // Recover local bounding box of the root node
1502 const SkdBox &box = getNode(0).getBoundingBox();
1503
1504 // Collect minimum and maximum coordinates of bounding boxes
1505 std::vector<int> count(m_nProcessors, 3);
1506
1507 std::vector<int> displacements(m_nProcessors);
1508 for (int i = 0; i < m_nProcessors; ++i) {
1509 displacements[i] = i*3;

1513 minBoxes[m_rank] = box.getBoxMin();

1515 count.data(), displacements.data(), MPI_DOUBLE, m_communicator);

1518 maxBoxes[m_rank] = box.getBoxMax();

1520 count.data(), displacements.data(), MPI_DOUBLE, m_communicator);
1521
1522 // Fill partition boxes
1523 m_partitionBoxes.resize(getPatch().getProcessorCount());
1524 for (int i = 0; i < m_nProcessors; ++i) {
1525 m_partitionBoxes[i] = SkdBox(minBoxes[i], maxBoxes[i]);
