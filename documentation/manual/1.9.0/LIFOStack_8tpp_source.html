315// ========================================================================== //

317// ========================================================================== //
318
319// Extract last element from the stack list --------------------------------- //
320item = STACK[TOPSTK - 1];
321TOPSTK--;
322
323// Resize stack list -------------------------------------------------------- //
324if (TOPSTK <= (long) STACK.size() - MAXSTK) {
325 decreaseSTACK();
