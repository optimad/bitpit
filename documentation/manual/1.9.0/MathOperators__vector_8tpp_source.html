917 const std::vector< T > &x,

919) {

924// VARIABLES DECLARATION //
925// ================================================================================== //
926

931
932// Counters
933int i, j;
934
935// ================================================================================== //
936// COMPUTE THE P-NORM //
937// ================================================================================== //
938if (p == 1) { return(norm1(x)); }
939if (p == 2) { return(norm2(x)); }
940
941if (n > 0) {
942 for (i = 0; i < n; i++) {
943 y = 1.0;
944 t = x[i];
945 for (j = 1; j <= p; j++) {
946 y = y*t;
947 } //next j
