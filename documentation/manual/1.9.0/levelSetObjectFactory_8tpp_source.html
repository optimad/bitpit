98template<typename kernel_t, template<typename> class narrow_band_cache_t>
99std::unique_ptr<LevelSetObject> LevelSetObjectFactory::_createImmutableObject(const kernel_t *kernel, LevelSetStorageType storageType, LevelSetObject *source)
100{
101 BITPIT_UNUSED(kernel);
102
103 switch (storageType) {
104
105 case LevelSetStorageType::SPARSE:
106 if (LevelSetCachedObject<narrow_band_cache_t<typename kernel_t::SparseStorageManager>> *cachedSource = dynamic_cast<LevelSetCachedObject<narrow_band_cache_t<typename kernel_t::SparseStorageManager>> *>(source)) {
107 return std::unique_ptr<LevelSetObject>(new LevelSetImmutableObject<narrow_band_cache_t<typename kernel_t::SparseStorageManager>>(cachedSource));
108 } else {
109 return std::unique_ptr<LevelSetObject>(new LevelSetImmutableObject<narrow_band_cache_t<typename kernel_t::SparseStorageManager>>(source));
110 }
111 break;
112
113 case LevelSetStorageType::DENSE:
114 if (LevelSetCachedObject<narrow_band_cache_t<typename kernel_t::DenseStorageManager>> *cachedSource = dynamic_cast<LevelSetCachedObject<narrow_band_cache_t<typename kernel_t::DenseStorageManager>> *>(source)) {
115 return std::unique_ptr<LevelSetObject>(new LevelSetImmutableObject<narrow_band_cache_t<typename kernel_t::DenseStorageManager>>(cachedSource));
116 } else {
117 return std::unique_ptr<LevelSetObject>(new LevelSetImmutableObject<narrow_band_cache_t<typename kernel_t::DenseStorageManager>>(source));
118 }
119 break;
120
121 default:
122 BITPIT_UNREACHABLE("Storage type not supported");
123
124 }
125
126 BITPIT_UNREACHABLE("Unable to create an immutable object from the specified object.");
127}
