566bool PetscManager::initialize(bool debug)
567{
568 // Early return if PETSc is already initialized
569 PetscBool isPetscInitialized;
570 PetscInitialized(&isPetscInitialized);
571 if (isPetscInitialized) {
572 return false;
573 }
574
575#if BITPIT_ENABLE_MPI==1
576 // Early return if MPI is already finalized
577 int isMPIFinalized;
578 MPI_Finalized(&isMPIFinalized);
579 if (isMPIFinalized) {
580 throw std::runtime_error("PETSc finalization cannot be called after MPI finaliation.");
581 }
582#endif
583
584 // Generate command line arguments
585 //
586 // The first argument is the executable name and it is set to a
587 // dummy value.
588 std::string help = "None";
589 std::string programName = "bitpit_petsc_manager";
590
591 int argc = 1 + m_options.size();
592 char **argv = new char*[argc + 1];
593 argv[0] = strdup(programName.data());
594 for (std::size_t i = 0; i < m_options.size(); i++) {
595 argv[1 + i] = strdup(m_options[i].data());
596 }
597 argv[argc] = nullptr;
598
599#if BITPIT_ENABLE_MPI==1
600 // Initialize MPI
601 int isMPIInitialized;
602 MPI_Initialized(&isMPIInitialized);
603 if (!isMPIInitialized) {
604 MPI_Init(&argc, &argv);
605 m_externalMPIInitialization = false;
606 }
607#endif
608
609 // Initialize PETSc
610 PetscInitialize(&argc, &argv, 0, help.data());
611 m_externalPETScInitialization = false;
612
613 // Enable log view
614 if (debug) {
615 enableLogView();
616 }
617
618 // Clean-up command line arguments
619 for (int i = 0; i < argc; ++i) {
620 free(argv[i]);
621 }
622
623 delete[] argv;
624
625 // Initialization completed
626 return true;
627}

650bool PetscManager::finalize()
651{
652#if BITPIT_ENABLE_MPI==1
653 // Early return if MPI is already finalized
654 int isMPIFinalized;
655 MPI_Finalized(&isMPIFinalized);
656 if (isMPIFinalized) {
657 return false;
658 }
659#endif
660
661 // Early return if PETSc was initialized externally
662 if (m_externalPETScInitialization) {
663 return false;
664 }
665
666 // Finalize PETSc
667 PetscBool isPetscFinalized;
668 PetscFinalized(&isPetscFinalized);
669 if (!isPetscFinalized) {
670 PetscFinalize();
671 }
672
673#if BITPIT_ENABLE_MPI==1
674 // Finalize MPI
675 if (!m_externalMPIInitialization) {
676 MPI_Finalize();
677 }
678#endif
679
680 return (!isPetscFinalized);
681}

1271void SystemSolver::solve()
1272{
1273 // Check if the system is assembled
1274 if (!isAssembled()) {
1275 throw std::runtime_error("Unable to solve the system. The system is not yet assembled.");
1276 }
1277
1278 // Prepare the KSP
1279 prepareKSP();
1280
1281 // Perfrom actions before KSP solution
1282 preKSPSolveActions();
1283
1284 // Force consistency
1285 if (m_forceConsistency) {
1286 removeNullSpaceFromRHS();
1287 }
1288
1289 // Solve the system
1290 if (!m_transpose) {
1291 m_KSPStatus.error = KSPSolve(m_KSP, m_rhs, m_solution);
1292 } else {
1293 m_KSPStatus.error = KSPSolveTranspose(m_KSP, m_rhs, m_solution);
1294 }
1295
1296 // Set solver info
1297 if (m_KSPStatus.error == 0) {
1298 KSPGetIterationNumber(m_KSP, &m_KSPStatus.its);
1299 KSPGetConvergedReason(m_KSP, &m_KSPStatus.convergence);
1300 } else {
1301 m_KSPStatus.its = -1;
1302 m_KSPStatus.convergence = KSP_DIVERGED_BREAKDOWN;
1303 }
1304
1305 // Perfrom actions after KSP solution
1306 postKSPSolveActions();
1307
1308 // Finalize the KSP
1309 finalizeKSP();
1310}

1398void SystemSolver::matrixCreate(const SystemMatrixAssembler &assembler)
1399{
1400 const PetscInt *rowReordering = nullptr;
1401 if (m_rowReordering) {
1402 ISGetIndices(m_rowReordering, &rowReordering);
1403 }
1404
1405 // Set block size
1406 //
1407 // When blocks are ignored, the PETSc matrix will be created with a unitary
1408 // block size, regardless of the blocks size of the assembler. If this is
1409 // the case, preallociation information provided by the assembler will be
1410 // properly expanded to have preallocation information for each matrix row.
1411 int assemblerBlockSize = assembler.getBlockSize();
1412
1413 int matrixBlockSize;
1414 if (m_flatten) {
1415 matrixBlockSize = 1;
1416 } else {
1417 matrixBlockSize = assembler.getBlockSize();
1418 }
1419
1420 int blockExpansionSize = (matrixBlockSize != assemblerBlockSize) ? assemblerBlockSize : 1;
1421
1422 // Create the matrix
1423 matrixCreate(matrixBlockSize);
1424
1425 // Get sizes
1426 long nAssemblerRows = assembler.getRowCount();
1427
1428 long nRowsElements = assembler.getRowElementCount();
1429 long nColsElements = assembler.getColElementCount();
1430
1431 long nGlobalRowsElements;
1432 long nGlobalColsElements;
1433#if BITPIT_ENABLE_MPI == 1
1434 nGlobalRowsElements = assembler.getRowGlobalElementCount();
1435 nGlobalColsElements = assembler.getColGlobalElementCount();
1436#else
1437 nGlobalRowsElements = nRowsElements;
1438 nGlobalColsElements = nColsElements;
1439#endif
1440
1441 MatSetSizes(m_A, nRowsElements, nColsElements, nGlobalRowsElements, nGlobalColsElements);
1442
1443 // Preallocation information
1444 long nPreallocationRows = blockExpansionSize * nAssemblerRows;
1445
1446 std::vector<int> d_nnz(nPreallocationRows, 0);
1447 for (long n = 0; n < nAssemblerRows; ++n) {
1448 long matrixRow = n;
1449 if (rowReordering) {
1450 matrixRow = rowReordering[matrixRow];
1451 }
1452
1453 int nAssemblerRowNZ = assembler.getRowNZCount(n);
1454
1455 long matrixRowOffset = matrixRow * blockExpansionSize;
1456 for (int i = 0; i < blockExpansionSize; ++i) {
1457 d_nnz[matrixRowOffset + i] = blockExpansionSize * nAssemblerRowNZ;
1458 }
1459 }
1460
1461
1462#if BITPIT_ENABLE_MPI == 1
1463 std::vector<int> o_nnz(nPreallocationRows, 0);
1464 if (m_partitioned) {
1465 long nAssemblerCols = assembler.getColCount();
1466
1467 long assemblerDiagonalBegin = assembler.getColGlobalOffset();
1468 long assemblerDiagonalEnd = assemblerDiagonalBegin + nAssemblerCols;
1469
1470 ConstProxyVector<long> assemblerRowPattern(static_cast<std::size_t>(0), assembler.getMaxRowNZCount());
1471 for (long n = 0; n < nAssemblerRows; ++n) {
1472 long matrixRow = n;
1473 if (rowReordering) {
1474 matrixRow = rowReordering[matrixRow];
1475 }
1476
1477
1478 assembler.getRowPattern(n, &assemblerRowPattern);
1479 int nAssemblerRowNZ = assemblerRowPattern.size();
1480
1481 long matrixRowOffset = matrixRow * blockExpansionSize;
1482 for (int k = 0; k < nAssemblerRowNZ; ++k) {
1483 long id = assemblerRowPattern[k];
1484 if (id < assemblerDiagonalBegin || id >= assemblerDiagonalEnd) {
1485 for (int i = 0; i < blockExpansionSize; ++i) {
1486 o_nnz[matrixRowOffset + i] += blockExpansionSize;
1487 d_nnz[matrixRowOffset + i] -= blockExpansionSize;
1488 }
1489 }
1490 }
1491 }
1492 }
1493#endif
1494
1495#if BITPIT_ENABLE_MPI == 1
1496 if (m_partitioned) {
1497 if (matrixBlockSize == 1) {
1498 MatMPIAIJSetPreallocation(m_A, 0, d_nnz.data(), 0, o_nnz.data());
1499 } else {
1500 MatMPIBAIJSetPreallocation(m_A, matrixBlockSize, 0, d_nnz.data(), 0, o_nnz.data());
1501 }
1502 } else
1503#endif
1504 {
1505 if (matrixBlockSize == 1) {
1506 MatSeqAIJSetPreallocation(m_A, 0, d_nnz.data());
1507 } else {
1508 MatSeqBAIJSetPreallocation(m_A, matrixBlockSize, 0, d_nnz.data());
1509 }
1510
1511 }
1512
1513 // Each process will only set values for its own rows
1514 MatSetOption(m_A, MAT_NO_OFF_PROC_ENTRIES, PETSC_TRUE);
1515
1516#if PETSC_VERSION_GE(3, 12, 0)
1517 // The first assembly will set a superset of the off-process entries
1518 // required for all subsequent assemblies. This avoids a rendezvous
1519 // step in the MatAssembly functions.
1520 MatSetOption(m_A, MAT_SUBSET_OFF_PROC_ENTRIES, PETSC_TRUE);
1521#endif
1522
1523 // Cleanup
1524 if (m_rowReordering) {
1525 ISRestoreIndices(m_rowReordering, &rowReordering);
1526 }
1527}

1616void SystemSolver::matrixUpdate(long nRows, const long *rows, const SystemMatrixAssembler &assembler)
1617{
1618 // Updating the matrix invalidates the KSP
1619 m_KSPDirty = true;
1620
1621 // Get block size
1622 const int matrixBlockSize = getBlockSize();
1623 const int assemblerBlockSize = assembler.getBlockSize();
1624 if (assemblerBlockSize != matrixBlockSize && matrixBlockSize != 1) {
1625 std::string message = "Unable to update the matrix.";
1626 message += " The block size of the assembler is not compatible with the block size of the system matrix.";
1627 throw std::runtime_error(message);
1628 }
1629
1630 // Initialize reordering
1631 const PetscInt *rowReordering = nullptr;
1632 if (m_rowReordering) {
1633 ISGetIndices(m_rowReordering, &rowReordering);
1634 }
1635
1636 const PetscInt *colReordering = nullptr;
1637 if (m_colReordering) {
1638 ISGetIndices(m_colReordering, &colReordering);
1639 }
1640
1641 // Global information
1642 PetscInt colGlobalBegin;
1643 PetscInt colGlobalEnd;
1644 MatGetOwnershipRangeColumn(m_A, &colGlobalBegin, &colGlobalEnd);
1645 colGlobalBegin /= assemblerBlockSize;
1646 colGlobalEnd /= assemblerBlockSize;
1647
1648 PetscInt rowGlobalOffset;
1649 MatGetOwnershipRange(m_A, &rowGlobalOffset, nullptr);
1650 rowGlobalOffset /= assemblerBlockSize;
1651
1652 // Get the options for assembling the matrix
1653 SystemMatrixAssembler::AssemblyOptions assemblyOptions = assembler.getOptions();
1654
1655 PetscBool matrixSortedFull = (assemblyOptions.full && assemblyOptions.sorted) ? PETSC_TRUE : PETSC_FALSE;
1656
1657#if PETSC_VERSION_GE(3, 12, 0)
1658 // Check if it is possible to speedup insertion of values
1659 //
1660 // The option MAT_SORTED_FULL means that each process provides exactly its
1661 // local rows; all column indices for a given row are passed in a single call
1662 // to MatSetValues(), preallocation is perfect, row oriented, INSERT_VALUES
1663 // is used. If this options is set to PETSC_TRUE, the function MatSetValues
1664 // will be faster.
1665 //
1666 // This options needs at least PETSc 3.12.
1667 MatSetOption(m_A, MAT_SORTED_FULL, matrixSortedFull);
1668#endif
1669
1670 // Update element values
1671 //
1672 // If the sizes of PETSc data types match the sizes of data types expected by
1673 // bitpit a direct update can be performed, otherwise the matrix is updated
1674 // using intermediate data storages.
1675 const long maxRowNZ = std::max(assembler.getMaxRowNZCount(), 0L);
1676 bool valuesDirectUpdate = (sizeof(double) == sizeof(PetscScalar));
1677
1678 ConstProxyVector<long> rowPattern(static_cast<std::size_t>(0), static_cast<std::size_t>(maxRowNZ));
1679 std::vector<PetscInt> petscRowPattern(maxRowNZ);
1680 std::vector<PetscInt> petscMatrixRowPattern(assemblerBlockSize * maxRowNZ);
1681
1682 ConstProxyVector<double> rowValues;
1683 std::vector<PetscScalar> petscRowValuesStorage;
1684 const PetscScalar *petscRowValues;
1685 if (!valuesDirectUpdate) {
1686 long maxRowNZElements = assemblerBlockSize * assemblerBlockSize * maxRowNZ;
1687 rowValues.set(ConstProxyVector<double>::INTERNAL_STORAGE, 0, maxRowNZElements);
1688 petscRowValuesStorage.resize(maxRowNZElements);
1689 petscRowValues = petscRowValuesStorage.data();
1690 }
1691
1692 for (long n = 0; n < nRows; ++n) {
1693 // Get row information
1694 long row;
1695 if (rows) {
1696 row = rows[n];
1697 } else {
1698 row = n;
1699 }
1700
1701 if (rowReordering) {
1702 row = rowReordering[row];
1703 }
1704
1705 const PetscInt globalRow = rowGlobalOffset + row;
1706
1707 // Check if it possible to perform a fast update
1708 //
1709 // A fast update allows to set all the values of a row at once (without
1710 // the need to get the row pattern), it can be performed if:
1711 // - the system matrix has already been assembled;
1712 // - the system matrix has a unitary block size;
1713 // - the assembler is providing all the values of the row;
1714 // - values provided by the assembler are sorted by ascending column.
1715 //
1716 // If fast update is used, row values will be set using a special PETSc
1717 // function (MatSetValuesRow) that allows to set all the values of a
1718 // row at once, without requiring the pattern of the row.
1719 //
1720 // Fast update is not related to the option MAT_SORTED_FULL, that option
1721 // is used to speedup the standard function MatSetValues (which still
1722 // requires the pattern of the row).
1723 bool fastUpdate = isAssembled() && (matrixBlockSize == 1) && (matrixSortedFull == PETSC_TRUE);
1724
1725 // Get row data
1726 if (fastUpdate) {
1727 assembler.getRowValues(n, &rowValues);
1728 } else {
1729 assembler.getRowData(n, &rowPattern, &rowValues);
1730 }
1731
1732 // Get values in PETSc format
1733 if (valuesDirectUpdate) {
1734 petscRowValues = reinterpret_cast<const PetscScalar *>(rowValues.data());
1735 } else {
1736 std::copy(rowValues.cbegin(), rowValues.cend(), petscRowValuesStorage.begin());
1737 }
1738
1739 if (fastUpdate) {
1740 // Set values
1741 if (assemblerBlockSize == 1) {
1742 MatSetValuesRow(m_A, globalRow, petscRowValues);
1743 } else {
1744 const int nMatrixRowValues = rowValues.size() / assemblerBlockSize;
1745
1746 PetscInt matrixGlobalRow = globalRow * assemblerBlockSize;
1747 const PetscScalar *petscMatrixRowValues = petscRowValues;
1748 for (int k = 0; k < assemblerBlockSize; ++k) {
1749 MatSetValuesRow(m_A, matrixGlobalRow, petscMatrixRowValues);
1750 ++matrixGlobalRow;
1751 petscMatrixRowValues += nMatrixRowValues;
1752 }
1753 }
1754 } else {
1755 // Get the size of the pattern
1756 const int rowPatternSize = rowPattern.size();
1757 if (rowPatternSize == 0) {
1758 continue;
1759 }
1760
1761 // Get pattern in PETSc format
1762 for (int k = 0; k < rowPatternSize; ++k) {
1763 long globalCol = rowPattern[k];
1764 if (colReordering) {
1765 if (globalCol >= colGlobalBegin && globalCol < colGlobalEnd) {
1766 long col = globalCol - colGlobalBegin;
1767 col = colReordering[col];
1768 globalCol = colGlobalBegin + col;
1769 }
1770 }
1771
1772 petscRowPattern[k] = globalCol;
1773 }
1774
1775 // Set data
1776 if (matrixBlockSize > 1) {
1777 MatSetValuesBlocked(m_A, 1, &globalRow, rowPatternSize, petscRowPattern.data(), petscRowValues, INSERT_VALUES);
1778 } else {
1779 if (assemblerBlockSize == 1) {
1780 MatSetValues(m_A, 1, &globalRow, rowPatternSize, petscRowPattern.data(), petscRowValues, INSERT_VALUES);
1781 } else {
1782 // Expand the pattern to get a values for each element
1783 const int matrixRowPatternSize = rowPatternSize * assemblerBlockSize;
1784 for (int k = 0; k < rowPatternSize; ++k) {
1785 int matrixPattenOffset = k * assemblerBlockSize;
1786 for (int n = 0; n < assemblerBlockSize; ++n) {
1787 petscMatrixRowPattern[matrixPattenOffset + n] = assemblerBlockSize * petscRowPattern[k] + n;
1788 }
1789 }
1790
1791 // Set data
1792 PetscInt matrixGlobalRow = globalRow * assemblerBlockSize;
1793 const PetscScalar *petscMatrixRowValues = petscRowValues;
1794 for (int k = 0; k < assemblerBlockSize; ++k) {
1795 MatSetValues(m_A, 1, &matrixGlobalRow, matrixRowPatternSize, petscMatrixRowPattern.data(), petscMatrixRowValues, INSERT_VALUES);
1796 ++matrixGlobalRow;
1797 petscMatrixRowValues += matrixRowPatternSize;
1798 }
1799 }
1800 }
1801 }
1802 }
1803
1804 // Let petsc assembly the matrix after the update
1805 MatAssemblyBegin(m_A, MAT_FINAL_ASSEMBLY);
1806 MatAssemblyEnd(m_A, MAT_FINAL_ASSEMBLY);
1807
1808 // Cleanup
1809 if (rowReordering) {
1810 ISRestoreIndices(m_rowReordering, &rowReordering);
1811 }
1812
1813 if (colReordering) {
1814 ISRestoreIndices(m_colReordering, &colReordering);
1815 }
1816}
