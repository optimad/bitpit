377template<typename stencil_t>
378void DiscretizationStencilSolverAssembler<stencil_t>::setMatrixSizes(long nRows, long nCols)
379#endif
380{
381 // Set system sizes
382 m_nRows = nRows;
383 m_nCols = nCols;
384
385#if BITPIT_ENABLE_MPI==1
386 // Global system sizes
387 m_nGlobalRows = nRows;
388 m_nGlobalCols = nCols;
389 if (partitioned) {
390 MPI_Allreduce(MPI_IN_PLACE, &m_nGlobalRows, 1, MPI_LONG, MPI_SUM, communicator);
391 MPI_Allreduce(MPI_IN_PLACE, &m_nGlobalCols, 1, MPI_LONG, MPI_SUM, communicator);
392 }
393
394 // Global offsets
395 m_globalRowOffset = 0;
396 m_globalColOffset = 0;
397 if (partitioned) {
398 int nProcessors;
399 MPI_Comm_size(communicator, &nProcessors);
400
401 std::vector<long> nRankRows(nProcessors);
402 MPI_Allgather(&m_nRows, 1, MPI_LONG, nRankRows.data(), 1, MPI_LONG, communicator);
403
404 std::vector<long> nRankCols(nProcessors);
405 MPI_Allgather(&m_nCols, 1, MPI_LONG, nRankCols.data(), 1, MPI_LONG, communicator);
406
407 int rank;
408 MPI_Comm_rank(communicator, &rank);
409 for (int i = 0; i < rank; ++i) {
410 m_globalRowOffset += nRankRows[i];
411 m_globalColOffset += nRankCols[i];
412 }
413 }
414#endif
415}

741template<typename stencil_t>
742template<typename U, typename std::enable_if<!std::is_fundamental<U>::value>::type *>
743void DiscretizationStencilSolverAssembler<stencil_t>::getValues(const stencil_t &stencil, ConstProxyVector<double> *values) const
744{
745 std::size_t stencilSize = stencil.size();
746 const typename stencil_t::weight_type *stencilWeightData = stencil.weightData();
747
748 int nBlockElements = m_blockSize * m_blockSize;
749 std::size_t nRowValues = m_blockSize * stencilSize;
750 std::size_t nValues = nBlockElements * stencilSize;
751 values->set(ConstProxyVector<double>::INTERNAL_STORAGE, nValues);
752 ConstProxyVector<double>::storage_pointer expandedValuesStorage = values->storedData();
753
754
755 for (std::size_t k = 0; k < stencilSize; ++k) {
756 const double *weightData = stencilWeightData[k].data();
757 for (int i = 0; i < m_blockSize; ++i) {
758 int weightOffset = linearalgebra::linearIndexRowMajor(i, 0, m_blockSize, m_blockSize);
759 int valuesOffset = linearalgebra::linearIndexRowMajor(i, m_blockSize * k, m_blockSize, nRowValues);
760
761 std::copy_n(weightData + weightOffset, m_blockSize, expandedValuesStorage + valuesOffset);
762 }
763 }
764}

833template<typename stencil_t>
834template<typename U, typename std::enable_if<!std::is_fundamental<U>::value>::type *>
835void DiscretizationStencilSolverAssembler<stencil_t>::getConstant(const stencil_t &stencil, bitpit::ConstProxyVector<double> *constant) const
836{
837 const typename stencil_t::weight_type &stencilConstant = stencil.getConstant();
838
839 constant->set(ConstProxyVector<double>::INTERNAL_STORAGE, m_blockSize);
840 ConstProxyVector<double>::storage_pointer constantStorage = constant->storedData();
841 for (int i = 0; i < m_blockSize; ++i) {
842 int offset_i = linearalgebra::linearIndexRowMajor(i, 0, m_blockSize, m_blockSize);
843
844 constantStorage[i] = 0;
845 for (int j = 0; j < m_blockSize; ++j) {
846 constantStorage[i] += getRawValue(stencilConstant, offset_i + j);
847 }
848 }
849}
