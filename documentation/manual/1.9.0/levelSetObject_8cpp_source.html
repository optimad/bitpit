310LevelSetIntersectionStatus LevelSetObject::intersectSurface(long id, LevelSetIntersectionMode mode) const{
311
312 double absoluteDistance = std::abs(getValue(id));
313 double distanceTolerance = m_kernel->getMesh()->getTol();
314
315 switch(mode){
316 case LevelSetIntersectionMode::FAST_GUARANTEE_TRUE:
317 {
318 double tangentSphere = m_kernel->computeCellTangentRadius(id) ;
319 if(utils::DoubleFloatingLessEqual()(absoluteDistance, tangentSphere, distanceTolerance, distanceTolerance)){
320 return LevelSetIntersectionStatus::TRUE;
321 } else {
322 return LevelSetIntersectionStatus::FALSE;
323 }
324
325 break;
326 }
327
328 case LevelSetIntersectionMode::FAST_GUARANTEE_FALSE:
329 {
330 double boundingSphere = m_kernel->computeCellBoundingRadius(id) ;
331 if(utils::DoubleFloatingGreater()(absoluteDistance, boundingSphere, distanceTolerance, distanceTolerance)){
332 return LevelSetIntersectionStatus::FALSE;
333 } else {
334 return LevelSetIntersectionStatus::TRUE;
335 }
336
337 break;
338 }
339
340 case LevelSetIntersectionMode::FAST_FUZZY:
341 {
342 double boundingSphere = m_kernel->computeCellBoundingRadius(id) ;
343 if(utils::DoubleFloatingGreater()(absoluteDistance, boundingSphere, distanceTolerance, distanceTolerance)){
344 return LevelSetIntersectionStatus::FALSE;
345 }
346
347 double tangentSphere = m_kernel->computeCellTangentRadius(id) ;
348 if(utils::DoubleFloatingLessEqual()(absoluteDistance, tangentSphere, distanceTolerance, distanceTolerance)){
349 return LevelSetIntersectionStatus::TRUE;
350 }
351
352 return LevelSetIntersectionStatus::CLOSE;
353
354 break;
355 }
356
357 case LevelSetIntersectionMode::ACCURATE:
358 {
359 double boundingSphere = m_kernel->computeCellBoundingRadius(id) ;
360 if(utils::DoubleFloatingGreater()(absoluteDistance, boundingSphere, distanceTolerance, distanceTolerance)){
361 return LevelSetIntersectionStatus::FALSE;
362 }
363
364 double tangentSphere = m_kernel->computeCellTangentRadius(id) ;
365 if(utils::DoubleFloatingLessEqual()(absoluteDistance, tangentSphere, distanceTolerance, distanceTolerance)){
366 return LevelSetIntersectionStatus::TRUE;
367 }
368
369 std::array<double,3> root = computeProjectionPoint(id);
370 std::array<double,3> normal = getGradient(id);
371 if( m_kernel->intersectCellPlane(id,root,normal, distanceTolerance) ){
372 return LevelSetIntersectionStatus::TRUE;
373 } else {
374 return LevelSetIntersectionStatus::FALSE;
375 }
376
377 break;
378 }
379 }
380
381 BITPIT_UNREACHABLE("cannot reach");

417void LevelSetObject::update( const std::vector<adaption::Info> &adaptionData, bool signedDistance ) {
418
419#if BITPIT_ENABLE_MPI
420 UpdateStrategy partitioningUpdateStrategy = getPartitioningUpdateStrategy();
421#endif
422
423 std::vector<long> pruneList ;
424 std::vector<long> updateList ;
425#if BITPIT_ENABLE_MPI
426 std::unordered_map<int, std::vector<long>> exchangeSendList ;
427 std::unordered_map<int, std::vector<long>> exchangeRecvList ;
428#endif
429 for( const adaption::Info &adaptionInfo : adaptionData){
430 if( adaptionInfo.entity != adaption::Entity::ENTITY_CELL){
431 continue;
432 }
433
434 switch (adaptionInfo.type) {
435
436#if BITPIT_ENABLE_MPI
437 case adaption::Type::TYPE_PARTITION_SEND:
438 if (partitioningUpdateStrategy == UPDATE_STRATEGY_EXCHANGE) {
439 exchangeSendList.insert({{adaptionInfo.rank,adaptionInfo.previous}}) ;
440 }
441 break;
442
443 case adaption::Type::TYPE_PARTITION_RECV:
444 if (partitioningUpdateStrategy == UPDATE_STRATEGY_EXCHANGE) {
445 exchangeRecvList.insert({{adaptionInfo.rank,adaptionInfo.current}}) ;
446 } else if (partitioningUpdateStrategy == UPDATE_STRATEGY_EVALUATE) {
447 updateList.insert(updateList.end(), adaptionInfo.current.begin(), adaptionInfo.current.end()) ;
448 }
449 break;
450#endif
451
452 default:
453 updateList.insert(updateList.end(), adaptionInfo.current.begin(), adaptionInfo.current.end()) ;
454 break;
455
456 }
457
458 pruneList.insert(pruneList.end(), adaptionInfo.previous.begin(), adaptionInfo.previous.end()) ;
459 }
460
461#if BITPIT_ENABLE_MPI
462 // Initialize data exchange
463 bool exchangeData = (!exchangeSendList.empty() || !exchangeRecvList.empty());
464 if (m_kernel->getMesh()->isPartitioned()) {
465 MPI_Allreduce(MPI_IN_PLACE, &exchangeData, 1, MPI_C_BOOL, MPI_LOR, m_kernel->getCommunicator()) ;
466 }
467
468 std::unique_ptr<DataCommunicator> dataCommunicator;
469 if (exchangeData) {
470 dataCommunicator = m_kernel->createDataCommunicator() ;
471 }
472
473 // Start data exchange
474 if (exchangeData) {
475 startExchange( exchangeSendList, dataCommunicator.get() ) ;
476 }
477#endif
478
479 // Prune narrow band data structures
480 if (!pruneList.empty()) {
481 pruneNarrowBand( pruneList ) ;
482 }
483
484#if BITPIT_ENABLE_MPI
485 // Complete data exchange
486 if (exchangeData) {
487 completeExchange( exchangeRecvList, dataCommunicator.get() ) ;
488 }
489#endif
490
491 // Update narrow band
492 if (!updateList.empty()) {
493 updateNarrowBand( updateList, signedDistance ) ;
494 }
495
496#if BITPIT_ENABLE_MPI
497 // Update data on ghost cells
498 exchangeGhosts() ;
499#endif
500}

625void LevelSetObject::enableVTKOutput( LevelSetWriteField fieldset, const std::string &objectName, bool enable) {
626
627 std::vector<LevelSetWriteField> fields;
628
629 if( fieldset==LevelSetWriteField::ALL){
630 fields.push_back(LevelSetWriteField::VALUE);
631 fields.push_back(LevelSetWriteField::GRADIENT);
632 fields.push_back(LevelSetWriteField::NORMAL);
633 fields.push_back(LevelSetWriteField::PART);
634
635 } else if ( fieldset==LevelSetWriteField::DEFAULT){
636 fields.push_back(LevelSetWriteField::VALUE);
637 fields.push_back(LevelSetWriteField::GRADIENT);
638
639 } else {
640 fields.push_back(fieldset);
641 }
642
643 VTK &vtkWriter = m_kernel->getMesh()->getVTK() ;
644 for( const LevelSetWriteField &field : fields){
645 // Update the list of enabled VTK outputs
646 if (enable) {
647 m_enabledVTKOutputs.insert(field) ;
648 } else {
649 m_enabledVTKOutputs.erase(field) ;
650 }
651
652
653 // Get name of the field
654 std::stringstream nameStream;
655 nameStream << "levelset";
656 switch(field){
657
658 case LevelSetWriteField::VALUE:
659 nameStream << "Value_" << objectName;
660 break;
661
662 case LevelSetWriteField::GRADIENT:
663 nameStream << "Gradient_" << objectName;
664 break;
665
666 case LevelSetWriteField::NORMAL:
667 nameStream << "Normal_" << objectName;
668 break;
669
670 case LevelSetWriteField::PART:
671 nameStream << "PartId_" << objectName;
672 break;
673
674 default:
675 std::cout << " field " << (int) field << std::endl;
676 throw std::runtime_error ("Unsupported value of field in LevelSetObject::addDataToVTK() ");
677 break;
678 }
679
680 std::string name = nameStream.str();
681
682 // Check if the state of the filed is already the requested one
683 if (enable == vtkWriter.hasData(name)) {
684 continue;
685 }
686
687 // Process the field
688 if (!enable) {
689 vtkWriter.removeData( name);
690 } else {
691 switch(field){
692
693 case LevelSetWriteField::VALUE:
694 vtkWriter.addData<double>( name, VTKFieldType::SCALAR, VTKLocation::CELL, this);
695 break;
696
697 case LevelSetWriteField::GRADIENT:
698 vtkWriter.addData<double>( name, VTKFieldType::VECTOR, VTKLocation::CELL, this);
699 break;
700
701 case LevelSetWriteField::NORMAL:
702 vtkWriter.addData<double>( name, VTKFieldType::VECTOR, VTKLocation::CELL, this);
703 break;
704
705 case LevelSetWriteField::PART:
706 vtkWriter.addData<int>( name, VTKFieldType::SCALAR, VTKLocation::CELL, this);
707 break;
708
709 default:
710 throw std::runtime_error ("Unsupported value of field in LevelSetObject::addDataToVTK() ");
711 break;
712
713 }
714 }
715 }

795void LevelSetObject::startExchange( const std::unordered_map<int,std::vector<long>> &sendList,
796 DataCommunicator *dataCommunicator){
797
798 // Fill the send buffer with the content from the LevelSetObject base
799 // class and the specific derived class.
800 for (const auto &entry : sendList) {
801 // Create an empty send
802 int rank = entry.first;
803 dataCommunicator->setSend(rank, 0);
804
805 // Write data in the buffer
806 SendBuffer &buffer = dataCommunicator->getSendBuffer(rank);
807 writeCommunicationBuffer(entry.second, buffer);
808 }
809
810 // Discover the receives
811 dataCommunicator->discoverRecvs();
812
813 // Start the sends
814 dataCommunicator->startAllRecvs();
815
816 // Start the sends
817 dataCommunicator->startAllSends();
818
819}

827void LevelSetObject::completeExchange( const std::unordered_map<int,std::vector<long>> &recvList,
828 DataCommunicator *dataCommunicator){
829
830 // Check which data communications have arrived. For those which are
831 // available start reading the databuffer into the data structure of
832 // LevelSetObject and its derived classes.
833 int nCompletedRecvs = 0;
834 while (nCompletedRecvs < dataCommunicator->getRecvCount()) {
835 int rank = dataCommunicator->waitAnyRecv();
836
837 RecvBuffer &dataBuffer = dataCommunicator->getRecvBuffer(rank);
838 readCommunicationBuffer(recvList.at(rank), dataBuffer);
839
840 ++nCompletedRecvs;
841 }
842
843 dataCommunicator->waitAllSends();
844 dataCommunicator->finalize();
845}
