672void POD::setSensorMask(const PiercedStorage<bool> & mask, VolumeKernel * mesh)
673{
674 m_sensorMask.unsetKernel();
675 m_sensorMask.setStaticKernel(&(m_podkernel->getMesh()->getCells()));
676
677 if (m_staticMesh || mesh == m_podkernel->getMesh()){
678
679 for (const Cell &cell : m_podkernel->getMesh()->getCells()) {
680 long id = cell.getId();
681 m_sensorMask[id] = mask[id];
682 }
683
684 }
685 else{
686
687 if (mesh == nullptr)
688 throw std::runtime_error ("POD: null mesh pointer passed to setSensorMask");
689
690 //If mapping dirty or not computed, compute mapping of input mesh on pod mesh
691 if (m_podkernel->isMapperDirty())
692 _computeMapper(mesh);
693
694 std::unordered_set<long> trueCells;
695 for (const Cell & cell : mesh->getCells()){
696 long id = cell.getId();
697 if (mask[id])
698 trueCells.insert(id);
699 }
700
701 //Map true cells
702 std::unordered_set<long> mappedCells = m_podkernel->mapCellsToPOD(&trueCells);
703
704 m_podkernel->mapBoolFieldToPOD(mask, mesh, &mappedCells, m_sensorMask);
705
706 }
707
708 fillListActiveIDs(m_sensorMask);
709 m_toUpdate = true;
710
711}

837{
838 if (m_runMode == RunMode::COMPUTE) {
839
840 // Evaluate mean field and mesh
841 log::cout() << "pod : computing mean field and pod mesh... " << std::endl;
842 evalMeanMesh();
843
844 // Initialize ID list
845 fillListActiveIDs(m_filter);
846
847 // Evaluate correlation matrices
848 log::cout() << "pod : computing correlation matrix... " << std::endl;
849 evalCorrelation();
850
851 // Compute eigenvalues and eigenvectors
852 log::cout() << "pod : computing eigenvectors... " << std::endl;
853 evalEigen();
854
855 // Compute POD modes
856 log::cout() << "pod : computing pod modes... " << std::endl;
857 evalModes();
858
859 } else if (m_runMode == RunMode::RESTORE) {
860
861 log::cout() << "pod : restore... " << std::endl;
862 restore();
863
864 }
865
866 // Dump pod
867 if (m_writeMode != WriteMode::NONE) {
868 log::cout() << "pod : dumping... " << std::endl;
869 dump();
870 }
871
872 // Evaluate reconstruction
873 evalReconstruction();
874
875}

880void POD::leave1out()
881{
882 setErrorMode(ErrorMode::COMBINED);
883 setReconstructionMode(ReconstructionMode::PROJECTION);
884 unsetLeave1outSnapshots();
885
886 // Evaluate mean field and mesh
887 log::cout() << "pod : computing pod mesh... " << std::endl;
888 evalMeanMesh();
889
890 // Initialize ID list
891 fillListActiveIDs(m_filter);
892
893 // Evaluate correlation matrices
894 log::cout() << "pod : computing correlation matrix... " << std::endl;
895 evalCorrelation();
896
897 // Initialize error field
898 initErrorMaps();
899
900 // Initialize help variables
901 std::vector<std::vector<double>> h_correlationMatrices=m_correlationMatrices;
902 std::vector<pod::SnapshotFile> h_database=m_database;
903 std::size_t h_nSnapshots=m_nSnapshots;
904 std::size_t nl1o=m_listActiveIDsLeave1out.size();
905
906 m_nSnapshots=h_nSnapshots-1;
907
908 for (std::size_t i=0; i<nl1o; ++i){
909 std::size_t id = m_listActiveIDsLeave1out[i];
910 log::cout() << ">> leave-1-out : removing snapshot " << m_database[id].directory+"/"+m_database[id].name<< std::endl;
911
912 m_reconstructionDatabase.clear();
913 m_nReconstructionSnapshots = 0;
914 addReconstructionSnapshot(m_database[id]);
915 m_database.erase(m_database.begin()+id);
916
917 for (std::size_t ifield = 0; ifield<m_nFields; ++ifield){
918 // get rid of id row
919 m_correlationMatrices[ifield].erase(m_correlationMatrices[ifield].begin()+id*h_nSnapshots, m_correlationMatrices[ifield].begin()+id*h_nSnapshots+h_nSnapshots);
920
921 // get rid of id column
922 std::size_t it=id;
923 for (std::size_t j=0; j<m_nSnapshots; ++j){
924 m_correlationMatrices[ifield].erase(m_correlationMatrices[ifield].begin()+it);
925 it= it-1+h_nSnapshots;
926 }
927 }
928
929 // Compute eigenvalues and eigenvectors
930 log::cout() << "pod : computing eigenvectors... " << std::endl;
931 evalEigen();
932
933 // Compute POD modes
934 log::cout() << "pod : computing pod modes... " << std::endl;
935 evalModes();
936
937 // Evaluate reconstruction & compute error fields
938 evalReconstruction();
939
940 // Reassignment for future usage
941 m_correlationMatrices=h_correlationMatrices;
942 m_database=h_database;
943 m_toUpdate=true;
944 }
945
946 //dump error
947 if (m_writeMode != WriteMode::NONE){
948 std::string ename = m_name + ".error";
949 dumpField(ename, m_errorMap);
950 }
951 m_nSnapshots=h_nSnapshots;
952}

978void POD::_evalMeanMesh()
979{
980
981 if (m_podkernel->getMesh() == nullptr){
982
983 //Read first mesh and use as meshPOD
984 std::unique_ptr<VolumeKernel> readMesh = m_podkernel->readMesh(m_database[0]);
985 m_podkernel->setMesh(std::move(readMesh));
986
987 //Dynamic mesh case
988 if (!m_staticMesh){
989 // Compute meshPOD (starting from inital mesh) and fill filter
990 for (std::size_t i = 1; i < m_nSnapshots; ++i) {
991 log::cout() << "pod : evaluation POD mesh - use snapshot " << i+1 << "/" << m_nSnapshots << std::endl;
992 std::unique_ptr<VolumeKernel> readMesh = m_podkernel->readMesh(m_database[i]);
993 m_podkernel->adaptMeshToMesh(m_podkernel->getMesh(), readMesh.get());
994 m_podkernel->clearMapper();
995 }
996 }
997 }
998
999 {
1000 // Read first snapshot to set n scalar and vector fields
1001 pod::PODField readf;
1002 readSnapshot(m_database[0], readf);
1003 }
1004
1005 //Compute cells volume
1006 m_podkernel->evalCellsVolume();
1007
1008 m_filter.unsetKernel();
1009 m_filter.setStaticKernel(&(m_podkernel->getMesh()->getCells()));
1010 m_filter.fill(true);
1011
1012
1013 if (m_useMean){
1014 m_mean.scalar = std::unique_ptr<pod::ScalarStorage>(new pod::ScalarStorage(m_nScalarFields, &m_podkernel->getMesh()->getCells()));
1015 m_mean.vector = std::unique_ptr<pod::VectorStorage>(new pod::VectorStorage(m_nVectorFields, &m_podkernel->getMesh()->getCells()));
1016 m_mean.scalar->fill(0.0);
1017 m_mean.vector->fill(std::array<double, 3>{{0.0, 0.0, 0.0}});
1018
1019 // Compute mean (starting from first snapshot) & filter+mask
1020 for (std::size_t i = 0; i < m_nSnapshots; ++i) {
1021 log::cout() << "pod : evaluation mean - use snapshot " << i+1 << "/" << m_nSnapshots << std::endl;
1022 pod::PODField readf;
1023 readSnapshot(m_database[i], readf);
1024 if (!m_staticMesh){
1025 //Compute mapping of read mesh on pod mesh
1026 _computeMapper(readf.mesh);
1027 readf = pod::PODField(m_podkernel->mapPODFieldToPOD(readf, nullptr));
1028 m_podkernel->clearMapper();
1029 }
1030
1031 for (const Cell &cell : m_podkernel->getMesh()->getCells()) {
1032 m_filter[cell.getId()] = (m_filter[cell.getId()]) && (readf.mask->at(cell.getId())); //N.B. mask == filter
1033 if (m_filter[cell.getId()]){
1034 for (std::size_t j = 0; j < m_nScalarFields; ++j)
1035 m_mean.scalar->at(cell.getId(),j) += readf.scalar->at(cell.getId(),j) / double(m_nSnapshots);
1036 for (std::size_t j = 0; j < m_nVectorFields; ++j)
1037 m_mean.vector->at(cell.getId(),j) += readf.vector->at(cell.getId(),j) / double(m_nSnapshots);
1038 }
1039 else{
1040 for (std::size_t j = 0; j < m_nScalarFields; ++j)
1041 m_mean.scalar->at(cell.getId(),j) = 0.0;
1042 for (std::size_t j = 0; j < m_nVectorFields; ++j)
1043 m_mean.vector->at(cell.getId(),j) = {{0.0, 0.0, 0.0}};
1044 }
1045 }
1046 }
1047 }
1048 else {
1049 // filter+mask
1050 for (std::size_t i = 0; i < m_nSnapshots; ++i) {
1051 pod::PODField readf;
1052 readSnapshot(m_database[i], readf);
1053 if (!m_staticMesh)
1054 {
1055 _computeMapper(readf.mesh);
1056 readf = pod::PODField(m_podkernel->mapPODFieldToPOD(readf, nullptr));
1057 m_podkernel->clearMapper();
1058 }
1059
1060 for (const Cell &cell : m_podkernel->getMesh()->getCells())
1061 m_filter[cell.getId()] = (m_filter[cell.getId()]) && (readf.mask->at(cell.getId())); //N.B. mask == filter
1062
1063 }
1064 }
1065
1066 setSensorMask(m_filter, m_podkernel->getMesh());
1067}

1075void POD::fillListActiveIDs(const PiercedStorage<bool> &mask)
1076{
1077
1078 //NOTE!! NO GHOSTS IN ACTIVEIDS (*)
1079 m_listActiveIDs.clear();
1080 std::unordered_set<long> listGhost;
1081 for (const Cell &cell : m_podkernel->getMesh()->getCells()) {

1706 std::vector<double> alambda(m_nSnapshots);
1707 for (std::size_t j = 0; j < m_nSnapshots*m_nSnapshots; ++j)
1708 Marr[j/m_nSnapshots+j%m_nSnapshots*m_nSnapshots] = m_correlationMatrices[i][j];
1709
1710 lapack_int info = LAPACKE_dsyev(LAPACK_COL_MAJOR, 'V', 'U', m_nSnapshots, Marr.data(), m_nSnapshots, alambda.data()); //lapacke
1711 if (info != 0) {
1712 log::cout() << "WARNING! algorithm convergence info " << info << std::endl;
1713 throw std::runtime_error("Unable to solve minimization problem.");
1714 }
1715
1716 // Check number of modes with energy level
1717 checkModeCount(alambda.data(), i);
1718
1719 for (std::size_t n = 0; n < m_nSnapshots; ++n)
1720 m_lambda[i][n] = alambda[m_nSnapshots - n - 1];
1721
1722 for (std::size_t n = 0; n < m_nSnapshots; ++n) {
1723 for (std::size_t p = 0; p < m_nSnapshots; ++p)
1724 m_podCoeffs[i][p][n] = Marr[N - m_nSnapshots * (p + 1) + n] * std::sqrt(std::abs(m_lambda[i][p]));
1725 }
1726 }
1727 if (m_errorMode != ErrorMode::COMBINED){
1728 // one coeff file for each mode, each line is related to a field, each term is related to a snapshot
1729 for (std::size_t ir=0; ir<m_nModes; ++ir) {
1730 std::ofstream out(m_directory + m_name + ".coeff."+std::to_string(ir)+".dat", std::ofstream::out);
1731 for (std::size_t i = 0; i < m_nFields; ++i)
1732 out << m_podCoeffs[i][ir] << std::endl;
1733
1734 out.close();
1735 }
1736
1737 std::ofstream outl(m_directory + m_name + ".lambda.dat", std::ofstream::out);
1738 for (std::size_t i = 0; i < m_nFields; ++i)
1739 outl << m_lambda[i] << std::endl;
1740
1741 outl.close();
1742 }
1743 }
1744
1745#if BITPIT_ENABLE_MPI
1746 long bufferSize = m_nModes;
1747 MPI_Bcast(&bufferSize, 1, MPI_LONG, 0, m_communicator);
1748 m_nModes = bufferSize;
1749 for (std::size_t i = 0; i < m_nFields; ++i) {
1750 for (std::size_t p=0; p<m_nModes; ++p)
1751 MPI_Bcast(m_podCoeffs[i][p].data(), m_nSnapshots, MPI_DOUBLE, 0, m_communicator);
1752
1753 MPI_Bcast(m_lambda[i].data(), m_nSnapshots, MPI_DOUBLE, 0, m_communicator);
1754 }
1755#endif
1756}

2817void POD::_evalReconstructionCoeffs(PiercedStorage<double> &fields,
2818 const std::vector<std::size_t> &scalarIds, const std::vector<std::size_t> &podscalarIds,
2819 const std::vector<std::array<std::size_t, 3>> &vectorIds, const std::vector<std::size_t> &podvectorIds)
2820{
2821 std::size_t nsf = scalarIds.size();
2822 std::size_t nvf = vectorIds.size();
2823
2824 // Difference field - m_mean
2825 diff(fields, m_mean, scalarIds, podscalarIds, vectorIds, podvectorIds, &m_listActiveIDs);
2826
2827 m_reconstructionCoeffs.clear();
2828 m_reconstructionCoeffs.resize(m_nFields, std::vector<double>(m_nModes, 0.0));
2829
2830 // Evaluate minimization matrices
2831 evalMinimizationMatrices();
2832
2833 // Evaluate RHS
2834 for (std::size_t ir = 0; ir < m_nModes; ++ir) {
2835 if (m_memoryMode == MemoryMode::MEMORY_LIGHT)
2836 readMode(ir);
2837
2838 std::unordered_set<long>::iterator it;
2839 for (it = m_listActiveIDs.begin(); it != m_listActiveIDs.end(); ++it) {
2840 long id = *it;
2841 std::size_t rawIndex = m_podkernel->getMesh()->getCells().getRawIndex(id);
2842 double *datag = fields.rawData(rawIndex);
2843 if (nsf) {
2844 double *modes = m_modes[ir].scalar->rawData(rawIndex);
2845 for (std::size_t ifield = 0; ifield < nsf; ++ifield) {
2846 double *modesi = modes + podscalarIds[ifield];
2847 double *datagi = datag + scalarIds[ifield];
2848 m_reconstructionCoeffs[ifield][ir] += (*datagi)*(*modesi)*getRawCellVolume(rawIndex);
2849 }
2850 }
2851 if (nvf) {
2852 std::array<double,3>* modev = m_modes[ir].vector->rawData(rawIndex);
2853 for (std::size_t ifield = 0; ifield < nvf; ++ifield) {
2854 std::array<double,3>* modevi = modev + podvectorIds[ifield];
2855 for (std::size_t j = 0; j < 3; ++j) {
2856 double *datagi = datag + vectorIds[ifield][j];
2857 m_reconstructionCoeffs[m_nScalarFields+ifield][ir] += ((*datagi)*(*modevi)[j])*getRawCellVolume(rawIndex);
2858 }
2859 }
2860 }
2861 }
2862 }
2863
2864#if BITPIT_ENABLE_MPI
2865 for (std::size_t i = 0; i < m_nFields; ++i)
2866 MPI_Allreduce(MPI_IN_PLACE, m_reconstructionCoeffs[i].data(), m_nModes, MPI_DOUBLE, MPI_SUM, m_communicator);
2867#endif

2869 // Solve
2870 solveMinimization(m_reconstructionCoeffs);
2871
2872 // Sum field and mean
2873 sum(fields, m_mean, scalarIds, podscalarIds, vectorIds, podvectorIds, &m_listActiveIDs);
2874}
