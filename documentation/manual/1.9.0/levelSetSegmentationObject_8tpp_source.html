319template<typename narrow_band_cache_t>
320void LevelSetSegmentationObject<narrow_band_cache_t>::computeNarrowBand( LevelSetCartesianKernel *levelsetKernel, bool signd){
321
322 // Get mesh information
323 const VolCartesian &mesh = *(levelsetKernel->getMesh() ) ;
324 int meshDimension = mesh.getDimension();
325 VolCartesian::MemoryMode meshMemoryMode = mesh.getMemoryMode();
326
327 ElementType meshCellType = mesh.getCellType();
328 const ReferenceElementInfo &meshCellTypeInfo = ReferenceElementInfo::getInfo(meshCellType);
329 int meshCellFaceCount = meshCellTypeInfo.nFaces;
330
331 // Get surface information
332 const SurfUnstructured &surface = getSurface();
333
334 // Define search radius
335 //
336 // Search radius should be equal to the maximum between the narrow band
337 // size and the radius of the bounding sphere. This guarantees that, when
338 // the narrow band size is equal or less than zero, the levelset will be
339 // evaluated on the cells that intersect the surface and on all their
340 // first neighbours.
341 double searchRadius = std::max(this->m_narrowBandSize, 2 * levelsetKernel->getCellBoundingRadius());
342
343 // Define mesh bounding box
344 //
345 // The bounding box is inflated be the search radius.
346 std::array<double,3> meshMinPoint;
347 std::array<double,3> meshMaxPoint;
348 mesh.getBoundingBox(meshMinPoint, meshMaxPoint) ;
349 for (int d = 0; d < meshDimension; ++d) {
350 meshMinPoint[d] -= searchRadius;
351 meshMaxPoint[d] += searchRadius;
352 }
353
354 // Initialize process list
355 //
356 // Process list is initialized with cells that are certainly inside the
357 // narrow band. Those cells are the one that contain the vertices of the
358 // segments or the intersection between the segments and the bounding box
359 // of the patch.
360 std::unordered_set<long> processList;
361
362 std::vector<std::array<double,3>> intersectionPoints;
363 std::vector<std::array<double,3>> segmentVertexCoords;
364 for (const Cell &segment : surface.getCells()) {
365 // Get segment info
366 //
367 // Since vertex information will be passed to the CG module, we need to
368 // get the vertices in counter-clockwise order.
369 ConstProxyVector<long> segmentVertexIds = surface.getFacetOrderedVertexIds(segment);
370 std::size_t nSegmentVertices = segmentVertexIds.size();
371
372 // Get segment coordinates
373 segmentVertexCoords.resize(nSegmentVertices);
374 surface.getVertexCoords(nSegmentVertices, segmentVertexIds.data(), segmentVertexCoords.data());
375
376 // Add to the process list the cells that contain the vertices of the
377 // segment or the intersection between the segment and the bounding box
378 // of the patch.
379 int nInnerVertices = 0;
380 for (const std::array<double,3> &vertexPoint : segmentVertexCoords) {
381 long cellId = mesh.locatePoint(vertexPoint);
382 if (cellId < 0) {
383 continue;
384 }
385
386 processList.insert(cellId);
387 ++nInnerVertices;
388 }
389
390 if (nInnerVertices == 0) {
391 if (CGElem::intersectBoxPolygon(meshMinPoint, meshMaxPoint, segmentVertexCoords, false, true, true, intersectionPoints, meshDimension)) {
392 for (const std::array<double,3> &intersectionPoint : intersectionPoints){
393 long cellId = mesh.locateClosestCell(intersectionPoint);
394 processList.insert(cellId);
395 }
396 }
397 }
398 }
399
400 // Evaluate the levelset within the narrow band
401 //
402 // The initial process list is gradually expanded considering all the
403 // neighbours with a distance less than the search radius.
404 narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
405
406 std::unordered_set<long> outsideNarrowBand;
407 while (!processList.empty()) {
408 // Get the cell to process
409 long cellId = *(processList.begin());
410 processList.erase(processList.begin());
411
412 // Find segment associated to the cell
413 std::array<double,3> cellCentroid = levelsetKernel->computeCellCentroid(cellId);
414
415 long segmentId;
416 double distance;
417 getSearchTree().findPointClosestCell(cellCentroid, searchRadius, &segmentId, &distance);
418 if(segmentId < 0){
419 outsideNarrowBand.insert(cellId);
420 continue;
421 }
422
423 // Evaluate levelset information
424 std::array<double, 3> gradient;
425 std::array<double, 3> normal;
426 int error = getSegmentInfo(cellCentroid, segmentId, signd, distance, gradient, normal);
427 if (error) {
428 throw std::runtime_error ("Unable to extract the levelset information from segment " + std::to_string(segmentId) + ".");
429 }
430
431
432 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId, true) ;
433 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
434
435 // Add cell neighbours to the process list
436 if (meshMemoryMode == VolCartesian::MEMORY_LIGHT) {
437 for (int face = 0; face < meshCellFaceCount; ++face) {
438 long neighId = mesh.getCellFaceNeighsLinearId(cellId, face);
439 if (neighId >= 0) {
440 if (!this->isInNarrowBand(neighId) && (outsideNarrowBand.count(neighId) == 0)) {
441 processList.insert(neighId);
442 }
443 }
444 }
445 } else {
446 const Cell &cell = mesh.getCell(cellId);
447 const long *neighbours = cell.getAdjacencies() ;
448 int nNeighbours = cell.getAdjacencyCount() ;
449 for (int n = 0; n < nNeighbours; ++n) {
450 long neighId = neighbours[n];
451 if (!this->isInNarrowBand(neighId) && (outsideNarrowBand.count(neighId) == 0)) {
452 processList.insert(neighId);
453 }
454 }
455 }
456 }
457}

471template<typename narrow_band_cache_t>
472void LevelSetSegmentationObject<narrow_band_cache_t>::computeNarrowBand( LevelSetKernel *levelsetKernel, bool signd){
473
474 // Get mesh information
475 const VolumeKernel &mesh = *(levelsetKernel->getMesh()) ;
476
477 // Get narrowband information
478 narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
479
480 // Evaluate levelset information for intersected cells
481 VolumeKernel::CellConstIterator cellBegin = mesh.cellConstBegin();
482 VolumeKernel::CellConstIterator cellEnd = mesh.cellConstEnd();
483
484 std::unordered_set<std::size_t> intersectedRawCellIds;
485 for (VolumeKernel::CellConstIterator cellItr = cellBegin; cellItr != cellEnd; ++cellItr) {
486 // Identify the segment associated with the cell
487 //
488 // The search radius is evaluated as the maximum value between the
489 // narroband size and the distance above which the cell will surely
490 // not intersect the surface. In this way, cells that intersect the
491 // surface are always included in the narrow band, even if their
492 // distance from the surface is greater than the narrow band size
493 // explicitly set by the user.
494 //
495 // If no segment is identified the cell is not processed.
496 long cellId = cellItr.getId();
497 std::array<double,3> cellCentroid = levelsetKernel->computeCellCentroid(cellId);
498 double cellBoundingRadius = levelsetKernel->computeCellBoundingRadius(cellId);
499
500 double searchRadius = std::max(this->m_narrowBandSize, cellBoundingRadius);
501
502 long segmentId;
503 double distance;
504 getSearchTree().findPointClosestCell(cellCentroid, searchRadius, &segmentId, &distance);
505 if(segmentId < 0){
506 continue;
507 }
508
509 // Evaluate levelset information
510 std::array<double,3> gradient;
511 std::array<double,3> normal;
512 int error = getSegmentInfo(cellCentroid, segmentId, signd, distance, gradient, normal);
513 if (error) {
514 throw std::runtime_error ("Unable to extract the levelset information from segment " + std::to_string(segmentId) + ".");
515 }
516
517 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId, true) ;
518 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
519
520 // Update the list of cells that intersects the surface
521 //
522 // When the narrowband size is not explicitly set, the cell will always
523 // intersects the surface because only cells that intersect the surface
524 // are considered, otherwise we need to check if the absolute distance
525 // associated with the cell is lower than the intersection distance.
526 if (this->m_narrowBandSize < 0 || cellBoundingRadius < std::abs(distance)) {
527 std::size_t cellRawId = cellItr.getRawIndex();
528 intersectedRawCellIds.insert(cellRawId);
529 }
530
531 }
532
533 // Process the neighbours of the cells that intersect the surface
534 //
535 // If a cell intersects the surface, we need to evaluate the levelset
536 // of all its neigbours.
537 for (std::size_t cellRawId : intersectedRawCellIds) {
538 // Compute cell projection point
539 VolumeKernel::CellConstIterator cellItr = mesh.getCells().rawFind(cellRawId);
540 std::array<double,3> cellProjectionPoint = this->computeProjectionPoint(cellItr.getId());
541
542 // Process cell adjacencies
543 const long *neighbours = cellItr->getAdjacencies() ;
544 int nNeighbours = cellItr->getAdjacencyCount() ;
545 for (int n = 0; n < nNeighbours; ++n) {
546 // Skip the neighbour if it has already been processed
547 //
548 // The neighbour may already have been processed either because
549 // it distance from the segmentation is within the search radius,
550 // or because is a neighbour of an intersected cells already
551 // processed.
552 long neighId = neighbours[n];
553 if( narrowBandCache->contains(neighId) ){
554 continue;
555 }
556
557 // Identify the segment associated with the neighbour
558 std::array<double,3> neighCentroid = levelsetKernel->computeCellCentroid(neighId);
559
560 double searchRadius = 1.05 * norm2(neighCentroid - cellProjectionPoint);
561
562 long segmentId;
563 double distance;
564 getSearchTree().findPointClosestCell(neighCentroid, searchRadius, &segmentId, &distance);
565 if (segmentId < 0) {
566 assert(false && "Should not pass here");
567 }
568
569 // Evaluate negihbour leveset information
570 std::array<double,3> gradient;
571 std::array<double,3> normal;
572 int error = getSegmentInfo(neighCentroid, segmentId, signd, distance, gradient, normal);
573 if (error) {
574 throw std::runtime_error ("Unable to extract the levelset information from segment " + std::to_string(segmentId) + ".");
575 }
576
577 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(neighId, true) ;
578 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
579 }
580 }
581}

596template<typename narrow_band_cache_t>
597void LevelSetSegmentationObject<narrow_band_cache_t>::updateNarrowBand( LevelSetKernel *levelsetKernel, const std::vector<long> &cellIds, bool signd){
598
599 VolumeKernel &mesh = *(levelsetKernel->getMesh()) ;
600 narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
601
602 std::vector<long> cellsOutsideNarrowband;
603
604 // Evaluate the levelset of the cells
605 //
606 // When searching for the segment associated to a cell, the search radius
607 // is evaluated as the maximum value between the narroband size and the
608 // distance above which the cell will surely not intersect the surface.
609 for( long cellId : cellIds ){
610
611 // Identify the segment associated with the cell
612 //
613 // The search radius is evaluated as the maximum value between the
614 // narroband size and the distance above which the cell will surely
615 // not intersect the surface. In this way, cells that intersect the
616 // surface are always included in the narrow band, even if their
617 // distance from the surface is greater than the narrow band size
618 // explicitly set by the user.
619 //
620 // If no segment is identified the cell is not processed.
621 std::array<double,3> centroid = levelsetKernel->computeCellCentroid(cellId);
622
623 double searchRadius = std::max(this->m_narrowBandSize, levelsetKernel->computeCellBoundingRadius(cellId));
624
625 long segmentId;
626 double distance;
627 getSearchTree().findPointClosestCell(centroid, searchRadius, &segmentId, &distance);
628 if (segmentId < 0) {
629 cellsOutsideNarrowband.push_back(cellId);
630 continue;
631 }
632
633 // Evaluate levelset information
634 std::array<double,3> gradient;
635 std::array<double,3> normal;
636 int error = getSegmentInfo(centroid, segmentId, signd, distance, gradient, normal);
637 if (error) {
638 throw std::runtime_error ("Unable to extract the levelset information from segment " + std::to_string(segmentId) + ".");
639 }
640
641 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId, true) ;
642 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
643 }
644
645 // Cells with neighbours that intersect the surface need to be added to
646 // the narrowband even if they don't intersect the surface themself or
647 // have a distance from the surface greater than the narroband size.
648 for( long cellId : cellsOutsideNarrowband){
649 const Cell &cell = mesh.getCell(cellId);
650
651 // Consider only cells with a neighbour that intersects the surface
652 //
653 // Care must be take to use only information from cells inside the
654 // narrow band, that's because values outside the narrowband are not
655 // up-to-date at this stage.
656 const long *neighbours = cell.getAdjacencies() ;
657 int nNeighbours = cell.getAdjacencyCount() ;
658
659 long intersectedNeighId = Cell::NULL_ID;
660 for (int n = 0; n < nNeighbours; ++n) {
661 long neighId = neighbours[n];
662 if (!this->isInNarrowBand(neighId)) {
663 continue;
664 }
665
666 if( this->intersectSurface(neighId,LevelSetIntersectionMode::FAST_GUARANTEE_FALSE) == LevelSetIntersectionStatus::TRUE){
667 intersectedNeighId = neighId;
668 break;
669 }
670 }
671
672 if (intersectedNeighId == Cell::NULL_ID) {
673 continue;
674 }
675
676 // Identify the segment associated with the cell
677 std::array<double,3> cellCentroid = levelsetKernel->computeCellCentroid(cellId);
678 std::array<double,3> neighProjectionPoint = this->computeProjectionPoint(intersectedNeighId);
679
680 double searchRadius = 1.05 * norm2(cellCentroid - neighProjectionPoint);
681
682 long segmentId;
683 double distance;
684 getSearchTree().findPointClosestCell(cellCentroid, searchRadius, &segmentId, &distance);
685 if (segmentId < 0) {
686 assert(false && "Should not pass here");
687 continue;
688 }
689
690 // Evaluate levelset information for the cell
691 std::array<double,3> gradient;
692 std::array<double,3> normal;
693 int error = getSegmentInfo(cellCentroid, segmentId, signd, distance, gradient, normal);
694 if (error) {
695 throw std::runtime_error ("Unable to extract the levelset information from segment " + std::to_string(segmentId) + ".");
696 }
697
698 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId, true) ;
699 narrowBandCache->set(narrowBandCacheItr, distance, gradient, segmentId, normal);
700 }
