72  struct CellPositionLess
73  {
74  CellPositionLess(PatchKernel &patch, bool native = true)
75  : m_patch(patch), m_native(native)
76  {
77  }
78 
79  bool operator()(const long &id_1, const long &id_2) const
80  {
81  std::array<double, 3> centroid_1;
82  std::array<double, 3> centroid_2;
83  if (m_native) {
84  centroid_1 = m_patch.evalCellCentroid(id_1);
85  centroid_2 = m_patch.evalCellCentroid(id_2);
86  } else {
87  centroid_1 = m_patch.PatchKernel::evalCellCentroid(id_1);
88  centroid_2 = m_patch.PatchKernel::evalCellCentroid(id_2);
89  }
90 
91  for (int k = 0; k < 3; ++k) {
92  if (std::abs(centroid_1[k] - centroid_2[k]) <= m_patch.getTol()) {
93  continue;
94  }
95 
96  return centroid_1[k] < centroid_2[k];
97  }
98 
99  // If we are here the two cell centroids coincide. It's not
100  // possible to define an order for the two cells.
101  std::ostringstream stream;
102  stream << "It was not possible to define an order for cells " << id_1 << " and " << id_2 << ". ";
103  stream << "The two cells have the same centroid.";
104  throw std::runtime_error (stream.str());
105  }
106 
107  PatchKernel &m_patch;
108  bool m_native;
109  };
