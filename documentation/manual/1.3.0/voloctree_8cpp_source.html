331 void VolOctree::setBoundingBox()
332 {
333  std::array<double, 3> minPoint;
334  std::array<double, 3> maxPoint;
335 
336  // Get the bounding box from the tree
337  m_tree->getBoundingBox(minPoint, maxPoint);
338 
339  // The tree is only evaluating the bounding box of the internal octants,
340  // we need to consider also ghosts cells.
341  for (auto ghostItr = ghostBegin(); ghostItr != ghostEnd(); ++ghostItr) {
342  const long *ghostConnect = ghostItr->getConnect();
343  int nGhostVertices = ghostItr->getVertexCount();
344  for (int i = 0; i < nGhostVertices; ++i) {
345  const std::array<double, 3> coords = m_vertices[ghostConnect[i]].getCoords();
346  for (int d = 0; d < 3; ++d) {
347  minPoint[d] = std::min(coords[d], minPoint[d]);
348  maxPoint[d] = std::max(coords[d], maxPoint[d]);
349  }
350  }
351  }
352 
353  // Set the bounding box
354  setBoundingBox(minPoint, maxPoint);
355 }

625 const std::vector<adaption::Info> VolOctree::sync(bool updateOctantMaps, bool trackChanges)
626 {
627  log::cout() << ">> Syncing patch..." << std::endl;
628 
629  // Detect if we are in import-from-scratch mode
630  //
631  // In import-from-scratch mode we start form an empty patch and we need
632  // to import all the octants of the tree. If we are importing the tree
633  // from scratch there are no cells to delete/renumber.
634  bool importFromScratch = (getCellCount() == 0);
635 
636  // Last operation on the tree
637  ParaTree::Operation lastTreeOperation = m_tree->getLastOperation();
638  if (lastTreeOperation == ParaTree::OP_ADAPT_UNMAPPED && !importFromScratch) {
639  throw std::runtime_error ("Unable to sync the patch after an unmapped adaption");
640  }
641 
642  // Info on the tree
643  long nOctants = m_tree->getNumOctants();
644  long nPreviousOctants = m_octantToCell.size();
645 
646  log::cout() << ">> Number of octants : " << nOctants << std::endl;
647 
648  // Info on the tree
649  long nGhostsOctants = m_tree->getNumGhosts();
650  long nPreviousGhosts = m_ghostToCell.size();
651 
652  // Initialize tracking data
653  adaption::InfoCollection adaptionData;
654 
655  // Current rank
656  int currentRank = -1;
657 #if BITPIT_ENABLE_MPI==1
658  currentRank = getRank();
659 #endif
660 
661  // Extract information for transforming the patch
662  //
663  // If there are no cells in the mesh we need to import all
664  // octants.
665  log::cout() << ">> Extract information for transforming the patch...";
666 
667  std::vector<bool> unmappedOctants(nPreviousOctants, true);
668  std::vector<OctantInfo> addedOctants;
669  std::vector<RenumberInfo> renumberedOctants;
670  std::vector<DeleteInfo> deletedOctants;
671 
672  addedOctants.reserve(nOctants + nGhostsOctants);
673  if (!importFromScratch) {
674  renumberedOctants.reserve(nPreviousOctants + nPreviousGhosts);
675  deletedOctants.reserve(nPreviousOctants + nPreviousGhosts);
676  }
677 
678  uint32_t treeId = 0;
679  while (treeId < (uint32_t) nOctants) {
680  // Octant mapping
681  std::vector<uint32_t> mapper_octantMap;
682  std::vector<bool> mapper_ghostFlag;
683  std::vector<int> mapper_octantRank;
684  if (!importFromScratch) {
685  m_tree->getMapping(treeId, mapper_octantMap, mapper_ghostFlag, mapper_octantRank);
686  }
687 
688  // Adaption type
689  adaption::Type adaptionType = adaption::TYPE_NONE;
690  if (importFromScratch) {
691  adaptionType = adaption::TYPE_CREATION;
692  } else if (lastTreeOperation == ParaTree::OP_ADAPT_MAPPED) {
693  bool isNewR = m_tree->getIsNewR(treeId);
694  if (isNewR) {
695  adaptionType = adaption::TYPE_REFINEMENT;
696  } else {
697  bool isNewC = m_tree->getIsNewC(treeId);
698  if (isNewC) {
699  adaptionType = adaption::TYPE_COARSENING;
700  } else if (treeId != mapper_octantMap.front()) {
701  adaptionType = adaption::TYPE_RENUMBERING;
702  }
703  }
704 #if BITPIT_ENABLE_MPI==1
705  } else if (lastTreeOperation == ParaTree::OP_LOADBALANCE || lastTreeOperation == ParaTree::OP_LOADBALANCE_FIRST) {
706  if (currentRank != mapper_octantRank.front()) {
707  adaptionType = adaption::TYPE_PARTITION_RECV;
708  } else if (treeId != mapper_octantMap.front()) {
709  adaptionType = adaption::TYPE_RENUMBERING;
710  }
711 #endif
712  }
713 
714  // If the octant cell has not been modified we can skip to the next
715  // octant.
716  if (adaptionType == adaption::TYPE_NONE) {
717  unmappedOctants[treeId] = false;
718  ++treeId;
719  continue;
720  }
721 
722  // Re-numbered cells just need to be added to the proper list.
723  //
724  // Renumbered cells are not tracked, because the re-numbering
725  // only happens inside VolOctree.
726  if (adaptionType == adaption::TYPE_RENUMBERING) {
727  uint32_t previousTreeId = mapper_octantMap.front();
728  OctantInfo previousOctantInfo(previousTreeId, !mapper_ghostFlag.front());
729  long cellId = getOctantId(previousOctantInfo);
730  renumberedOctants.emplace_back(cellId, treeId);
731  unmappedOctants[previousTreeId] = false;
732 
733  // No more work needed, skip to the next octant
734  ++treeId;
735  continue;
736  }
737 
738  // Handle other kind of adaption
739  //
740  // New octants need to be imported into the patch,
741  // whereas cells associated to previous octants
742  // need to be removed.
743  //
744  // If the user want to track adaption, adaption
745  // data needs to be filled.
746 
747  // Current tree ids that will be imported
748  long nCurrentTreeIds;
749  if (importFromScratch) {
750  nCurrentTreeIds = nOctants - treeId;
751  } else if (adaptionType == adaption::TYPE_REFINEMENT) {
752  nCurrentTreeIds = pow(2, getDimension());
753  } else {
754  nCurrentTreeIds = 1;
755  }
756 
757  const long lastCurrentTreeId = treeId + nCurrentTreeIds;
758  for (int currentTreeId = treeId; currentTreeId < lastCurrentTreeId; ++currentTreeId) {
759  addedOctants.emplace_back(currentTreeId, true);
760  }
761 
762  // Cells that will be removed
763  //
764  // Mark the cells associated to previous local octants for deletion.
765  int nPreviousTreeIds;
766  if (!importFromScratch) {
767  nPreviousTreeIds = mapper_octantMap.size();
768  for (int k = 0; k < nPreviousTreeIds; ++k) {
769 #if BITPIT_ENABLE_MPI==1
770  // Only local cells can be deleted
771  if (mapper_octantRank[k] != currentRank) {
772  continue;
773  }
774 #endif
775 
776  // Mark previous octant for deletion
777  uint32_t previousTreeId = mapper_octantMap[k];
778  OctantInfo previousOctantInfo(previousTreeId, !mapper_ghostFlag[k]);
779  long cellId = getOctantId(previousOctantInfo);
780  deletedOctants.emplace_back(cellId, adaptionType);
781 
782  unmappedOctants[previousTreeId] = false;
783  }
784  } else {
785  nPreviousTreeIds = 0;
786  }
787 
788  // Adaption tracking
789  //
790  // The adaption info associated to the octants that has been received
791  // from external partitions will contain the current octants sorted by
792  // their tree id (we are looping over the octants in that order), this
793  // is the same order that will be used on the processor that has sent
794  // the octants. Since the order is the same, the two processors are able
795  // to exchange cell data without any additional extra communication
796  // (they already know the list of cells for which data is needed and
797  // the order in which these data will be sent).
798  if (trackChanges) {
799  // Rank assocated to the adaption info
800  int rank = currentRank;
801 #if BITPIT_ENABLE_MPI==1
802  if (adaptionType == adaption::TYPE_PARTITION_RECV) {
803  rank = mapper_octantRank[0];
804  }
805 #endif
806 
807  // Get the adaption info
808  std::size_t infoId = adaptionData.create(adaptionType, adaption::ENTITY_CELL, rank);
809  adaption::Info &adaptionInfo = adaptionData[infoId];
810 
811  // Current status
812  //
813  // We don't know the id of the current status, because those
814  // cells are not yet in the mesh. Store the trre id, and
815  // make the translation later.
816  //
817  // WARNING: tree id are uint32_t wherase adaptionInfo stores
818  // id as long.
819  adaptionInfo.current.reserve(nCurrentTreeIds);
820  auto addedOctantsIter = addedOctants.cend() - nCurrentTreeIds;
821  while (addedOctantsIter != addedOctants.cend()) {
822  adaptionInfo.current.emplace_back();
823  long &adaptionId = adaptionInfo.current.back();
824  adaptionId = (*addedOctantsIter).id;
825 
826  addedOctantsIter++;
827  }

830  //
831  // A coarsening can merge togheter cells of different processors.
832  // However, since the coarsening is limited to one level, the
833  // previous cells will always be internal or among the ghost of
834  // the current processor.
835  int nPreviousCellIds = mapper_octantMap.size();
836  adaptionInfo.previous.reserve(nPreviousCellIds);
837  for (int k = 0; k < nPreviousCellIds; ++k) {
838  long previousCellId;
839 #if BITPIT_ENABLE_MPI==1
840  if (mapper_octantRank[k] != currentRank) {
841  previousCellId = Cell::NULL_ID;
842  } else
843 #endif
844  {
845  OctantInfo previousOctantInfo(mapper_octantMap[k], !mapper_ghostFlag[k]);
846  previousCellId = getOctantId(previousOctantInfo);
847  }
848 
849  adaptionInfo.previous.emplace_back();
850  long &adaptionId = adaptionInfo.previous.back();
851  adaptionId = previousCellId;
852  }
853  }
854 
855  // Incremente tree id
856  treeId += nCurrentTreeIds;
857  }
858 
859  log::cout() << " Done" << std::endl;
860 
861 #if BITPIT_ENABLE_MPI==1
862  // New ghost octants need to be added
863  for (uint32_t treeId = 0; treeId < (uint32_t) nGhostsOctants; ++treeId) {
864  addedOctants.emplace_back(treeId, false);
865  }
866 #endif
867 
868  // Remove octants that are no more in the tree
869  if (!importFromScratch) {
870 #if BITPIT_ENABLE_MPI==1
871  // Cells that have been send to other processors need to be removed
872  std::unordered_map<int, std::array<uint32_t, 4>> sendOctants = m_tree->getSentIdx();
873  for (const auto &rankEntry : sendOctants) {
874  int rank = rankEntry.first;
875

877  if (rank == currentRank) {
878  deletionType = adaption::TYPE_DELETION;
879  } else {
880  deletionType = adaption::TYPE_PARTITION_SEND;
881  }
882 
883  for (int k = 0; k < 2; ++k) {
884  uint32_t beginTreeId = rankEntry.second[2 * k];
885  uint32_t endTreeId = rankEntry.second[2 * k + 1];
886  for (uint32_t treeId = beginTreeId; treeId < endTreeId; ++treeId) {
887  OctantInfo octantInfo(treeId, true);
888  long cellId = getOctantId(octantInfo);
889  deletedOctants.emplace_back(cellId, deletionType, rank);
890  unmappedOctants[treeId] = false;
891  }
892  }
893  }
894 
895  // Previous ghosts cells need to be removed
896  if (nPreviousGhosts > 0) {
897  clearGhostOwners(true);
898 
899  for (uint32_t ghostTreeId = 0; ghostTreeId < nPreviousGhosts; ++ghostTreeId) {
900  OctantInfo ghostOctantInfo(ghostTreeId, false);
901  long ghostId = getOctantId(ghostOctantInfo);
902  deletedOctants.emplace_back(ghostId, adaption::TYPE_DELETION);
903  }
904  }
905 #endif
906 
907  // Remove unmapped octants
908  //
909  // A coarsening that merges cells from different processors, can leave, on
910  // the processors which own the ghost octants involved in the coarsening,
911  // some octants that are not mapped.
912  for (uint32_t previousTreeId = 0; previousTreeId < nPreviousOctants; ++previousTreeId) {
913  if (unmappedOctants[previousTreeId]) {
914  OctantInfo octantInfo = OctantInfo(previousTreeId, true);
915  long cellId = getOctantId(octantInfo);
916  deletedOctants.emplace_back(cellId, adaption::TYPE_DELETION);
917  }
918  }
919  }

921  // Enable advanced editing
922  setExpert(true);
923 
924  // Reset cell-to-octant and octant-to-cell map
925  if (updateOctantMaps) {
926  log::cout() << ">> Resetting cell-to-octant and octant-to-cell maps...";
927 
928  updateCellOctantMaps(deletedOctants, renumberedOctants, addedOctants);
929  std::vector<RenumberInfo>().swap(renumberedOctants);
930 
931  log::cout() << " Done" << std::endl;
932  }
933

935  StitchInfo stitchInfo;
936  if (deletedOctants.size() > 0) {
937  log::cout() << ">> Removing non-existing cells...";
938 
939  // Track changes
940  //
941  // The adaption info associated to the octants that has been sent
942  // to external partitions will contain the current octants sorted by
943  // their tree id (they were added to the deleted octants list in that
944  // order), this is the same order that will be used on the processor
945  // that has received the octants. Since the order is the same, the two
946  // processors are able to exchange cell data without any additional
947  // extra communication (they already know the list of cells for which
948  // data is needed and the order in which these data will be sent).
949  if (trackChanges) {

951  std::unordered_set<long> sendAdaptionInfo;
952  std::unordered_set<long> removedInterfaces;
953  for (const DeleteInfo &deleteInfo : deletedOctants) {
954  // Cell id
955  long cellId = deleteInfo.cellId;
956 
957  // Adaption tracking
958  //
959  // Only cells deleted from a real deletion or a partition send
960  // needs to be tracked here, the other cells will be tracked
961  // where the adaption that deleted the cell is tracked.
962  adaption::Type adaptionType = deleteInfo.trigger;
963  bool adaptionIsDeletion = (adaptionType == adaption::TYPE_DELETION);
964  bool adaptionIsSend = (adaptionType == adaption::TYPE_PARTITION_SEND);
965 
966  if (adaptionIsDeletion || adaptionIsSend) {
967  int rank = deleteInfo.rank;
968  std::size_t adaptionInfoId = adaptionData.create(adaptionType, adaption::ENTITY_CELL, rank);
969  adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
970  adaptionInfo.previous.emplace_back();
971  long &deletedId = adaptionInfo.previous.back();
972  deletedId = cellId;
973 
974  // Keep track of adaption info for the send cells
975  if (adaptionIsSend) {
976  sendAdaptionInfo.insert(adaptionInfoId);
977  }
978  }
979 
980  // List of deleted interfaces
981  const Cell &cell = m_cells.at(cellId);
982  long nCellInterfaces = cell.getInterfaceCount();
983  const long *interfaces = cell.getInterfaces();
984  for (int k = 0; k < nCellInterfaces; ++k) {
985  long interfaceId = interfaces[k];
986  if (interfaceId >= 0) {
987  removedInterfaces.insert(interfaceId);
988  }
989  }
990  }
991 
992 #if BITPIT_ENABLE_MPI==1
993  // Sort sent cells
994  //
995  // We cannot use native functions to evaluate the position of the
996  // cells because the octants associated to the cells no longer
997  // exist on the octree. The cells are still there, therefore we
998  // can evaluate the cell positions using generic patch functions.
999  for (int adaptionInfoId : sendAdaptionInfo) {
1000  adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
1001  std::sort(adaptionInfo.previous.begin(), adaptionInfo.previous.end(), CellPositionLess(*this, false));
1002  }
1003 #endif
1004 
1005  // Adaption info for the deleted interfaces
1006  std::size_t adaptionInfoId = adaptionData.create(adaption::TYPE_DELETION, adaption::ENTITY_INTERFACE, currentRank);
1007  adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
1008  for (const long &interfaceId : removedInterfaces) {
1009  adaptionInfo.previous.emplace_back();
1010  long &deletedInterfaceId = adaptionInfo.previous.back();
1011  deletedInterfaceId = interfaceId;
1012  }
1013  }
1014 
1015  // Delete cells
1016  stitchInfo = deleteCells(deletedOctants);
1017 
1018  log::cout() << " Done" << std::endl;
1019  log::cout() << ">> Cells removed: " << deletedOctants.size() << std::endl;
1020  }
1021 
1022  std::vector<DeleteInfo>().swap(deletedOctants);
1023 
1024  // Import added octants
1025  std::vector<long> createdCells;
1026  if (addedOctants.size() > 0) {
1027  log::cout() << ">> Importing new octants...";
1028 
1029  createdCells = importCells(addedOctants, stitchInfo);
1030 
1031  log::cout() << " Done" << std::endl;
1032  log::cout() << ">> Octants imported: " << addedOctants.size() << std::endl;
1033  }
1034 
1035  StitchInfo().swap(stitchInfo);
1036 
1037  // Rebuild the ghost information
1038 #if BITPIT_ENABLE_MPI==1
1039  buildGhostExchangeData();
1040 #endif
1041 
1042  // Disable advanced editing
1043  setExpert(false);
1044 
1045  // Track mesh adaption
1046  if (trackChanges) {
1047  // Complete mesh adaption info for the cells
1048  for (auto &adaptionInfo : adaptionData.data()) {
1049  if (adaptionInfo.entity != adaption::ENTITY_CELL) {
1050  continue;
1051  }
1052 
1053  // Map ids of the added cells
1054  int nCurrentIds = adaptionInfo.current.size();
1055  for (int k = 0; k < nCurrentIds; ++k) {
1056  long cellId = m_octantToCell.at(adaptionInfo.current[k]);
1057  adaptionInfo.current[k] = cellId;
1058  }
1059 
1060 #if BITPIT_ENABLE_MPI==1
1061  // Sort received cells
1062  //
1063  // To match the sorting done on the procesor that sent the cells,
1064  // we don't use the native functions to evaluate the position of
1065  // the cells.
1066  adaption::Type adaptionType = adaptionInfo.type;
1067  if (adaptionType == adaption::TYPE_PARTITION_RECV) {
1068  std::sort(adaptionInfo.current.begin(), adaptionInfo.current.end(), CellPositionLess(*this, false));
1069  }
1070 #endif
1071  }
1072 
1073  // Track created ghosts cells
1074 #if BITPIT_ENABLE_MPI==1
1075  if (nGhostsOctants > 0) {
1076  std::size_t adaptionInfoId = adaptionData.create(adaption::TYPE_CREATION, adaption::ENTITY_CELL, currentRank);
1077  adaption::Info &adaptionInfo = adaptionData[adaptionInfoId];
1078 
1079  adaptionInfo.current.reserve(nGhostsOctants);
1080  auto cellIterator = m_cellToGhost.cbegin();
1081  while (cellIterator != m_cellToGhost.cend()) {
1082  adaptionInfo.current.emplace_back();
1083  long &adaptionId = adaptionInfo.current.back();
1084  adaptionId = cellIterator->first;
1085 
1086  cellIterator++;
1087  }
1088  }
1089 #endif
1090 
1091  // Track created interfaces
1092  if (createdCells.size() > 0) {
1093  // List of unique interfaces that have been created
1094  std::unordered_set<long> createdInterfaces;
1095  for (const auto &cellId : createdCells) {
1096  const Cell &cell = m_cells.at(cellId);
1097  long nCellInterfaces = cell.getInterfaceCount();
1098  const long *interfaces = cell.getInterfaces();
1099  for (int k = 0; k < nCellInterfaces; ++k) {
1100  long interfaceId = interfaces[k];
1101  if (interfaceId >= 0) {
1102  createdInterfaces.insert(interfaceId);
1103  }
1104  }
1105  }
1106 
1107  // Adaption info
1108  std::size_t infoId = adaptionData.create(adaption::TYPE_CREATION, adaption::ENTITY_INTERFACE, currentRank);
1109  adaption::Info &adaptionInfo = adaptionData[infoId];
1110  for (const long &interfaceId : createdInterfaces) {
1111  adaptionInfo.current.emplace_back();
1112  long &createdInterfaceId = adaptionInfo.current.back();
1113  createdInterfaceId = interfaceId;
1114  }
1115  }
1116  }
1117 
1118  // Done
1119  return adaptionData.dump();
1120 }

1282 VolOctree::StitchInfo VolOctree::deleteCells(std::vector<DeleteInfo> &deletedOctants)
1283 {
1284  // Info of the cells
1285  int nCellVertices = m_cellTypeInfo->nVertices;
1286  const std::vector<std::vector<int>> &cellLocalFaceConnect = m_cellTypeInfo->faceConnect;
1287 
1288  // Info on the faces
1289  const int &nInterfaceVertices = m_interfaceTypeInfo->nVertices;
1290 
1291  // List of cells ot delete
1292  std::unordered_set<long> deadCells;
1293  deadCells.reserve(deletedOctants.size());
1294  for (const DeleteInfo &deleteInfo : deletedOctants) {
1295  deadCells.insert(deleteInfo.cellId);
1296  }
1297 
1298  // Delete the cells
1299  std::unordered_set<long> deadVertices;
1300  std::unordered_set<long> deadInterfaces;
1301  std::unordered_set<long> danglingCells;
1302  for (long cellId : deadCells) {
1303  const Cell &cell = m_cells[cellId];
1304 
1305  // List vertices to remove
1306  //
1307  // For now, all cell vertices will be listed. Later, the vertex of
1308  // the dangling faces will be removed from the list.
1309  int nCellVertices = cell.getVertexCount();
1310  for (int k = 0; k < nCellVertices; ++k) {
1311  long vertexId = cell.getVertex(k);
1312  deadVertices.insert(vertexId);
1313  }
1314 
1315  // List of interfaces to delete
1316  //
1317  // All the interfaces of the cell will be deleted, this means that the
1318  // neighbours that are not deleted will have a face not connected to
1319  // anything. Those faces are called dangling faces and a cell with
1320  // dangling faces is called dangling cell.
1321  int nCellInterfaces = cell.getInterfaceCount();
1322  const long *interfaces = cell.getInterfaces();
1323  for (int k = 0; k < nCellInterfaces; ++k) {
1324  long interfaceId = interfaces[k];
1325  if (interfaceId < 0) {
1326  continue;
1327  }
1328 
1329  // Interfaces has to be considered just once
1330  if (deadInterfaces.count(interfaceId) > 0) {
1331  continue;
1332  }
1333 
1334  // Find if the face associated to the interface will be dangling
1335  Interface &interface = m_interfaces[interfaceId];
1336 
1337  int danglingSide = -1;
1338  if (!interface.isBorder()) {
1339  if (deadCells.count(interface.getOwner()) == 0) {
1340  danglingSide = 0;
1341  } else if (deadCells.count(interface.getNeigh()) == 0) {
1342  danglingSide = 1;
1343  }
1344  }
1345 
1346  // Handle dangling faces
1347  if (danglingSide >= 0) {
1348  // Info on the dangling face
1349  long danglingCellId;
1350  long danglingNeighId;
1351  int danglingCellFace;
1352  if (danglingSide == 0) {
1353  danglingCellId = interface.getOwner();
1354  danglingNeighId = interface.getNeigh();
1355  danglingCellFace = interface.getOwnerFace();
1356  } else {
1357  danglingCellId = interface.getNeigh();
1358  danglingNeighId = interface.getOwner();
1359  danglingCellFace = interface.getNeighFace();
1360  }
1361 
1362  Cell &danglingCell = m_cells[danglingCellId];
1363  danglingCells.insert(danglingCellId);
1364 
1365  // Since the dangling cell will not be deleted, we have to
1366  // updated its interface and adjacency data structures.
1367  int cellInterfaceIndex = danglingCell.findInterface(danglingCellFace, interfaceId);
1368  danglingCell.deleteInterface(danglingCellFace, cellInterfaceIndex);
1369 
1370  int cellAdjacencyIndex = danglingCell.findAdjacency(danglingCellFace, danglingNeighId);
1371  danglingCell.deleteAdjacency(danglingCellFace, cellAdjacencyIndex);
1372  }
1373 
1374  // Add the interface to the list of interfaces to delete
1375  deadInterfaces.insert(interfaceId);
1376  }
1377 
1378  // Delete cell
1379  deleteCell(cellId, false, true);
1380  }
1381 
1382  m_cells.flush();
1383 
1384  // Delete the interfaces
1385  for (long interfaceId : deadInterfaces) {
1386  deleteInterface(interfaceId, false, true);
1387  }
1388 
1389  m_interfaces.flush();
1390 
1391  // All the vertices belonging to the dangling cells has to be kept
1392  //
1393  // The vertices of the dangling faces need to be kept because there are
1394  // still cells using them. However it's not enough to consider only the
1395  // vertices on the dangling faces, we have to consider the vertices of
1396  // the whole cell. That's because we may need to keep vertices on the
1397  // edges of the cell, and those vertices may not be on interfaces of the
1398  // dangling faces.
1399  //
1400  // Morover we need to build a map between the patch numbering and the
1401  // octree numbering of the vertices of the dangling cells. This map will
1402  // be used when imprting the octants to stitch the imported octants to
1403  // the existing cells.
1404  StitchInfo stitchVertices;
1405  for (const long cellId : danglingCells) {
1406  // Vertices of the cell
1407  const Cell &cell = m_cells[cellId];
1408  const long *cellConnect = cell.getConnect();
1409 
1410  OctantInfo octantInfo = getCellOctant(cellId);
1411  Octant *octant = getOctantPointer(octantInfo);
1412 
1413  for (int k = 0; k < nCellVertices; ++k) {
1414  long vertexId = cellConnect[k];
1415  uint64_t vertexTreeMorton = m_tree->getNodeMorton(octant, k);
1416  stitchVertices.insert({vertexTreeMorton, vertexId});
1417  deadVertices.erase(vertexId);
1418  }
1419 
1420  // Vertices of all other interfaces left of the cell
1421  int nCellInterfaces = cell.getInterfaceCount();
1422  const long *interfaces = cell.getInterfaces();
1423  for (int k = 0; k < nCellInterfaces; ++k) {
1424  long interfaceId = interfaces[k];
1425  if (interfaceId < 0) {
1426  continue;
1427  }
1428 
1429  const Interface &interface = m_interfaces[interfaceId];
1430  if (interface.isBorder()) {
1431  continue;
1432  }
1433 
1434  long ownerId = interface.getOwner();
1435  int ownerFace = interface.getOwnerFace();
1436 
1437  const Cell &ownerCell = m_cells[ownerId];
1438  const long *ownerCellConnect = ownerCell.getConnect();
1439 
1440  OctantInfo ownerOctantInfo = getCellOctant(ownerId);
1441  Octant *ownerOctant = getOctantPointer(ownerOctantInfo);
1442 
1443  const std::vector<int> &localFaceConnect = cellLocalFaceConnect[ownerFace];
1444  for (int k = 0; k < nInterfaceVertices; ++k) {
1445  long vertexId = ownerCellConnect[localFaceConnect[k]];
1446  uint64_t vertexTreeMorton = m_tree->getNodeMorton(ownerOctant, localFaceConnect[k]);
1447  stitchVertices.insert({vertexTreeMorton, vertexId});
1448  deadVertices.erase(vertexId);
1449  }
1450  }
1451  }
1452 
1453  // Delete the vertices
1454  for (long vertexId : deadVertices) {
1455  deleteVertex(vertexId, true);
1456  }
1457 
1458  m_vertices.flush();
1459 
1460  // Done
1461  return stitchVertices;
