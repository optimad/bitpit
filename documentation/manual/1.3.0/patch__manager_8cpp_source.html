69 int PatchManager::registerPatch(PatchKernel *patch, int id)
70 {
71  if (id >= 0) {
72  if (m_idGenerator.isIdAssigned(id)) {
73  throw std::runtime_error ("A patch with the same id already exists");
74  }
75  } else {
76  id = m_idGenerator.generateId();
77  }
78 
79  patch->setId(id);
80  m_patchIds[patch] = id;
81  m_patchOrder.push_back(patch);
82 
83  return id;
84 }

91 void PatchManager::unregisterPatch(PatchKernel *patch)
92 {
93  auto iterator = m_patchIds.find(patch);
94  if (iterator == m_patchIds.end()) {
95  throw std::runtime_error ("The patch to be unregistered does not exist");
96  }
97 
98  int id = iterator->second;
99  m_idGenerator.trashId(id);
100 
101  m_patchIds.erase(iterator);
102  for (auto itr = m_patchOrder.begin(); itr != m_patchOrder.end(); ++itr) {
103  if (*itr == patch) {
104  m_patchOrder.erase(itr);
105  break;
106  }
107  }
108 }
