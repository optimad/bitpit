148 void LevelSetSegmentation::setSegmentation( SurfUnstructured *segmentation){
149 
150  std::vector<std::array<double,3>> vertexNormal, vertexGradient ;
151 
152  m_segmentation = segmentation;
153  m_dimension = m_segmentation->getSpaceDimension() ;
154 
155  int i, nV;
156  long segId ;
157  double norm, tol = m_segmentation->getTol() ;
158 
159  for( auto & segment : m_segmentation->getCells() ){
160 
161  segId = segment.getId() ;
162  nV = segment.getVertexCount() ;
163 
164  vertexNormal.resize(nV) ;
165  vertexNormal.shrink_to_fit() ;
166 
167  vertexGradient.resize(nV) ;
168  vertexGradient.shrink_to_fit() ;
169 
170  for(i=0; i<nV; ++i){
171  vertexGradient[i] = m_segmentation->evalVertexNormal(segId,i) ;
172  vertexNormal[i] = m_segmentation->evalLimitedVertexNormal(segId,i,m_featureAngle) ;
173  }
174 
175  m_vertexGradient.insert({{segId,vertexGradient}}) ;
176 
177  norm = 0. ;
178  for(i=0; i<nV; ++i){
179  norm += norm2(vertexGradient[i] - vertexNormal[i]) ;
180  }
181 
182  if( norm >= tol ){
183  m_vertexNormal.insert({{segId,vertexNormal}}) ;
184  }
185 
186  }

372 std::unordered_set<long> LevelSetSegmentation::createSegmentInfo( LevelSetKernel *visitee,
373  const double &search,
374  SegmentToCellMap &segmentToCellMap ){
375 
376  log::cout() << " Creating segment info... " << std::endl;
377 
378  // Count the number of segments that will be processed for each cell
379  std::unordered_map<long, int> nSegmentsPerCell;
380  for ( auto &entry : segmentToCellMap ) {
381  const std::vector<long> &cellList = entry.second ;
382 
383  for ( long cell : cellList ) {
384  auto countItr = nSegmentsPerCell.find( cell );
385  if ( countItr == nSegmentsPerCell.end() ) {
386  countItr = nSegmentsPerCell.insert( { cell, 0 } ).first;
387  }
388 
389  *(countItr)++;
390  }
391  }
392 
393  // Create the needed dat strucures
394  std::unordered_set<long> newSegInfo ;
395  std::vector<std::array<double,3>> cloud ;
396 
397  // Add the segments info
398  for ( auto mapItr = segmentToCellMap.begin(); mapItr != segmentToCellMap.end(); ) {
399  // Segment data
400  long segment = mapItr->first ;
401  std::vector<std::array<double,3>> VS = getSimplexVertices( segment ) ;
402 
403  // Cell data
404  const std::vector<long> &cellList = mapItr->second ;
405  size_t cellListCount = cellList.size();
406 
407  cloud.resize( cellListCount ) ;
408  for ( size_t k = 0; k < cellListCount; ++k ) {
409  cloud[k] = visitee->computeCellCentroid( cellList[k] ) ;
410  }
411 
412  // Eval distances
413  std::vector<double> distancesFromSegment = CGElem::distanceCloudSimplex( cloud, VS );
414  for ( size_t k = 0; k < cellListCount; ++k ) {
415  // Discard segments with a distance greater than the narrow band
416  double segmentDistance = distancesFromSegment[k];
417  if ( segmentDistance > search ) {
418  continue ;
419  }
420 
421  // Get or create the segment's info
422  long cell = cellList[k] ;
423  auto segInfoItr = m_seg.find( cell ) ;
424  if ( segInfoItr == m_seg.end() ) {
425  int segmentCount = nSegmentsPerCell.at(cell);
426  segInfoItr = m_seg.emplace( cell );
427  segInfoItr->segments.reserve( segmentCount );
428  segInfoItr->distances.reserve( segmentCount );
429 
430  newSegInfo.insert( cell ) ;
431  }
432 
433  // Add the segment info
434  segInfoItr->segments.push_back(segment) ;
435  segInfoItr->distances.push_back(segmentDistance) ;
436  }
437 
438  mapItr = segmentToCellMap.erase( mapItr );
439  }
440 
441  // Order the segments from the closes to the farthest from the body
442  std::vector<size_t> distanceRank;
443  std::vector<size_t> segmentRank;
444  for ( long id : newSegInfo ) {
445  auto segInfoItr = m_seg.find( id ) ;
446  std::vector<long> &segments = segInfoItr->segments;
447  size_t nSegments = segments.size();
448 
449  // Evaluate ranks
450  distanceRank.resize(nSegments);
451  for (size_t k = 0; k < nSegments; ++k) {
452  distanceRank[k] = k;
453  }
454 
455  std::vector<double> &distances = segInfoItr->distances;
456  std::sort(distanceRank.begin(), distanceRank.end(), DistanceComparator(distances) ) ;
457 
458  segmentRank.resize(nSegments);
459  for (size_t k = 0; k < nSegments; ++k) {
460  segmentRank[k] = distanceRank[k];
461  }
462 
463  // Order vectors
464  utils::reorderVector(distanceRank, distances, nSegments) ;
465  distances.shrink_to_fit();
466 
467  utils::reorderVector(segmentRank, segments, nSegments) ;
468  segments.shrink_to_fit();
469  }
470 
471  return newSegInfo ;
472 }

480 void LevelSetSegmentation::updateSegmentList( const double &search) {
481 
482 
483  log::cout() << " Updating segment list for cells inside narrow band... " << std::endl;
484 
485  bitpit::PiercedVector<SegInfo>::iterator segEnd = m_seg.end();
486  for (bitpit::PiercedVector<SegInfo>::iterator segItr = m_seg.begin(); segItr != segEnd; ++segItr) {
487  // The filter radius is the narrow band size plus the cell circumcenter.
488  long cellId = segItr.getId() ;
489  double filterRadius = search + m_kernelPtr->computeCellCircumcircle(cellId);
490 
491  // Starting from the farthest segment (the last in the list) we loop
492  // backwards until we find the first segment with a distance less
493  // that the specified narrow band size.
494  //
495  // We know that the cell is inside the narrow band, because all the
496  // cells outside the narrow band have already been removed. Therefore
497  // we need to perform the check up to the second segment (the first one
498  // is in the narrow band).
499  std::vector<double> &distances = segItr->distances;
500  size_t nCurrentSegments = distances.size();
501 
502  size_t nSegmentsToKeep = 1;
503  for( size_t k = nCurrentSegments - 1; k >= 1; --k) {
504  double distance = distances[k];
505  if ( distance <= filterRadius ){
506  nSegmentsToKeep = k + 1;
507  break;
508  }
509  }
510 
511  if (nSegmentsToKeep != nCurrentSegments) {
512  distances.resize(nSegmentsToKeep);
513  distances.shrink_to_fit();
514 
515  std::vector<long> &segments = segItr->segments;
516  segments.resize(nSegmentsToKeep);
517  segments.shrink_to_fit();
518  }
519  }
520 }

570 void LevelSetSegmentation::infoFromSimplex( const std::array<double,3> &p, const long &i, double &d, double &s, std::array<double,3> &x, std::array<double,3> &n ) const {
571 
572  std::array<double,3> g ;
573 
574  Cell &cell = m_segmentation->getCell(i) ;
575  int nV = cell.getVertexCount() ;
576 
577  auto itrNormal = m_vertexNormal.find(i) ;
578  auto itrGradient = m_vertexGradient.find(i) ;
579  assert( itrGradient != m_vertexGradient.end() ) ;
580 
581  if( nV == 1){
582  long id = cell.getVertex(0) ;
583  d = norm2( p- m_segmentation->getVertexCoords(id) ) ;
584  g.fill(0.) ;
585  n.fill(0.) ;
586 
587  } else if( nV == 2){
588  long id0 = cell.getVertex(0) ;
589  long id1 = cell.getVertex(1) ;
590 
591  std::array<double,2> lambda ;
592  int flag ;
593 
594  d= CGElem::distancePointSegment( p, m_segmentation->getVertexCoords(id0), m_segmentation->getVertexCoords(id1), x, lambda, flag ) ;
595 
596  g = p-x;
597  g /= norm2(g);
598 
599  n = lambda[0] *itrGradient->second[0] ;
600  n += lambda[1] *itrGradient->second[1] ;
601  n /= norm2(n) ;
602 
603  g *= sign(dotProduct(g,n));
604 
605  if( itrNormal != m_vertexNormal.end() ){
606  n = lambda[0] *itrNormal->second[0] ;
607  n += lambda[1] *itrNormal->second[1] ;
608  n /= norm2(n) ;
609 
610  double kappa ;
611  maxval(lambda,kappa);
612  kappa = 1. -kappa;
613 
614  n *= kappa;
615  n += (1.-kappa)*g;
616  n /= norm2(n);
617 
618  }
619 
620 
621  } else if (nV == 3){
622  long id0 = cell.getVertex(0) ;
623  long id1 = cell.getVertex(1) ;
624  long id2 = cell.getVertex(2) ;
625 
626  std::array<double,3> lambda ;
627  int flag ;
628 
629  d= CGElem::distancePointTriangle( p, m_segmentation->getVertexCoords(id0), m_segmentation->getVertexCoords(id1), m_segmentation->getVertexCoords(id2), x, lambda, flag ) ;
630 
631  g = p-x;
632  g /= norm2(g);
633 
634  n = lambda[0] *itrGradient->second[0] ;
635  n += lambda[1] *itrGradient->second[1] ;
636  n += lambda[2] *itrGradient->second[2] ;
637 
638  g *= sign(dotProduct(g,n));
639 
640  if( itrNormal != m_vertexNormal.end() ){
641  n = lambda[0] *itrNormal->second[0] ;
642  n += lambda[1] *itrNormal->second[1] ;
643  n += lambda[2] *itrNormal->second[2] ;
644  n /= norm2(n) ;
645 
646  double kappa ;
647  maxval(lambda,kappa);
648  kappa = 1. -kappa;
649 
650  n *= kappa;
651  n += (1.-kappa)*g;
652  n /= norm2(n);
653 
654  }
655 
656 
657 
658  } else{
659  log::cout() << " simplex not supported in LevelSetSegmentation::infoFromSimplex " << nV << std::endl ;
660 
661  }
662 
663  s = sign( dotProduct(g, p - x) );
664 
665 }

673 bool LevelSetSegmentation::seedNarrowBand( LevelSetCartesian *visitee, std::vector<std::array<double,3>> &VS, std::vector<int> &I){
674 
675  VolCartesian &mesh = *(static_cast<VolCartesian*>(visitee->getMesh()));
676 
677  bool found(false) ;
678  int dim( mesh.getDimension() ) ;
679  std::array<double,3> B0, B1;
680  std::vector<std::array<double,3>> VP ;
681 
682  mesh.getBoundingBox(B0, B1) ;
683 
684  for( int i=0; i<dim; ++i){
685  B0[i] -= getSizeNarrowBand() ;
686  B1[i] += getSizeNarrowBand() ;
687  }
688 
689  I.clear() ;
690 
691  for( const auto &P : VS){
692  if( CGElem::intersectPointBox( P, B0, B1, dim ) ) {
693  I.push_back( mesh.locateClosestCell(P) );
694  found = true ;
695  }
696  }
697 
698  if( !found && CGElem::intersectBoxSimplex( B0, B1, VS, VP, dim ) ) {
699  for( const auto &P : VP){
700  I.push_back( mesh.locateClosestCell(P) );
701  found = true ;
702  }
703  }
704 
705  return found ;
706 }

802 LevelSetSegmentation::SegmentToCellMap LevelSetSegmentation::extractSegmentToCellMap( LevelSetCartesian *visitee, const double &RSearch ){
803 
804  VolumeKernel &mesh = *(visitee->getMesh() ) ;
805  std::vector<std::array<double,3>> VS(3);
806 
807  std::vector< int > stack, temp ;
808  std::vector< std::array<double,3> > cloud ;
809 
810  std::vector<double> d;
811  std::vector<double>::iterator vit;
812 
813  std::vector<int> flag( mesh.getCellCount(), -1);
814 
815  int i, N( m_segmentation->getCellCount() );
816 
817  std::vector<long> neighs ;
818 
819  stack.reserve(128) ;
820  temp.reserve(128) ;
821 
822  SegmentToCellMap segmentToCellMap;
823  segmentToCellMap.reserve(m_segmentation->getCellCount());
824 
825  log::cout() << " Extracting segment-to-cell map from octree patch... " << std::endl;
826 
827  // --------------------------------------------------------------------------
828  // COMPUTE THE SDF VALUE AT EACH MESH POINT //
829  //
830  for (i = 0; i < N; i++) {
831 
832  std::vector<long> &cellList = segmentToCellMap[i] ;
833 
834  // Segments vertex ------------------------------------------------------ //
835  VS = getSimplexVertices( i ) ;
836  seedNarrowBand( visitee, VS, stack );
837 
838 
839  //-----------------------------------------------------------------
840  //Propagate from seed
841  size_t stackSize = stack.size();
842  while (stackSize > 0) {
843 
844  // Extract point from lifo
