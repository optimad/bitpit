274 void VolCartesian::setDiscretization(const std::array<int, 3> &nCells)
275 {
276  // Spacing
277  for (int n = 0; n < getDimension(); ++n) {
278  // Initialize cells
279  m_nCells1D[n] = nCells[n];
280  m_cellSpacings[n] = (m_maxCoords[n] - m_minCoords[n]) / m_nCells1D[n];
281 
282  m_cellCenters[n].resize(m_nCells1D[n]);
283  for (int i = 0; i < m_nCells1D[n]; i++) {
284  m_cellCenters[n][i] = m_minCoords[n] + (0.5 + i) * m_cellSpacings[n];
285  }
286 
287  log::cout() << " - Cell count along direction " << n << " : " << m_nCells1D[n] << "\n";
288 
289  // Initialize vertices
290  m_nVertices1D[n] = m_nCells1D[n] + 1;
291 
292  m_vertexCoords[n].resize(m_nVertices1D[n]);
293  for (int i = 0; i < m_nVertices1D[n]; i++) {
294  m_vertexCoords[n][i] = m_minCoords[n] + i * m_cellSpacings[n];
295  }
296  }
297 
298  if (!isThreeDimensional()) {
299  m_nCells1D[Vertex::COORD_Z] = 0;
300  m_cellSpacings[Vertex::COORD_Z] = 0.;
301 
302  m_nVertices1D[Vertex::COORD_Z] = 0;
303  }
304 
305  log::cout() << std::endl;
306 
307  // Count the total number of vertices
308  m_nVertices = 1;
309  for (int n = 0; n < getDimension(); ++n) {
310  m_nVertices *= m_nVertices1D[n];
311  }
312  log::cout() << " - Total vertex count: " << m_nVertices << "\n";
313 
314  // Count the total number of cells
315  m_nCells = 1;
316  for (int n = 0; n < getDimension(); ++n) {
317  m_nCells *= m_nCells1D[n];
318  }
319  log::cout() << " - Total cell count: " << m_nCells << "\n";
320 
321  // Count the total number of interfaces
322  m_nInterfaces = 0;
323  for (int n = 0; n < getDimension(); n++) {
324  std::array<int, 3> interfaceCount1D = getInterfaceCountDirection(n);
325 
326  int nDirectionInterfaces = 1;
327  for (int n = 0; n < getDimension(); n++) {
328  nDirectionInterfaces *= interfaceCount1D[n];
329  }
330  m_nInterfaces += nDirectionInterfaces;
331  }
332  log::cout() << " - Total interface count: " << m_nInterfaces << "\n";
333 
334  // Cell volume
335  initializeCellVolume();
336 
337  // Interface area
338  initializeInterfaceArea();
339 
340  // Create cells, vertices and interfaces
341  if (getMemoryMode() == MemoryMode::MEMORY_NORMAL) {
342  update();
343  }
344 }

566 const std::vector<adaption::Info> VolCartesian::_updateAdaption(bool trackAdaption)
567 {
568  log::cout() << ">> Updating cartesian mesh\n";
569 
570  // Enable advanced editing
571  setExpert(true);
572 
573  // Reset the mesh
574  //
575  // We need to reset only the generic data of the patch, therefore we call
576  // the 'reset' implementation of the kernel.
577  VolumeKernel::reset();
578 
579  // Definition of the mesh
580  addVertices();
581  addCells();
582  addInterfaces();
583 
584  // Disable advanced editing
585  setExpert(false);
586 
587  // Adaption info
588  std::vector<adaption::Info> adaptionData;
589  if (trackAdaption) {
590  adaptionData.emplace_back();
591  adaption::Info &adaptionCellInfo = adaptionData.back();
592  adaptionCellInfo.type = adaption::TYPE_CREATION;
593  adaptionCellInfo.entity = adaption::ENTITY_CELL;
594  adaptionCellInfo.current.reserve(m_cells.size());
595  for (auto &cell : m_cells) {
596  adaptionCellInfo.current.emplace_back();
597  long &cellId = adaptionCellInfo.current.back();
598  cellId = cell.getId();
599  }
600 
601  adaptionData.emplace_back();
602  adaption::Info &adaptionInterfaceInfo = adaptionData.back();
603  adaptionInterfaceInfo.type = adaption::TYPE_CREATION;
604  adaptionInterfaceInfo.entity = adaption::ENTITY_INTERFACE;
605  adaptionInterfaceInfo.current.reserve(m_interfaces.size());
606  for (auto &interface : m_interfaces) {
607  adaptionInterfaceInfo.current.emplace_back();
608  long &interfaceId = adaptionInterfaceInfo.current.back();
609  interfaceId = interface.getId();
610  }
611  } else {
612  adaptionData.emplace_back();
613  }
614 
615  // Updating the adaption brings the patch is in normal memory mode
616  setMemoryMode(MemoryMode::MEMORY_NORMAL, false);
617 
618  // Done
619  return adaptionData;
620 }

743 void VolCartesian::addInterfacesDirection(const int &direction)
744 {
745  log::cout() << " >> Creating interfaces normal to direction " << direction << "\n";
746 
747  // Info on the interfaces
748  ElementInfo::Type interfaceType = getInterfaceType();
749 
750  const ElementInfo &interfaceTypeInfo = ElementInfo::getElementInfo(interfaceType);
751  const int nInterfaceVertices = interfaceTypeInfo.nVertices;
752  std::array<int, 3> interfaceCount1D = getInterfaceCountDirection(direction);
753 
754  // Counters
755  std::array<int, 3> counters = {{0, 0, 0}};
756  int &i = counters[Vertex::COORD_X];
757  int &j = counters[Vertex::COORD_Y];
758  int &k = counters[Vertex::COORD_Z];
759 
760  // Creation of the interfaces
761  for (k = 0; (isThreeDimensional()) ? (k < interfaceCount1D[Vertex::COORD_Z]) : (k <= 0); k++) {
762  for (j = 0; j < interfaceCount1D[Vertex::COORD_Y]; j++) {
763  for (i = 0; i < interfaceCount1D[Vertex::COORD_X]; i++) {
764  InterfaceIterator interfaceIterator = VolumeKernel::addInterface(interfaceType);
765  Interface &interface = *interfaceIterator;
766 
767  // Owner id
768  std::array<int, 3> ownerIJK;
769  for (int n = 0; n < 3; n++) {
770  ownerIJK[n] = counters[n];
771  }
772  if (counters[direction] == (interfaceCount1D[direction] - 1)) {
773  ownerIJK[direction] -= 1;
774  }
775 
776  long ownerId = getCellLinearId(ownerIJK);
777 
778  // Neighbour id
779  long neighId;
780  if (counters[direction] != 0 && counters[direction] != interfaceCount1D[direction] - 1) {
781  std::array<int, 3> neighIJK;
782  for (int n = 0; n < 3; n++) {
783  neighIJK[n] = counters[n];
784  }
785  neighIJK[direction] -= 1;
786 
787  neighId = getCellLinearId(neighIJK);
788  } else {
789  neighId = Element::NULL_ID;
790  }
791 
792  // Owner data
793  Cell &owner = m_cells[ownerId];
794 
795  int ownerFace = 2 * direction;
796  if (counters[direction] == interfaceCount1D[direction] - 1) {
797  ownerFace++;
798  }
799 
800  interface.setOwner(owner.getId(), ownerFace);
801  owner.setInterface(ownerFace, 0, interface.getId());
802  owner.setAdjacency(ownerFace, 0, neighId);
803 
804  // Neighbour data
805  if (counters[direction] != 0 && counters[direction] != interfaceCount1D[direction] - 1) {
806  Cell &neigh = m_cells[neighId];
807 
808  int neighFace = 2 * direction + 1;
809 
810  interface.setNeigh(neigh.getId(), neighFace);
811  neigh.setInterface(neighFace, 0, interface.getId());
812  neigh.setAdjacency(neighFace, 0, ownerId);
813  } else {
814  interface.unsetNeigh();
815  }
816 
817  // Connectivity
818  std::unique_ptr<long[]> connect = std::unique_ptr<long[]>(new long[nInterfaceVertices]);
819  if (direction == Vertex::COORD_X) {
820  connect[0] = getVertexLinearId(i, j, k);
821  connect[1] = getVertexLinearId(i, j + 1, k);
822  if (isThreeDimensional()) {
823  connect[2] = getVertexLinearId(i, j + 1, k + 1);
824  connect[3] = getVertexLinearId(i, j, k + 1);
825  }
826  } else if (direction == Vertex::COORD_Y) {
827  connect[0] = getVertexLinearId(i, j, k);
828  connect[1] = getVertexLinearId(i + 1, j, k);
829  if (isThreeDimensional()) {
830  connect[2] = getVertexLinearId(i + 1, j, k + 1);
831  connect[3] = getVertexLinearId(i, j, k + 1);
832  }
833  } else if (direction == Vertex::COORD_Z) {
834  connect[0] = getVertexLinearId(i, j, k);
835  connect[1] = getVertexLinearId(i + 1, j, k);
836  if (isThreeDimensional()) {
837  connect[2] = getVertexLinearId(i + 1, j + 1, k);
838  connect[3] = getVertexLinearId(i, j + 1, k);
839  }
840  }
841 
842  interface.setConnect(std::move(connect));
843  }
844  }
845  }
