250 uint64_t VTKUnstructuredGrid::calcSizeConnectivity( ){
251 
252  uint64_t nconn(0) ;
253 
254  std::fstream str ;
255  std::fstream::pos_type position_appended;
256  std::string line;
257  char c_ ;
258  uint32_t nbytes32 ;
259  uint64_t nbytes64 ;
260 
261  str.open( m_fh.getPath( ), std::ios::in ) ;
262 
263  //Read appended data
264  //Go to the initial position of the appended section
265  while( getline(str, line) && (! bitpit::utils::keywordInString( line, "<AppendedData")) ){}
266 
267  str >> c_;
268  while( c_ != '_') str >> c_;
269 
270  position_appended = str.tellg();
271 
272 
273  str.close();
274  str.clear();
275 
276  //Open in binary for read
277  str.open( m_fh.getPath( ), std::ios::in | std::ios::binary);
278 
279  if( m_geometry[3].getCodification() == VTKFormat::APPENDED ){
280  str.seekg( position_appended) ;
281  str.seekg( m_geometry[3].getOffset(), std::ios::cur) ;
282 
283  if( m_headerType== "UInt32") {
284  genericIO::absorbBINARY( str, nbytes32 ) ;
285  nconn = nbytes32 /VTKTypes::sizeOfType( m_geometry[3].getDataType() ) ;
286  }
287 
288  if( m_headerType== "UInt64") {
289  genericIO::absorbBINARY( str, nbytes64 ) ;
290  nconn = nbytes64 /VTKTypes::sizeOfType( m_geometry[3].getDataType() ) ;
291  }
292  }
293 
294 
295  //Read geometry
296  if( m_geometry[3].getCodification() == VTKFormat::ASCII ){
297  str.seekg( m_geometry[3].getPosition() ) ;
298 
299  std::string line ;
300  std::vector<uint64_t> temp;
301 
302  nconn = 0 ;
303 
304  getline( str, line) ;
305  while( ! bitpit::utils::keywordInString(line,"/DataArray") ) {
306 
307  temp.clear() ;
308  bitpit::utils::convertString( line, temp) ;
309  nconn += temp.size() ;
310  }
311 
312 
313  }
314 
315  str.close();
316 
317  return nconn ;

353 
354  //Closing Piece

356  str << " </UnstructuredGrid>" << std::endl;
357 
358  //Appended Section
359 
360  str << " <AppendedData encoding=\"raw\">" << std::endl;
361  str << "_" ;

423 void VTKUnstructuredGrid::readMetaInformation( ){
424 
425  std::fstream str;
426  std::string line, temp;
427 
428  std::fstream::pos_type position;
429 
430  str.open( m_fh.getPath( ), std::ios::in ) ;
431 
432  getline( str, line);
433  while( ! bitpit::utils::keywordInString( line, "<VTKFile")){
434  getline(str, line);
435  }
436 
437  if( bitpit::utils::getAfterKeyword( line, "header_type", '\"', temp) ){
438  setHeaderType( temp) ;
439  }
440 
441  while( ! bitpit::utils::keywordInString( line, "<Piece")){
442  getline(str, line);
443  }
444 
445  bitpit::utils::getAfterKeyword( line, "NumberOfPoints", '\"', temp) ;
446  bitpit::utils::convertString( temp, m_points );
447 
448  bitpit::utils::getAfterKeyword( line, "NumberOfCells", '\"', temp) ;
449  bitpit::utils::convertString( temp, m_cells );
450 
451 
452  position = str.tellg() ;
453  readDataHeader( str ) ;
454 
455 
456  for( auto &field : m_geometry ){
457  str.seekg( position) ;
458  if( ! readDataArray( str, field ) ) {
459  log::cout() << field.getName() << " DataArray not found" << std::endl ;
460  }
461  }
462 
463 
464  str.close() ;
465 
466  if( m_homogeneousType == VTKElementType::UNDEFINED) {
467  setDimensions( m_cells, m_points, calcSizeConnectivity() ) ;
468  } else {
469  setDimensions( m_cells, m_points ) ;
470  }

500  entries = m_points *static_cast<int>(VTKFieldType::VECTOR) ;
501

503  entries = m_cells ;
504

506  entries = m_cells ;
507 
508  } else if( name == "connectivity"){
509  entries = m_connectivity ;
510 
511  } else{
512 
513  VTKLocation location( field.getLocation() ) ;
514  assert( location != VTKLocation::UNDEFINED) ;
515 
516  if( location == VTKLocation::CELL ){
517  entries = m_cells ;
518 
519  } else if( location == VTKLocation::POINT ){
520  entries = m_points ;
521 
522  }
523 
524  VTKFieldType fieldType( field.getFieldType() ) ;
525  assert( fieldType != VTKFieldType::UNDEFINED) ;
526 
527  entries *= static_cast<uint64_t>(fieldType) ;
528 
529  }
