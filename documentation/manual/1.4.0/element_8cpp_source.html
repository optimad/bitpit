165 void Element::Tesselation::importPolygon(const std::vector<int> &vertexIds)
166 {
167  int nVertices = vertexIds.size();
168  if (nVertices == 3 || nVertices == 4) {
169  m_nTiles++;
170  m_types.push_back(nVertices == 3 ? ElementType::TRIANGLE : ElementType::QUAD);
171  m_connects.push_back(vertexIds);
172 
173  return;
174  }
175 
176  // Add the centroid
177  std::array<double, 3> centroid = {{0., 0., 0.}};
178  for (int k = 0; k < nVertices; ++k) {
179  centroid += m_coordinates[vertexIds[k]];
180  }
181  centroid = centroid / double(nVertices);
182 
183  int centroidId = importVertexCoordinates(std::move(centroid));
184 
185  // Decompose the polygon in triangles
186  //
187  // Each triangle is composed by the two vertices of a side and by the
188  // centroid.
189  ElementType tileType = ElementType::TRIANGLE;
190  int nTileVertices = ReferenceElementInfo::getInfo(tileType).nVertices;
191  int nSideVertices = ReferenceElementInfo::getInfo(ElementType::LINE).nVertices;
192 
193  int nSides = nVertices;
194  m_types.resize(m_nTiles + nSides, tileType);
195  m_connects.resize(m_nTiles + nSides, std::vector<int>(nTileVertices));
196  for (int i = 0; i < nSides; ++i) {
197  m_nTiles++;
198  for (int k = 0; k < nSideVertices; ++k) {
199  m_connects[m_nTiles - 1][k] = vertexIds[(i + k) % nVertices];
200  }
201  m_connects[m_nTiles - 1][nSideVertices] = centroidId;
202  }
203 }

211 void Element::Tesselation::importPolyhedron(const std::vector<int> &vertexIds, const std::vector<std::vector<int>> &faceVertexIds)
212 {
213  int nFaces = faceVertexIds.size();
214  int nVertices = vertexIds.size();
215 
216  // Generate the tesselation of the surface
217  int nInitialTiles = m_nTiles;
218  for (int i = 0; i < nFaces; ++i) {
219  importPolygon(faceVertexIds[i]);
220  }
221  int nFinalTiles = m_nTiles;
222 
223  // Add the centroid of the element to the tesselation
224  std::array<double, 3> centroid = {{0., 0., 0.}};
225  for (int k = 0; k < nVertices; ++k) {
226  centroid += m_coordinates[vertexIds[k]];
227  }
228  centroid = centroid / double(nVertices);
229 
230  int centroidTesselationId = importVertexCoordinates(std::move(centroid));
231 
232  // Decompose the polyhedron in prisms and pyramids.
233  //
234  // Each prism/pyramid has a tile of the surface tesselation as the
235  // base and the centroid of the element as the apex. Since we have
236  // already generated the surface tesselation, we can "convert" the
237  // two-dimensional tiles of that tesselation in three-dimensional
238  // tiles to obtain the volume tesselation.
239  for (int i = nInitialTiles; i < nFinalTiles; ++i) {
240  // Change the tile type
241  ElementType &tileType = m_types[i];
242  if (tileType == ElementType::TRIANGLE) {
243  tileType = ElementType::TETRA;
244  } else if (tileType == ElementType::QUAD) {
245  tileType = ElementType::PYRAMID;
246  } else {
247  BITPIT_UNREACHABLE("Unsupported tile");
248  throw std::runtime_error ("Unsupported tile");
249  }
250 
251  // Fix the order of the connectivity the match the order of the
252  // three-dimensional element
253  if (tileType == ElementType::TETRA) {
254  std::swap(m_connects[i][0], m_connects[i][2]);
255  } else if (tileType == ElementType::PYRAMID) {
256  std::swap(m_connects[i][1], m_connects[i][3]);
257  }
258 
259  // Add the centroid to the tile connectivity
260  m_connects[i].push_back(centroidTesselationId);
261  }
262 }

739 ConstProxyVector<int> Element::getFaceLocalConnect(const int &face) const
740 {
741  switch (m_type) {
742 
743  case (ElementType::POLYGON):
744  {
745  int connectSize = getConnectSize();
746 
747  int faceConnectSize = getFaceVertexCount(face);
748  std::vector<int> localFaceConnect(faceConnectSize);
749  for (int i = 0; i < faceConnectSize; ++i) {
750  int localVertexId = (face + i) % connectSize;
751 
752  localFaceConnect[i] = localVertexId;
753  }
754 
755  return ConstProxyVector<int>(std::move(localFaceConnect));
756  }
757 
758  case (ElementType::POLYHEDRON):
759  {
760  ElementType faceType = getFaceType(face);
761  ConstProxyVector<long> faceVertexIds = getFaceVertexIds(face);
762  int nFaceVertices = faceVertexIds.size();
763  bool faceHasReferenceInfo = ReferenceElementInfo::hasInfo(faceType);
764 
765  int faceConnectSize = nFaceVertices;
766  int localVertexOffset = 0;
767  if (!faceHasReferenceInfo) {
768  ++faceConnectSize;
769  ++localVertexOffset;
770  }
771 
772  std::vector<int> localFaceConnect(faceConnectSize);
773  if (!faceHasReferenceInfo) {
774  localFaceConnect[0] = nFaceVertices;
775  }
776 
777  // The list of the vertices in sorted, therfore we can search the
778  // vertices using a binary search.
779  ConstProxyVector<long> vertexIds = getVertexIds();
780  for (int k = 0; k < nFaceVertices; ++k) {
781  int vertexId = faceVertexIds[k];
782  auto localVertexIdItr = std::lower_bound(vertexIds.begin(), vertexIds.end(), vertexId);
783  assert(localVertexIdItr != vertexIds.end() && *localVertexIdItr == vertexId);
784  localFaceConnect[localVertexOffset + k] = std::distance(vertexIds.begin(), localVertexIdItr);
785  }
786 
787  return ConstProxyVector<int>(std::move(localFaceConnect));
788  }
789 
790  default:
791  {
792  assert(m_type != ElementType::UNDEFINED);
793 
794  const std::vector<int> &localFaceConnect = getInfo().faceConnect[face];
795 
796  return ConstProxyVector<int>(localFaceConnect.data(), localFaceConnect.size());
797  }
798 
799  }
800 }

808 ConstProxyVector<long> Element::getFaceConnect(int face) const
809 {
810  const long *connectivity = getConnect();
811 
812  switch (m_type) {
813 
814  case (ElementType::POLYGON):
815  {
816  int connectSize = getConnectSize();
817 
818  int facePos = 1 + face;
819  int faceConnectSize = getFaceVertexCount(face);
820  if (facePos + faceConnectSize <= connectSize) {
821  return ConstProxyVector<long>(connectivity + facePos, faceConnectSize);
822  } else {
823  std::vector<long> faceConnect(faceConnectSize);
824  for (int i = 0; i < faceConnectSize; ++i) {
825  int position = facePos + i;
826  if (position >= connectSize) {
827  position = position % connectSize + 1;
828  }
829 
830  faceConnect[i] = connectivity[position];
831  }
832 
833  return ConstProxyVector<long>(std::move(faceConnect));
834  }
835  }
836 
837  case (ElementType::POLYHEDRON):
838  {
839  ElementType faceType = getFaceType(face);
840 
841  int facePos = getFaceStreamPosition(face);
842  int faceConnectSize = connectivity[facePos];
843  int faceConnectBegin = facePos + 1;
844  if (!ReferenceElementInfo::hasInfo(faceType)) {
845  faceConnectSize++;
846  faceConnectBegin--;
847  }
848 
849  return ConstProxyVector<long>(connectivity + faceConnectBegin, faceConnectSize);
850  }
851 
852  default:
853  {
854  assert(m_type != ElementType::UNDEFINED);
855 
856  ConstProxyVector<int> localFaceConnect = getFaceLocalConnect(face);
857  int faceConnectSize = localFaceConnect.size();
858 
859  std::vector<long> faceConnect(faceConnectSize);
860  for (int k = 0; k < faceConnectSize; ++k) {
861  int localVertexId = localFaceConnect[k];
862  long vertexId = connectivity[localVertexId];
863  faceConnect[k] = vertexId;
864  }
865 
866  return ConstProxyVector<long>(std::move(faceConnect));
867  }
868 
869  }
