371 void LevelSet::addProcessingOrder( int objectId ) {
372 
373  bool primary = m_objects.at(objectId)->isPrimary() ;
374 
375  if(primary){
376  std::vector<int>::iterator orderItr = m_order.begin() ;
377  bool iterate( orderItr != m_order.end()) ;
378 
379  while(iterate){
380  int id = *orderItr ;
381  if( m_objects.at(id)->isPrimary() ){
382  ++orderItr ;
383  iterate = orderItr != m_order.end() ;
384  } else {
385  iterate = false;
386  }
387  }
388 
389  m_order.insert(orderItr,objectId) ;
390 
391  } else {
392  m_order.push_back(objectId) ;
393 
394  }
395

554 void LevelSet::partition( const std::vector<adaption::Info> &mapper ){
555 
556  assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::partition()");
557 
558  // Set the communicator
559  m_kernel->initializeCommunicator();
560 
561  // Udate the cache of the kernel
562  m_kernel->updateGeometryCache( mapper ) ;
563 
564  // Compile send and receive lists
565  std::unordered_map<int,std::vector<long>> sendList, recvList ;
566  for( const auto &event : mapper){
567  if( event.entity != adaption::Entity::ENTITY_CELL){
568  continue;
569  }
570 
571  if( event.type == adaption::Type::TYPE_PARTITION_SEND){
572  sendList.insert({{event.rank,event.previous}}) ;
573  } else if( event.type == adaption::Type::TYPE_PARTITION_RECV){
574  recvList.insert({{event.rank,event.current}}) ;
575  }
576  }
577 
578  // Communicate according to new partitioning
579  for( int objectId : m_order){
580  auto &visitor = *(m_objects.at(objectId)) ;
581 
582  visitor.communicate( sendList, recvList, &mapper ) ;
583  }
584 
585  for( int objectId : m_order){
586  auto &visitor = *(m_objects.at(objectId)) ;
587 
588  visitor.exchangeGhosts();
589  }
590 
591 }
