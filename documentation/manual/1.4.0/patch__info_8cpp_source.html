167 void PatchNumberingInfo::_extract()
168 {
169  long consecutiveId;
170 
171 #if BITPIT_ENABLE_MPI==1
172  // Initialize communications
173  size_t exchangeDataSize = sizeof(consecutiveId);
174  std::unique_ptr<DataCommunicator> dataCommunicator;
175 
176  if (m_patch->getProcessorCount() > 1) {
177  // Create the data communicator
178  dataCommunicator = std::unique_ptr<DataCommunicator>(new DataCommunicator(m_patch->getCommunicator()));
179 
180  // Set and start the receives
181  for (const auto entry : m_patch->getGhostExchangeTargets()) {
182  const int rank = entry.first;
183  const auto &list = entry.second;
184 
185  dataCommunicator->setRecv(rank, list.size() * exchangeDataSize);
186  dataCommunicator->startRecv(rank);
187  }
188  }
189 #endif
190 
191 #if BITPIT_ENABLE_MPI==1
192  // Get the internal count of all the partitions
193  m_nGlobalInternals.resize(m_patch->getProcessorCount());
194  if (m_patch->getProcessorCount() > 1) {
195  long nLocalInternals = m_patch->getInternalCount();
196  MPI_Allgather(&nLocalInternals, 1, MPI_LONG, m_nGlobalInternals.data(), 1, MPI_LONG, m_patch->getCommunicator());
197  } else {
198  m_nGlobalInternals[0] = m_patch->getInternalCount();
199  }
200 #endif
201 
202  // Evalaute the consecutive id of the internal cells
203  if (m_patch->getInternalCount() > 0) {
204  std::map<long,long> nativeIds;
205  for (auto itr = m_patch->internalConstBegin(); itr != m_patch->internalConstEnd(); ++itr) {
206  long id = itr.getId();
207  long nativeId = m_patch->_getCellNativeIndex(id);
208 
209  nativeIds.insert({nativeId, id});
210  }
211 
212 #if BITPIT_ENABLE_MPI==1
213  consecutiveId = getCellGlobalCountOffset();
214 #else
215  consecutiveId = 0;
216 #endif
217  for (auto itr = nativeIds.begin(); itr != nativeIds.end(); ++itr) {
218  m_cellLocalToConsecutiveMap.insert({itr->second, consecutiveId++});
219  }
220 
221  std::map<long,long>().swap(nativeIds);
222  }
223 
224 #if BITPIT_ENABLE_MPI==1
225  // Communicate the consecutive id of the ghost cells
226  if (m_patch->getProcessorCount() > 1) {
227  // Set and start the sends
228  for (const auto entry : m_patch->getGhostExchangeSources()) {
229  const int rank = entry.first;
230  auto &list = entry.second;
231 
232  dataCommunicator->setSend(rank, list.size() * exchangeDataSize);
233  SendBuffer &buffer = dataCommunicator->getSendBuffer(rank);
234  for (long id : list) {
235  buffer << m_cellLocalToConsecutiveMap.at(id);
236  }
237  dataCommunicator->startSend(rank);
238  }
239 
240  // Receive the consecutive ids of the ghosts
241  int nCompletedRecvs = 0;
242  while (nCompletedRecvs < dataCommunicator->getRecvCount()) {
243  int rank = dataCommunicator->waitAnyRecv();
244  const auto &list = m_patch->getGhostExchangeTargets(rank);
245 
246  RecvBuffer &buffer = dataCommunicator->getRecvBuffer(rank);
247  for (long id : list) {
248  buffer >> consecutiveId;
249  m_cellLocalToConsecutiveMap.insert({id, consecutiveId});
250  }
251 
252  ++nCompletedRecvs;
253  }
254 
255  // Wait for the sends to finish
256  dataCommunicator->waitAllSends();
257  }
258 #endif
259 }
