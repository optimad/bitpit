228  m_partitionRangeGlobalIdx[p] = m_globalNumOctants - 1;
229  m_partitionLastDesc[p] = lastDescMorton;
230  m_partitionFirstDesc[p] = firstDescMorton;

380  void ParaTree::dump(std::ostream &stream, bool full)
381  {
382  // Version
383  utils::binary::write(stream, getDumpVersion());
384 
385  // Tree data
386  utils::binary::write(stream, getNproc());
387 
388  utils::binary::write(stream, getDim());
389 
390  utils::binary::write(stream, getSerial());
391  utils::binary::write(stream, getMaxDepth());
392  utils::binary::write(stream, getStatus());
393  utils::binary::write(stream, getBalanceCodimension());
394 
395  for (int i = 0; i < m_global.m_nfaces; i++) {
396  utils::binary::write(stream, getPeriodic(i));
397  }
398 
399  // Octant data
400  uint32_t nOctants = getNumOctants();
401  utils::binary::write(stream, nOctants);
402 
403  uint32_t nGlobalOctants = getGlobalNumOctants();
404  utils::binary::write(stream, nGlobalOctants);
405 
406  for (uint32_t i = 0; i < nOctants; i++) {
407  const Octant &octant = m_octree.m_octants[i];
408 
409  utils::binary::write(stream, octant.getLevel());
410  utils::binary::write(stream, octant.getX());
411  utils::binary::write(stream, octant.getY());
412  utils::binary::write(stream, octant.getZ());
413 
414  for (size_t k = 0; k < octant.m_info.size(); ++k) {
415  utils::binary::write(stream, (bool) octant.m_info[k]);
416  }
417 
418  utils::binary::write(stream, octant.getBalance());
419  utils::binary::write(stream, octant.getMarker());
420  }
421 
422  // Information about partitioning
423  for (int k = 0; k < m_nproc; ++k) {
424  utils::binary::write(stream, m_partitionFirstDesc[k]);
425  }
426 
427  for (int k = 0; k < m_nproc; ++k) {
428  utils::binary::write(stream, m_partitionLastDesc[k]);
429  }
430 
431  for (int k = 0; k < m_nproc; ++k) {
432  utils::binary::write(stream, m_partitionRangeGlobalIdx[k]);
433  }
434 
435  // Extended information (mapping, ...)
436  utils::binary::write(stream, full);
437  if (full) {
438  utils::binary::write(stream, m_lastOp);
439  if (m_lastOp == OP_ADAPT_MAPPED){
440  for (auto idx : m_mapIdx) {
441  utils::binary::write(stream, idx);
442  }
443 
444  utils::binary::write(stream, m_octree.m_lastGhostBros.size());
445  for (auto lastGhostBrother : m_octree.m_lastGhostBros) {
446  utils::binary::write(stream, lastGhostBrother);
447  }
448  }
449  else if (m_lastOp == OP_LOADBALANCE || m_lastOp == OP_LOADBALANCE_FIRST){
450  for (int i = 0; i < m_nproc; ++i) {
451  utils::binary::write(stream, m_partitionRangeGlobalIdx0[i]);
452  }
453  }
454  }
455 
456  }

464  void ParaTree::restore(std::istream &stream)
465  {
466  // Version
467  int version;
468  utils::binary::read(stream, version);
469  if (version != getDumpVersion()) {
470  throw std::runtime_error ("The version of the file does not match the required version");
471  }
472 
473  // Check if the number of processors matches
474  int nProcs;
475  utils::binary::read(stream, nProcs);
476  if (nProcs != m_nproc) {
477  throw std::runtime_error ("The restart was saved with a different number of processors.");
478  }
479 
480  // Initialize the tree
481  uint8_t dimension;
482  utils::binary::read(stream, dimension);
483 
484  m_octree.initialize(dimension);
485  m_trans.initialize(dimension);
486  reinitialize(dimension, m_log->getName());
487  reset(false);
488 
489  // Set tree properties
490  utils::binary::read(stream, m_serial);
491  utils::binary::read(stream, m_maxDepth);
492  utils::binary::read(stream, m_status);
493 
494  bool balanceCodimension;
495  utils::binary::read(stream, balanceCodimension);
496  setBalanceCodimension(balanceCodimension);
497 
498  for (int i = 0; i < m_global.m_nfaces; i++) {
499  bool periodicBorder;
500  utils::binary::read(stream, periodicBorder);
501  if (periodicBorder){
502  setPeriodic(i);
503  }
504  }
505 
506  // Restore octants
507  uint32_t nOctants;
508  utils::binary::read(stream, nOctants);
509  m_octree.m_sizeOctants = nOctants;
510 
511  uint32_t nGlobalOctants;
512  utils::binary::read(stream, nGlobalOctants);
513  m_globalNumOctants = nGlobalOctants;
514 
515  m_octree.m_octants.clear();
516  m_octree.m_octants.reserve(nOctants);
517  for (uint32_t i = 0; i < nOctants; i++) {
518  // Create octant
519  uint8_t level;
520  utils::binary::read(stream, level);
521 
522  uint32_t x;
523  utils::binary::read(stream, x);
524 
525  uint32_t y;
526  utils::binary::read(stream, y);
527 
528  uint32_t z;
529  utils::binary::read(stream, z);
530 
531  Octant octant(false, m_dim, level, x, y, z);
532 
533  // Set octant info
534  for (size_t k = 0; k < octant.m_info.size(); ++k) {
535  bool bit;
536  utils::binary::read(stream, bit);
537  octant.m_info.set(k, bit);
538  }
539 
540  // Set octant 2:1 balance
541  bool balance21;
542  utils::binary::read(stream, balance21);
543  octant.setBalance(balance21);
544 
545  // Set marker
546  int8_t marker;
547  utils::binary::read(stream, marker);
548  octant.setMarker(marker);
549 
550  // Add octant to the list
551  m_octree.m_octants.push_back(std::move(octant));
552  }
553 
554  m_octree.updateLocalMaxDepth();
555 
556  // Set first last descendant
557  m_partitionFirstDesc.resize(m_nproc);
558  for (int k = 0; k < m_nproc; ++k) {
559  uint64_t descendant;
560  utils::binary::read(stream, descendant);
561  m_partitionFirstDesc[k] = descendant;
562  }
563  m_octree.m_firstDescMorton = m_partitionFirstDesc[m_rank];
564 
565  m_partitionLastDesc.resize(m_nproc);
566  for (int k = 0; k < m_nproc; ++k) {
567  uint64_t descendant;
568  utils::binary::read(stream, descendant);
569  m_partitionLastDesc[k] = descendant;
570  }
571  m_octree.m_lastDescMorton = m_partitionLastDesc[m_rank];
572 
573  // Set partitions and parallel information
574  m_partitionRangeGlobalIdx.resize(m_nproc);
575  for (int k = 0; k < m_nproc; ++k) {
576  uint64_t rangeGlobalIdx;
577  utils::binary::read(stream, rangeGlobalIdx);
578  m_partitionRangeGlobalIdx[k] = rangeGlobalIdx;
579  }
580 
581 #if BITPIT_ENABLE_MPI==1
582  if (!m_serial) {
583  setPboundGhosts();
584  }
585 #endif
586 
587  // Full restore (i.e. restore with mapper of last operation)
588  m_mapIdx.clear();
589 
590  for (int i = 0; i < m_nproc; ++i) {
591  m_partitionRangeGlobalIdx0[i] = 0;
592  }
593 
594  bool full;
595  utils::binary::read(stream, full);
596  if (full) {
597  utils::binary::read(stream, m_lastOp);
598  if (m_lastOp == OP_ADAPT_MAPPED) {
599  m_mapIdx.resize(m_octree.m_octants.size());
600  for (size_t i = 0; i < m_octree.m_octants.size(); ++i) {
601  utils::binary::read(stream, m_mapIdx[i]);
602  }
603 
604  size_t lastGhostBrosSize;
605  utils::binary::read(stream, lastGhostBrosSize);
606  m_octree.m_lastGhostBros.resize(lastGhostBrosSize);
607  for (size_t i = 0; i < lastGhostBrosSize; ++i) {
608  utils::binary::read(stream, m_octree.m_lastGhostBros[i]);
609  }
610  }
611  else if (m_lastOp == OP_LOADBALANCE || m_lastOp == OP_LOADBALANCE_FIRST){
612  for (int i = 0; i < m_nproc; ++i) {
613  utils::binary::read(stream, m_partitionRangeGlobalIdx0[i]);
614  }
615  }
616  }
617  else {
618  m_lastOp = OP_INIT;
619  }
620  }

734  throw std::runtime_error ("PABLO communicator can be set just once");

738  if (communicator == MPI_COMM_NULL) {

740  }

743  //
744  // No library routine should use MPI_COMM_WORLD as the communicator;

768  void
769  ParaTree::replaceComm(MPI_Comm communicator, MPI_Comm *previousCommunicator)
770  {
771  // The communicator has to be already set
772  if (!isCommSet()) {
773  throw std::runtime_error ("PABLO communicator is not set");
774  }
775 
776  // Check if the communicator is valid
777  //
778  // The communicator should be equaivalent to the one currently set,
779  // i.e., the rank of the processes have to be the same in both
780  // communicators.
781  int updatedRank;
782  MPI_Comm_rank(communicator, &updatedRank);
783  int currentRank = getRank();
784 
785  int isValid = (updatedRank == currentRank) ? 1 : 0;
786  MPI_Allreduce(MPI_IN_PLACE, &isValid, 1, MPI_INT, MPI_LAND, m_comm);
787  if (!isValid) {
788  throw std::runtime_error ("New communicator is not valid");
789  }
790 
791  // Handle previous communicator
792  if (previousCommunicator) {
793  *previousCommunicator = m_comm;
794  m_comm = MPI_COMM_NULL;
795  } else {
796  freeComm();
797  }
798 
799  // Set the communicator
800  setComm(communicator);
801  }
