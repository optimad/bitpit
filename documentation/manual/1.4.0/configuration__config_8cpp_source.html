359 void Config::dump(std::ostream &out, int indentLevel) const
360 {
361  const int INDENT_SIZE = 2;
362 
363  std::string padding(INDENT_SIZE, ' ');
364  std::string indent(INDENT_SIZE * indentLevel, ' ');
365 
366  out << std::endl;
367  out << indent << "Options..." << std::endl;
368  if (getOptionCount() > 0) {
369  for (auto &entry : getOptions()) {
370  out << indent << padding << entry.first << " = " << entry.second << std::endl;
371  }
372  } else {
373  out << indent << padding << "No options." << std::endl;
374  }
375 
376  ++indentLevel;
377  for (auto &entry : getSections()) {
378  out << std::endl;
379  out << indent << padding << "::: Section " << entry.first << " :::" << std::endl;
380  entry.second->dump(out, indentLevel);
381  }
382 }
