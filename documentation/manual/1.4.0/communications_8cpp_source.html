244 void DataCommunicator::discoverSends()
245 {
246  // Cancel current sends
247  clearAllSends();
248 
249  // Send the data sizes with a synchronous send
250  int nRecvs = getRecvCount();
251 
252  std::vector<long> discoverSizes(nRecvs, 0);
253  std::vector<MPI_Request> discoverRequests(nRecvs, MPI_REQUEST_NULL);
254  for (int i = 0; i < nRecvs; ++i) {
255  int rank = m_recvRanks[i];
256  RecvBuffer &buffer = m_recvBuffers[i];
257  discoverSizes[i] = buffer.getSize();
258  MPI_Issend(discoverSizes.data() + i, 1, MPI_LONG, rank, m_discoverTag, m_communicator, discoverRequests.data() + i);
259  }
260 
261  // Receive the data sizes and set the sends
262  MPI_Request exchangeCompletedRequest = MPI_REQUEST_NULL;
263  while (true) {
264  // If there are messagea available receive them and set the sends
265  int messageAvailable = 1;
266  while (messageAvailable) {
267  MPI_Status status;
268  MPI_Iprobe(MPI_ANY_SOURCE, m_discoverTag, m_communicator, &messageAvailable, &status);
269  if (messageAvailable) {
270  long dataSize;
271  MPI_Recv(&dataSize, 1, MPI_LONG, status.MPI_SOURCE, m_discoverTag, m_communicator, MPI_STATUS_IGNORE);
272  setSend(status.MPI_SOURCE, dataSize);
273  }
274  }
275 
276  // If all the sends are complete notify it
277  if (exchangeCompletedRequest == MPI_REQUEST_NULL) {
278  int discoverSendsCompleted;
279  MPI_Testall(discoverRequests.size(), discoverRequests.data(), &discoverSendsCompleted, MPI_STATUSES_IGNORE);
280  if (discoverSendsCompleted) {
281  MPI_Ibarrier(m_communicator, &exchangeCompletedRequest);
282  }
283  }
284 
285  // If all sends are completed, check if also the other processes have
286  // completed the sends. Sice these are synchronous sends, they will
287  // be makred as completed only when the corresponding receive has
288  // completed. When all processes have completed the send/recevies
289  // all sizes have been exchanged.
290  if (exchangeCompletedRequest != MPI_REQUEST_NULL) {
291  int exchangeCompleted = 0;
292  MPI_Test(&exchangeCompletedRequest, &exchangeCompleted, MPI_STATUS_IGNORE);
293  if (exchangeCompleted) {
294  break;
295  }
296  }
297  }
298 }

364 void DataCommunicator::clearSend(int rank)
365 {
366  if (m_sendIds.count(rank) == 0) {
367  return;
368  }
369 
370  // Cancel waiting send
371  cancelSend(rank);
372 
373  // Remove the send associated to the specified rank
374  int id = m_sendIds[rank];
375  m_sendIds.erase(rank);
376  for (auto &entry : m_sendIds) {
377  if (entry.second > id) {
378  entry.second--;
379  }
380  }
381 
382  m_sendRanks.erase(m_sendRanks.begin() + id);
383  m_sendRequests.erase(m_sendRequests.begin() + id);
384  m_sendBuffers.erase(m_sendBuffers.begin() + id);
385 }
