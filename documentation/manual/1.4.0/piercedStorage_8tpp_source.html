338 template<typename value_t, typename id_t>
339 void PiercedStorage<value_t, id_t>::commitSyncAction(const PiercedSyncAction &action)
340 {
341  switch (action.type) {
342 
343  case PiercedSyncAction::TYPE_CLEAR:
344  {
345  rawClear(true);
346  break;
347  }
348 
349  case PiercedSyncAction::TYPE_RESERVE:
350  {
351  rawReserve(action.info[PiercedSyncAction::INFO_SIZE]);
352  break;
353  }
354 
355  case PiercedSyncAction::TYPE_RESIZE:
356  {
357  rawResize(action.info[PiercedSyncAction::INFO_SIZE]);
358  break;
359  }
360 
361  case PiercedSyncAction::TYPE_SHRINK_TO_FIT:
362  {
363  rawShrinkToFit();
364  break;
365  }
366 
367  case PiercedSyncAction::TYPE_REORDER:
368  {
369  rawReorder(*action.data);
370  break;
371  }
372 
373  case PiercedSyncAction::TYPE_APPEND:
374  {
375  // Since we are increasing the sotrage by an element at the time
376  // calling a reserve will hurt performance badly because this will
377  // prevent the automatic reallocation of the storage.
378  rawEmplaceBack();
379  break;
380  }
381 
382  case PiercedSyncAction::TYPE_INSERT:
383  {
384  // Since we are increasing the sotrage by an element at the time
385  // calling a reserve will hurt performance badly because this will
386  // prevent the automatic reallocation of the storage.
387  rawEmplace(action.info[PiercedSyncAction::INFO_POS]);
388  break;
389  }
390 
391  case PiercedSyncAction::TYPE_OVERWRITE:
392  {
393  break;
394  }
395 
396  case PiercedSyncAction::TYPE_MOVE_APPEND:
397  {
398  // Since we are increasing the sotrage by an element at the time
399  // calling a reserve will hurt performance badly because this will
400  // prevent the automatic reallocation of the storage.
401  rawPushBack(std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
402  rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
403  break;
404  }
405 
406  case PiercedSyncAction::TYPE_MOVE_INSERT:
407  {
408  // Since we are increasing the sotrage by an element at the time
409  // calling a reserve will hurt performance badly because this will
410  // prevent the automatic reallocation of the storage.
411  rawInsert(action.info[PiercedSyncAction::INFO_POS_SECOND], 1, std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
412  rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
413  break;
414  }
415 
416  case PiercedSyncAction::TYPE_MOVE_OVERWRITE:
417  {
418  rawSet(action.info[PiercedSyncAction::INFO_POS_SECOND], std::move(rawAt(action.info[PiercedSyncAction::INFO_POS_FIRST])));
419  rawEmreplace(action.info[PiercedSyncAction::INFO_POS_FIRST]);
420  break;
421  }
422 
423  case PiercedSyncAction::TYPE_PIERCE:
424  {
425  // Nothing to do. To improve performance the element will not be
426  // cleared.
427  break;
428  }
429 
430  case PiercedSyncAction::TYPE_SWAP:
431  {
432  rawSwap(action.info[PiercedSyncAction::INFO_POS_FIRST], action.info[PiercedSyncAction::INFO_POS_SECOND]);
433  break;
434  }
435 
436  default:
437  {
438  throw std::runtime_error("Undefined synchronization action.");
439  break;
440  }
441 
442  }
443 }

523 template<typename value_t, typename id_t>
524 void PiercedStorage<value_t, id_t>::rawReorder(const std::vector<std::size_t> &permutations)
525 {
526  std::size_t storageRawSize = rawSize();
527  assert(permutations.size() == storageRawSize);
528 
529  // Evaluate the permutation of the fields
530  std::vector<std::size_t> fieldPermutations(storageRawSize * m_nFields);
531  for (std::size_t pos = 0; pos < storageRawSize; ++pos) {
532  std::size_t posOffset = pos * m_nFields;
533  std::size_t permutationOffset = permutations[pos] * m_nFields;
534  for (std::size_t k = 0; k < m_nFields; ++k) {
535  fieldPermutations[posOffset + k] = permutationOffset + k;
536  }
537  }
538 
539  // Sort the fields
540  utils::reorderVector<value_t>(fieldPermutations, m_fields, storageRawSize * m_nFields);
541 }

702 template<typename value_t, typename id_t>
703 void PiercedStorage<value_t, id_t>::swap(PiercedStorage &x) noexcept
704 {
705  // It is only possible to swap two storages with the same number of field.
706  // If this condition is not fulfilled we can not continue. However, we
707  // cannot throw an exception because the function is declared snoexcept.
708  if (x.getFieldCount() != getFieldCount()) {
709  std::cout << "It is only possible to swap storages with the same number of fields." << std::endl;
710  assert(false);
711  exit(EXIT_FAILURE);
712  }
713 
714  PiercedSyncSlave::swap(x);
715  std::swap(x.m_nFields, m_nFields);
716  std::swap(x.m_fields, m_fields);
717  std::swap(x.m_kernel, m_kernel);
718  std::swap(x.m_const_kernel, m_const_kernel);
719 }
