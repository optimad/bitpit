144  }

147  // Space dimension
148  utils::binary::write(stream, getSpaceDimension());
149 
150  // Save the vertices
151  utils::binary::write(stream, getVertexCount());
152

154  utils::binary::write(stream, vertex.getId());
155 
156  std::array<double, 3> coords = vertex.getCoords();

158  utils::binary::write(stream, coords[1]);
159  utils::binary::write(stream, coords[2]);
160  }
161 
162  // Save the cells
163  utils::binary::write(stream, getInternalCount());
164  utils::binary::write(stream, getGhostCount());
165 
166  for (const Cell &cell: m_cells) {
167  const ReferenceElementInfo &cellInfo = cell.getInfo();
168 
169  utils::binary::write(stream, cell.getId());
170  utils::binary::write(stream, cell.getPID());
171  utils::binary::write(stream, cellInfo.type);
172 
173  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
174  int nCellVertices = cellVertexIds.size();
175  for (int i = 0; i < nCellVertices; ++i) {
176  utils::binary::write(stream, cellVertexIds[i]);

186 void SurfUnstructured::_restore(std::istream &stream)
187 {
188 #if BITPIT_ENABLE_MPI==1
189  // Restore works only for serial calculations
190  if (getProcessorCount() != 1) {
191  throw std::runtime_error ("Restore of surfunstructured is implemented only for serial calculations.");
192  }
193 #endif
194 
195  // Space dimension
196  int spaceDimension;
197  utils::binary::read(stream, spaceDimension);
198  setSpaceDimension(spaceDimension);
199 
200  // Restore the vertices
201  long nVertices;
202  utils::binary::read(stream, nVertices);
203 
204  reserveVertices(nVertices);
205  for (long i = 0; i < nVertices; ++i) {
206  long id;
207  utils::binary::read(stream, id);
208 
209  std::array<double, 3> coords;
210  utils::binary::read(stream, coords[0]);
211  utils::binary::read(stream, coords[1]);
212  utils::binary::read(stream, coords[2]);
213 
214  addVertex(coords, id);
215  }
216 
217  // Restore the cells
218  long nInternals;
219  utils::binary::read(stream, nInternals);
220 
221  long nGhosts;
222  utils::binary::read(stream, nGhosts);
223 
224  long nCells = nInternals + nGhosts;
225 
226  reserveCells(nCells);
227  for (long i = 0; i < nCells; ++i) {
228  long id;
229  utils::binary::read(stream, id);
230 
231  int PID;
232  utils::binary::read(stream, PID);
233 
234  ElementType type;
235  utils::binary::read(stream, type);
236  const ReferenceElementInfo &cellInfo = ReferenceElementInfo::getInfo(type);
237 
238  int nCellVertices = cellInfo.nVertices;
239  std::vector<long> connect(nCellVertices, Vertex::NULL_ID);
240  for (int k = 0; k < nCellVertices; ++k) {
241  utils::binary::read(stream, connect[k]);
242  }
243 
244  CellIterator cellIterator = addCell(type, true, connect, id);
245  cellIterator->setPID(PID);
246  }
247 
248  // Build ghost exchange data
249 #if BITPIT_ENABLE_MPI==1
250  if (getProcessorCount()) {
251  buildGhostExchangeData();
252  }
253 #endif
254 }

286 void SurfUnstructured::extractEdgeNetwork(SurfUnstructured &net)
287 {
288  // ====================================================================== //
289  // VARIABLES DECLARATION //
290  // ====================================================================== //
291 
292  // Local variables
293  bool check;
294  int n_faces, n_adj;
295  long id;
296 
297  // Counters
298  int i, j;
299  vector<int>::const_iterator i_;
300  vector<long>::iterator j_;
301  VertexIterator v_, ve_ = vertexEnd();
302  CellIterator c_, ce_ = cellEnd();
303 
304  // ====================================================================== //
305  // INITIALIZE DATA STRUCTURE //
306  // ====================================================================== //
307  net.reserveCells(net.getCellCount() + countFaces());
308  net.reserveVertices(net.getVertexCount() + getVertexCount());
309 
310  // ====================================================================== //
311  // ADD VERTEX TO net //
312  // ====================================================================== //
313  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
314  net.addVertex(v_->getCoords(), v_->getId());
315  } //next v_
316 
317  // ====================================================================== //
318  // ADD EDGES //
319  // ====================================================================== //
320  for (c_ = cellBegin(); c_ != ce_; ++c_) {
321  id = c_->getId();
322  n_faces = c_->getFaceCount();
323  ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
324  for (i = 0; i < n_faces; ++i) {
325  check = true;
326  n_adj = c_->getAdjacencyCount(i);
327  for (j = 0; j < n_adj; ++j) {
328  check = check && (id > c_->getAdjacency(i, j));
329  } //next j
330  if (check) {
331  // Get edge type
332  ElementType edgeType = c_->getFaceType(i);
333 
334  // Get edge connect
335  ConstProxyVector<long> faceConnect = c_->getFaceConnect(i);
336  int faceConnectSize = faceConnect.size();
337 
338  std::unique_ptr<long[]> edgeConnect = std::unique_ptr<long[]>(new long[faceConnectSize]);
339  for (int k = 0; k < faceConnectSize; ++k) {
340  edgeConnect[k] = faceConnect[k];
341  }
342 
343  // Add edge
344  net.addCell(edgeType, true, std::move(edgeConnect));
345  }
346  } //next i
347  } //next c_
348 
349  return;
350 }

425 unsigned short SurfUnstructured::importSTL(STLObj &STL, int PIDOffset, bool PIDSquash)
426 {
427  // ====================================================================== //
428  // OPEN STL FILE //
429  // ====================================================================== //
430  STL.open("in");
431  if (STL.err != 0) {
432  return STL.err;
433  }
434 
435  // ====================================================================== //
436  // LOAD ALL SOLID FROM THE STL FILE //
437  // ====================================================================== //
438  int pid = PIDOffset;
439  if (!PIDSquash) {
440  --pid;
441  }
442 
443  while (true) {
444  // ====================================================================== //
445  // LOAD SOLID FROM THE STL FILE //
446  // ====================================================================== //
447  int nVertex = 0;
448  int nSimplex = 0;
449  std::vector<std::array<double, 3>> vertexList;
450  std::vector<std::array<double, 3>> normalList;
451  std::vector<std::array<int, 3>> connectivityList;
452 
453  STL.loadSolid(nVertex, nSimplex, vertexList, normalList, connectivityList);
454  if (nVertex == 0) {
455  break;
456  }
457 
458  // ====================================================================== //
459  // PID OF THE SOLID //
460  // ====================================================================== //
461  if (!PIDSquash) {
462  ++pid;
463  }
464 
465  // ====================================================================== //
466  // PREPARE MESH FOR DATA IMPORT //
467  // ====================================================================== //
468  reserveVertices(getVertexCount() + nVertex);
469  reserveCells(m_nInternals + m_nGhosts + nSimplex);
470 
471  // ====================================================================== //
472  // ADD VERTICES TO MESH //
473  // ====================================================================== //
474  vector<array<double, 3>>::const_iterator v_, ve_;
475 
476  std::vector<long> vertexMap;
477  vertexMap.reserve(nVertex);
478 
479  long v_counter = 0;
480  ve_ = vertexList.cend();
481  for (v_ = vertexList.cbegin(); v_ != ve_; ++v_) {
482  VertexIterator i_ = addVertex(*v_);
483  vertexMap[v_counter] = i_->getId();
484  ++v_counter;
485  } //next v_
486 
487  // ====================================================================== //
488  // ADD CELLS TO MESH //
489  // ====================================================================== //
490  vector<array<int,3>>::const_iterator c_, ce_;
491  array<int,3>::const_iterator w_, we_;
492 
493  ce_ = connectivityList.cend();
494  for (c_ = connectivityList.cbegin(); c_ != ce_; ++c_) {
495  // Remap STL connectivity
496  int n_v = c_->size();
497  std::vector<long> connect(n_v, Vertex::NULL_ID);
498  we_ = c_->cend();
499  int i = 0;
500  for (w_ = c_->cbegin(); w_ < we_; ++w_) {
501  connect[i] = vertexMap[*w_];
502  ++i;
503  } //next w_
504 
505  // Add cell
506  CellIterator cellIterator = addCell(getSTLFacetType(n_v), true, connect);
507  cellIterator->setPID(pid);
508  } //next c_
509 
510  // ====================================================================== //
511  // Multi-body STL files are supported only in ASCII mode //
512  // ====================================================================== //
513  if (STL.stl_type) {
514  break;
515  }
516  }
517 
518  // ====================================================================== //
519  // CLOSE STL FILE //
520  // ====================================================================== //
521  STL.close("in");
522 
523  return 0;
524 }

584 unsigned short SurfUnstructured::exportSTLSingle(const string &stl_name, const bool &isBinary, bool exportInternalsOnly)
585 {
586  // ====================================================================== //
587  // VARIABLES DECLARATION //
588  // ====================================================================== //
589 
590  // Local variables
591  int nVertex;
592  int nSimplex;
593  vector<array<double, 3>> vertexList;
594  vector<array<double, 3>> normalList;
595  vector<array<int,3>> connectivityList;
596  unordered_map<long, long> vertexMap;
597  array<int,3> dummyIntArray;
598 
599  // Counters
600  int v_count ,j;
601  vector<array<double, 3>>::iterator i_;
602  vector<array<int,3>>::iterator j_;
603  array<int,3>::iterator k_, ke_;
604  VertexIterator v_, ve_;
605  CellIterator c_, cb_, ce_;
606 
607  // ====================================================================== //
608  // INITIALIZE DATA STRUCTURE //
609  // ====================================================================== //
610  dummyIntArray.fill(0) ;
611 
612  nSimplex = m_nInternals;
613  if (!exportInternalsOnly) nSimplex += m_nGhosts;
614  vertexList.resize(getVertexCount());
615  normalList.resize(nSimplex);
616  connectivityList.resize(nSimplex, dummyIntArray);
617 
618  // ====================================================================== //
619  // CREATE VERTEX LIST //
620  // ====================================================================== //
621  i_ = vertexList.begin();
622  ve_ = vertexEnd();
623  v_count = 0;
624  for (v_ = vertexBegin(); v_ != ve_; ++v_) {
625 
626  // Store vertex coordinates
627  *i_ = v_->getCoords();
628  vertexMap[v_->getId()] = v_count;
629 
630  // Update counters
631  ++v_count;
632  ++i_;
633 
634  } //next v_
635  nVertex = getVertexCount();
636 
637  // ====================================================================== //
638  // CREATE CONNECTIVITY //
639  // ====================================================================== //
640  if (exportInternalsOnly) {
641  cb_ = internalBegin();
642  ce_ = internalEnd();
643  }
644  else {
645  cb_ = cellBegin();
646  ce_ = cellEnd();
647  }
648  i_ = normalList.begin();
649  j_ = connectivityList.begin();
650  for (c_ = cb_; c_ != ce_; ++c_) {
651 
652  // Build normals
653  *i_ = std::move(evalFacetNormal(c_->getId()));
654 
655  // Build connectivity
656  ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
657 
658  ke_ = j_->end();
659  j = 0;
660  for (k_ = j_->begin(); k_ != ke_; ++k_) {
661  *k_ = vertexMap[cellVertexIds[j]];
662  ++j;
663  } //next k_
664 
665  // Update counters
666  ++j_;
667  ++i_;
668  } //next c_
669 
670  // ====================================================================== //
671  // EXPORT STL DATA //
672  // ====================================================================== //
673  STLObj STL(stl_name, isBinary);
674  STL.save("", nVertex, nSimplex, vertexList, normalList, connectivityList);
675 
676  return 0;
677 }

690 unsigned short SurfUnstructured::exportSTLMulti(const string &stl_name, bool exportInternalsOnly)
691 {
692  int nTotVertex;
693  vector<array<double, 3>> totVertexList;
694  unordered_map<long, long> vertexMap;
695 
696  int nLocSimplex;
697  vector<array<double, 3>> normalLocList;
698  vector<array<int,3>> connectivityLocList;
699 
700  STLObj STL(stl_name, false);
701 
702  STL.open("out");
703  if (STL.err != 0) {
704  return STL.err;
705  }
706 
707  // Create the vertex map
708  nTotVertex = getVertexCount();
709  totVertexList.resize(nTotVertex);
710  unsigned int count_v = 0;
711  for(const Vertex &v: getVertices()){
712  totVertexList[count_v] = v.getCoords();
713  vertexMap[v.getId()] = count_v;
714  ++count_v;
715  }
716 
717  // Export the internal cells
718  for(int pid : getInternalPIDs()){
719  std::vector<long> cells = getInternalsByPID(pid);
720 
721  nLocSimplex = (int) cells.size();
722  connectivityLocList.resize(nLocSimplex);
723  normalLocList.resize(nLocSimplex);
724 
725  vector<array<int,3>>::iterator itC = connectivityLocList.begin();
726  vector<array<double, 3>>::iterator itN = normalLocList.begin();
727 
728  // Fill local connectivity and normals structures
729  for (auto id : cells) {
730  // Fill connectivity
731  const Cell &cell = getCell(id);
732  for (int iloc = 0; iloc<3; ++iloc) {
733  (*itC)[iloc] = vertexMap[cell.getConnect()[iloc]];
734  }
735 
736  // Fill normal
737  *itN = std::move(evalFacetNormal(cell.getId()));
738 
739  // Increment iterators
740  itC++;
741  itN++;
742  }
743 
744  // Write the solid associated to the current PID
745  STL.saveSolid(std::to_string(pid), nTotVertex, nLocSimplex, totVertexList, normalLocList, connectivityLocList);
746  }
747 
748  // Export ghost cells
749  long nGhosts = getGhostCount();
750  if (!exportInternalsOnly && nGhosts > 0) {
751  nLocSimplex = nGhosts;
752  connectivityLocList.resize(nLocSimplex);
753  normalLocList.resize(nLocSimplex);
754 
755  vector<array<int,3>>::iterator itC = connectivityLocList.begin();
756  vector<array<double, 3>>::iterator itN = normalLocList.begin();
757 
758  // Fill local connectivity and normals structures
759  CellConstIterator endItr = ghostConstEnd();
760  for (CellConstIterator itr = ghostConstBegin(); itr != endItr; ++itr) {
761  // Fill connectivity
762  for (int iloc = 0; iloc<3; ++iloc) {
763  (*itC)[iloc] = vertexMap[itr->getConnect()[iloc]];
764  }
765 
766  // Fill normal
767  *itN = std::move(evalFacetNormal(itr.getId()));
768 
769  // Increment iterators
770  itC++;
771  itN++;
772  }
773 
774  // Write the solid associated to the ghosts
775  STL.saveSolid("ghosts", nTotVertex, nLocSimplex, totVertexList, normalLocList, connectivityLocList);
776  }
777 
778  STL.close("out");
779 
780  return 0;
781 }

822 unsigned short SurfUnstructured::importDGF(const string &dgf_name, int PIDOffset, bool PIDSquash)

863  for (c_ = simplex_list.begin(); c_ != ce_; ++c_) {
864  ie_ = c_->end();
865  for (i_ = c_->begin(); i_ != ie_; ++i_) {
866  *i_ = vertex_map[*i_];
867  } //next i_
868  } //next c_
869

871  int k;
872  for (c_ = simplex_list.begin(), k = 0; c_ != ce_; ++c_, ++k) {
873  // Create cell
874  i_ = c_->begin();
875  connect.resize(c_->size(), Vertex::NULL_ID);
876  je_ = connect.end();
877  for (j_ = connect.begin(); j_ != je_; ++j_) {
878  *j_ = *i_;
879  ++i_;
880  } //next j_
881  CellIterator cellIterator = addCell(getDGFFacetType(c_->size()), true, connect);
882 
883  // Set cell PID
884  int cellPID = PIDOffset;
885  if (!PIDSquash) {
886  cellPID += simplex_PID[k];
887  }
888 
889  cellIterator->setPID(cellPID);
890  } //next c_
891 
892  return 0;
893 }
